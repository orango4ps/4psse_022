               OnPreDataItem=BEGIN

               OnAfterGetRecord=BEGIN
                                  IF ItemSeries <> '' THEN
                                    ItemCode := NoSeriesMgt.GetNextNo(ItemSeries,0D,TRUE)

                                  IF STRLEN(ItemCode) > MAXSTRLEN(EplanItemRec."No.") THEN
                                    CurrReport.BREAK

                                    CreateEplanItem;
