                    IF FilterCostCenter <> '' THEN

                      IF (EmployeeRec."No." <> locUserSetup."Employee No.") AND (NOT locUserSetup."Access own Hours only") AND (locUserSetup."Department Authoriz. Active")

                        IF NOT CheckEmployee(EmployeeRec."No.", locUserSetup."Department Code Filter") THEN BEGIN
                      ///***
                          //>>141103 ITERO.GH
                          //IF First THEN BEGIN
                          //  FilterEmployee := EmployeeRec."No.";

                          //END ELSE
                          //  FilterEmployee += '|' + EmployeeRec."No.";

                             ( DimensionCode <> GLSetup."Global Dimension 2 Code" ) THEN BEGIN

                             lvDefaultDim.SETRANGE("Table ID" , 5200 ); //Employee
                             lvDefaultDim.SETRANGE("No."                  , EmployeeRec."No.");
                             lvDefaultDim.SETRANGE("Dimension Code"       , DimensionCode );
                             lvDefaultDim.SETRANGE("Dimension Value Code" , DimensionValueCode );
                             IF lvDefaultDim.FINDSET THEN
                                IF First THEN BEGIN
                                    FilterEmployee := EmployeeRec."No.";
                                    First := FALSE
                                END ELSE
                                    FilterEmployee += '|' + EmployeeRec."No.";

                            END
                          ELSE BEGIN
                            IF First THEN BEGIN
                               FilterEmployee := EmployeeRec."No.";
                               First := FALSE
                             END ELSE
                               FilterEmployee += '|' + EmployeeRec."No.";
                           END;
                          //<<141103 ITERO.GH
                        END;
                      END;
                    UNTIL EmployeeRec.NEXT = 0;

                    //+++4PSSE.MK 140218
                    IF FilterEmployee = '' THEN
                      CurrReport.SKIP;
                    //---4PSSE.MK

                  END;

                  IF FromWeek <> 0 THEN
                    PeriodStart := DWY2DATE(1, FromWeek, FromYear);
                  IF ToWeek <> 0 THEN
                    PeriodEnd := DWY2DATE(7, ToWeek, ToYear);

                  //>>REP-100-024
                  IF FromDate <> 0D THEN BEGIN
                     PeriodStart := FromDate;
                     PeriodEnd   := ToDate;
                  END;
                  //<<REP-100-024

                  EntryNo := 1;
                  // *** Creates records from the hour accounting line ****
                  IF ShowOpenHours THEN
                    CreateOpenHours;


                  //**** Create from posted lines *** //       //////////////////////////////
                  IF ShowPostedHours THEN
                    CreatedPostedHours;


                  // *** Create temptable and present information***
                  TEMPHourAccoutBuffer.SETCURRENTKEY("Last Name","First Name", Date);

                  IF FilterHours <> '' THEN
                    TEMPHourAccoutBuffer.SETFILTER(Hours, FilterHours);
                  IF FilterAmount <> '' THEN
                    TEMPHourAccoutBuffer.SETFILTER(Amount, FilterAmount);
                  IF FilterWageCostCode <> '' THEN
                    TEMPHourAccoutBuffer.SETFILTER("Wage Component /Cost Code", FilterWageCostCode);

                  // >> REP-100-032
                  //** If the report is filtering on date instead of weeks and year a second filer
                  //** is need to be make to remove the hours that is not in the correct date period
                  IF FromDate <> 0D THEN BEGIN
                    IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
                     TEMPHourAccoutBuffer.SETRANGE(Date, PeriodStart, PeriodEnd)
                    ELSE IF (PeriodStart <> 0D) THEN
                      TEMPHourAccoutBuffer.SETFILTER(Date, '%1..', PeriodStart)
                    ELSE IF (PeriodEnd <> 0D) THEN
                      TEMPHourAccoutBuffer.SETFILTER(Date, '..%1', PeriodEnd);
                  END;
                  // << REP-100-032
                END;

    PROCEDURE CreateOpenHours@1100285507();
    VAR
      HourAccountingLine@1100285506 : Record 11012039;

    BEGIN
      // >> REP-100-032 ITERO.MK
      IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
        //>>141104 ENH050
        //HourAccounting2.SETRANGE("Posting Date", PeriodStart, PeriodEnd)
        HourAccounting2.SETRANGE("Document Date", PeriodStart, PeriodEnd)
        //<<141104 ENH050
      ELSE IF (PeriodStart <> 0D) THEN
        //>>141104 ENH050
        //HourAccounting2.SETFILTER("Posting Date", '%1..', PeriodStart)
        HourAccounting2.SETFILTER("Document Date", '%1..', PeriodStart)
        //<<141104 ENH050
      ELSE IF (PeriodEnd <> 0D) THEN
        //>>141104 ENH050
        //HourAccounting2.SETFILTER("Posting Date", '..%1', PeriodEnd);
        HourAccounting2.SETFILTER("Document Date", '..%1', PeriodEnd);
        //<<141104 ENH050
      IF FilterEmployee <> '' THEN
        HourAccounting2.SETFILTER("Employee No.", FilterEmployee);
      IF HourAccounting2.FINDSET(FALSE) THEN REPEAT

        //HourAccountingLine.SETCURRENTKEY(Year,Week,"Project No.","Employee No.");

        //IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
        //  HourAccountingLine.SETRANGE("Posting Date", PeriodStart, PeriodEnd)
        //ELSE IF (PeriodStart <> 0D) THEN
        //  HourAccountingLine.SETFILTER("Posting Date", '%1..', PeriodStart)
        //ELSE IF (PeriodEnd <> 0D) THEN
        //  HourAccountingLine.SETFILTER("Posting Date", '..%1', PeriodEnd);
        HourAccountingLine.RESET;
        HourAccountingLine.SETRANGE(Year, HourAccounting2.Year);
        HourAccountingLine.SETRANGE(Week, HourAccounting2.Week);
        HourAccountingLine.SETRANGE("Employee No.", HourAccounting2."Employee No.");
