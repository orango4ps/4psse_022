    PROCEDURE BudgetURL@1100525004() : Text;
    VAR

      HyperlinkManagement@1100525002 : Codeunit 11012351;
    BEGIN

      Job.COPYFILTER("Project Filter", BudgetLine."Project No.");

      EXIT(HyperlinkManagement.CreatePageURLWithFilter(RecRef));
    END;
