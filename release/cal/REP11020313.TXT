    LOCAL PROCEDURE FillShipToAddress@1100528403(ItemJournalLine@1100528400 : Record 83;Job@1100528401 : Record 11072003;VAR ShipToAddress@1100528402 : ARRAY [8] OF Text[100]);
    VAR
      ProjectLocation@1100528403 : Record 11012011;
      FormatAddress@1100528404 : Codeunit 365;
      Location@1100528405 : Record 14;
    BEGIN
      CLEAR(ShipToAddress);

      IF ItemJournalLine."Ship To Location" = '' THEN BEGIN
        FormatAddress.Project(ShipToAddress, Job);
        EXIT;
      END;

      IF ItemJournalLine.Deliver = ItemJournalLine.Deliver::"Project Location" THEN BEGIN
        IF ProjectLocation.GET(
          ItemJournalLine."Job No.",
          ItemJournalLine."Ship To Location")
        THEN
          FormatAddress.ProjectLoc(ShipToAddress, ProjectLocation);
        EXIT;
      END;

      IF ItemJournalLine.Deliver = ItemJournalLine.Deliver::Location THEN BEGIN
        IF Location.GET(ItemJournalLine."Ship To Location") THEN
          FormatAddress.Location(ShipToAddress, Location);
        EXIT;
      END;
    END;

    LOCAL PROCEDURE FillVars@1100525004(ItemJournalLine@1100525000 : Record 83);
    VAR
      Employee@1100525001 : Record 5200;
    BEGIN
      ProjectOrServiceOrderNo := '';
      ProjectOrServiceOrderNo_Caption := '';
      ProjectOrServiceOrderDescription := '';
      ProjectOrOrderManager := '';
      ProjectOrOrderManagerCaption := '';
      ProjectOrOrderManagerDescription := '';

      IF ItemJournalLine."Job No." <> '' THEN BEGIN
        ProjectOrServiceOrderNo := ItemJournalLine."Job No.";
        ProjectOrServiceOrderNo_Caption := ItemJournalLine.FIELDCAPTION("Job No.");
        ProjectOrServiceOrderDescription := Job.Description;
        ProjectOrOrderManager := Job."Project Manager";
        ProjectOrOrderManagerCaption := Job.FIELDCAPTION("Project Manager");
      END;

      IF ItemJournalLine."Service Order No." <> '' THEN BEGIN
        ProjectOrServiceOrderNo := ItemJournalLine."Service Order No.";
        ProjectOrServiceOrderNo_Caption := ItemJournalLine.FIELDCAPTION("Service Order No.");
        ProjectOrServiceOrderDescription := ServiceOrder.Description;
        ProjectOrOrderManager := ServiceOrder."Order Manager";
        ProjectOrOrderManagerCaption := ServiceOrder.FIELDCAPTION("Order Manager");
      END;

      IF ProjectOrOrderManager <> '' THEN
        IF Employee.GET(ProjectOrOrderManager) THEN
          ProjectOrOrderManagerDescription := Employee."Full Name";
    END;

    BEGIN
    END.
