OBJECT Codeunit 1304 Sales-Quote to Invoice Yes/No
{
  OBJECT-PROPERTIES
  {
    Date=;
    Time=;
    Version List=NAVW110.00;
  }
  PROPERTIES
  {
    TableNo=36;
    OnRun=VAR
            InvoiceSalesHeader@1001 : Record 36;
            SalesQuoteToInvoice@1000 : Codeunit 1305;
            OfficeMgt@1002 : Codeunit 1630;
          BEGIN
            TESTFIELD("Document Type","Document Type"::Quote);
            IF GUIALLOWED THEN
              IF NOT CONFIRM(ConfirmConvertToInvoiceQst,FALSE) THEN
                EXIT;

            SalesQuoteToInvoice.RUN(Rec);
            SalesQuoteToInvoice.GetSalesInvoiceHeader(InvoiceSalesHeader);

            COMMIT;

            IF GUIALLOWED THEN
              IF OfficeMgt.AttachAvailable THEN
                PAGE.RUN(PAGE::"Sales Invoice",InvoiceSalesHeader)
              ELSE
                IF CONFIRM(STRSUBSTNO(OpenNewInvoiceQst,InvoiceSalesHeader."No."),TRUE) THEN
                  PAGE.RUN(PAGE::"Sales Invoice",InvoiceSalesHeader);
          END;

  }
  CODE
  {
    VAR
      ConfirmConvertToInvoiceQst@1000 : TextConst 'ENU=Do you want to convert the quote to an invoice?;NOR=Vil du konvertere tilbudet til en faktura?;SVE=Vill du konvertera offerten till en faktura?';
      OpenNewInvoiceQst@1001 : TextConst 'ENU=The quote has been converted to invoice %1. Do you want to open the new invoice?;NOR=Tilbudet er konvertert til faktura %1. Vil du †pne den nye fakturaen?;SVE=Offerten har konverterats till faktura %1. Vill du ”ppna den nya fakturan?';

    BEGIN
    END.
  }
}
