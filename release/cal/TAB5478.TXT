    PROCEDURE UpdateItemId@1();
    VAR
      Item@1000 : Record 27;
    BEGIN
      IF ("No." = '') OR (Type <> Type::Item) THEN BEGIN

      END;

        EXIT;

      "Item Id" := Item.Id;
    END;

    [External]
    PROCEDURE UpdateAccountId@12();
    VAR
      GLAccount@1000 : Record 15;
    BEGIN
      IF ("No." = '') OR (Type <> Type::"G/L Account") THEN BEGIN

        EXIT;
      END;

    END;

    [External]
    PROCEDURE UpdateNo@5();
    VAR
      Item@1001 : Record 27;
      GLAccount@1000 : Record 15;
    BEGIN
      CASE Type OF
        Type::Item:
          BEGIN
            Item.SETRANGE(Id,"Item Id");
            IF NOT Item.FINDFIRST THEN
              EXIT;

            "No." := Item."No.";
          END;
        Type::"G/L Account":
          BEGIN
            GLAccount.SETRANGE(Id,"Account Id");
            IF NOT GLAccount.FINDFIRST THEN
              EXIT;

            "No." := GLAccount."No.";
          END;
      END;
    END;
