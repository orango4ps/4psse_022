    PROCEDURE GetReportFilter@1100525000() : Text;

    PROCEDURE CheckBookMarks@1100525008(iText@1100525001 : Text[250];ContractorNo@1100525007 : Integer) oText : Text[250];
    VAR
      lvSep@1100525002 : Text[30];
      lvPos@1100525000 : Integer;
      lvLen@1100525006 : Integer;
      lvCounter@1100525003 : Integer;
      lvText001@1100525004 : TextConst 'DEU=..%1;ENU=%1);NLD=%1)';
    BEGIN
      IF STRPOS(iText, ')') = 0 THEN
        EXIT(iText);


      REPEAT
        lvCounter := lvCounter + 1;
        lvSep := STRSUBSTNO(lvText001, lvCounter);
        lvPos := STRPOS(iText, lvSep);
        IF lvPos <> 0 THEN BEGIN
          lvLen := STRLEN(iText) - lvPos;
          oText := COPYSTR(iText, lvPos+2, lvLen);
          IF oText <> '' THEN
            EXIT(iText);  //legenda
          oText := COPYSTR(iText, 1, lvPos-1);
          oText := DELCHR(oText, '>', ' ');
          lvLen := STRLEN(oText);
          oText := COPYSTR(iText, 1, lvLen);
          CASE lvCounter OF
            1:
              BEGIN
                CASE ContractorNo OF
                  1: oText := oText + ' ' + EstRec."Contractor Name";
                  2: oText := oText + ' ' + EstRec."Contractor 2 Name";
                  3: oText := oText + ' ' + EstRec."Contractor 3 Name";
                END;
              END;
            2:
              BEGIN
                CASE ContractorNo OF
                  1: oText := oText + ' ' + EstRec."Contractor City";
                  2: oText := oText + ' ' + EstRec."Contractor 2 City";
                  3: oText := oText + ' ' + EstRec."Contractor 3 City";
                END;
              END;
            3:
              BEGIN
                PageHeaderFill.SETRANGE("Specification No.", EstRec."Construction Spec. No.");
                PageHeaderFill.SETRANGE(Section, PageHeaderFill.Section::"Bid Form");
                IF PageHeaderFill.FINDSET THEN BEGIN
                  REPEAT
                    lvLen := MAXSTRLEN(oText) - STRLEN(oText);
                    oText := oText + ' ' + COPYSTR(PageHeaderFill.Text, 1, lvLen);
                  UNTIL PageHeaderFill.NEXT = 0;
                END;
              END;
            4: oText := oText + ' ' + FormatDec(EstRec."Contract Amount (Actual)");
            5: oText := oText + ' ' + ConvertAmnt.Zegge(ROUND(EstRec."Contract Amount (Actual)", 0.01));
            6: oText := oText + ' ' + FormatDec(VatBaseAmnt);
            7: oText := oText + ' ' + ConvertAmnt.Zegge(ROUND(VatBaseAmnt, 0.01));
            //8: nvt
            //9: ??
          END;
          oText := oText + '   ' + lvSep;
          EXIT(oText);
        END;
      UNTIL lvCounter = 10;

      EXIT(iText);
    END;

    PROCEDURE CheckSignature@1100525001(Text@1100525000 : Text[250]) Result : Text[250];
    BEGIN
      Result := Text;

      IF COPYSTR(DELCHR(Text, '='), 1, STRLEN('Gedaantede')) = 'Gedaantede' THEN // ouch! :(
        Result := STRSUBSTNO('Gedaan te %1 de %2,',
          EstRec."Contractor City",
          FORMAT(TenderDate, 0, '<Day> <Month Text> <Year4>'));  //X001
    END;

    LOCAL PROCEDURE GetBookMarkByTableField@1100525005(iText@1100525000 : Text[250]) rTxtContent : Text;
    BEGIN
      //C037233
      WITH "Constr. Spec. Bid Form (RAW)" DO BEGIN
        IF "Field No." <> 0 THEN BEGIN
          CASE "Table No." OF
            "Table No."::CompInfo: rTxtContent := iText + ' ' + GetSubstituteFromSourceTable(CompanyInformation);
            "Table No."::Estimate: rTxtContent := iText + ' ' + GetSubstituteFromSourceTable(EstRec);
          END;
        END ELSE BEGIN
          CASE "Table Function" OF
            "Table Function"::ContractAmountText: rTxtContent := iText + ' ' + ConvertAmnt.Zegge(ROUND(EstRec."Contract Amount (Actual)", 0.01));
            "Table Function"::VatAmount: rTxtContent := iText + ' ' + FormatDec(VatBaseAmnt);
            "Table Function"::VatAmountText: rTxtContent := iText + ' ' + ConvertAmnt.Zegge(ROUND(VatBaseAmnt, 0.01));
        END;
        END;
      END;
      EXIT(rTxtContent);
    END;

    END;

    BEGIN
