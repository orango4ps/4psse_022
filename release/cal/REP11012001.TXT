    PROCEDURE GetReportFilter@1100525000() : Text;

    PROCEDURE CheckSameProperty@1100485002(VAR SubTotal@1100485000 : Decimal;VAR SubTotalAmount@1100485001 : Decimal;VAR SubTotalHours@1100525000 : Decimal) : Decimal;
    VAR
      BudgetLine@1210190000 : Record 11012001;
    BEGIN
      //db, 01-07-05
      WITH "Budget Line" DO BEGIN
        SubTotal := Quantity;
        SubTotalAmount := "Amount (LCY)";
        BudgetLine.COPY("Budget Line");

        BudgetLine.SETRANGE("Project No.", "Project No.");
        BudgetLine.SETRANGE(Element, Element);
        BudgetLine.SETRANGE("Extension Contract", "Extension Contract");
        BudgetLine.SETRANGE("Item No.", "Item No.");
        BudgetLine.SETRANGE("Basic Item", "Basic Item");
        BudgetLine.SETRANGE(Manufacturer, Manufacturer);
        BudgetLine.SETRANGE("Trade Item", "Trade Item");
        BudgetLine.SETRANGE("Vendor (Trade Item)", "Vendor (Trade Item)");
        BudgetLine.SETRANGE("Cost Object", "Cost Object");
        BudgetLine.SETRANGE(Description, Description);
        BudgetLine.SETRANGE("Description 2", "Description 2");
        BudgetLine.SETRANGE("Unit of Measure", "Unit of Measure");
        BudgetLine.SETRANGE("External Text", "External Text");
        BudgetLine.SETRANGE("Price (LCY)", "Price (LCY)");
        BudgetLine.SETRANGE("Time Quantity", "Time Quantity");
        BudgetLine.SETRANGE("Unit of Time", "Unit of Time");
        BudgetLine.SETFILTER("Line No.", '<>%1', "Line No.");
        IF BudgetLine.FINDSET(FALSE, FALSE) THEN BEGIN
          REPEAT
            TmpBudgetLine := BudgetLine;
            IF TmpBudgetLine.INSERT THEN;

            SubTotal := SubTotal + BudgetLine.Quantity;
            SubTotalAmount := SubTotalAmount + BudgetLine."Amount (LCY)";
            SubTotalHours := SubTotalHours + BudgetLine.Hours;

          UNTIL BudgetLine.NEXT = 0;
        END;
      END;
    END;

    END;
