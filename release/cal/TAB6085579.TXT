        OptionString := OptionString + ',' + Text003;

            EXIT(TempTemplate."No.");

      Template.SETRANGE("Category Code",DocCatCode);
      Template.SETRANGE(Type,Template.Type::Master);
      Template.FINDFIRST;

        IF PAGE.RUNMODAL(0,Template) <> ACTION::LookupOK THEN
          EXIT;
