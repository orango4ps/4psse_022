               DataItemTable=Table11012472;
               DataItemTableView=SORTING(Project No.,Principal,Production Date,Extension Contract,Revision Rule,Tender Paragraph No.);
               PrintOnlyIfDetail=No;
               OnPreDataItem=BEGIN
                               CurrReport.CREATETOTALS("Production Value (FCY)");

                               //Clear variables for printing of formulas

                               CLEAR(FormulaCode);

               OnAfterGetRecord=VAR
                                  i@1100525003 : Integer;
                                  j@1100525002 : Integer;
                                  FormulaFound@1100525001 : Boolean;
                                  EmptyEntryFound@1100525000 : Boolean;
                                BEGIN
                                  //Fill variables for printing of formulas

                                  IF "Revision Rule" = '' THEN
                                    //Only process revision rule lines overhere
                                    EXIT;

                                  IF "Extension Contract" = '' THEN BEGIN
                                    //only one formule per project
                                    FormulaCodeProject := "Revision Rule";
                                    EXIT;
                                  END;

                                  //Process lines for extension contracts
                                  //Look whether there is another extension contract that uses the same formulas
                                  FormulaFound := FALSE;
                                  i := 0;
                                  WHILE NOT FormulaFound AND (i <= LastFormula) DO BEGIN
                                    i += 1;
                                    IF FormulaCode[i] = "Revision Rule" THEN
                                      FormulaFound := TRUE;
                                  END;

                                  IF FormulaFound THEN BEGIN
                                    //Formula is found, register contract under formula
                                    j := 0;
                                    WHILE NOT EmptyEntryFound AND (j <= 1000) DO BEGIN
                                      j += 1;
                                      IF ContractNumbers[i,j] = '' THEN
                                        EmptyEntryFound := TRUE;
                                    END;
                                    IF NOT EmptyEntryFound THEN
                                      ERROR(ErrMsg002, "Project No.");
                                    ContractNumbers[i,j] := "Extension Contract";
                                  END ELSE BEGIN
                                    //This is a new formula, add it
                                    LastFormula += 1;
                                    IF LastFormula > 20 THEN
                                      ERROR(ErrMsg001, "Project No.");
                                    FormulaCode[LastFormula] := "Revision Rule";
                                    ContractNumbers[LastFormula,1] := "Extension Contract";
                                  END;

                                  PrintInstalMot := TRUE;
                                  IF ("Installment Motivation".Status <> "Installment Motivation".Status::Approved) OR ("Installment Motivation"."Approved By Name" = '') THEN
                                    PrintInstalMot := FALSE;
                                END;

    { 1100525008;2;DataItem;FormulaContractLoop;
               DataItemTable=Table2000000026;
               DataItemTableView=SORTING(Number)
                                 WHERE(Number=FILTER(1..));
               OnAfterGetRecord=VAR
                                  DoBreak@1100525000 : Boolean;
                                BEGIN
                                  IF (FormulaContractLoop.Number > 1000) THEN
                                    CurrReport.BREAK;
                                  IF (ContractNumbers[FormulaLoop.Number, FormulaContractLoop.Number] = '') THEN
                                    CurrReport.BREAK;

                                  IF (FormulaCode[FormulaLoop.Number] = '') THEN BEGIN
                                    //can occur if there is a formula for the project but not for a extension contract
                                    CurrReport.SKIP
                                  END ELSE
                                    RevRuleRec.GET(FormulaCode[FormulaLoop.Number]);

                                  DoBreak := FALSE;
                                  ContractList := '';
                                  WHILE NOT DoBreak DO BEGIN
                                    ContractList += ContractNumbers[FormulaLoop.Number,FormulaContractLoop.Number];
                                    FormulaContractLoop.Number += 1;
                                    DoBreak := (FormulaContractLoop.Number > 1000);
                                    IF NOT DoBreak THEN
                                      DoBreak := (ContractNumbers[FormulaLoop.Number, FormulaContractLoop.Number] = '');
                                    IF NOT DoBreak THEN
                                      //there are more contracts to handle, add a ', '
                                      ContractList +=', ';
                                      //there is approximately room for 100 characters on a line, if we are about to execeed the 100, go to the next line.
                                    //note that the length of the Contract Code plus a ', ' is 12.
                                    IF NOT DoBreak THEN
                                      DoBreak := (STRLEN(ContractList) > (100 - 12));
                                    IF DoBreak THEN
                                      //minus one, the navision reporting engine will add one in the next iteration
                                      FormulaContractLoop.Number -= 1;
                                  END;
                                END;
                                 }
