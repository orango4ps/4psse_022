    LOCAL PROCEDURE GetReportName@1100525003() : Text;

    END;

    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Work Order".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', "Work Order".TABLECAPTION, "Work Order".GETFILTERS));
    END;

    LOCAL PROCEDURE SetFiltersOnWorkOrder@1100529601();
    BEGIN
      "Work Order".SETRANGE("Source Company", COMPANYNAME);

      CASE SourceType OF
        SourceType::Project:
          "Work Order".SETRANGE("Source Type", "Work Order"."Source Type"::Project);
        SourceType::ServiceOrder:
          "Work Order".SETRANGE("Source Type", "Work Order"."Source Type"::ServiceOrder);
        SourceType::Request:
          "Work Order".SETRANGE("Source Type", "Work Order"."Source Type"::Request);
      END;
      IF (SourceType <> SourceType::" ") AND (SourceNo <> '') THEN
        "Work Order".SETRANGE("Source No.", SourceNo);
    END;

    LOCAL PROCEDURE WorkOrderCanBeRemoved@1100529600() : Boolean;
    BEGIN
      MessageText := '';
      IF NOT DeleteWhenSourceIsFinished THEN
        EXIT(TRUE);

      CASE "Work Order"."Source Type" OF
        "Work Order"."Source Type"::Project:
          IF Project.GET("Work Order"."Source No.") THEN BEGIN
            IF Project."Project Status" = Project."Project Status"::Finished THEN
              EXIT(TRUE);
            MessageText := Text003;
            EXIT(FALSE);
          END;
        "Work Order"."Source Type"::ServiceOrder:
          IF ServiceOrder.GET("Work Order"."Source No.") THEN BEGIN
            IF ServiceOrder.Status = ServiceOrder.Status::Finished THEN
              EXIT(TRUE);
            MessageText := Text003;
            EXIT(FALSE);
          END;
        "Work Order"."Source Type"::Request:
          IF ResourceRequest.GET("Work Order"."Source No.") THEN BEGIN
            IF ResourceRequest.Status = ResourceRequest.Status::Finished THEN
              EXIT(TRUE);
            MessageText := Text003;
            EXIT(FALSE);
          END;
      END;

      MessageText := Text002;
      EXIT(FALSE);
    END;

    BEGIN
    END.
