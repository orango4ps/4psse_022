    LOCAL PROCEDURE InsertTmpReportLine@1100529600(ReportingDate@1100529604 : Date;SeqNo@1100529603 : Integer;PeriodEndingDate@1100529602 : Date;ServiceContractNo@1100529601 : Code[20];ContractControlPeriodCode@1100529600 : Code[10];StartingDate@1100529607 : Date;EndingDate@1100529608 : Date;DateFinished@1100529609 : Date;AddText@1100529610 : Boolean;LineType@1100529605 : Integer);
    BEGIN
      IF NOT TmpReportLine.GET(
        FORMAT(ReportingDate),
        FORMAT(SeqNo),
        FORMAT(PeriodEndingDate),
        ServiceContractNo,
        ContractControlPeriodCode)
      THEN BEGIN
        TmpReportLine.INIT;
        TmpReportLine."Key Code1" := FORMAT(ReportingDate);
        TmpReportLine."Key Code2" := FORMAT(SeqNo);
        TmpReportLine."Key Code3" := FORMAT(PeriodEndingDate);
        TmpReportLine."Key Code4" := ServiceContractNo;
        TmpReportLine."Key Code5" := ContractControlPeriodCode;

        TmpReportLine.Int1 := LineType;
        IF AddText THEN BEGIN
          CASE LineType OF
            1: TmpReportLine.Text1 := TmpServiceContractPreclosure.TABLECAPTION;
            2: TmpReportLine.Text1 := TmpServiceContractPreclLine.TABLECAPTION;
            3: TmpReportLine.Text1 := TmpServiceContrPreclDetail.TABLECAPTION;
          END;

          TmpReportLine.Date1 := ReportingDate;
          TmpReportLine.Date2 := PeriodEndingDate;

          IF SeqNo <> 0 THEN
            TmpReportLine.Code1 := FORMAT(SeqNo);
          TmpReportLine.Code2 := ServiceContractNo;
          TmpReportLine.Code3 := ContractControlPeriodCode;

          TmpReportLine.Date3 := StartingDate;
          TmpReportLine.Date4 := EndingDate;
          TmpReportLine.Date5 := DateFinished;
        END;
        TmpReportLine.INSERT;
      END;
    END;

    BEGIN
    END.
