    LOCAL PROCEDURE RemoveVisibleResourcesOfUnknownUsers@1100528605();
    VAR
      VisibleResourceWOP@1100528600 : Record 11229285;
      UserSetup@1100528601 : Record 91;
    BEGIN
      IF VisibleResourceWOP.FINDSET THEN
        REPEAT
          VisibleResourceWOP.CALCFIELDS("Resource Company");
          IF UserSetup.CURRENTCOMPANY <> VisibleResourceWOP."Resource Company" THEN
            UserSetup.CHANGECOMPANY(VisibleResourceWOP."Resource Company");
          IF NOT UserSetup.GET(VisibleResourceWOP."User ID") THEN
            VisibleResourceWOP.DELETE;
        UNTIL VisibleResourceWOP.NEXT = 0;
    END;

    BEGIN
    END.
