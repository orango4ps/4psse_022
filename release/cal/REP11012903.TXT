    PROCEDURE CalculateRubricBudget@1210190011() : Decimal;
    BEGIN
      WITH "Investm. Proposal Rubric Line" DO BEGIN
        CASE Type OF
          Type::Costs,
          Type::Revenues,
          Type::"Percentage Costs",
          Type::"Percentage Revenues",
          Type::"Costs as Perc. Revenues",
          Type::"Revenues as Perc. Costs",
          Type::"Costs to Distr. as Perc. Rev.",
          Type::"Perc. (Sub)Totals",  //*15625.n
          Type::Subtotal,
          Type::Balance:
            EXIT(ProjectDevelopmentFunctions.CalculateBudgetByRubric("Investm. Proposal Rubric Line", 0D, 0D, ReferenceDate));
          Type::ROS:
            EXIT(ProjectDevelopmentFunctions.CalculateTotalRos("Project No.", Version, ReferenceDate, TRUE));
          Type::EPV:
            EXIT(ProjectDevelopmentFunctions.CalculateTotalEPV("Project No.", Version, ReferenceDate, PeriodType, TRUE));
          Type::PRI:
            EXIT(ProjectDevelopmentFunctions.CalculateTotalPRI("Project No.", Version, ReferenceDate, PeriodType, TRUE));
        END;
        EXIT(0);
      END;
    END;

    PROCEDURE CalculatePeriodAmounts@1210190006();
    BEGIN
      WITH "Investm. Proposal Rubric Line" DO BEGIN
        CLEAR(AmntArray);
        FOR Counter := 1 TO NoOfColumn DO BEGIN
          CASE Type OF
            Type::Costs,
            Type::Revenues,
            Type::"Percentage Costs",
            Type::"Percentage Revenues",
            Type::"Costs as Perc. Revenues",
            Type::"Revenues as Perc. Costs",
            Type::"Costs to Distr. as Perc. Rev.",
            Type::Subtotal,
            Type::Balance:
              AmntArray[Counter] := ProjectDevelopmentFunctions.CalculateBudgetByRubric("Investm. Proposal Rubric Line",
                                          StartDateArray[Counter], GetEndDate(Counter), ReferenceDate);
            Type::EPV:
              BEGIN
                Date_EndPeriod := EndDateArray[Counter];
                IF Counter = NoOfColumn THEN
                  Date_EndPeriod := 0D;  //* Hierdoor EPV vanaf laatste periode (kolom 'Period..')
                AmntArray[Counter] := ProjectDevelopmentFunctions.CalculatePeriodEPV(
                  "Project No.", Version, ReferenceDate, PeriodType,
                  StartDateArray[Counter], Date_EndPeriod, SplitType::Budget, TRUE);
              END;
          END;
        END;
      END;
    END;
