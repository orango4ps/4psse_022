    PROCEDURE CalculateOutstanding@1000000000();
    VAR
      PurchLine@1000000000 : Record 39;
    BEGIN
      TempNextToApprove.RESET;
      TempNextToApprove.DELETEALL;
      TempDocLineApp.RESET;
      TempDocLineApp.DELETEALL;

      OutStanding := 0;

      PurchLine.RESET;
      PurchLine.SETRANGE("Document Type","Purchase Header"."Document Type");
      PurchLine.SETRANGE("Document No.","Purchase Header"."No.");
      IF OnlyIncludeGL THEN
        PurchLine.SETRANGE(Type, PurchLine.Type::"G/L Account");
      IF ExcludeNonExFlowLines THEN
        PurchLine.SETRANGE(Approve, TRUE);
      IF PurchLine.FINDSET THEN BEGIN
        REPEAT
          BuildNextApprover(PurchLine);
          IF CheckUserGroupFilter THEN BEGIN
            IF CalcOutstandingAmt(PurchLine,OutStanding) THEN
              CreatePrintLine(PurchLine);
          END;
        UNTIL PurchLine.NEXT = 0;

        IF "Purchase Header"."Currency Code" <> '' THEN BEGIN
          IF CurrencyDate = CurrencyDate::"Currency Factor from PO" THEN
            OutStanding := OutStanding / "Purchase Header"."Currency Factor"
          ELSE
            OutStanding :=
              ROUND(
                CurrExchRate.ExchangeAmtFCYToLCY(0,"Purchase Header"."Job No.", //4PS
                  WORKDATE,"Purchase Header"."Currency Code",
                  OutStanding,CurrExchRate.ExchangeRate(0,"Purchase Header"."Job No.",CurrDateToUse,"Purchase Header"."Currency Code",FALSE),FALSE)); // 4PS
        END;
      END;
    END;

    PROCEDURE CalcOutstandingAmt@1100285002(_PurchLine@1100285002 : Record 39;VAR _NetAmount@1100285000 : Decimal) : Boolean;
    VAR
      TotalQtyAlreadyMatched@1100285006 : Decimal;
      TotalAmtAlreadyMatched@1100285005 : Decimal;
      ReceivedPerc@1100285003 : Decimal;
      LineNetAmount@1100285001 : Decimal;
      TotalAmtAlreadyMatchedLCY@1100285004 : Decimal;
      ExflowGetOrder@1100285007 : Codeunit 12013608;
      bMultiCurr@1100285008 : Boolean;
    BEGIN
      IF CheckSkipLine(_PurchLine) THEN
        EXIT(FALSE);

      // Order amount or Invoices/Credit Memos
      IF (TypeofAmount = TypeofAmount::"Order Amount") OR NOT (_PurchLine."Document Type" IN
                                                                [_PurchLine."Document Type"::Order,
                                                                 _PurchLine."Document Type"::"Return Order"]) THEN BEGIN
        LineNetAmount := (_PurchLine.Quantity * _PurchLine."Direct Unit Cost" * (1 - _PurchLine."Line Discount %" / 100))

      END
      ELSE BEGIN
      // Only Orders with a received amount
        ReceivedPerc := 0;
        IF _PurchLine."Quantity Received" <> 0 THEN
          IF _PurchLine.Quantity < _PurchLine."Quantity Received" THEN
            ReceivedPerc := 1
          ELSE
            ReceivedPerc := ROUND(_PurchLine."Quantity Received" / _PurchLine.Quantity,0.0001);

        LineNetAmount := ReceivedPerc * (_PurchLine.Quantity * _PurchLine."Direct Unit Cost" * (1 - _PurchLine."Line Discount %" / 100))
      END;

      IF (LineNetAmount <> 0 ) AND (_PurchLine."Document Type" IN [_PurchLine."Document Type"::Order,
                                                             _PurchLine."Document Type"::"Return Order"])
      THEN BEGIN
        IF ReduceOrderAmountWithOpenDoc AND
           (_PurchLine."Document Type" IN [_PurchLine."Document Type"::Order,_PurchLine."Document Type"::"Return Order"]) THEN BEGIN

          ExFlow.CalcMatchedTotalsPurchLine(_PurchLine,TotalQtyAlreadyMatched,TotalAmtAlreadyMatched,TotalAmtAlreadyMatchedLCY);
        END
        ELSE
          ExFlow.CalcPostedMatchedPurchLine(_PurchLine,TotalQtyAlreadyMatched,TotalAmtAlreadyMatched,TotalAmtAlreadyMatchedLCY);

        IF ExFlowSetup."Order Applies-to" = ExFlowSetup."Order Applies-to"::"Purch Order Lines" THEN
          bMultiCurr := ExflowGetOrder.MultiCurr(_PurchLine."No.",_PurchLine."Line No.","Purchase Header"."Currency Code");

        IF bMultiCurr THEN BEGIN

          IF ("Purchase Header"."Currency Code" <> '') THEN BEGIN
            IF CurrencyDate = CurrencyDate::"Currency Factor from PO" THEN
              LineNetAmount :=  LineNetAmount / "Purchase Header"."Currency Factor"
              ELSE
                OutStanding :=
                  ROUND(
                    CurrExchRate.ExchangeAmtFCYToLCY(0,"Purchase Header"."Job No.", // 4PS
                      WORKDATE,"Purchase Header"."Currency Code",
                      LineNetAmount,CurrExchRate.ExchangeRate(0,"Purchase Header"."Job No.",CurrDateToUse,"Purchase Header"."Currency Code",FALSE),FALSE));
      END;

          LineNetAmount := LineNetAmount - TotalAmtAlreadyMatchedLCY;

          IF "Purchase Header"."Currency Code" <> '' THEN BEGIN
            IF CurrencyDate = CurrencyDate::"Currency Factor from PO" THEN
              LineNetAmount :=  LineNetAmount * "Purchase Header"."Currency Factor"
              ELSE
                OutStanding :=
                  ROUND(
                    CurrExchRate.ExchangeAmtLCYToFCY(0,"Purchase Header"."Job No.", // 4PS
                      WORKDATE,"Purchase Header"."Currency Code",
                      LineNetAmount,CurrExchRate.ExchangeRate(0,"Purchase Header"."Job No.",CurrDateToUse,"Purchase Header"."Currency Code",FALSE),FALSE)); // 4PS
          END;
        END ELSE BEGIN
          IF LineNetAmount < 0 THEN    //2649
             LineNetAmount := LineNetAmount - TotalAmtAlreadyMatched  //2649
          ELSE BEGIN  //2649
            LineNetAmount := LineNetAmount - TotalAmtAlreadyMatched;
            IF LineNetAmount < 0 THEN
             LineNetAmount := 0;
          END;  //2649
        END;
      END;

      _NetAmount := _NetAmount + LineNetAmount;

      //IF LineNetAmount > 0 THEN
      IF LineNetAmount <> 0 THEN
        EXIT(TRUE)
      ELSE
        EXIT(FALSE);
    END;

    PROCEDURE CheckSkipLine@1100285003(_PurchLine@1100285000 : Record 39) : Boolean;
    BEGIN
      IF _PurchLine."Fully Matched" THEN
        EXIT(TRUE);

      IF _PurchLine.Type = _PurchLine.Type::" " THEN
        EXIT(TRUE);

      IF _PurchLine."No." = '' THEN
        EXIT(TRUE);

      //348884
      IF _PurchLine."VAT Calculation Type" = _PurchLine."VAT Calculation Type"::"Full VAT" THEN
        EXIT(TRUE);
      //348884

      IF ExcludeNonExFlowLines AND NOT _PurchLine.Approve THEN
        EXIT(TRUE);

      IF MatchedAgainstOrder <> MatchedAgainstOrder::All THEN BEGIN
        IF (_PurchLine."Document Type" = _PurchLine."Document Type"::Invoice) OR
           (_PurchLine."Document Type" = _PurchLine."Document Type"::"Credit Memo") THEN
          IF MatchedAgainstOrder = MatchedAgainstOrder::"Exclude matched against order" THEN BEGIN
            IF (_PurchLine."Receipt No." <> '') OR (_PurchLine."Return Shipment No." <> '') OR (_PurchLine."ExFlow Order No." <> '') THEN
              EXIT(TRUE);
          END
          ELSE
            IF MatchedAgainstOrder = MatchedAgainstOrder::"Only matched against order" THEN
              IF (_PurchLine."Receipt No." = '') AND (_PurchLine."Return Shipment No." = '') AND (_PurchLine."ExFlow Order No." = '') THEN
                EXIT(TRUE);
      END;

      IF _PurchLine.Quantity = 0 THEN
        EXIT(TRUE);

    PROCEDURE CreatePrintLine@1100285009(_PurchLine@1100285000 : Record 39);
    VAR
      DocLine@1100285001 : Record 12013609;
      DocLineDim@1100285004 : Record 12013611;
    BEGIN
      CASE RetrieveLinesFrom OF
        RetrieveLinesFrom::"Purchase Line":
          BEGIN
            PrintTempPurchLine.INIT;
            PrintTempPurchLine.TRANSFERFIELDS(_PurchLine);
            PrintTempPurchLine.INSERT;
          END;
        RetrieveLinesFrom::"Ex Document Line":
          BEGIN
            DocLine.RESET;
            DocLine.SETCURRENTKEY("Document No.");
            DocLine.SETRANGE("Document Type", _PurchLine."Document Type");
            DocLine.SETRANGE("Document No.", _PurchLine."Document No.");

                    PrintTempPurchLine."Shortcut Dimension 1 Code" := DocLineDim."Dimension Value";
                  IF GLSetup."Global Dimension 2 Code" = DocLineDim."Dimension Code" THEN
                    PrintTempPurchLine."Shortcut Dimension 2 Code" := DocLineDim."Dimension Value";
                UNTIL DocLineDim.NEXT = 0;
              //4PS
              PrintTempPurchLine."Job No." := DocLine."Job No.";
              PrintTempPurchLine."Sales Order No." := DocLine."Service Order No.";
              //4PS

              PrintTempPurchLine.INSERT;
            END;

            IF NOT NewLinesAdded THEN BEGIN
              DocLine.RESET;
              DocLine.SETCURRENTKEY("Document No.");
              DocLine.SETRANGE("Document Type", _PurchLine."Document Type");
              DocLine.SETRANGE("Document No.", _PurchLine."Document No.");
              DocLine.SETRANGE("New Line", TRUE);

                REPEAT
                  PrintTempPurchLine.INIT;
                  PrintTempPurchLine.TRANSFERFIELDS(_PurchLine);
                  PrintTempPurchLine."Line No." := DocLine."Line No.";
                  PrintTempPurchLine.Type := DocLine."Line Type";
                  PrintTempPurchLine."No." := DocLine."No.";
                  IF DocLine.Description <> '' THEN
                    PrintTempPurchLine.Description := DocLine.Description;

                  PrintTempPurchLine.Quantity := DocLine.Quantity;
                  PrintTempPurchLine."Direct Unit Cost" := DocLine."Direct Unit Cost";

                  DocLineDim.SETCURRENTKEY("Document Line ID");
                  DocLineDim.SETRANGE("Document Line ID", DocLine.ID);
                  IF DocLineDim.FINDSET THEN
                    REPEAT
                      IF GLSetup."Global Dimension 1 Code" = DocLineDim."Dimension Code" THEN
                        PrintTempPurchLine."Shortcut Dimension 1 Code" := DocLineDim."Dimension Value";
                      IF GLSetup."Global Dimension 2 Code" = DocLineDim."Dimension Code" THEN
                        PrintTempPurchLine."Shortcut Dimension 2 Code" := DocLineDim."Dimension Value";
                    UNTIL DocLineDim.NEXT = 0;
                  //4PS
                  PrintTempPurchLine."Job No." := DocLine."Job No.";
                  PrintTempPurchLine."Sales Order No." := DocLine."Service Order No.";
                  //4PS

                  PrintTempPurchLine.INSERT;
                UNTIL DocLine.NEXT = 0;

              NewLinesAdded := TRUE;
            END;
          END;
      END;
    END;

    PROCEDURE CheckUserGroupFilter@1100285006() : Boolean;
    VAR
      UserGroupFound@1100285000 : Boolean;
    BEGIN
      IF NOT UserGroupFilterExist THEN
        EXIT(TRUE);

      TempNextToApprove.RESET;

      NoOfApprovers := TempNextToApprove.COUNT;

      IF NoOfApprovers = 0 THEN
        EXIT(FALSE);

      IF NoOfApprovers <= 1 THEN BEGIN
        IF TempNextToApprove.FIND('-') THEN BEGIN
          TempExUserGroup.SETRANGE(Code, TempNextToApprove.Approver);
          IF NOT TempExUserGroup.FIND('-') THEN
            EXIT(FALSE)
          ELSE
            EXIT(TRUE);
        END;
      END
      ELSE BEGIN
        TempNextToApprove.FIND('-');
        UserGroupFound := FALSE;
        REPEAT
          TempExUserGroup.SETRANGE(Code, TempNextToApprove.Approver);
          IF TempExUserGroup.FIND('-') THEN
            UserGroupFound := TRUE;
        UNTIL (TempNextToApprove.NEXT = 0) OR UserGroupFound;

        IF NOT UserGroupFound THEN
          EXIT(FALSE)
        ELSE
          EXIT(TRUE);
      END;
    END;
