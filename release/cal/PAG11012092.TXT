    PROCEDURE DrillDownAllowed@1();
    VAR
      Job@1100525002 : Record 11072003;
      FiltGroup@1100525001 : Integer;
      DetermineProgressDate@1100525000 : Codeunit 11012014;
    BEGIN
      //C039890 - Progress dates and prognosis dates may differ for each subproject.
      ProgressRec.RESET;
      IF Job.GET("No.") THEN BEGIN
        FOR FiltGroup := 4 DOWNTO 0 DO BEGIN
          FILTERGROUP(FiltGroup);
          Job.FILTERGROUP(FiltGroup);
          Job.SETFILTER("Period Filter",GETFILTER("Period Filter"));
          Job.SETFILTER("Cumulation Filter",GETFILTER("Cumulation Filter"));
          Job.SETFILTER("Progress Filter",GETFILTER("Progress Filter"));
          Job.SETFILTER(PrevProgressFilter,GETFILTER(PrevProgressFilter));
        END;
        DetermineProgressDate.DetermineBothDates(Job,TRUE);
        ProgressRec.SETRANGE("Main Project No.", "Main Project");
        ProgressRec.SETRANGE("Project No.",Job."No.");
        ProgressRec.SETFILTER("Progress Date", Job.GETFILTER("Progress Filter"));
        IF ProgressRec.FINDSET THEN BEGIN
          REPEAT
           ProgressRec.MARK(TRUE);
          UNTIL ProgressRec.NEXT=0;
        END;
      END;

      IF "Single/Main/Sub Project" = "Single/Main/Sub Project"::"Main Project" THEN BEGIN
        //This is a main project, also add the allowed amounts/hours of the sub projects
        Job.RESET;
        Job.SETCURRENTKEY("Main Project");
        Job.SETFILTER("Main Project", "No.");
        //take filter into account because in some cases we do not want to cumulate over the
        //subprojects (e.g. in report 11012028)
        Job.SETFILTER("No.", GETFILTER("Project Filter"));
        //skip the main project now
        Job.SETFILTER("Single/Main/Sub Project",'%1', Rec."Single/Main/Sub Project"::"Sub Project");
        IF Job.FINDSET(FALSE, FALSE) THEN BEGIN
          REPEAT
            FOR FiltGroup := 4 DOWNTO 0 DO BEGIN
              FILTERGROUP(FiltGroup);
              Job.FILTERGROUP(FiltGroup);
              Job.SETFILTER("Period Filter",GETFILTER("Period Filter"));
              Job.SETFILTER("Cumulation Filter",GETFILTER("Cumulation Filter"));
              Job.SETFILTER("Progress Filter",GETFILTER("Progress Filter"));
              Job.SETFILTER(PrevProgressFilter,GETFILTER(PrevProgressFilter));
            END;
            DetermineProgressDate.DetermineBothDates(Job,TRUE);
            ProgressRec.SETRANGE("Main Project No.", "Main Project");
            ProgressRec.SETRANGE("Project No.",Job."No.");
            ProgressRec.SETFILTER("Progress Date", Job.GETFILTER("Progress Filter"));
            IF ProgressRec.FINDSET THEN BEGIN
              REPEAT
                ProgressRec.MARK(TRUE);
              UNTIL ProgressRec.NEXT=0;
            END;
          UNTIL Job.NEXT = 0;
        END;
      END;
      ProgressRec.SETRANGE("Project No.");
      ProgressRec.SETRANGE("Progress Date");
      ProgressRec.MARKEDONLY(TRUE);
      ProgressForm.SETTABLEVIEW(ProgressRec);
      ProgressForm.RUN;
      //C039890.en
    END;
