OBJECT Codeunit 11072048 Insurance Jnl.-B.Post
{
  OBJECT-PROPERTIES
  {
    Date=;
    Time=;
    Version List=NAVW17.00,4PS14.00 (5655);
  }
  PROPERTIES
  {
    TableNo=11072091;
    OnRun=BEGIN
            InsuranceJnlBatch.COPY(Rec);
            Code;
            COPY(InsuranceJnlBatch);
          END;

  }
  CODE
  {
    VAR
      Text000@1000 : TextConst 'DEU=Mchten Sie die Buch.-Bltter buchen?;ENU=Do you want to post the journals?;NLD=Wilt u de dagboeken boeken?;NOR=Vil du bokfre journalene?;SVE=Vill du bokfra journalerna?';
      Text001@1001 : TextConst 'DEU=Die Buch.-Bltter sind gebucht worden.;ENU=The journals were successfully posted.;NLD=De dagboeken werden probleemloos geboekt.;NOR=Kladdene ble bokfrt p riktig mte.;SVE=Journalerna r bokfrda.';
      Text002@1002 : TextConst 'DEU="Es konnten nicht alle Buch.-Bltter gebucht werden. ";ENU="It was not possible to post all of the journals. ";NLD="Niet alle dagboeken konden geboekt worden. ";NOR="Det er ikke mulig  bokfre alle kladdene. ";SVE="Det var inte mjligt att bokfra alla journaler. "';
      Text003@1003 : TextConst 'DEU=Die nicht erfolgreich gebuchten Buch.-Bltter sind jetzt markiert.;ENU=The journals that were not successfully posted are now marked.;NLD=De dagboeken die niet zonder problemen konden geboekt worden, zijn gemarkeerd.;NOR=Kladdene som ikke ble bokfrt, er merket.;SVE=Journalerna som inte blev bokfrda r nu markerade.';
      InsuranceJnlTempl@1004 : Record 11072090;
      InsuranceJnlBatch@1005 : Record 11072091;
      InsuranceJnlLine@1006 : Record 11072092;
      InsuranceJnlPostBatch@1007 : Codeunit 11072046;
      JournalWithErrors@1008 : Boolean;

    LOCAL PROCEDURE Code@1();
    BEGIN
      WITH InsuranceJnlBatch DO BEGIN
        InsuranceJnlTempl.GET("Journal Template Name");
        InsuranceJnlTempl.TESTFIELD("Force Posting Report",FALSE);

        IF NOT CONFIRM(Text000,FALSE) THEN
          EXIT;

        FIND('-');
        REPEAT
          InsuranceJnlLine."Journal Template Name" := "Journal Template Name";
          InsuranceJnlLine."Journal Batch Name" := Name;
          InsuranceJnlLine."Line No." := 1;

          CLEAR(InsuranceJnlPostBatch);
          IF InsuranceJnlPostBatch.RUN(InsuranceJnlLine) THEN
            MARK(FALSE)
          ELSE BEGIN
            MARK(TRUE);
            JournalWithErrors := TRUE;
          END;
        UNTIL NEXT = 0;

        IF NOT JournalWithErrors THEN
          MESSAGE(Text001)
        ELSE
          MESSAGE(
            Text002 +
            Text003);

        IF NOT FIND('=><') THEN BEGIN
          RESET;
          FILTERGROUP := 2;
          SETRANGE("Journal Template Name","Journal Template Name");
          FILTERGROUP := 0;
          Name := '';
        END;
      END;
    END;

    BEGIN
    END.
  }
}
