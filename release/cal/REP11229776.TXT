    PROCEDURE ProcessSelection@1210190000();
    VAR
      EstLine@1210190000 : Record 11072072;
      RcpLine@1210190002 : Record 11072075;
      EstLineTotal@1100409001 : Record 11072077;
      LastLino@1210190001 : Integer;
    BEGIN
      LastLino :=0;
      EstLine.COPY("Estimate Line");
      IF EstLine.FINDSET THEN BEGIN
        REPEAT
          IF EstLine.Recipe <> '' THEN BEGIN
            RcpLine.SETRANGE("Estimate No.", EstLine."Estimate No.");
            RcpLine.SETRANGE("Sub-Estimate No.", EstLine."Sub-Estimate No.");
            RcpLine.SETRANGE("Part Group", EstLine."Part Group");
            RcpLine.SETRANGE(Part, EstLine.Part);
            RcpLine.SETRANGE("Estimate Line", EstLine."Line No.");
            IF RcpLine.FINDSET THEN BEGIN
              REPEAT
                TempLine.INIT;
                TempLine."Estimate No." := RcpLine."Estimate No.";
                //TempLine."Sub-Estimate No." := RcpLine."Sub-Estimate No.";
                //TempLine."Part Group" := RcpLine."Part Group";
                //TempLine.Part := RcpLine.Part;
                //TempLine."Estimate Line No." := RcpLine."Estimate Line";
                TempLine.Description := RcpLine.Description;
                IF RcpLine."Ignore Recipe Quantity" THEN
                 TempLine.Quantity := RcpLine.Quantity * EstLine."Part Quantity"
                ELSE
                  TempLine.Quantity := RcpLine.Quantity * EstLine."Part Quantity" * EstLine.Quantity;
                TempLine."Unit of Measure" := RcpLine."Unit of Measure";
                TempLine."Project No." := RcpLine."Project No.";
                TempLine."Project Element" := RcpLine.Element;
                TempLine.Code := RcpLine.Code;
                IF RcpLine.Hours <> 0 THEN BEGIN
                  LastLino := LastLino + 1;
                  TempLine."Line No." := LastLino;
                  TempLine."Cost Object" := RcpLine."Cost Object Labor";
                  TempLine.Hours := RcpLine."Cumulative Hours";
                  TempLine."Labor Amount" := RcpLine."Cumulative Labor";
                  FillSortCode(TempLine);
                  TempLine.INSERT;
                END;
                IF RcpLine."Net Cost Amount" <> 0 THEN BEGIN
                  IF RcpLine."Cost Type (Price)" = RcpLine."Cost Type (Price)"::Material THEN BEGIN
                    LastLino := LastLino + 1;
                    TempLine."Line No." := LastLino;
                    TempLine."Cost Object" := RcpLine."Cost Object Price";
                    TempLine."Material Amount" := RcpLine."Cumulative Net Amount";
                    FillSortCode(TempLine);
                    TempLine.INSERT;
                  END;
                  IF RcpLine."Cost Type (Price)" = RcpLine."Cost Type (Price)"::Subcontracting THEN BEGIN
                    LastLino := LastLino + 1;
                    TempLine."Line No." := LastLino;
                    TempLine."Cost Object" := RcpLine."Cost Object Price";
                    TempLine."Subcontracting Amount" := RcpLine."Cumulative Net Amount";
                    FillSortCode(TempLine);
                    TempLine.INSERT;
                  END;
                  IF RcpLine."Cost Type (Price)" = RcpLine."Cost Type (Price)"::Plant THEN BEGIN
                    LastLino := LastLino + 1;
                    TempLine."Line No." := LastLino;
                    TempLine."Cost Object" := RcpLine."Cost Object Price";
                    TempLine."Plant Amount" := RcpLine."Cumulative Net Amount";
                    FillSortCode(TempLine);
                    TempLine.INSERT;
                  END;
                  IF RcpLine."Cost Type (Price)" = RcpLine."Cost Type (Price)"::Sundry THEN BEGIN
                    LastLino := LastLino + 1;
                    TempLine."Line No." := LastLino;
                    TempLine."Cost Object" := RcpLine."Cost Object Price";
                    TempLine."Sundry Amount" := RcpLine."Cumulative Net Amount";
                    FillSortCode(TempLine);
                    TempLine.INSERT;
                  END;
                END;
              UNTIL RcpLine.NEXT = 0;
            END;
          END ELSE BEGIN
            TempLine.INIT;
            TempLine."Estimate No." := EstLine."Estimate No.";
            //TempLine."Sub-Estimate No." := EstLine."Sub-Estimate No.";
            //TempLine."Part Group" := EstLine."Part Group";
            //TempLine.Part := EstLine.Part;
            //TempLine."Estimate Line No." := EstLine."Line No.";
            TempLine.Description := EstLine.Description;
            TempLine.Quantity := EstLine.Quantity * EstLine."Part Quantity";
            TempLine."Unit of Measure" := EstLine."Unit of Measure";
            TempLine."Project No." := EstLine."Project No.";
            TempLine."Project Element" := EstLine.Element;
            TempLine.Code := EstLine.Code;
            EstLineTotal.GET(EstLine."Estimate No.", EstLine."Sub-Estimate No.", EstLine."Part Group", EstLine.Part, EstLine."Line No.");
            IF EstLine.Hours <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Labor";
              TempLine.Hours := EstLineTotal."Cumulative Hours";
              TempLine."Labor Amount" := EstLineTotal."Cumulative Labor";
              FillSortCode(TempLine);
              TempLine.INSERT;
            END;
            IF EstLine."Material Cost Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Price";
              TempLine."Material Amount" := EstLineTotal."Cumulative Material";
              FillSortCode(TempLine);
              TempLine.INSERT;
            END;
            IF EstLine."Subcontracting Cost Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Price";
              TempLine."Subcontracting Amount" := EstLineTotal."Cumulative Subcontracting";
              FillSortCode(TempLine);
              TempLine.INSERT;
            END;
            IF EstLine."Plant Cost Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Price";
              TempLine."Plant Amount" := EstLineTotal."Cumulative Plant";
              FillSortCode(TempLine);
              TempLine.INSERT;
            END;
            IF EstLine."Sundry Cost Amount" <> 0 THEN BEGIN
              LastLino := LastLino + 1;
              TempLine."Line No." := LastLino;
              TempLine."Cost Object" := EstLine."Cost Object Price";
              TempLine."Sundry Amount" := EstLineTotal."Cumulative Sundry";
              FillSortCode(TempLine);
              TempLine.INSERT;
            END;
          END;
        UNTIL EstLine.NEXT = 0;
      END;
    END;

    LOCAL PROCEDURE FillSortCode@1100525001(VAR TempLine@1100525000 : Record 11012174);
    BEGIN
      //db, 01-04-16
      CASE SortLines OF
        SortLines::ProjectElement: TempLine.Code := TempLine."Project Element";
        SortLines::CostObject: TempLine.Code := TempLine."Cost Object";
        //SortLines::CodingSystem: TempLine.Code := TempLine.Code;
      END;
    END;

    LOCAL PROCEDURE StartProcessEstimate@1100529400();
    VAR
      Estimate@1100529400 : Record 11012151;
      UpdateMgt@1100529401 : Codeunit 11012229;
    BEGIN
      //C020140
      IF NOT ProcessEstimate THEN EXIT;

      Estimate.SETFILTER("No.", "Estimate Line".GETFILTER("Estimate No."));
      IF Estimate.FINDSET THEN BEGIN
        REPEAT
          UpdateMgt.UpdateTotals(Estimate."No.", '', '', TRUE);
        UNTIL Estimate.NEXT = 0;
      END;
    END;

    BEGIN
    END.
