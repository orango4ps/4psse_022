    LOCAL PROCEDURE GetReportName@1100525003() : Text;
    BEGIN
      EXIT(DELSTR(CurrReport.OBJECTID, 1, STRPOS(CurrReport.OBJECTID, ' ')));
    END;

    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      ProjectFilter0@1100529601 : Text;
      ProjectFilter2@1100529600 : Text;
    BEGIN
      ProjectFilter0 := Project.GETFILTERS;
      Project.FILTERGROUP(2);
      ProjectFilter2 := Project.GETFILTERS;
      Project.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Project.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Project.TABLECAPTION, ProjectFilter0));
    END;

    LOCAL PROCEDURE GetPrognosisDate@1100529602();
    VAR
      Prognosis@1100529600 : Record 11012034;
    BEGIN
      PrognosisDate := FilterDate;
      PrevPrognosisDate := 0D;

      Prognosis.SETRANGE("Project No.", Project."No.");
      Prognosis.SETRANGE("Prognosis Date", 0D, FilterDate);
      Prognosis.SETRANGE("Select For Reporting", TRUE);
      IF Prognosis.FINDLAST THEN BEGIN
        PrognosisDate := Prognosis."Prognosis Date";

        Prognosis.SETFILTER("Prognosis Date", '<%1', PrognosisDate);
        IF Prognosis.FINDLAST THEN
          PrevPrognosisDate := Prognosis."Prognosis Date";
      END;
    END;

    LOCAL PROCEDURE GetReportLines@1100529600();
    VAR
      PrognosisLine@1100529600 : Record 11012035;
      ProjectPurchaseAction@1100529601 : Record 11012026;
      TmpCostObject@1100529602 : TEMPORARY Record 349;
      TextFunctionManagement@1100527700 : Codeunit 11012357;
      PurchOrderWasCreated@1100529603 : Boolean;
      DescrArray@1100527650 : ARRAY [5] OF Text;
    BEGIN
      TmpReportLine.RESET;
      TmpReportLine.DELETEALL;
      TmpCommentLine.RESET;
      TmpCommentLine.DELETEALL;

      PrognosisLine.SETRANGE("Project No.", Project."No.");
      PrognosisLine.SETRANGE("Prognosis Date", PrognosisDate);
      IF PrognosisLine.ISEMPTY THEN
        EXIT;
      PrognosisLine.SETFILTER("Previous Date Filter", FORMAT(PrevPrognosisDate));
      PrognosisLine.SETAUTOCALCFIELDS("Cost Object Description", "Prev Prognosis End Result", "Project Related Name");

      ProjectPurchaseAction.SETRANGE("Project No.", Project."No.");
      ProjectPurchaseAction.SETFILTER("Default Cost Object Purchase", '<>%1', '');
      IF ProjectPurchaseAction.FINDSET THEN BEGIN
        REPEAT
          TmpCostObject.SETRANGE(Code, ProjectPurchaseAction."Default Cost Object Purchase");
          IF TmpCostObject.ISEMPTY THEN BEGIN
            TmpCostObject.Code := ProjectPurchaseAction."Default Cost Object Purchase";
            TmpCostObject.INSERT;

            PrognosisLine.SETRANGE("Cost Object", ProjectPurchaseAction."Default Cost Object Purchase");
            IF PrognosisLine.FINDSET THEN BEGIN
              CostTypeInt := PrognosisLine."Cost Type";
              CLEAR(TmpReportLine);
              TmpReportLine."Key Code1" := PurchActionCode;
              TmpReportLine."Key Code2" := FORMAT(CostTypeInt);
              TmpReportLine."Key Code3" := DetailLineCode;
              TmpReportLine."Key Code4" := PrognosisLine."Cost Object";
              TmpReportLine.Int2 := 1;
              TmpReportLine.Text1 := PrognosisLine."Project Related Name";
              IF TmpReportLine.Text1 = '' THEN
                TmpReportLine.Text1 := PrognosisLine."Cost Object Description";

              PurchOrderWasCreated := PurchOrderCreated(ProjectPurchaseAction."Default Cost Object Purchase");
              REPEAT
                InsertComment(PrognosisLine.Comment);

                TmpReportLine.Dec1 += PrognosisLine."Original Budget Amount";
                TmpReportLine.Dec2 += PrognosisLine.Extension + PrognosisLine.Options;
                TmpReportLine.Dec3 += PrognosisLine.Adjustments;
                TmpReportLine.Dec4 += PrognosisLine."Budget Amount";

                TmpReportLine.Dec5 += PrognosisLine."Budget Amount";
                TmpReportLine.Dec6 += PrognosisLine."Purchase Amount";
                IF PurchOrderWasCreated THEN
                  TmpReportLine.Dec7 += PrognosisLine."Budget Amount" - PrognosisLine."Purchase Amount" - PrognosisLine."Total Risc Coverage Amount";

                TmpReportLine.Dec8 += PrognosisLine."Actual Costs (Purchase Action)";
                TmpReportLine.Dec9 += PrognosisLine."Actual Costs" - PrognosisLine."Actual Costs (Purchase Action)";
                TmpReportLine.Dec10 += PrognosisLine.Amount;
                TmpReportLine.Dec11 += PrognosisLine."Prognosis End Result";
                TmpReportLine.Dec12 += PrognosisLine."Prognosis End Result" - PrognosisLine."Prev Prognosis End Result";
              UNTIL PrognosisLine.NEXT = 0;
              TmpReportLine.INSERT;
            END;
          END;
        UNTIL ProjectPurchaseAction.NEXT = 0;
        PrognosisLine.SETRANGE("Cost Object");
      END;

      PrognosisLine.FINDSET;
      REPEAT
        TmpCostObject.SETRANGE(Code, PrognosisLine."Cost Object");
        IF TmpCostObject.ISEMPTY THEN BEGIN
          CostTypeInt := PrognosisLine."Cost Type";

          TmpReportLine.SETRANGE("Key Code1", OtherCostsCode);
          TmpReportLine.SETRANGE("Key Code2", FORMAT(CostTypeInt));
          TmpReportLine.SETRANGE("Key Code3", DetailLineCode);
          TmpReportLine.SETRANGE("Key Code4", PrognosisLine."Cost Object");

          IF NOT TmpReportLine.FINDFIRST THEN BEGIN
            CLEAR(TmpReportLine);
            TmpReportLine."Key Code1" := OtherCostsCode;
            TmpReportLine.Text2 := FORMAT(PrognosisLine."Cost Type");
            TmpReportLine."Key Code2" := FORMAT(CostTypeInt);
            TmpReportLine."Key Code3" := DetailLineCode;
            TmpReportLine."Key Code4" := PrognosisLine."Cost Object";
            TmpReportLine.Int2 := CostTypeInt + 2;
            TmpReportLine.Text1 := PrognosisLine."Project Related Name";
            IF TmpReportLine.Text1 = '' THEN
              TmpReportLine.Text1 := PrognosisLine."Cost Object Description";
            TmpReportLine.INSERT;
          END;

          InsertComment(PrognosisLine.Comment);

          TmpReportLine.Dec1 += PrognosisLine."Original Budget Amount";
          TmpReportLine.Dec2 += PrognosisLine.Extension + PrognosisLine.Options;
          TmpReportLine.Dec3 += PrognosisLine.Adjustments;
          TmpReportLine.Dec4 += PrognosisLine."Budget Amount";

          TmpReportLine.Dec6 += PrognosisLine."Purchase Amount";
          TmpReportLine.Dec7 += 0 - PrognosisLine."Purchase Amount" - PrognosisLine."Total Risc Coverage Amount";

          TmpReportLine.Dec8 += PrognosisLine."Actual Costs (Purchase Action)";
          TmpReportLine.Dec9 += PrognosisLine."Actual Costs" - PrognosisLine."Actual Costs (Purchase Action)";
          TmpReportLine.Dec10 += PrognosisLine.Amount;
          TmpReportLine.Dec11 += PrognosisLine."Prognosis End Result";
          TmpReportLine.Dec12 += PrognosisLine."Prognosis End Result" - PrognosisLine."Prev Prognosis End Result";

          TmpReportLine.MODIFY;
        END;
      UNTIL PrognosisLine.NEXT = 0;

      TmpReportLine.RESET;
      TmpReportLine.SETRANGE(Dec1, 0);
      TmpReportLine.SETRANGE(Dec2, 0);
      TmpReportLine.SETRANGE(Dec3, 0);
      TmpReportLine.SETRANGE(Dec4, 0);
      TmpReportLine.SETRANGE(Dec5, 0);
      TmpReportLine.SETRANGE(Dec6, 0);
      TmpReportLine.SETRANGE(Dec7, 0);
      TmpReportLine.SETRANGE(Dec8, 0);
      TmpReportLine.SETRANGE(Dec9, 0);
      TmpReportLine.SETRANGE(Dec10, 0);
      TmpReportLine.SETRANGE(Dec11, 0);
      TmpReportLine.SETRANGE(Dec12, 0);
      TmpReportLine.DELETEALL;
      TmpReportLine.RESET;

      CalcEstimateAmounts;
      CreateTotalCorrectionZeroLines;
      CreateTotalLines;

      TmpReportLine.SETRANGE("Key Code3", DetailLineCode);
      IF TmpReportLine.FINDSET THEN
        REPEAT
          CLEAR(DescrArray);
          IF PrintPrognosisRemarks THEN
            TextFunctionManagement.Text2Array(TmpReportLine.Text1, 26, DescrArray)
          ELSE
            TextFunctionManagement.Text2Array(TmpReportLine.Text1, 20, DescrArray);
          TmpReportLine.SplitByDescrArray(TmpReportLine, DescrArray, DescrRounding);
          IF PrintPrognosisRemarks THEN BEGIN
            TmpCommentLine.SETRANGE("Key Code1", TmpReportLine."Key Code1");
            TmpCommentLine.SETRANGE("Key Code2", TmpReportLine."Key Code2");
            TmpCommentLine.SETRANGE("Key Code3", TmpReportLine."Key Code3");
            TmpCommentLine.SETRANGE("Key Code4", TmpReportLine."Key Code4");
            IF TmpCommentLine.FINDSET THEN
              REPEAT
                TmpReportLine := TmpCommentLine;
                TmpReportLine.INSERT;
              UNTIL TmpCommentLine.NEXT = 0;
          END;
        UNTIL TmpReportLine.NEXT = 0;
      TmpReportLine.RESET;
    END;

    LOCAL PROCEDURE CalcEstimateAmounts@1100529604();
    VAR
      ProjectLoc@1100529602 : Record 11072003;
      PrognosisLine@1100529604 : Record 11012035;
      CostControlMgt@1100529600 : Codeunit 11012006;
      PrognosisMgt@1100529601 : Codeunit 11020217;
      CostType@1100529606 : ' ,Labor,Material,Subcontracting,Plant,Sundry,Revenues';
      CostObjectFilter@1100529603 : Text;
    BEGIN
      TmpReportLine.INIT;

      PrognosisMgt.CalcEstimateResultPrognosis(Project."No.", PrognosisDate, CostType, CostObjectFilter,
        TmpReportLine.Dec1, TmpReportLine.Dec2, TmpReportLine.Dec16, TmpReportLine.Dec3);
      TmpReportLine.Dec2 += TmpReportLine.Dec16;
      TmpReportLine.Dec4 := TmpReportLine.Dec1 + TmpReportLine.Dec2 + TmpReportLine.Dec3;

      IF CostObjectFilter <> '' THEN BEGIN
        PrognosisLine.SETRANGE("Project No.", Project."No.");
        PrognosisLine.SETRANGE("Prognosis Date", PrognosisDate);
        IF CostType > 0 THEN
          PrognosisLine.SETRANGE("Cost Type", CostType - 1);
        PrognosisLine.SETFILTER("Cost Object", CostObjectFilter);
        PrognosisLine.CALCSUMS("Prognosis End Result");
        TmpReportLine.Dec11 := PrognosisLine."Prognosis End Result";
        TmpReportLine.Dec12 := PrognosisLine."Prognosis End Result" - PrognosisLine."Prev Prognosis End Result";
      END;

      IF NOT ReportLineHasValues THEN BEGIN
        CostObjectFilter := '';
        ProjectLoc.GET(Project."No.");
        ProjectLoc.SETFILTER("Cumulation Filter", '..%1', PrognosisDate);
        ProjectLoc.CALCFIELDS("Contract Amount", "Extension Contract Amount", "Options Contract Amount");
        CostControlMgt.CalcProjectBudget(Project."No.", PrognosisDate,
          TmpReportLine.Dec1, TmpReportLine.Dec2, TmpReportLine.Dec16, TmpReportLine.Dec3);
        TmpReportLine.Dec1 := ProjectLoc."Contract Amount" - TmpReportLine.Dec1;
        TmpReportLine.Dec2 := ProjectLoc."Extension Contract Amount" + ProjectLoc."Options Contract Amount" - TmpReportLine.Dec2 - TmpReportLine.Dec16;
        TmpReportLine.Dec4 := TmpReportLine.Dec1 + TmpReportLine.Dec2 + TmpReportLine.Dec3;
        TmpReportLine.Dec11 := CostControlMgt.CalcTotalEstimateResult(Project."No.", PrognosisDate);
        IF PrevPrognosisDate <> 0D THEN
          TmpReportLine.Dec12 := TmpReportLine.Dec11 - CostControlMgt.CalcTotalEstimateResult(Project."No.", PrevPrognosisDate);
      END;

      IF ReportLineHasValues THEN BEGIN
        TmpReportLine."Key Code1" := EstimateLineCode;
        TmpReportLine.Text1 := TxtEstimateResult;
        TmpReportLine.Bool1 := TRUE;
        TmpReportLine.Bool2 := TRUE;
        TmpReportLine.INSERT;
      END;

      IF CostObjectFilter <> '' THEN BEGIN
        IF CostType <> 0 THEN BEGIN
          CostTypeInt := CostType - 1;
          TmpReportLine.SETRANGE("Key Code2", FORMAT(CostTypeInt));
        END;
        TmpReportLine.SETFILTER("Key Code4", CostObjectFilter);
        TmpReportLine.DELETEALL;
        TmpReportLine.RESET;
      END;
    END;

    LOCAL PROCEDURE CreateTotalCorrectionZeroLines@1100527702();
    VAR
      LocTotalTmpReportLine@1100527700 : TEMPORARY Record 11020579;
    BEGIN
      IF ShowZeroLinesTotalBudget THEN
        EXIT;

      TmpReportLine.RESET;
      TmpReportLine.SETRANGE(Dec4, 0);
      IF TmpReportLine.ISEMPTY THEN
        EXIT;

      TmpReportLine.FINDSET;
      REPEAT
        IF (TmpReportLine."Key Code1" = PurchActionCode) OR (TmpReportLine."Key Code1" = OtherCostsCode) THEN BEGIN
          TmpReportLine.SETRANGE("Key Code1", TmpReportLine."Key Code1");
          IF TmpReportLine."Key Code1" = OtherCostsCode THEN
            TmpReportLine.SETRANGE("Key Code2", TmpReportLine."Key Code2");
          TmpReportLine.CALCSUMS(Dec1, Dec2, Dec3, Dec4, Dec5, Dec6, Dec7, Dec8, Dec9, Dec10, Dec11, Dec12);
          LocTotalTmpReportLine := TmpReportLine;
          IF TmpReportLine."Key Code1" = PurchActionCode THEN
            LocTotalTmpReportLine."Key Code2" := LastLineCostTypeCode;
          LocTotalTmpReportLine."Key Code3" := ZeroTotalBudgetLineCode;
          LocTotalTmpReportLine.Text1 := STRSUBSTNO(TxtTotalCorrZeroLines, SELECTSTR(TmpReportLine.Int2, TxtZeroLineTypes));

          TmpReportLine.SETRANGE("Key Code2");

      UNTIL TmpReportLine.NEXT = 0;

      TmpReportLine.DELETEALL;
      TmpReportLine.RESET;

      IF LocTotalTmpReportLine.FINDSET THEN
        REPEAT
          TmpReportLine := LocTotalTmpReportLine;
          TmpReportLine.INSERT;
        UNTIL LocTotalTmpReportLine.NEXT = 0;
    END;

    LOCAL PROCEDURE CreateTotalLines@1100529606();
    VAR
      TmpReportLineLoc@1100529600 : TEMPORARY Record 11020579;
    BEGIN
      TmpReportLine.RESET;
      IF TmpReportLine.ISEMPTY THEN
        EXIT;

      TmpReportLine.FINDSET;
      REPEAT
        TmpReportLine.SETRANGE("Key Code1", TmpReportLine."Key Code1");
        IF TmpReportLine."Key Code1" = OtherCostsCode THEN
          TmpReportLine.SETRANGE("Key Code2", TmpReportLine."Key Code2");
        TmpReportLine.CALCSUMS(Dec1, Dec2, Dec3, Dec4, Dec5, Dec6, Dec7, Dec8, Dec9, Dec10, Dec11, Dec12);
        TmpReportLineLoc := TmpReportLine;
        TmpReportLineLoc.INSERT;
        TmpReportLine.FINDLAST;
        TmpReportLine.SETRANGE("Key Code1");
        TmpReportLine.SETRANGE("Key Code2");
      UNTIL TmpReportLine.NEXT = 0;

      TmpReportLineLoc.FINDSET;
      REPEAT
        IF TmpReportLineLoc."Key Code1" <> EstimateLineCode THEN BEGIN
          TmpReportLine := TmpReportLineLoc;
          IF TmpReportLine."Key Code1" = OtherCostsCode THEN
            IF ShowCostObjectDetails OR (ShowLaborCostsDetails AND (TmpReportLine."Key Code2" = '0')) THEN
              TmpReportLine.Text1 := STRSUBSTNO(TxtTotal, TmpReportLine.Text2)
            ELSE
              TmpReportLine.Text1 := TmpReportLine.Text2
          ELSE BEGIN
            TmpReportLine."Key Code2" := LastLineCostTypeCode;
            IF ShowPurchaseDetails THEN
              TmpReportLine.Text1 := STRSUBSTNO(TxtTotal, TxtContract)
            ELSE
              TmpReportLine.Text1 := TxtContract;
          END;
          TmpReportLine."Key Code3" := TotalLineCode;
          TmpReportLine.Bool1 := TRUE;
          TmpReportLine.Bool2 := TRUE;
          TmpReportLine.INSERT;
        END;
      UNTIL TmpReportLineLoc.NEXT = 0;

      TmpReportLineLoc.CALCSUMS(Dec1, Dec2, Dec3, Dec4, Dec5, Dec6, Dec7, Dec8, Dec9, Dec10, Dec11, Dec12);
      TmpReportLine := TmpReportLineLoc;
      TmpReportLine."Key Code1" := GrandTotalLineCode;
      TmpReportLine.Text1 := TxtGrandTotal;
      TmpReportLine.Bool1 := TRUE;
      TmpReportLine.Bool2 := TRUE;
      TmpReportLine.INSERT;

      IF NOT ShowPurchaseDetails THEN BEGIN
        TmpReportLine.SETRANGE("Key Code1", PurchActionCode);
        TmpReportLine.SETFILTER("Key Code3", '%1|%2', DetailLineCode, ZeroTotalBudgetLineCode);
        TmpReportLine.DELETEALL;
      END;

      IF NOT ShowLaborCostsDetails THEN BEGIN
        TmpReportLine.SETRANGE("Key Code1", OtherCostsCode);
        TmpReportLine.SETRANGE("Key Code2", '0');
        TmpReportLine.SETFILTER("Key Code3", '%1|%2', DetailLineCode, ZeroTotalBudgetLineCode);
        TmpReportLine.DELETEALL;
      END;

      IF NOT ShowCostObjectDetails THEN BEGIN
        TmpReportLine.SETRANGE("Key Code1", OtherCostsCode);
        TmpReportLine.SETFILTER("Key Code2", '<>0');
        TmpReportLine.SETFILTER("Key Code3", '%1|%2', DetailLineCode, ZeroTotalBudgetLineCode);
        TmpReportLine.DELETEALL;
      END;

      IF NOT ShowZeroLinesPrevPrognosis THEN BEGIN
        TmpReportLine.SETFILTER("Key Code1", '<>%1&<>%2', EstimateLineCode, GrandTotalLineCode);
        TmpReportLine.SETRANGE("Key Code2");
        TmpReportLine.SETFILTER("Key Code3", '<>%1', ZeroTotalBudgetLineCode);
        TmpReportLine.SETRANGE(Dec12, 0);
        TmpReportLine.DELETEALL;
      END;

      TmpReportLine.RESET;
    END;

    LOCAL PROCEDURE GetFilterDate@1100529608();
