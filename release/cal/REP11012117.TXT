    LOCAL PROCEDURE CalculateLineAmounts@1210190000();
    VAR
      lvTmpQuoteCompRec2@1210190000 : TEMPORARY Record 11012063;
    BEGIN
      CLEAR(QuoteAmntArray);
      IF QuoteCounter < 1 THEN
        EXIT;

      lvTmpQuoteCompRec2.COPY(TempQuoteComparison);  //* Save

      WITH TempQuoteComparison DO BEGIN
        RESET;
        SETCURRENTKEY("Line Type", "Comparison Quote", "Quote No.", "Line No.", "Quote Line No.");
        SETRANGE("Line Type", "Line Type"::"Quote Line");
        SETRANGE("Comparison Quote", lvTmpQuoteCompRec2."Comparison Quote");
        SETRANGE("Line No.", lvTmpQuoteCompRec2."Line No.");
        FOR ForCounter := 1 TO QuoteCounter DO BEGIN
          SETRANGE("Quote No.", QuoteNoArray[ForCounter]);
            CALCSUMS(Amount);
            QuoteAmntArray[ForCounter] := Amount;
        END;
      END;

      TempQuoteComparison.COPY(lvTmpQuoteCompRec2);  //* Restore
    END;
