    PROCEDURE CheckRequisitionJobLedg@1210190006() : Boolean;
    BEGIN
      WITH "Job Ledger Entry" DO BEGIN
        //If "Entry No." = 0 then the source is the Purchase Order Control Line Table else the Job Ledger Entry Table
        IF "Job No." <> ProjRec."No." THEN
          IF NOT ProjRec.GET("Job No.") THEN
            EXIT(FALSE);

        IF ProjRec."Project Status" >= ProjRec."Project Status"::"Administrative Finished" THEN
          EXIT(FALSE);

        IF "Extension Contract" <> '' THEN BEGIN
          IF NOT ExtRec.GET("Job No.", "Extension Contract") THEN
            EXIT(FALSE);
          IF ExtRec."Settlement Method" <> ExtRec."Settlement Method"::"Cost Plus" THEN
            EXIT(FALSE);
        END ELSE
          IF ProjRec."Settlement Method" <> ProjRec."Settlement Method"::"Cost Plus" THEN
            EXIT(FALSE);

        IF (ProjectTypeRec.Code = '') OR (ProjectTypeRec.Code <> ProjRec."Project Type") THEN BEGIN
          ProjRec.TESTFIELD("Project Type");
          ProjectTypeRec.GET(ProjRec."Project Type");
        END;

        IF NOT ProjectTypeRec."Post Req. Revenue Cost Plus" THEN
          EXIT(FALSE);

        IF "Entry No." <> 0 THEN BEGIN  // from job ledger else from posted hours or purchase control
          SETRANGE("Date Filter", 0D, PostingDate);
          CALCFIELDS("Quantity Invoiced");
          CALCFIELDS("Amount Invoiced (LCY)");
          IF (Quantity <> 0) THEN BEGIN
            IF (ABS("Quantity Invoiced") >= ABS(Quantity)) THEN
              EXIT(FALSE);
          END ELSE BEGIN
           IF "Amount Invoiced (LCY)" <> 0 THEN
            EXIT(FALSE);
          END;
        END;

        IF "Cost Type" = "Cost Type"::Labor THEN
          IF NOT CheckSetupCostPlusHours(TABLENAME) THEN
            EXIT(FALSE);

      END;

      EXIT(TRUE);
    END;

    PROCEDURE CheckSetupCostPlusHours@1100485012(lTableName@1100485000 : Text[50]) : Boolean;
    VAR
      PostProjLedgEntry@1100485001 : Boolean;
    BEGIN
      WITH ProjSetup DO
        IF (ProjRec."Small Project" AND
            ("Small Project Cost Plus Hours" = "Small Project Cost Plus Hours"::"Job Ledger Entries"))
          OR
           (NOT ProjRec."Small Project" AND
            ("Project Cost Plus Hours from" = "Project Cost Plus Hours from"::"Job Ledger Entries"))
        THEN
          PostProjLedgEntry := TRUE
        ELSE
          PostProjLedgEntry := FALSE;

      CASE lTableName OF
        "Job Ledger Entry".TABLENAME:
          EXIT(PostProjLedgEntry);
        "Posted Hour Line".TABLENAME,
        "Purchase Order Control Line".TABLENAME:
          EXIT(NOT PostProjLedgEntry);
      END;
    END;

    PROCEDURE CheckDocument@1210190000(DocSeries@1210190003 : Code[20];DocNo@1210190000 : Code[20];LineNo@1210190001 : Integer;VAR GrossPrice@1210190005 : Decimal;VAR NetPrice@1210190006 : Decimal;VAR Discount@1210190007 : Decimal);
    VAR
      PurchInvLine@1210190002 : Record 123;
      PurchSetup@1210190004 : Record 312;
    BEGIN
      PurchSetup.GET;
      IF DocSeries <> PurchSetup."Posted Invoice Nos." THEN EXIT;

      IF PurchInvLine.GET(DocNo, LineNo) THEN BEGIN
        Discount := PurchInvLine."Line Discount %";
        NetPrice := PurchInvLine."Unit Cost (LCY)";
        //basis is bedrag uit post, dus exclusief indirecte kosten)
        GrossPrice := NetPrice;
        IF Discount <> 100 THEN
          GrossPrice := NetPrice * 100 / (100 - Discount);
        NetPrice := 0; //Basic Price Sales is not filled with Cost Price anymore
      END;
    END;

    PROCEDURE PostRequisitions@1100485005();
    VAR
      EntryNo@1100485000 : Integer;
    BEGIN
      WITH GenJnlLine DO BEGIN

        TempGenJnlLineBuffer.RESET;
        IF NOT TempGenJnlLineBuffer.FIND('-') THEN
          EXIT;

        //Fill general data
        INIT;
        "Journal Template Name" := GenLedgerSetup."Requisitions Journal Template";
        "Account Type" := "Account Type"::"G/L Account";
        "Posting Date" := PostingDate;
        "Document Date" := PostingDate;
        IF GenJnlTemplate."No. Series" <> '' THEN BEGIN
          CLEAR(NoSeriesMgt);
          "Document No." := NoSeriesMgt.GetNextNo(GenJnlTemplate."No. Series","Posting Date",TRUE);
        END ELSE
          "Document No." := DocumentNo;
        "Source Code" := GenJnlTemplate."Source Code";
        "Reason Code" := GenJnlTemplate."Reason Code";
        "Posting No. Series" := GenJnlTemplate."Posting No. Series";
        "System-Created Entry" := TRUE;

        //Post General Ledger Entries
        LineNo := 0;
        IF TempGenJnlLineBuffer.FIND('-') THEN
          REPEAT
            LineNo := LineNo + 10000;
            "Line No." := LineNo;
            VALIDATE("Account No.", TempGenJnlLineBuffer."Account No.");
            VALIDATE(Amount, TempGenJnlLineBuffer.Amount);
            VALIDATE("VAT Prod. Posting Group",'');
            VALIDATE("Shortcut Dimension 1 Code", TempGenJnlLineBuffer."Shortcut Dimension 1 Code");
            PostGenJnl;
          UNTIL TempGenJnlLineBuffer.NEXT = 0;

        TempBalGenJnlLineBuffer.RESET;
        IF TempBalGenJnlLineBuffer.FIND('-') THEN
          REPEAT
            LineNo := LineNo + 10000;
            "Line No." := LineNo;
            VALIDATE("Account No.", TempBalGenJnlLineBuffer."Account No.");
            VALIDATE(Amount, TempBalGenJnlLineBuffer.Amount);
            VALIDATE("VAT Prod. Posting Group",'');
            VALIDATE("Shortcut Dimension 1 Code",TempBalGenJnlLineBuffer."Shortcut Dimension 1 Code");
            PostGenJnl;
          UNTIL TempBalGenJnlLineBuffer.NEXT = 0;

        //Reverse post on next day
        IF TempGenJnlLineBuffer.FIND('-') THEN
          REPEAT
            LineNo := LineNo + 10000;
            "Line No." := LineNo;
            "Posting Date" := CALCDATE('<1D>',PostingDate);
            VALIDATE("Account No.", TempGenJnlLineBuffer."Account No.");
            VALIDATE(Amount, -TempGenJnlLineBuffer.Amount);
            VALIDATE("VAT Prod. Posting Group",'');
            VALIDATE("Shortcut Dimension 1 Code", TempGenJnlLineBuffer."Shortcut Dimension 1 Code");
           PostGenJnl;
          UNTIL TempGenJnlLineBuffer.NEXT = 0;

    PROCEDURE RequisitionAmountJobLedg@1100485000() ReqAmt : Decimal;
    VAR

      WITH "Job Ledger Entry" DO BEGIN

        //Find the Cost Plus Principal
        lProjPrincipalRec.SETRANGE("Project No.", "Job No.");
        lProjPrincipalRec.SETRANGE("Cost Plus Customer",TRUE);

        lProjCostPlusEntry."Extension Contract" := "Extension Contract";

        END;

        lProjCostPlusEntry."Unit of Measure" := "Unit of Measure Code";
        lProjCostPlusEntry."Employee No." := "Employee No.";
        DimMgt.GetDimValueRec(2, "Global Dimension 2 Code", lDimValRec, FALSE, "Job No.");
        lProjCostPlusEntry."Cost Object" := "Global Dimension 2 Code";
        lProjCostPlusEntry."Wage Component" := "Wage Component";
        lProjCostPlusEntry."Cost Price (LCY)" := "Unit Cost (LCY)";
        lProjCostPlusEntry.GetBasicPrice;

        IF CustRec."No." <> lProjCostPlusEntry.Principal THEN
          IF NOT CustRec.GET(lProjCostPlusEntry.Principal) THEN
            CLEAR(CustRec);

        IF ((CustRec."No." <> '') AND (CustRec."Item Price Cost Plus Project" = CustRec."Item Price Cost Plus Project"::GrossMin)) OR
           ((CustRec."No." = '') AND (ProjSetup."Item Price Cost Plus Entry" = ProjSetup."Item Price Cost Plus Entry"::GrossMin)) THEN
        BEGIN
          //evaluate unit cost as gross price item minus sales discount:
          lProjCostPlusEntry.Manufacturer := Manufacturer;
          lProjCostPlusEntry."Vendor (Trade Item)" := "Vendor (Trade Item)";
          IF "Trade Item" <> '' THEN
            lProjCostPlusEntry.VALIDATE("Trade Item", "Trade Item")
          ELSE
            IF "Basic Item" <> '' THEN
              lProjCostPlusEntry.VALIDATE("Basic Item", "Basic Item")
            ELSE
              IF "Item No." <> '' THEN
                lProjCostPlusEntry.VALIDATE("Item No.", "Item No.")
              ELSE
                CheckDocument("No. Series", "Document No.", "Document Line No.",
                  lProjCostPlusEntry."Gross Price (LCY)", lProjCostPlusEntry."Basic Price (LCY)", lProjCostPlusEntry."Sales Discount % (Item)");
          lProjCostPlusEntry.VALIDATE("Unit of Measure","Unit of Measure Code");

        END;
        lProjCostPlusEntry."Cost Object" := "Global Dimension 2 Code";
        lProjCostPlusEntry."Item No." := "Item No.";
        lProjCostPlusEntry."Basic Item" := "Basic Item";
        lProjCostPlusEntry."Trade Item" := "Trade Item";
        lProjCostPlusEntry.Manufacturer := Manufacturer;
        lProjCostPlusEntry."Vendor (Trade Item)" := "Vendor (Trade Item)";
        IF (lDimValRec."Cost Type" <> lDimValRec."Cost Type"::Labor) THEN
          IF (NOT lProjCostPlusEntry.CheckItemLine()) THEN
            lProjCostPlusEntry.GetBasicPrice;
        lProjCostPlusEntry."Surcharge %" := lProjCostPlusEntry.GetSurcharge;
        lProjCostPlusEntry.VALIDATE("Basic Price (LCY)");
        IF "Sales Price Purch. Order" <> 0 THEN
          lProjCostPlusEntry.VALIDATE("Sales Price (LCY)", "Sales Price Purch. Order");

        IF ("Entry No." <> 0) AND (Quantity <> 0) THEN BEGIN
          CostAmt := "Total Cost (LCY)" * lvQuantity/Quantity;
        END ELSE BEGIN
          CostAmt := "Total Cost (LCY)";
        END;


        ReqAmt := ROUND(lProjCostPlusEntry.Quantity * lProjCostPlusEntry.DimensionalFactor * lProjCostPlusEntry."Sales Price (LCY)");

        EXIT(ReqAmt);
      END;
    END;

        IF "Entry No." <> 0 THEN BEGIN  // from service ledger else from posted hours or purchase control
          IF (("Document Type" = "Document Type"::"Sales Invoice") OR

        IF (ServTypeRec.Code = '') OR (ServTypeRec.Code <> lvType) THEN
          ServTypeRec.GET(lvType);

        IF NOT ServTypeRec."Post Req. Revenue Cost Plus" THEN
          EXIT(FALSE);

        IF "Entry No." <> 0 THEN BEGIN  // from service ledger else from posted hours or purchase control
          SETRANGE("Date Filter", 0D, PostingDate);
          CALCFIELDS("Quantity Invoiced");
          CALCFIELDS("Amount Invoiced");
          IF (Quantity <> 0) THEN BEGIN
            IF (ABS("Quantity Invoiced") >= ABS(Quantity)) THEN
              EXIT(FALSE);
          END ELSE BEGIN
           IF "Amount Invoiced" <> 0 THEN
            EXIT(FALSE);
          END;

        END;

      END;

      EXIT(TRUE);
    END;
