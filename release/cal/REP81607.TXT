    PROCEDURE CreateBuffer@1100285501();
    VAR
      ContBusRel@1100285503 : Record 5054;
      Cont@1100285504 : Record 5050;
      CustLedgerEntry@1100285500 : Record 21;
      VendorLedgerEntry@1100285501 : Record 25;
      Contact@1100285502 : Record 5050;
      First@1100285505 : Boolean;
    BEGIN
      EntryNo := 1;

      CustLedgerEntry.SETCURRENTKEY("Customer No.","Posting Date","Currency Code");

      // *** Registration No. Filter *** //
      IF FilterRegNo <> '' THEN BEGIN
        First := TRUE;
        Cont.SETFILTER("COC Registration No.", FilterRegNo);
        Cont.SETRANGE(Type, Cont.Type::Company);

        IF Cont.FINDSET THEN REPEAT

          IF First THEN
            FilterContact := Cont."No."
          ELSE
            FilterContact += '|' + Cont."No.";

          First := FALSE;
        UNTIL Cont.NEXT = 0;

        IF FilterContact <> '' THEN BEGIN
          ContBusRel.SETFILTER("Contact No.", FilterContact);
          ContBusRel.SETRANGE("Link to Table", ContBusRel."Link to Table"::Customer);

          FilterContact := '';
          First := TRUE;
          IF ContBusRel.FINDSET THEN REPEAT

            IF First THEN
              FilterContact += ContBusRel."No."
            ELSE
              FilterContact += '|' + ContBusRel."No.";

            First := FALSE;
          UNTIL ContBusRel.NEXT = 0;

        END;

        //+++4PSSE.MK 140218
        IF FilterContact = '' THEN
          CurrReport.SKIP;
        //---4PSSE.MK 140218

        CustLedgerEntry.SETFILTER("Customer No.", FilterContact);
      END;

      // *** Date Filter *** //
      IF (FromPostingDate <> 0D) AND (ToPostingDate <> 0D) THEN
        CustLedgerEntry.SETRANGE("Posting Date", FromPostingDate, ToPostingDate)
      ELSE IF (FromPostingDate <> 0D) THEN
        CustLedgerEntry.SETFILTER("Posting Date", '%1..', FromPostingDate)
      ELSE IF (ToPostingDate <> 0D) THEN
        CustLedgerEntry.SETFILTER("Posting Date", '..%1', ToPostingDate);

      // *** Cost Center Filter *** //
      IF DimensionValueCode <> '' THEN
        CustLedgerEntry.SETFILTER("Global Dimension 1 Code", FilterCostCenter);
      // ***

      IF CustLedgerEntry.FINDSET THEN REPEAT
        CustLedgerEntry.CALCFIELDS("Remaining Amount");

        IF CustLedgerEntry."Remaining Amount" <> 0 THEN BEGIN

          WITH TEMPBufferLedgerEntry DO BEGIN
            INIT;
            "Entry No." := EntryNo;

            ContBusRel.RESET;
            Cont.RESET;
            ContBusRel.SETCURRENTKEY("Link to Table", "No.");
            ContBusRel.SETRANGE("Link to Table",ContBusRel."Link to Table"::Customer);
            ContBusRel.SETRANGE("No.", CustLedgerEntry."Customer No.");
            IF ContBusRel.FINDFIRST THEN BEGIN
              ContBusRel.CALCFIELDS("Contact Name");

              IF Cont.GET(ContBusRel."Contact No.") THEN BEGIN
                "Registration No.":= Cont."COC Registration No.";
                "Registration Name" := Cont.Name;
              END;
            END;

            "Sales Invoice No." := CustLedgerEntry."Document No.";
            "Account Receivables" := CustLedgerEntry."Remaining Amount";
            "Posting Date" := CustLedgerEntry."Posting Date";
            INSERT;
          END;

          EntryNo += 1;

        END;

      UNTIL CustLedgerEntry.NEXT = 0;
      //*****************************************************
      Cont.RESET;
      ContBusRel.RESET;

      VendorLedgerEntry.SETCURRENTKEY("Vendor No.","Posting Date","Currency Code");

      // *** Registration No. Filter *** //
      IF FilterRegNo <> '' THEN BEGIN
        Cont.SETFILTER("COC Registration No.", FilterRegNo);
        Cont.SETRANGE(Type, Cont.Type::Company);

        First := TRUE;
        IF Cont.FINDSET THEN REPEAT

          IF First THEN
            FilterContact := Cont."No."
          ELSE
            FilterContact += '|' + Cont."No.";

          First := FALSE;
        UNTIL Cont.NEXT = 0;

        IF FilterContact <> '' THEN BEGIN
          ContBusRel.SETFILTER("Contact No.", FilterContact);
          ContBusRel.SETRANGE("Link to Table", ContBusRel."Link to Table"::Vendor);

          FilterContact := '';
          First := TRUE;
          IF ContBusRel.FINDSET THEN REPEAT

            IF First THEN
              FilterContact += ContBusRel."No."
            ELSE
              FilterContact += '|' + ContBusRel."No.";

            First := FALSE;
          UNTIL ContBusRel.NEXT = 0;

          //+++4PSSE.MK 140218
          IF FilterContact = '' THEN
            CurrReport.SKIP;
          //---4PSSE.MK 140218

        END;
        VendorLedgerEntry.SETFILTER("Vendor No.", FilterContact);
      END;

      // *** Date Filter *** //
      IF (FromPostingDate <> 0D) AND (ToPostingDate <> 0D) THEN
        VendorLedgerEntry.SETRANGE("Posting Date", FromPostingDate, ToPostingDate)
      ELSE IF (FromPostingDate <> 0D) THEN
        VendorLedgerEntry.SETFILTER("Posting Date", '%1..', FromPostingDate)
      ELSE IF (ToPostingDate <> 0D) THEN
        VendorLedgerEntry.SETFILTER("Posting Date", '..%1', ToPostingDate);

      // *** Cost Center Filter *** //
      IF DimensionValueCode <> '' THEN
        VendorLedgerEntry.SETFILTER("Global Dimension 1 Code", FilterCostCenter);
      // ***

      IF VendorLedgerEntry.FINDSET(FALSE) THEN REPEAT

        VendorLedgerEntry.CALCFIELDS("Remaining Amount");

        IF VendorLedgerEntry."Remaining Amount" <> 0 THEN BEGIN
          WITH TEMPBufferLedgerEntry DO BEGIN
            INIT;
            "Entry No." := EntryNo;


            ContBusRel.SETCURRENTKEY("Link to Table", "No.");
            ContBusRel.SETRANGE("Link to Table",ContBusRel."Link to Table"::Vendor);
            ContBusRel.SETRANGE("No.", VendorLedgerEntry."Vendor No.");
            IF ContBusRel.FINDFIRST THEN BEGIN
              ContBusRel.CALCFIELDS("Contact Name");

              IF Cont.GET(ContBusRel."Contact No.") THEN BEGIN
                "Registration No.":= Cont."COC Registration No.";
                "Registration Name" := Cont.Name;
              END;
            END;

            "Purchase Invoice No." := VendorLedgerEntry."Document No.";
            "Account Payables" := VendorLedgerEntry."Remaining Amount";
            "Posting Date" := VendorLedgerEntry."Posting Date";
            INSERT;

          END;

          EntryNo += 1;

        END;

      UNTIL VendorLedgerEntry.NEXT = 0;
    END;

    PROCEDURE GetReportFilter1@1100525000() : Text;
    VAR
      FilterString@1100285000 : Text;
    BEGIN

      IF (FromPostingDate <> 0D) OR (ToPostingDate <> 0D) THEN BEGIN
        FilterString := TEMPBufferLedgerEntry.FIELDCAPTION("Posting Date") + ': ' + FORMAT(FromPostingDate)
                        + ' ... ' + FORMAT(ToPostingDate);
      END;
      {
        FilterString := TextDateFilter;

        IF (FromWeek <> 0) AND (ToWeek <> 0) THEN
          FilterString += ' '+ TextWeek +' '+ FORMAT(FromWeek) +' '+ TextYear +' '+ FORMAT(FromYear) +' ... '+
                                TextWeek +' '+ FORMAT(ToWeek) +' '+ TextYear +' '+ FORMAT(ToYear)
        ELSE IF (FromWeek <> 0) THEN
          FilterString += ' '+ TextWeek +' '+ FORMAT(FromWeek) +' '+ TextYear +' '+ FORMAT(FromYear) +' ... '
        ELSE IF (ToWeek <> 0) THEN
          FilterString += ' ' + TextWeek +' '+ FORMAT(ToWeek) +' '+ TextYear +' '+ FORMAT(ToYear);

      END ELSE
        FilterString := '';
      }
      EXIT(FilterString);
    END;
