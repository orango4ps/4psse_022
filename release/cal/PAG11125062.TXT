    LOCAL PROCEDURE NameOnFormat@19039177();
    BEGIN
      IF Type = Type::Company THEN
        NameEmphasize := TRUE
      ELSE BEGIN
        Cont.SETCURRENTKEY("Company Name", "Company No.", Type, Surname);
        IF ("Company No." <> '') AND (NOT MARKEDONLY) AND (CURRENTKEY = Cont.CURRENTKEY)
        THEN
          NameIndent := 1;
      END;
    END;
