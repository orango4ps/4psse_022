    LOCAL PROCEDURE GetDimText@1080000() DimText : Text[120];
    BEGIN
      WITH DimSetEntry DO BEGIN
        CLEAR(DimText);
        Continue := FALSE;
        REPEAT
          OldDimText := DimText;
          IF DimText = '' THEN
            DimText := STRSUBSTNO('%1 - %2',"Dimension Code","Dimension Value Code")
          ELSE
            DimText :=
              STRSUBSTNO(
                '%1: %2 - %3',DimText,"Dimension Code","Dimension Value Code");
          IF STRLEN(DimText) > MAXSTRLEN(OldDimText) THEN BEGIN
            DimText := OldDimText;
            Continue := TRUE;
            EXIT;
          END;
        UNTIL NEXT = 0;
        EXIT(DimText);
      END;
    END;
