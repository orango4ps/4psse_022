    PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF Estimate.GETFILTERS <> '' THEN

    PROCEDURE CheckBookMarks@1100525002(iText@1100525001 : Text[250]) oText : Text[250];
    VAR
      lvSep@1100525002 : Text[30];
      lvPos@1100525000 : Integer;
      lvLen@1100525006 : Integer;
      lvCounter@1100525003 : Integer;
      lvText001@1100525004 : TextConst 'DEU=..%1;ENU=%1);NLD=%1)';
    BEGIN
      IF STRPOS(iText, ')') = 0 THEN
        EXIT(iText);

      REPEAT
        lvCounter := lvCounter + 1;
        lvSep := STRSUBSTNO(lvText001, lvCounter);
        lvPos := STRPOS(iText, lvSep);
        IF lvPos <> 0 THEN BEGIN
          lvLen := STRLEN(iText) - lvPos;
          oText := COPYSTR(iText, lvPos+2, lvLen);
          IF oText <> '' THEN
            EXIT(iText);  //legenda
          oText := COPYSTR(iText, 1, lvPos-1);
          oText := DELCHR(oText, '>', ' ');
          lvLen := STRLEN(oText);
          oText := COPYSTR(iText, 1, lvLen);
          CASE lvCounter OF
            1: oText := oText + ' ' + Estimate."Contractor Name";
            2: oText := oText + ' ' + Estimate."Contractor City";
          END;
          oText := oText + ' ' + lvSep;
          EXIT(oText);
        END;
      UNTIL lvCounter = 3;

      EXIT(iText);
    END;

    PROCEDURE PrintBlankLine@1100525016(VAR EstimateElementRAW@1100525001 : Record 11125679) : Boolean;
    VAR
      EstimateElementRAW2@1100525000 : Record 11125679;
    BEGIN
      CASE TRUE OF
        STRLEN(EstimateElementRAW."Tender Paragraph No.") = 4:
          BEGIN
            EstimateElementRAW2.COPY(EstimateElementRAW);
            IF EstimateElementRAW2.NEXT = 0 THEN
              EXIT(FALSE);
            EXIT(STRLEN(EstimateElementRAW2."Tender Paragraph No.") <> 6);
          END;
        STRLEN(EstimateElementRAW."Tender Paragraph No.") < 6:
          EXIT(TRUE);
        COPYSTR(EstimateElementRAW."Tender Paragraph No.", 1, 1) = '9':
          BEGIN
            EstimateElementRAW2.COPY(EstimateElementRAW);
            IF EstimateElementRAW2.NEXT = 0 THEN
              EXIT(FALSE);
            IF STRLEN(EstimateElementRAW2."Tender Paragraph No.") <> 6 THEN
              EXIT(TRUE);
            EXIT(COPYSTR(EstimateElementRAW."Tender Paragraph No.", 2, 1) <> COPYSTR(EstimateElementRAW2."Tender Paragraph No.", 2, 1));
          END;
      ELSE
        EstimateElementRAW2.COPY(EstimateElementRAW);
        IF EstimateElementRAW2.NEXT = 0 THEN
          EXIT(FALSE);

        EXIT(STRLEN(EstimateElementRAW2."Tender Paragraph No.") <> 6);
      END;
    END;
