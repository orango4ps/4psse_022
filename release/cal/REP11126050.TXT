    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      ProjectFilter0@1100529601 : Text;
      ProjectFilter2@1100529600 : Text;
    BEGIN
      ProjectFilter0 := Project.GETFILTERS;
      Project.FILTERGROUP(2);
      ProjectFilter2 := Project.GETFILTERS;
      Project.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Project.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Project.TABLECAPTION, ProjectFilter0));
    END;

    LOCAL PROCEDURE GetReportLines@1100529600();
    VAR
      PrognosisLine@1100529600 : Record 11012035;
      DescrArray@1100527651 : ARRAY [5] OF Text;
      TextFunctionManagement@1100527650 : Codeunit 11012357;
    BEGIN
      TmpReportLine.RESET;
      TmpReportLine.DELETEALL;

      TmpTotalReportLine.DELETEALL;
      TmpTotalReportLine.INIT;
      ClearTotalValues;

      PrognosisLine.SETRANGE("Project No.", Project."No.");
      PrognosisLine.SETRANGE("Prognosis Date", PrognosisDate);
      PrognosisLine.SETRANGE("Cost Type", PrognosisLine."Cost Type"::Labor);
      IF PrognosisLine.ISEMPTY THEN
        EXIT;

      PrognosisLine.SETAUTOCALCFIELDS("Cost Object Description", "Project Related Name");
      PrognosisLine.FINDSET;
      REPEAT
        IF NOT TmpReportLine.GET(PrognosisLine."Cost Object") THEN BEGIN
          TmpReportLine.INIT;
          TmpReportLine."Key Code1" := PrognosisLine."Cost Object";
          TmpReportLine.Text1 := PrognosisLine."Project Related Name";
          IF TmpReportLine.Text1 = '' THEN
            TmpReportLine.Text1 := PrognosisLine."Cost Object Description";
          TmpReportLine.INSERT;
        END;

        TmpReportLine.Dec1 += PrognosisLine."Budget Hours";
        TmpReportLine.Dec2 += PrognosisLine."Budget Amount";
        TmpReportLine.Dec3 += PrognosisLine."Allowed Costs";
        TmpReportLine.Dec5 += PrognosisLine."Actual Hours";
        TmpReportLine.Dec6 += PrognosisLine."Actual Costs";
        TmpReportLine.Dec13 += PrognosisLine."Allowed Hours";
        TmpReportLine.Dec7 += PrognosisLine."Allowed Hours" - PrognosisLine."Actual Hours";
        TmpReportLine.Dec8 += PrognosisLine."Allowed Costs" - PrognosisLine."Actual Costs";
        TmpReportLine.Dec11 += PrognosisLine.Amount;
        TmpReportLine.Dec12 += PrognosisLine."Prognosis End Result";

        TmpReportLine.MODIFY;

        TotalBudgetAmount += PrognosisLine."Budget Amount";
        TotalBudgetHours += PrognosisLine."Budget Hours";

        TotalAllowedHours += PrognosisLine."Allowed Hours";
        TotalAllowedCosts += PrognosisLine."Allowed Costs";

        TotalActualCosts += PrognosisLine."Actual Costs";
        TotalActualHours += PrognosisLine."Actual Hours";
        IF IsTravelHourCostCode(PrognosisLine."Cost Object") THEN
          TotalActualTravelHours += PrognosisLine."Actual Hours"
        ELSE
          TotalActualWorkedHours += PrognosisLine."Actual Hours";

      UNTIL PrognosisLine.NEXT = 0;

      TmpReportLine.FINDSET;
      REPEAT
        IF TmpReportLine.Dec1 <> 0 THEN
          TmpReportLine.Dec1 := TmpReportLine.Dec2 / TmpReportLine.Dec1
        ELSE
          TmpReportLine.Dec1 := 0;

        IF TmpReportLine.Dec13 <> 0 THEN
          TmpReportLine.Dec13 := TmpReportLine.Dec3 / TmpReportLine.Dec13
        ELSE
          TmpReportLine.Dec13 := 0;

        IF TmpReportLine.Dec2 <> 0 THEN
          TmpReportLine.Dec4 := TmpReportLine.Dec3 / TmpReportLine.Dec2
        ELSE
          TmpReportLine.Dec4 := 0;

        IF TmpReportLine.Dec5 <> 0 THEN
          TmpReportLine.Dec5 := TmpReportLine.Dec6 / TmpReportLine.Dec5
        ELSE
          TmpReportLine.Dec5 := 0;

        IF TmpReportLine.Dec13 <> 0 THEN
          TmpReportLine.Dec9 := TmpReportLine.Dec7 * TmpReportLine.Dec13
        ELSE
          TmpReportLine.Dec9 := TmpReportLine.Dec7 * TmpReportLine.Dec1;

        IF TmpReportLine.Dec13 <> 0 THEN
          TmpReportLine.Dec10 := (TmpReportLine.Dec13 - TmpReportLine.Dec5) * TmpReportLine.Dec5
        ELSE
          IF TmpReportLine.Dec1 <> 0 THEN
            TmpReportLine.Dec10 := (TmpReportLine.Dec1 - TmpReportLine.Dec5) * TmpReportLine.Dec5;

        IF ReportLineHasValues THEN BEGIN
          TmpReportLine.MODIFY;

          TmpTotalReportLine.Dec2 += TmpReportLine.Dec2;
          TmpTotalReportLine.Dec3 += TmpReportLine.Dec3;
          TmpTotalReportLine.Dec6 += TmpReportLine.Dec6;
          TmpTotalReportLine.Dec7 += TmpReportLine.Dec7;
          TmpTotalReportLine.Dec8 += TmpReportLine.Dec8;
          TmpTotalReportLine.Dec11 += TmpReportLine.Dec11;
          TmpTotalReportLine.Dec12 += TmpReportLine.Dec12;
        END ELSE
          TmpReportLine.DELETE;
      UNTIL TmpReportLine.NEXT = 0;

      CalcTotals;
      TmpTotalReportLine.Dec9 += TotalEfficiencyDiff;
      TmpTotalReportLine.Dec10 += TotalPriceDiff;
      TmpTotalReportLine.INSERT;

      IF TmpReportLine.FINDSET THEN
        REPEAT
          CLEAR(DescrArray);
          TextFunctionManagement.Text2Array(TmpReportLine.Text1, 20, DescrArray);
          TmpReportLine.SplitByDescrArray(TmpReportLine, DescrArray, DescrRounding);
        UNTIL TmpReportLine.NEXT = 0;
    END;

    LOCAL PROCEDURE ReportLineHasValues@1100529603() : Boolean;
    BEGIN

      IF TmpReportLine.Dec1 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec2 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec3 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec4 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec5 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec6 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec7 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec8 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec9 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec10 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec11 <> 0 THEN
        EXIT(TRUE);
      IF TmpReportLine.Dec12 <> 0 THEN
        EXIT(TRUE);
      EXIT(FALSE);
    END;

    LOCAL PROCEDURE CalcTotals@1100529610();
    BEGIN
      IF TotalBudgetHours <> 0 THEN
        TotalBudgetRate := TotalBudgetAmount / TotalBudgetHours;

      IF TotalAllowedHours <> 0 THEN
        TotalProgressRate := TotalAllowedCosts / TotalAllowedHours;

      IF TotalActualWorkedHours <> 0 THEN
        TotalActualRate := TotalActualCosts / TotalActualWorkedHours;

      IF TotalProgressRate <> 0 THEN
        TotalEfficiencyDiff := TmpTotalReportLine.Dec7 * TotalProgressRate
      ELSE
        TotalEfficiencyDiff := TmpTotalReportLine.Dec7 * TotalBudgetRate;

      IF TotalProgressRate <> 0 THEN
        TotalPriceDiff := (TotalProgressRate - TotalActualRate) * TotalActualWorkedHours
      ELSE
        TotalPriceDiff := (TotalBudgetRate - TotalActualRate) * TotalActualWorkedHours;
    END;

    LOCAL PROCEDURE IsTravelHourCostCode@1100529605(CostCode@1100529600 : Code[20]) : Boolean;
    VAR
      AvgHourRateCalculation@1100529601 : Record 11126040;
    BEGIN

        EXIT(FALSE);
      AvgHourRateCalculation.FINDFIRST;
      EXIT(NOT AvgHourRateCalculation."In Avg Hour Rate Calculation");
    END;

      TotalPriceDiff := 0;
    END;

    LOCAL PROCEDURE GetFilterDate@1100529607();
    VAR
      Prognosis@1100529600 : Record 11012034;
    BEGIN
      IF ProjectNo = '' THEN BEGIN
        FilterDate := 0D;
        EXIT;
      END;

      IF NOT Prognosis.GET(ProjectNo, FilterDate) THEN
        FilterDate := 0D
      ELSE
        IF NOT Prognosis."Select For Reporting" THEN
          FilterDate := 0D;

      IF FilterDate = 0D THEN BEGIN
        Prognosis.SETRANGE("Project No.", ProjectNo);
        Prognosis.SETRANGE("Select For Reporting", TRUE);
        IF Prognosis.FINDLAST THEN
          FilterDate := Prognosis."Prognosis Date";
        EXIT;
      END;
    END;

    LOCAL PROCEDURE RoundAmount@1100529634(Value@1102 : Decimal) : Text;
    BEGIN
      CASE RoundingFactor OF
        RoundingFactor::None:
          EXIT(FORMAT(Value, 0, NormalFormatString));
        RoundingFactor::"1":
          EXIT(FORMAT((ROUND(Value, 1))));
        RoundingFactor::"1000":
          EXIT(FORMAT(ROUND(Value / 1000, 0.1), 0, Text010));
        RoundingFactor::"1000000":
          EXIT(FORMAT(ROUND(Value / 1000000, 0.1), 0, Text010));
      END;
    END;
