    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;

    PROCEDURE FillOptionArray@1100525002(OptionType@1100525001 : Option);
    VAR
      OptionCounter@1100525000 : Integer;
      NoOfOptions@1100525002 : Integer;
      MaxArrayNo@1100525003 : Integer;
      ExtraSeparatorLen@1100525004 : Integer;
    BEGIN
      ArrayNo := 0;
      CLEAR(OptionArray);

      AccidentSelectedOption.RESET;
      AccidentSelectedOption.SETRANGE("Accident No.", "Accident QA"."No.");
      AccidentSelectedOption.SETRANGE(Type, OptionType);
      IF (NOT AccidentSelectedOption.FINDSET) THEN
        EXIT;

      ArrayNo := 1;
      MaxArrayNo := ARRAYLEN(OptionArray);
      ExtraSeparatorLen := 3;
      OptionCounter := 0;
      NoOfOptions := AccidentSelectedOption.COUNT;
      REPEAT
        OptionCounter := OptionCounter + 1;
        AccidentSelectedOption.CALCFIELDS(Description);
        IF (OptionCounter = NoOfOptions) THEN
          ExtraSeparatorLen := 0;
        IF (STRLEN(AccidentSelectedOption.Description) + STRLEN(OptionArray[ArrayNo]) + ExtraSeparatorLen) >
          MAXSTRLEN(OptionArray[ArrayNo])
        THEN BEGIN
          IF ArrayNo >= MaxArrayNo THEN
            EXIT;
          ArrayNo := ArrayNo + 1;
        END;
        IF (OptionArray[ArrayNo] <> '') THEN
