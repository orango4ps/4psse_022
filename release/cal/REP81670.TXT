    PROCEDURE CreateBuffer@1100285501();
    VAR
      HourAccountingLine@1100285501 : Record 11012039;
      PostedHourLine@1100285503 : Record 11012085;
      Employee@1100285502 : Record 5200;
      HoursConsentLine@1100285509 : Record 11020437;
      PostedHoursConsentLine@1100285507 : Record 11020439;
      UserSetup@1100285511 : Record 91;
      Job@1100285100 : Record 11072003;
      ConsentDeniedBy@1100285510 : Text[30];
      Hours@1100285506 : Decimal;
      CountDays@1100285504 : Integer;
      HourAccounting@1100285508 : Record 11012038;
      PostedHourHeader@1100285505 : Record 11012084;
    BEGIN
      EntryNo := 1;
      // *** Creates records from the hour accounting line ****
      IF ShowOpenHours THEN BEGIN
        //>> 151119
        IF (FilterEmployee <> '') AND (ProcjectNo = '') THEN  BEGIN
          HourAccountingLine.SETCURRENTKEY(Year,Week,"Employee No.","Project No.");

          IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
            HourAccountingLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          ELSE IF (PeriodStart <> 0D) THEN
            HourAccountingLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          ELSE IF (PeriodEnd <> 0D) THEN
            HourAccountingLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);
          IF FilterEmployee <> '' THEN
            HourAccountingLine.SETFILTER("Employee No.", FilterEmployee);
          IF ProcjectNo <> '' THEN
            HourAccountingLine.SETFILTER("Project No.", ProcjectNo);


        END ELSE BEGIN
          HourAccountingLine.SETCURRENTKEY(Year,Week,"Project No.","Employee No.");

          IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
            HourAccountingLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          ELSE IF (PeriodStart <> 0D) THEN
            HourAccountingLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          ELSE IF (PeriodEnd <> 0D) THEN
            HourAccountingLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);
          IF ProcjectNo <> '' THEN
            HourAccountingLine.SETFILTER("Project No.", ProcjectNo);
          IF FilterEmployee <> '' THEN
            HourAccountingLine.SETFILTER("Employee No.", FilterEmployee);
        END;
        // << 151119

        IF WageCostCode <> '' THEN
          HourAccountingLine.SETFILTER("Wage Component", WageCostCode);
        // 151116 >>
        HourAccountingLine.SETRANGE("Component Type", HourAccountingLine."Component Type"::Hours);
        // 151116 <<
        IF HourAccountingLine.FINDSET(FALSE) THEN REPEAT

          CountDays := 1;
          REPEAT

            CASE CountDays OF

              1: BEGIN
                IF HourAccountingLine.Monday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Monday);
                  EntryNo += 1;
                END;
              END;

              2: BEGIN
                IF HourAccountingLine.Tuesday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Tuesday);
                  EntryNo += 1;
                END;
              END;

              3: BEGIN
                IF HourAccountingLine.Wednesday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Wednesday);
                  EntryNo += 1;
                END;
              END;

              4: BEGIN
                IF HourAccountingLine.Thursday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Thursday);
                  EntryNo += 1;
                END;
              END;

              5: BEGIN
                IF HourAccountingLine.Friday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Friday);
                  EntryNo += 1;
                END;
              END;

              6: BEGIN
                IF HourAccountingLine.Saturday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Saturday);
                  EntryNo += 1;
                END;
              END;

              7: BEGIN
                IF HourAccountingLine.Sunday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, HourAccountingLine.Sunday);
                  EntryNo += 1;
                END;
              END;

            END;
            CountDays += 1;
          UNTIL CountDays = 8;

        UNTIL HourAccountingLine.NEXT = 0;

      END;

      //**** Create from posted lines *** //
      IF ShowPostedHours THEN BEGIN

        //>> 151119
        IF (FilterEmployee <> '') AND (ProcjectNo = '') THEN BEGIN
          PostedHourLine.SETCURRENTKEY(Year,Week,"Employee No.","Project No.");

          IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
            PostedHourLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          ELSE IF (PeriodStart <> 0D) THEN
            PostedHourLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          ELSE IF (PeriodEnd <> 0D) THEN
            PostedHourLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);

          IF FilterEmployee <> '' THEN
            PostedHourLine.SETFILTER("Employee No.", FilterEmployee);
          IF ProcjectNo <> '' THEN
            PostedHourLine.SETFILTER("Project No.", ProcjectNo);
        END ELSE BEGIN
          PostedHourLine.SETCURRENTKEY(Year,Week,"Project No.","Employee No.");

          IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
            PostedHourLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          ELSE IF (PeriodStart <> 0D) THEN
            PostedHourLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          ELSE IF (PeriodEnd <> 0D) THEN
            PostedHourLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);

          IF ProcjectNo <> '' THEN
            PostedHourLine.SETFILTER("Project No.", ProcjectNo);
          IF FilterEmployee <> '' THEN
            PostedHourLine.SETFILTER("Employee No.", FilterEmployee);
        END;
        // << 151119

        // 151116 >>
        IF WageCostCode <> '' THEN
          PostedHourLine.SETFILTER("Wage Component", WageCostCode);
        PostedHourLine.SETRANGE("Component Type", PostedHourLine."Component Type"::Hours);
        // 151116 <<
        IF PostedHourLine.FINDSET(FALSE) THEN REPEAT

          WITH TEMPHourAccoutBuffer DO BEGIN
            INIT;
            "Entry No." := EntryNo;

            IF Employee.GET(PostedHourLine."Employee No.") THEN BEGIN
              "Last Name" := Employee."Last Name";
              "First Name" := Employee."First Name";
              "No." := PostedHourLine."Employee No.";
              "Agda No." := Employee."Salary No.";
            END;

            Week := PostedHourLine.Week;

            IF PostedHourLine.Monday <> 0 THEN
              Hours := PostedHourLine.Monday
            ELSE IF PostedHourLine.Tuesday <> 0 THEN
              Hours := PostedHourLine.Tuesday
            ELSE IF PostedHourLine.Wednesday <> 0 THEN
              Hours := PostedHourLine.Wednesday
            ELSE IF PostedHourLine.Thursday <> 0 THEN
              Hours := PostedHourLine.Thursday
            ELSE IF PostedHourLine.Friday <> 0 THEN
              Hours := PostedHourLine.Friday
            ELSE IF PostedHourLine.Saturday <> 0 THEN
              Hours := PostedHourLine.Saturday
            ELSE IF PostedHourLine.Sunday <> 0 THEN
              Hours := PostedHourLine.Sunday;

            Date := PostedHourLine."Actual date of line";
            "Project No." := PostedHourLine."Project No.";
            PostedHourLine.CALCFIELDS("Project Description", "Description Wage Component");
            "Project Description" := PostedHourLine."Project Description";

            "Wage Component /Cost Code" := PostedHourLine."Wage Component";
            "Wage Comp. /Cost Code Descrip." := PostedHourLine."Description Wage Component";

            //Amount := PostedHourLine."Amount (LCY)";
            "Unit Cost" := PostedHourLine."Unit Cost (LCY)";
            Amount := "Unit Cost" * Hours;

            IF Job.GET(PostedHourLine."Project No.") THEN BEGIN
              Job.CALCFIELDS("Name Project Manager");
              "Project Manager" := Job."Name Project Manager";
              "Main Project" := Job."Main Project"
            END;


            //***Get Concent Lines , can be in either the posted or the regular consent lines ***
            HoursConsentLine.RESET;
            HoursConsentLine.SETRANGE(Year, PostedHourLine.Year);
            HoursConsentLine.SETRANGE(Week, PostedHourLine.Week);
            HoursConsentLine.SETRANGE(Employee, PostedHourLine."Employee No.");
            HoursConsentLine.SETRANGE("Line No.", PostedHourLine."Line No.");
            HoursConsentLine.SETRANGE(LineType, HoursConsentLine.LineType::Normal);

            IF HoursConsentLine.FINDSET(FALSE) THEN BEGIN

              REPEAT
