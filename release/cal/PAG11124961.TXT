    PROCEDURE GetStandardText@1210190000();
    VAR
      TxtRec@1210190003 : Record 7;
      LineTxtRec@1210190002 : Record 280;
      CommLineRec@1210190001 : Record 11020314;
      CurrLino@1210190000 : Integer;
      lvText11012000@1100485001 : TextConst 'DEU="Nicht genÅgend Raum zum ZwischenfÅgen von Textzeilen; nicht alle Textzeilen wurden hinzugefÅgt.";ENU=Not enough space to insert text lines, not all text lines are added.;NLD=Niet voldoende ruimte om tekstregels tussen te voegen, niet alle tekstregels zijn toegevoegd.;NOR=Det finnes ikke tilstrekkelig med plass for Ü sette inn tekstrader. ikke alle tekstrader legges til.;SVE=Det finns inte tillrÑckligt med utrymme fîr att infoga textrader. Inte alla textrader lÑggs till.';
    BEGIN
      //**4PS
      IF Code = '' THEN
        EXIT;

      CurrPage.SAVERECORD;
      CurrLino := "Line No.";
      IF TxtRec.GET(Code) THEN BEGIN
        IF (TxtRec.Description <> '') THEN BEGIN
          Comment := COPYSTR(TxtRec.Description, 1, MAXSTRLEN(Comment));
          MODIFY;
        END;
        LineTxtRec.SETRANGE("Table Name", LineTxtRec."Table Name"::"Standard Text");
        LineTxtRec.SETRANGE("No.", Code);
        IF LineTxtRec.FIND('-') THEN BEGIN
          REPEAT
            CommLineRec := Rec;
            CommLineRec.Comment := COPYSTR(LineTxtRec.Text, 1, MAXSTRLEN(CommLineRec.Comment));
            CommLineRec."Line No." := CurrLino + 1;
            CommLineRec.Date := 0D;
            CommLineRec.Code := '';
            IF NOT CommLineRec.INSERT THEN BEGIN
              MESSAGE(lvText11012000);
              EXIT;
            END;
            CurrLino := CommLineRec."Line No.";
          UNTIL LineTxtRec.NEXT = 0;
        END;
      END;
      CurrPage.UPDATE(TRUE);
    END;
