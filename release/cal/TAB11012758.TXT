    LOCAL PROCEDURE CreatePlantLocation@1100485007(TryoutQuote@1100529000 : Record 11012758;PlantCompany@1100485010 : Text[30]) : Code[20];
    VAR
      MarketingSetup@1100485005 : Record 5079;
      PlantSetup@1100485004 : Record 11012550;
      CompanyData@1100485006 : Record 11020674;
      Customer@1100485003 : Record 18;
      PlantLocation@1100485001 : Record 11012554;
      PlantLocation2@1100485002 : Record 11012554;
      CompanyPlantLocNoSeries@1100529002 : Record 11229940;
      NoSeriesMgt@1100529001 : Codeunit 396;
    BEGIN
      //* From a Try-out Quote in the current company create a plant location in the 'Plant Company' this
      //* can be another company, so do not use 'VALIDATE'/'INSERT/MODIFY(TRUE)' on the Plant Location table!
      WITH TryoutQuote DO BEGIN
        TESTFIELD(Code);
        MarketingSetup.GET;
        MarketingSetup.TESTFIELD("Build Pl.Loc.Code T-out Qt. fr");

        IF NOT CompanyData.GET(CURRENTCOMPANY) THEN
          CompanyData."Company Name" := CURRENTCOMPANY;
        IF (CompanyData."Company No." = '') AND
           (MarketingSetup."Build Pl.Loc.Code T-out Qt. fr" = MarketingSetup."Build Pl.Loc.Code T-out Qt. fr"::CompNoTryoutNo)
        THEN
          CompanyData.TESTFIELD("Company No.");
        IF MarketingSetup."Build Pl.Loc.Code T-out Qt. fr" = MarketingSetup."Build Pl.Loc.Code T-out Qt. fr"::NoSeries THEN BEGIN
          CompanyPlantLocNoSeries.CHANGECOMPANY(PlantCompany);
          IF NOT CompanyPlantLocNoSeries.GET(CURRENTCOMPANY) THEN BEGIN
            CompanyPlantLocNoSeries.INIT;
            CompanyPlantLocNoSeries."Company Name" := CURRENTCOMPANY;
          END;
          //* Nos. for projects, because plant location for try-out will eventually become a plant location for a project
          IF CompanyPlantLocNoSeries."Plant Location Nos. Projects" = '' THEN
            CompanyPlantLocNoSeries.FIELDERROR("Plant Location Nos. Projects",
              STRSUBSTNO(Text008, PlantSetup.FIELDCAPTION("Plant Location Nos."), PlantSetup.TABLECAPTION, PlantCompany));
        END;

        PlantSetup.CHANGECOMPANY(PlantCompany);
        PlantSetup.GET;

        PlantLocation2.CHANGECOMPANY(PlantCompany);

        PlantLocation.CHANGECOMPANY(PlantCompany);
        PlantLocation.INIT;
        PlantLocation.InitRecord;
        CASE MarketingSetup. "Build Pl.Loc.Code T-out Qt. fr" OF
          MarketingSetup."Build Pl.Loc.Code T-out Qt. fr"::TryoutNo:
            PlantLocation.Code := Code;
          MarketingSetup."Build Pl.Loc.Code T-out Qt. fr"::CompNoTryoutNo:
            PlantLocation.Code := CompanyData."Company No." + Code;
          MarketingSetup."Build Pl.Loc.Code T-out Qt. fr"::NoSeries:
            BEGIN
              IF PlantCompany <> COMPANYNAME THEN
                NoSeriesMgt.SetFromOtherCompany(PlantCompany);
              PlantLocation.Code := NoSeriesMgt.GetNextNo(CompanyPlantLocNoSeries."Plant Location Nos. Projects", 0D, TRUE);
              PlantLocation."No. Series" := CompanyPlantLocNoSeries."Plant Location Nos. Projects";
            END;
          ELSE
            PlantLocation.Code := '';
        END;
        PlantLocation.TESTFIELD(Code);
        IF PlantLocation2.GET(PlantLocation.Code) THEN
          ERROR(Text009, PlantLocation.Code, PlantCompany, PlantLocation2."Try-out Quote No.", PlantLocation2."Company Name");

        PlantLocation."Creation Date" := TODAY;
        PlantLocation.Depot :=  FALSE;
        PlantLocation."Try-out Quote No." :=  Code;
        IF PlantCompany <> CURRENTCOMPANY THEN BEGIN
          PlantLocation."Company Name" := CURRENTCOMPANY;
          //Fill Customer only if project and plant location taken from different company
          CompanyData.TESTFIELD("Customer No.");
          Customer.CHANGECOMPANY(PlantCompany);
          IF NOT Customer.GET(CompanyData."Customer No.") THEN
            Customer.INIT;
          PlantLocation."Customer No." := CompanyData."Customer No.";
          PlantLocation."VAT Bus. Posting Group" := Customer."VAT Bus. Posting Group";
          PlantLocation."Gen. Bus. Posting Group" := PlantSetup."Default Gen. Bus.Posting Group";
          IF PlantSetup."Def. Loc. Bus. P.Grp of Cust." THEN
            PlantLocation."Gen. Bus. Posting Group" := Customer."Gen. Bus. Posting Group";
        END;
        IF PlantLocation."Customer No." = '' THEN BEGIN
          PlantLocation."Gen. Bus. Posting Group" := PlantSetup."Default Gen. Bus.Posting Group";
          PlantLocation."VAT Bus. Posting Group" := PlantSetup."Default VAT Bus.Posting Group";
        END;
        PlantLocation.TransferTryOutQuoteData(TryoutQuote);
        PlantLocation.NormalizePostCodeHouseNo;//DP00195
        PlantLocation.INSERT;

        EXIT(PlantLocation.Code);
      END;
    END;

    BEGIN
    {
      //20061009/4PS : Modified field ["Copy Address from"] added option [CompanyVisit,Person,PersonVisit]
      DP01090: flowfield Estimate Total expired; use function GetCostAmount or GetSalesAmount (table Estimate)
    }
    END.
