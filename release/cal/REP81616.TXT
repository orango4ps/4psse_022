    PROCEDURE CalcValues@1100285510();
    BEGIN
      CalcBalanceAtDate := 0;
      CalcNetChange     := 0;

      DimSetEntry2.COPYFILTERS(DimSetEntry);
      IF GLEntry.FINDSET(FALSE) THEN
      REPEAT
        DimSetEntry2.SETRANGE("Dimension Set ID",GLEntry."Dimension Set ID");
        IF NOT DimSetEntry2.ISEMPTY THEN BEGIN
         CalcNetChange     := CalcNetChange     + GLEntry.Amount;
        END;
      UNTIL GLEntry.NEXT=0;

      GLEntry.SETFILTER("Posting Date"            , FilterAtDate);
      IF GLEntry.FINDSET(FALSE) THEN
      REPEAT
        DimSetEntry2.SETRANGE("Dimension Set ID",GLEntry."Dimension Set ID");
        IF NOT DimSetEntry2.ISEMPTY THEN BEGIN
         CalcBalanceAtDate := CalcBalanceAtDate + GLEntry.Amount;
        END;
      UNTIL GLEntry.NEXT=0;
      GLEntry.SETRANGE("Posting Date");
      IF Filter_Date <> '' THEN
        GLEntry.SETFILTER("Posting Date"            , Filter_Date);
    END;

    PROCEDURE CalcBudgetValues@1100285514();
    BEGIN
      CalcBudgetedAmount:=0;
      CalcBudgetAtDate:= 0;

      IF GLBudgetEntry.FIND('-') THEN REPEAT
        DimSetEntry2.COPYFILTERS(DimSetEntry);
        DimSetEntry2.SETRANGE("Dimension Set ID",GLEntry."Dimension Set ID");
        IF DimSetEntry2.FINDFIRST THEN BEGIN
          CalcBudgetedAmount := CalcBudgetedAmount + GLBudgetEntry.Amount;
          CalcBudgetAtDate   := CalcBudgetAtDate   + GLBudgetEntry.Amount;
        END;
      UNTIL GLBudgetEntry.NEXT=0;
    END;
