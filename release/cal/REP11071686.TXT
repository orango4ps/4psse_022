    LOCAL PROCEDURE SendAppointment@1100409020();
    VAR
      ServiceOrder@1100409000 : Record 11012823;
      ServiceOrderExtension@1100409001 : Record 11071727;
      SCCustSourceTypeSetup@1100528600 : Record 11071800;
    BEGIN
      MessageString := STRSUBSTNO(Text006);

      ServiceOrder.GET(TempServiceOrder."No.");
      ServiceOrder.MARK(TRUE);
      ServiceOrder.SETRECFILTER;
      ServiceOrderExtension.GetServOrderExtension(ServiceOrder."No.");
      IF (ServiceOrderExtension."Date Appointm. Confirm. Sent" <> 0D) AND
         ((ServiceOrderExtension."Resend Appointm. Confirmation") OR ResendAll)
      THEN BEGIN
        ServiceOrderExtension."Date Appointm. Confirm. Sent" := 0D;
        ServiceOrderExtension.UpdateServOrderExtension(ServiceOrder."No.");
      END;
      IF (ServiceOrderExtension."Date Appointm. Confirm. Sent" = 0D) OR
         (ServiceOrderExtension."Resend Appointm. Confirmation")
      THEN BEGIN
        IF NOT SCCustSourceTypeSetup.GET(SCCustSourceTypeSetup.Type::Customer, ServiceOrder."Bill-to Customer No.", ServiceOrder."Source Type") THEN
          SCCustSourceTypeSetup.INIT;
        CASE SCCustSourceTypeSetup."Appointments by" OF
          SCCustSourceTypeSetup."Appointments by"::Letter: BEGIN
            IF (ServiceOrderExtension."Appointment Letter Code" = '') THEN
              ServiceOrderExtension."Appointment Letter Code" := SCCustSourceTypeSetup."Appointment Letter Code";
            IF (OverruleAppointmentLetterCode <> '') THEN
              ServiceOrderExtension."Appointment Letter Code" := OverruleAppointmentLetterCode;
            IF (ServiceOrderExtension."Resend Appointm. Confirmation") THEN
              ServiceOrderExtension."Resend Appointm. Confirmation" := FALSE;
            ServiceOrderExtension.UpdateServOrderExtension(ServiceOrder."No.");
            IF (ServiceOrderExtension."Appointment Letter Code" <> '') THEN BEGIN
              ConfirmPlanning(ServiceOrder);
              MessageString := STRSUBSTNO(Text004);

              ServiceOrderLetter := ServiceOrder;
              ServiceOrderLetter.MARK(TRUE);
            END ELSE
              MessageString := STRSUBSTNO(Text008, SCCustSourceTypeSetup.TABLECAPTION, SCCustSourceTypeSetup.Code,
                SCCustSourceTypeSetup.FIELDCAPTION("Appointment Letter Code"));
          END;
          SCCustSourceTypeSetup."Appointments by"::"E-Mail": BEGIN
            ConfirmPlanning(ServiceOrder);
            MessageString := STRSUBSTNO(Text005);

            ServiceOrderEmail := ServiceOrder;
            ServiceOrderEmail.MARK(TRUE);
          END;
          ELSE
            MessageString := STRSUBSTNO(Text008, SCCustSourceTypeSetup.TABLECAPTION, SCCustSourceTypeSetup.Code,
              SCCustSourceTypeSetup.FIELDCAPTION("Appointments by"));
        END;
      END ELSE
        MessageString := STRSUBSTNO(Text010, SCCustSourceTypeSetup.TABLECAPTION, SCCustSourceTypeSetup.Code,
          SCCustSourceTypeSetup.FIELDCAPTION("Appointments by"));
    END;
