                                          gLineText := Description;

                                    IF gVendLedgerEntry.FIND('-') THEN
                                      IF gVendLedgerEntry.COUNT = 1 THEN BEGIN
                                        gVendLedgerEntry.CALCFIELDS("Remaining Amt. (LCY)");
                                        gAmountLCY := gVendLedgerEntry."Remaining Amt. (LCY)";
                                        IF ("Document Type" = "Document Type"::Invoice) AND
                                           ("Due Date" <= "Pmt. Discount Date")
                                        THEN BEGIN
                                          IF gVendLedgerEntry."Remaining Amount" <> gVendLedgerEntry."Remaining Amt. (LCY)" THEN
