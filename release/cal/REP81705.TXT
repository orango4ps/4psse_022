    PROCEDURE CalcullateTotalBalanceAtDate@1100525000();
    VAR
      lDate@1100525001 : Record 2000000007;
      I@1100525000 : Integer;
    BEGIN
      CLEAR(TotalAccBalance);
      lDate.SETRANGE("Period Type",lDate."Period Type"::Month);
      lDate.SETRANGE("Period Start",DMY2DATE(1,1,Year),DMY2DATE(1,12,Year));
      lDate.FINDSET;
      REPEAT
        GLAccount.SETRANGE("Date Filter",lDate."Period End");
        GLAccount.CALCFIELDS("Balance at Date");
        I += 1;
        TotalAccBalance[I] := GLAccount."Balance at Date";
      UNTIL lDate.NEXT = 0;
    END;
