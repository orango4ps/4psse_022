    PROCEDURE GetReportFilter1@1100525000() : Text;
    VAR
      ProjectFilter0@1100525000 : Text;
      ProjectFilter2@1100525001 : Text;
    BEGIN
      ProjectFilter0 := Job.GETFILTERS;
      Job.FILTERGROUP(2);
      ProjectFilter2 := Job.GETFILTERS;
      Job.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Job.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Job.TABLECAPTION, ProjectFilter0));
    END;
