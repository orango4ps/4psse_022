    LOCAL PROCEDURE GetReportName@1100525003() : Text;
    BEGIN
      EXIT(DELSTR(CurrReport.OBJECTID, 1, STRPOS(CurrReport.OBJECTID, ' ')));
    END;

    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "QA Deviation".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', "QA Deviation".TABLECAPTION, "QA Deviation".GETFILTERS));
    END;

    LOCAL PROCEDURE FillProjManagerAndPrincipName@1100529401(VAR ProjectManagerName@1100525000 : Text[100];VAR PrincipalName@1100525001 : Text[100]);
    VAR
      Job@1100525002 : Record 11072003;
    BEGIN
      ProjectManagerName := '';
      PrincipalName := '';

      IF ("QA Deviation"."Source Type" = "QA Deviation"."Source Type"::Project) AND
         ("QA Deviation"."Source No." <> '')
      THEN BEGIN
        IF (Job.GET("QA Deviation"."Source No.")) THEN BEGIN
          Job.CALCFIELDS(Job."Name Project Manager", Job."Principal Name");
          ProjectManagerName := Job."Name Project Manager";
          PrincipalName := Job."Principal Name";
        END;
      END;
    END;

    LOCAL PROCEDURE FillOptionArray@1100529400(OptionType@1100525001 : Option);
    VAR
      OptionCounter@1100525000 : Integer;
      NoOfOptions@1100525002 : Integer;
      MaxArrayNo@1100525003 : Integer;
      ExtraSeparatorLen@1100525004 : Integer;
    BEGIN
      CLEAR(OptionArray);
      ArrayNo := 0;

      DeviationSelectedOption.RESET;
      DeviationSelectedOption.SETRANGE("Deviation No.", "QA Deviation"."No.");
      DeviationSelectedOption.SETRANGE(Type, OptionType);
      IF (NOT DeviationSelectedOption.FINDSET) THEN
        EXIT;

      ArrayNo := 1;
      MaxArrayNo := ARRAYLEN(OptionArray);
      ExtraSeparatorLen := 3;
      OptionCounter := 0;
      NoOfOptions := DeviationSelectedOption.COUNT;
      REPEAT
        OptionCounter := OptionCounter + 1;
        DeviationSelectedOption.CALCFIELDS(Description);
        IF (OptionCounter = NoOfOptions) THEN
          ExtraSeparatorLen := 0;
        IF (STRLEN(DeviationSelectedOption.Description) + STRLEN(OptionArray[ArrayNo]) + ExtraSeparatorLen) >
          MAXSTRLEN(OptionArray[ArrayNo])
        THEN BEGIN
          IF (ArrayNo >= MaxArrayNo) THEN
            EXIT;
          ArrayNo := ArrayNo + 1;
        END;
        IF (OptionArray[ArrayNo] <> '') THEN
          OptionArray[ArrayNo] := OptionArray[ArrayNo] + ' ';
        OptionArray[ArrayNo] := OptionArray[ArrayNo] + DeviationSelectedOption.Description;
        IF (OptionCounter < NoOfOptions) THEN
          OptionArray[ArrayNo] := OptionArray[ArrayNo] + ' /';
      UNTIL (DeviationSelectedOption.NEXT = 0);
    END;

    BEGIN
    END.
