    VAR

    LOCAL PROCEDURE GetStyleExpr@1100527702(ProjectCF@1100527703 : Record 11012635);
    VAR
      ProjectCFDate@1100527700 : Record 11012630;
      ProjectCFDateCompany@1100527704 : Record 11012630;
      ColumnNo@1100527701 : Integer;
      StyleExpr@1100527702 : ARRAY [32] OF Text;
      Priority@1100527705 : Integer;
    BEGIN
      FOR ColumnNo := 1 TO ARRAYLEN(MATRIX_MatrixRecords) DO BEGIN
        Priority := 0;
        StyleExpr[ColumnNo] := 'Standard';
        ProjectCFDate.SETRANGE("Record Type",ProjectCFDate."Record Type"::"Cash Flow");
        ProjectCFDate.SETRANGE("Project No.",ProjectCF."Project No.");
        ProjectCFDate.SETRANGE("Cash Flow",ProjectCF."Cash Flow");
        ProjectCFDate.SETRANGE(Date,MATRIX_MatrixRecords[ColumnNo]."Period Start",MATRIX_MatrixRecords[ColumnNo]."Period End");
        IF ProjectCFDate.FINDSET THEN
          REPEAT
            IF ProjectCFDateCompany.GET(ProjectCFDateCompany."Record Type"::Company,'',0DT,ProjectCFDate.Code,ProjectCFDate."Template Code") THEN
              IF ProjectCFDateCompany."Color Mark" <> 0 THEN
                IF ProjectCFDateCompany."Color Mark Priority" > Priority THEN BEGIN
                  Priority := ProjectCFDateCompany."Color Mark Priority";
                  CASE ProjectCFDateCompany."Color Mark" OF
                    ProjectCFDateCompany."Color Mark"::Blue:
                      StyleExpr[ColumnNo] := 'StandardAccent';
                    ProjectCFDateCompany."Color Mark"::"Blue Bold":
                      StyleExpr[ColumnNo] := 'StrongAccent';
                    ProjectCFDateCompany."Color Mark"::"Blue Italic":
                      StyleExpr[ColumnNo] := 'AttentionAccent';
                    ProjectCFDateCompany."Color Mark"::"Red Italic":
                      StyleExpr[ColumnNo] := 'Attention';
                    ProjectCFDateCompany."Color Mark"::"Red Bold Italic":
                      StyleExpr[ColumnNo] := 'Unfavorable';
                    ProjectCFDateCompany."Color Mark"::Green:
                      StyleExpr[ColumnNo] := 'Favorable';
                    ProjectCFDateCompany."Color Mark"::Orange:
                      StyleExpr[ColumnNo] := 'Ambiguous';
                    ProjectCFDateCompany."Color Mark"::Grey:
                      StyleExpr[ColumnNo] := 'Subordinate';
                  END;
                END;
          UNTIL (ProjectCFDate.NEXT = 0);
      END;

      HandleStyleExprArray(StyleExpr);
    END;
