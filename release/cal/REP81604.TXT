    PROCEDURE CreateBuffer@1100285501();
    VAR
      HourAccountingLine@1100285501 : Record 11012039;
      PostedHourLine@1100285503 : Record 11012085;
      Employee@1100285502 : Record 5200;
      WageComponent@1100285505 : Record 11012014;
      Days@1100285508 : 'Monday,Tuesday,Wednesday,Thursday,Friday,Saturday,Sunday';
      Hours@1100285506 : Decimal;
      EntryNo@1100285500 : Integer;
      CountDays@1100285504 : Integer;
      RecRef@1100285507 : RecordRef;
    BEGIN
      EntryNo := 1;

      IF ShowOpenHours THEN BEGIN

        HourAccountingLine.SETCURRENTKEY(Year,Week,"Project No.","Employee No.");
        IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
          //>>141104 ENH050
          //HourAccountingLine.SETRANGE("Posting Date", PeriodStart, PeriodEnd)
          HourAccountingLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          //<<141104 ENH050
        ELSE IF (PeriodStart <> 0D) THEN
          //>>141104 ENH050
          //HourAccountingLine.SETFILTER("Posting Date", '%1..', PeriodStart)
          HourAccountingLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          //<<141104 ENH050

        ELSE IF (PeriodEnd <> 0D) THEN
          //>>141104 ENH050
          //HourAccountingLine.SETFILTER("Posting Date", '..%1', PeriodEnd);
          HourAccountingLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);
          //<<141104 ENH050

        IF FilterProject <> '' THEN
          HourAccountingLine.SETFILTER("Project No.", FilterProject)
      // >> ITERO.MK
        ELSE
          HourAccountingLine.SETFILTER("Project No.", '<>%1', '');
      // << ITERO.MK


        IF FilterEmployee <> '' THEN
          HourAccountingLine.SETFILTER("Employee No.", FilterEmployee);

        // >> ITERO.MK
        //IF WageCostCode <> '' THEN
        //  HourAccountingLine.SETFILTER("Wage Component", WageCostCode);
        //<< ITERO.MK

        IF HourAccountingLine.FINDSET(FALSE) THEN REPEAT

          CountDays := 1;
          REPEAT

            CASE CountDays OF

              1: BEGIN
                IF HourAccountingLine.Monday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Monday, HourAccountingLine.Monday);
                  EntryNo += 1;
                END;
              END;

              2: BEGIN
                IF HourAccountingLine.Tuesday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Tuesday, HourAccountingLine.Tuesday);
                  EntryNo += 1;
                END;
              END;

              3: BEGIN
                IF HourAccountingLine.Wednesday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Wednesday, HourAccountingLine.Wednesday);
                  EntryNo += 1;
                END;
              END;

              4: BEGIN
                IF HourAccountingLine.Thursday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Thursday, HourAccountingLine.Thursday);
                  EntryNo += 1;
                END;
              END;

              5: BEGIN
                IF HourAccountingLine.Friday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Friday, HourAccountingLine.Friday);
                  EntryNo += 1;
                END;
              END;

              6: BEGIN
                IF HourAccountingLine.Saturday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Saturday, HourAccountingLine.Saturday);
                  EntryNo += 1;
                END;
              END;

              7: BEGIN
                IF HourAccountingLine.Sunday <> 0 THEN BEGIN
                  CreateFromHoursAccountingLine(HourAccountingLine, EntryNo, Days::Sunday, HourAccountingLine.Sunday);
                  EntryNo += 1;
                END;
              END;


            END;
            CountDays += 1;
          UNTIL CountDays = 8;

        UNTIL HourAccountingLine.NEXT = 0;

      END;


      //**** Create from posted lines *** //
      IF ShowPostedHours THEN BEGIN

        PostedHourLine.SETCURRENTKEY(Year,Week,"Project No.","Employee No.");

        IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
          //>>141104 ENH050
          //PostedHourLine.SETRANGE("Posting Date", PeriodStart, PeriodEnd)
          PostedHourLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          //<<141104 ENH050
        ELSE IF (PeriodStart <> 0D) THEN
          //>>141104 ENH050
          //PostedHourLine.SETFILTER("Posting Date", '%1..', PeriodStart)
          PostedHourLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          //<<141104 ENH050
        ELSE IF (PeriodEnd <> 0D) THEN
          //>>141104 ENH050
          //PostedHourLine.SETFILTER("Posting Date", '..%1', PeriodEnd);
          PostedHourLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);
          //<<141104 ENH050

        IF FilterProject <> '' THEN
          PostedHourLine.SETFILTER("Project No.", FilterProject)
      // >> ITERO.MK
        ELSE
          PostedHourLine.SETFILTER("Project No.", '<>%1', '');
      // << ITERO.MK

        IF FilterEmployee <> '' THEN
          PostedHourLine.SETFILTER("Employee No.", FilterEmployee);

        // >> ITERO.MK
        //IF WageCostCode <> '' THEN
        //  HourAccountingLine.SETFILTER("Wage Component", WageCostCode);
        //<< ITERO.MK

        IF PostedHourLine.FINDSET(FALSE) THEN REPEAT

          WITH TEMPHourAccoutBuffer DO BEGIN
            INIT;
            "Entry No." := EntryNo;

            IF Employee.GET(PostedHourLine."Employee No.") THEN BEGIN
