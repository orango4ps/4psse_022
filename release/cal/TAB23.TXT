    PROCEDURE GetVendorNoOpenCard@56(VendorText@1000 : Text[100];ShowVendorCard@1006 : Boolean) : Code[20];
    VAR
      Vendor@1001 : Record 23;
      MarketingSetup@1100527650 : Record 5079;
      VendorNo@1002 : Code[20];
      NoFiltersApplied@1007 : Boolean;
      VendorWithoutQuote@1003 : Text;
      VendorFilterFromStart@1004 : Text;
      VendorFilterContains@1005 : Text;
    BEGIN
      IF VendorText = '' THEN
        EXIT('');

      IF STRLEN(VendorText) <= MAXSTRLEN(Vendor."No.") THEN
        IF Vendor.GET(VendorText) THEN
          EXIT(Vendor."No.");

      VendorWithoutQuote := CONVERTSTR(VendorText,'''','?');

      Vendor.SETFILTER(Name,'''@' + VendorWithoutQuote + '''');
      IF Vendor.FINDFIRST THEN
        EXIT(Vendor."No.");
      Vendor.SETRANGE(Name);

      VendorFilterFromStart := '''@' + VendorWithoutQuote + '*''';

      Vendor.FILTERGROUP := -1;
      //**4PS.so
      //Vendor.SETFILTER("No.",VendorFilterFromStart);
      //Vendor.SETFILTER(Name,VendorFilterFromStart);
      //IF Vendor.FINDFIRST THEN
      //  EXIT(Vendor."No.");
      //**4PS.eo

      VendorFilterContains := '''@*' + VendorWithoutQuote + '*''';

      Vendor.SETFILTER("No.",VendorFilterContains);
      Vendor.SETFILTER(Name,VendorFilterContains);
      Vendor.SETFILTER(City,VendorFilterContains);
      Vendor.SETFILTER(Contact,VendorFilterContains);
      Vendor.SETFILTER("Phone No.",VendorFilterContains);
      Vendor.SETFILTER("Post Code",VendorFilterContains);

      //IF Vendor.COUNT = 0 THEN //**4PS.o
      MarkVendorsWithSimilarName(Vendor,VendorText);
      //**4PS.sn
      MarketingSetup.GET;
      IF MarketingSetup."Tradenames in Name Validation" THEN
        EXIT(GetVendorNoByNameAndTradename(VendorText, Vendor, ShowVendorCard));
      //**4PS.en

      IF Vendor.COUNT = 1 THEN BEGIN
        Vendor.FINDFIRST;
        EXIT(Vendor."No.");
      END;

      IF NOT GUIALLOWED THEN
        ERROR(SelectVendorErr);

      IF Vendor.COUNT = 0 THEN BEGIN
        IF Vendor.WRITEPERMISSION THEN
          CASE STRMENU(STRSUBSTNO('%1,%2',STRSUBSTNO(CreateNewVendTxt,VendorText),SelectVendTxt),1,VendNotRegisteredTxt) OF
            0:
              ERROR(SelectVendorErr);
            1:
              EXIT(CreateNewVendor(COPYSTR(VendorText,1,MAXSTRLEN(Vendor.Name)),ShowVendorCard));
          END;
        Vendor.RESET;
        NoFiltersApplied := TRUE;
      END;

      IF ShowVendorCard THEN
        VendorNo := PickVendor(Vendor,NoFiltersApplied)
      ELSE
        EXIT('');

      IF VendorNo <> '' THEN
        EXIT(VendorNo);

      ERROR(SelectVendorErr);
    END;

    LOCAL PROCEDURE MarkVendorsWithSimilarName@33(VAR Vendor@1001 : Record 23;VendorText@1000 : Text);
    VAR
      TypeHelper@1002 : Codeunit 10;
      VendorCount@1003 : Integer;
      VendorTextLenght@1004 : Integer;
      Treshold@1005 : Integer;
    BEGIN
      IF VendorText = '' THEN
        EXIT;
      IF STRLEN(VendorText) > MAXSTRLEN(Vendor.Name) THEN
        EXIT;
      VendorTextLenght := STRLEN(VendorText);
      Treshold := VendorTextLenght DIV 5;
      IF Treshold = 0 THEN
        EXIT;
      Vendor.RESET;
      Vendor.ASCENDING(FALSE); // most likely to search for newest Vendors
      IF Vendor.FINDSET THEN
        REPEAT
          VendorCount += 1;
          IF ABS(VendorTextLenght - STRLEN(Vendor.Name)) <= Treshold THEN
            IF TypeHelper.TextDistance(UPPERCASE(VendorText),UPPERCASE(Vendor.Name)) <= Treshold THEN
              Vendor.MARK(TRUE);
        UNTIL Vendor.MARK OR (Vendor.NEXT = 0) OR (VendorCount > 1000);
      Vendor.MARKEDONLY(TRUE);
    END;

    LOCAL PROCEDURE CheckDuplicateCOCNosForVendor@1100528502(COCRegistrationNo@1000 : Text[20];Number@1002 : Code[20]);
    VAR
      Vendor@1003 : Record 23;
      Check@1004 : Boolean;
      Finish@1001 : Boolean;
      t@1005 : Text[250];
    BEGIN
      //**4PS
      // Copied from Table 381 (CheckVendor)
      Check := TRUE;
      t := '';
      Vendor.SETRANGE("COC Registration No.",COCRegistrationNo);
      Vendor.SETFILTER("No.",'<>%1',Number);
      IF Vendor.FINDSET THEN BEGIN
        Check := FALSE;
        Finish := FALSE;
        REPEAT
          IF Vendor."No." <> Number THEN
            IF t = '' THEN
              t := Vendor."No."
            ELSE
              IF STRLEN(t) + STRLEN(Vendor."No.") + 5 <= MAXSTRLEN(t) THEN
                t := t + ', ' + Vendor."No."
              ELSE BEGIN
                t := t + '...';
                Finish := TRUE;
              END;
        UNTIL (Vendor.NEXT = 0) OR Finish;
      END;
      IF NOT Check THEN
        MESSAGE(Text11012009,FIELDCAPTION("COC Registration No."),t);
    END;

    PROCEDURE GetVendorNoByNameAndTradename@1100527650(VAR VendorName@1000 : Text[50];VAR VendorMarked@1100525000 : Record 23;ShowVendorCard@1100525001 : Boolean) : Code[20];
    VAR
      TempVendor@1001 : TEMPORARY Record 23;
      VendorNo@1002 : Code[20];
    BEGIN
      //**4PS
      FillTempVendorIncludingTradenames(TempVendor, VendorName, VendorMarked);
      TempVendor.SETRANGE(Name,VendorName);
      IF TempVendor.COUNT = 1 THEN BEGIN
        TempVendor.FINDFIRST;
        VendorName := TempVendor.Name;
        EXIT(TempVendor."Temporary Vendor No.");
      END;

      CLEAR(TempVendor);
      TempVendor.SETFILTER(Name,'''@*' + VendorName + '*''');
      CASE TempVendor.COUNT OF
        0: ERROR(SelectVendorErr);
        1:
          BEGIN
            TempVendor.FINDFIRST;
            VendorName := TempVendor.Name;
            EXIT(TempVendor."Temporary Vendor No.");
          END;
        ELSE BEGIN
          COMMIT;
          IF ShowVendorCard THEN BEGIN
            VendorNo := PickVendorIncludingTradenames(TempVendor);
            IF VendorNo = '' THEN
              ERROR(SelectVendorErr);
            VendorName := TempVendor.Name;
            EXIT(VendorNo);
          END ELSE
            EXIT('');
        END;
      END;
    END;

    LOCAL PROCEDURE FillTempVendorIncludingTradenames@1100527652(VAR TempVendor@1100525000 : TEMPORARY Record 23;VendorName@1100528802 : Text;VAR VendorMarked@1100525002 : Record 23);
    VAR
      Vendor@1100525001 : Record 23;
      Contact@1100525003 : Record 5050;
      ContactTradename@1100528800 : Record 11229299;
      counter@1100525005 : Integer;
    BEGIN
      //**4PS
      IF VendorMarked.FINDSET THEN
        REPEAT
          TempVendor.TRANSFERFIELDS(VendorMarked);
          TempVendor."Temporary Vendor No." := VendorMarked."No.";
          TempVendor.INSERT;
        UNTIL VendorMarked.NEXT = 0;

      ContactTradename.SETFILTER(Tradename,'''@*' + VendorName + '*''');
      IF ContactTradename.FINDSET THEN
        REPEAT
          IF Contact.GET(ContactTradename."Contact No.") THEN
            IF Contact.GetVendorByContact(Contact, Vendor) THEN BEGIN
              TempVendor.INIT;
              TempVendor.TRANSFERFIELDS(Vendor);
              TempVendor."No." := COPYSTR(Vendor."No.", 1, 16) + 'TN' + FORMAT(counter);
              TempVendor.INSERT;
              TempVendor.Name := ContactTradename.Tradename;
              TempVendor."Temporary Vendor No." := Vendor."No.";
              TempVendor.MODIFY;
              counter += 1;
            END;
        UNTIL ContactTradename.NEXT = 0;
    END;
