    LOCAL PROCEDURE ValidateInvoiceDiscountAmount@30();
    VAR
      PurchaseHeader@1000 : Record 38;
      ConfirmManagement@1001 : Codeunit 27;
    BEGIN
      PurchaseHeader.GET("Document Type","Document No.");
      IF PurchaseHeader.InvoicedLineExists THEN

          EXIT;

    END;
