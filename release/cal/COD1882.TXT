    [Integration]

    END;

    LOCAL PROCEDURE RaiseEventForEveryCompany@4();
    VAR
      Company@1000 : Record 2000000006;
    BEGIN
      IF Company.FINDSET THEN
        REPEAT
          OnClearConfiguration(Company.Name);
        UNTIL Company.NEXT = 0;
      OnClearConfiguration('');
    END;

    BEGIN
    END.
