    PROCEDURE CalcHrsMobile@1100525004(IStartTime@1100525000 : Time;IEndTime@1100525001 : Time;VAR OHours@1100525002 : Decimal);
    VAR
      HoursMob@1100525007 : Decimal;
    BEGIN
      OHours := 0;

      IF (IEndTime > IStartTime) THEN BEGIN
        HoursMob := (IEndTime - IStartTime) / (1000*60*60);
        HoursMob := ROUND(HoursMob, 0.01);
        IF HoursMob > 0 THEN
          OHours := HoursMob;
      END;
    END;

    PROCEDURE GetWageComponent@1100528801(DebriefHourAccounting@1100526002 : Record 11012685) : Code[10];
    VAR
      CostCodeTable@1100528802 : Record 11012083;
      MobileSetup@1100526003 : Record 11012676;
      WorkOrder@1100526000 : Record 11229279;
      HourLineTemplateLine@1100526001 : Record 11012078;
      ReceivingCompany@1100526004 : Text;
    BEGIN
      IF DebriefHourAccounting."Wage Component" <> '' THEN
        EXIT(DebriefHourAccounting."Wage Component");

      IF WorkOrder.GET(DebriefHourAccounting."Work Order No.") THEN
        ReceivingCompany := WorkOrder."Source Company"
      ELSE
        ReceivingCompany := COMPANYNAME;

      CostCodeTable.CHANGECOMPANY(ReceivingCompany);
      IF CostCodeTable.GET(DebriefHourAccounting."Cost Code") THEN
        EXIT(CostCodeTable.Component);

      IF DebriefHourAccounting."Hour Line Template Code" <> '' THEN BEGIN
        IF NOT MobileSetup.GET THEN
          EXIT;
        IF MobileSetup."Hours Destination" = MobileSetup."Hours Destination"::"Hour Accounting" THEN
          HourLineTemplateLine.SETRANGE(Source, HourLineTemplateLine.Source::"Hour Accounting")
        ELSE
          HourLineTemplateLine.SETRANGE(Source, HourLineTemplateLine.Source::"Site Mngmnt. Hours");
        HourLineTemplateLine.SETRANGE("Receiving Company", ReceivingCompany);
        HourLineTemplateLine.SETRANGE("Template Code", DebriefHourAccounting."Hour Line Template Code");
        HourLineTemplateLine.SETFILTER("Start Date", '<=%1', DebriefHourAccounting.Date);
        IF HourLineTemplateLine.FINDLAST THEN BEGIN
          IF (HourLineTemplateLine."Wage Component" <> '') THEN
            EXIT(HourLineTemplateLine."Wage Component");
          HourLineTemplateLine.CALCFIELDS(Type);
          IF (HourLineTemplateLine.Type = HourLineTemplateLine.Type::Indirect) AND (HourLineTemplateLine."Object No." <> '') THEN BEGIN
            IF CostCodeTable.GET(HourLineTemplateLine."Object No.") THEN
              EXIT(CostCodeTable.Component);
          END;
        END;
      END;
    END;

    BEGIN
    END.
