    PROCEDURE FindPeriod@3(Which@1000 : 'Previous,Next');
    BEGIN
      CASE Which OF
        Which::Previous:
          "Starting Date" := CALCDATE('<-1W>',"Starting Date");
        Which::Next:
          "Starting Date" := CALCDATE('<+1W>',"Starting Date");
      END;
      MODIFY;
    END;
