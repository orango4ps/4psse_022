    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      ProjectFilter0@1100529601 : Text;
      ProjectFilter2@1100529600 : Text;
    BEGIN
      ProjectFilter0 := Project.GETFILTERS;
      Project.FILTERGROUP(2);
      ProjectFilter2 := Project.GETFILTERS;
      Project.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Project.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Project.TABLECAPTION, ProjectFilter0));
    END;

    LOCAL PROCEDURE GetComments@1100529600();
    VAR
      Prognosis@1100529600 : Record 11012034;
      PrognosisLine@1100529601 : Record 11012035;
      CodeLoc@1100529602 : Code[10];
    BEGIN
      TmpReportLine.RESET;
      TmpReportLine.DELETEALL;

      PrognosisDate := FilterDate;
      Prognosis.SETRANGE("Project No.", Project."No.");
      Prognosis.SETRANGE("Prognosis Date", 0D, FilterDate);
      IF Prognosis.ISEMPTY THEN
        EXIT;

      Prognosis.FINDLAST;
      PrognosisDate := Prognosis."Prognosis Date";
      CodeLoc := '00000';

      IF Prognosis.Comment <> '' THEN
        AddReportLine(CodeLoc, '', '', Prognosis.Comment);

      PrognosisLine.SETRANGE("Project No.", Project."No.");
      PrognosisLine.SETRANGE("Prognosis Date", PrognosisDate);
      PrognosisLine.SETFILTER(Comment, '<>%1', '');
      PrognosisLine.SETAUTOCALCFIELDS("Cost Object Description", "Project Related Name");
      IF PrognosisLine.FINDSET THEN
        REPEAT
          IF PrognosisLine."Project Related Name" = '' THEN
            AddReportLine(CodeLoc, PrognosisLine."Cost Object", PrognosisLine."Cost Object Description", PrognosisLine.Comment)
          ELSE
            AddReportLine(CodeLoc, PrognosisLine."Cost Object", PrognosisLine."Project Related Name", PrognosisLine.Comment);
        UNTIL PrognosisLine.NEXT = 0;
    END;

    LOCAL PROCEDURE AddReportLine@1100529604(VAR Code1@1100529600 : Code[10];Code2@1100529601 : Code[20];CostObjectDescription@1100529602 : Text[250];CommentTxt@1100529603 : Text[250]);
    BEGIN

      TmpReportLine."Key Code2" := Code2;
      TmpReportLine.Text1 := CostObjectDescription;
      TmpReportLine.Text2 := CommentTxt;
      TmpReportLine.INSERT;
    END;

    LOCAL PROCEDURE GetFilterDate@1100529608();
    VAR
      Prognosis@1100529600 : Record 11012034;
    BEGIN
      IF ProjectNo = '' THEN BEGIN
        FilterDate := 0D;
        EXIT;
      END;

      IF NOT Prognosis.GET(ProjectNo, FilterDate) THEN
        FilterDate := 0D
      ELSE
        IF NOT Prognosis."Select For Reporting" THEN
          FilterDate := 0D;

      IF FilterDate = 0D THEN BEGIN
        Prognosis.SETRANGE("Project No.", ProjectNo);
        Prognosis.SETRANGE("Select For Reporting", TRUE);
        IF Prognosis.FINDLAST THEN
          FilterDate := Prognosis."Prognosis Date";
        EXIT;
      END;
    END;
