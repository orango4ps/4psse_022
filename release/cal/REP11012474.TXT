    LOCAL PROCEDURE GetReportFilter@1100529600() : Text;
    VAR
      ProjectFilter0@1100529601 : Text;
      ProjectFilter2@1100529600 : Text;
    BEGIN
      ProjectFilter0 := Project.GETFILTERS;
      Project.FILTERGROUP(2);
      ProjectFilter2 := Project.GETFILTERS;

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN

        EXIT(STRSUBSTNO('%1: %2, %3', Project.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Project.TABLECAPTION, ProjectFilter0));
    END;

    LOCAL PROCEDURE GetTmpCostCompAmtLine@1100529602();
    BEGIN
      TmpCostCompAmounts.SETRANGE("Key Code1", "Tender Paragraph Distribution"."Cost Component");
      IF TmpCostCompAmounts.FINDFIRST THEN
        EXIT;

      TmpCostCompAmounts.INIT;
      TmpCostCompAmounts."Key Code1" := "Tender Paragraph Distribution"."Cost Component";
      CalcCostObjAmounts;
      TmpCostCompAmounts.INSERT;
    END;

    LOCAL PROCEDURE CalcCostObjAmounts@1210190001();
    VAR
      CostObjectControl@1100529609 : Record 11012047;
      CostControlMgt@1100529610 : Codeunit 11012006;
      Allowed@1100529608 : Decimal;
      AllowedHours@1100529607 : Decimal;
      PrevAllowed@1100529606 : Decimal;
      PrevAllowedHours@1100529605 : Decimal;
      Purchase@1100529604 : Decimal;
      EstimPurchResult@1100529603 : Decimal;
      PurchCost@1100529602 : Decimal;
      ActualPurchResult@1100529601 : Decimal;
      TotPurchResult@1100529600 : Decimal;
    BEGIN
      CostObjectControl.SETRANGE("Project No.", Project."No.");
      CostObjectControl.SETRANGE("Cost Component", "Tender Paragraph Distribution"."Cost Component");
      IF CostObjectControl.ISEMPTY THEN
        EXIT;

      CostObjectControl.FINDSET;
      REPEAT
        CostObjectControl.SETFILTER("Period Filter", Project.GETFILTER("Period Filter"));
        CostObjectControl.SETFILTER("Cumulation Filter", Project.GETFILTER("Cumulation Filter"));
        CostObjectControl.SETFILTER("Progress Filter", Project.GETFILTER("Progress Filter"));
        CostObjectControl.SETFILTER(PrevProgressFilter, Project.GETFILTER(PrevProgressFilter));
        CostObjectControl.SETFILTER("Prognosis Filter", Project.GETFILTER("Prognosis Filter"));
        CostObjectControl.SETFILTER(PrevPrognosisFilter, Project.GETFILTER(PrevPrognosisFilter));
        CostObjectControl.SETFILTER("Purchase Action Filter", Project.GETFILTER("Purchase Action Filter"));

        CostControlMgt.CalcFieldsCostObj(CostObjectControl);

        TmpCostCompAmounts.Dec1 += CostObjectControl."Budget Hours";
        TmpCostCompAmounts.Dec2 += CostObjectControl."Actual Hours";
        CostControlMgt.AllowedAndPurchCostObj(
          CostObjectControl,
          Allowed, AllowedHours, PrevAllowed, PrevAllowedHours,
          Purchase, EstimPurchResult, PurchCost, ActualPurchResult, TotPurchResult, TRUE);
        TmpCostCompAmounts.Dec3 += AllowedHours;
        TmpCostCompAmounts.Dec4 += CostObjectControl."Allowed Hours";
        TmpCostCompAmounts.Dec5 += AllowedHours - CostObjectControl."Actual Hours";
        TmpCostCompAmounts.Dec6 += CostObjectControl.Budget;
        TmpCostCompAmounts.Dec7 += CostControlMgt.TotCostCostObj(CostObjectControl, FALSE);
        TmpCostCompAmounts.Dec8 += Allowed;
        TmpCostCompAmounts.Dec9 += CostControlMgt.PurchResultCostObj(CostObjectControl, FALSE);
      UNTIL CostObjectControl.NEXT = 0;
    END;

    LOCAL PROCEDURE CalcTotRevenueAmt@1100529601();
    VAR
      TenderParagrDistr@1100529600 : Record 11012479;
    BEGIN
      TenderParagrDistrLinesCount := 0;
      TenderParagrDistrTotalLinesCount := 0;
      TotRevenueAmt := 0;
      CheckRevenueAmt := 0;
      DiffRevenueAmt := 0;

      TenderParagrDistr.SETRANGE("Project No.", "Tender Paragraph by Principal"."Project No.");
      TenderParagrDistr.SETRANGE(Principal, "Tender Paragraph by Principal".Principal);
      TenderParagrDistr.SETRANGE("Contract No.", "Tender Paragraph by Principal"."Contract No.");
      TenderParagrDistr.SETRANGE("Tender Paragraph", "Tender Paragraph by Principal"."No.");
      TenderParagrDistr.SETFILTER("Distribution Percentage", '<>0');
      IF TenderParagrDistr.ISEMPTY THEN
        EXIT;

      TenderParagrDistr.CALCSUMS("Distribution Percentage");
      TotRevenueAmt := ROUND("Tender Paragraph by Principal"."Total Amount (LCY)" * TenderParagrDistr."Distribution Percentage" / 100, 0.01);

      TenderParagrDistrTotalLinesCount := TenderParagrDistr.COUNT;
    END;

    LOCAL PROCEDURE SetDateFilters@1100529604();

      IF Project.GETFILTER("Period Filter") = '' THEN
        Project.SETFILTER("Period Filter", Project.GETFILTER("Cumulation Filter"));
    END;

    BEGIN
    END.
