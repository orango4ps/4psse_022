    PROCEDURE SendCreditMemoElectronic@1210190009(SelectSendOption@1100528701 : Boolean;FromBatchPost@1100528203 : Boolean);
    VAR
      SalesCrMemoHeader@1210190002 : Record 114;
      SelectSendOptionSalesHdr@1210190004 : Page 11126492;
      SendCrMemoGS1@1100528201 : Report 11012398;
      SendCrMemoDigipoort@1100528202 : Report 11229763;
      SalesRecSetup@1100285000 : Record 311;
      GLSetup@1100285501 : Record 98;
      CreateDKElectronicInvoices@1000 : Report 11128571;
      SvefakturaSetup@111280000 : Record 11128035;
    BEGIN
      //**4PS, DP00141, Digipoort Invoice
      SalesCrMemoHeader.COPY(Rec);
      SalesCrMemoHeader.SETRECFILTER;

      IF (SalesCrMemoHeader."Electronic Invoicing" = SalesCrMemoHeader."Electronic Invoicing"::Print)
      THEN BEGIN
        IF GUIALLOWED THEN
          MESSAGE(Text11012001,
            SalesCrMemoHeader.TABLECAPTION, SalesCrMemoHeader.FIELDCAPTION("Electronic Invoicing"),
            SalesCrMemoHeader."Electronic Invoicing");
        ERROR('');
      END;

      IF SelectSendOption THEN BEGIN
        SelectSendOptionSalesHdr.SetSelectEmail(
          SalesCrMemoHeader."Electronic Invoicing" =
          SalesCrMemoHeader."Electronic Invoicing"::"E-Mail");
        SelectSendOptionSalesHdr.SetSelectGLNXML(
          SalesCrMemoHeader."Electronic Invoicing" =
          SalesCrMemoHeader."Electronic Invoicing"::"GLN (XML)");
        SelectSendOptionSalesHdr.SetSelectDigipoort(
          SalesCrMemoHeader."Electronic Invoicing" =
          SalesCrMemoHeader."Electronic Invoicing"::eVerbinding);
          // 140702 <<
        SelectSendOptionSalesHdr.SetSelectSvefaktura(
          SalesCrMemoHeader."Electronic Invoicing" =
          SalesCrMemoHeader."Electronic Invoicing"::"Svefaktura (XML)");
          //140702 >>
      //>>4PSSE.DK
        SelectSendOptionSalesHdr.SetSelectOIOUBL(
          SalesCrMemoHeader."Electronic Invoicing" =
          SalesCrMemoHeader."Electronic Invoicing"::"OIOUBL (DK)");
      //<<4PSSE.DK
        IF (SelectSendOptionSalesHdr.RUNMODAL <> ACTION::LookupOK) THEN
          EXIT;
      END;

      //User must have option to stop the send option after the menu-choice in the page/form
      //For this "NOT SelectSendOption" can be used. because with selectoption the choice to stop is already there
      IF ((NOT SelectSendOption) AND
          (SalesCrMemoHeader."Electronic Invoicing" = SalesCrMemoHeader."Electronic Invoicing"::"E-Mail")) OR
         (SelectSendOptionSalesHdr.GetSelectEmail)
      THEN BEGIN
        IF FromBatchPost THEN
          SalesCrMemoHeader.EmailRecordFromBatchPost
        ELSE
          SalesCrMemoHeader.EmailRecords(TRUE);
      END;

      GLSetup.GET;  //NAVNO

      IF ((NOT SelectSendOption) AND
          (NOT GLSetup."Norwegian Localization Active") AND //NAVNO
          (SalesCrMemoHeader."Electronic Invoicing" = SalesCrMemoHeader."Electronic Invoicing"::"GLN (XML)")) OR
         (SelectSendOptionSalesHdr.GetSelectGLNXML)
      THEN BEGIN
        SendCrMemoGS1.USEREQUESTPAGE := NOT SelectSendOption;
        SendCrMemoGS1.SetReportSelection(FromBatchPost);
        SendCrMemoGS1.SETTABLEVIEW(SalesCrMemoHeader);
        SendCrMemoGS1.RUNMODAL;
      END;

      IF ((NOT SelectSendOption) AND
          (SalesCrMemoHeader."Electronic Invoicing" = SalesCrMemoHeader."Electronic Invoicing"::eVerbinding)) OR
         (SelectSendOptionSalesHdr.GetSelectDigipoort)

    PROCEDURE PrintCrMemoIncludingAttachments@1100528407(FromBatch@1100528402 : Boolean);
    VAR
      SalesCrMemoHeader@1100528400 : Record 114;
      TempReportSelections@1100528401 : TEMPORARY Record 77;
      TempEmailAttachment@1100528404 : TEMPORARY Record 11229294;
      DocumentPrint@1100528403 : Codeunit 229;
      PDFFileManagement@1100528405 : Codeunit 11020228;
      UserInteractionAllowed@1100528406 : Boolean;
    BEGIN
      //**4PS
      UserInteractionAllowed := NOT FromBatch;
      SalesCrMemoHeader.COPY(Rec);
      SalesCrMemoHeader.SETRECFILTER;
      DocumentPrint.GetReportSelectionForPrintUsage(
        TempReportSelections.Usage::"S.Cr.Memo", SalesCrMemoHeader."Bill-to Customer No.",
        UserInteractionAllowed, TempReportSelections);
      TempReportSelections.FINDSET;
      REPEAT
        CreatePDF(TempReportSelections."Report ID", TRUE, TempEmailAttachment, SalesCrMemoHeader);
      UNTIL TempReportSelections.NEXT = 0;
      AddRelatedDocuments(
        TempEmailAttachment, SalesCrMemoHeader, TRUE, FALSE);
      PDFFileManagement.MergePDFEmailAttachmentsOnServer(TempEmailAttachment);

      IF TempEmailAttachment.FINDFIRST THEN BEGIN
        TempEmailAttachment."Attachment Name" :=
          STRSUBSTNO('%1 %2.pdf', DocTxt, SalesCrMemoHeader."No.");
        IF UserInteractionAllowed THEN
          TempEmailAttachment.DownloadPdfAttachment
        ELSE
          TempEmailAttachment.PrintPdfAttachment;
        ERASE(TempEmailAttachment."Attachment Server File Path");
      END;
    END;
