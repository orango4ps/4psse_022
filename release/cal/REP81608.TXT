    PROCEDURE CreateInvoicedData@1100285001();
    VAR
      JobsSetup@1100285005 : Record 315;
      VendorLedgerEntry@1100285000 : Record 25;
      Text100@1100285006 : TextConst 'ENU=Adjustment Journal Batch must have a value in Jobs Setup.;NOR=Justering Journalbunke m† ha en verdi i Jobbinnstillinger';
      PurchInvLine@1100285002 : Record 123;
      PurchCrMemoLine@1100285011 : Record 125;
      Division@1100285008 : Code[20];
      Region@1100285009 : Code[20];
      First@1100285014 : Boolean;
      Invoice@1100285010 : Boolean;
    BEGIN
      TempVendorPurchaseBuffer.DELETEALL;
      JobsSetup.GET;

      //*** Create Invoice Data ***
      IF VendorFilter <> '' THEN
        PurchInvLine.SETFILTER("Buy-from Vendor No.", VendorFilter);
      IF PostDate <> '' THEN
        PurchInvLine.SETFILTER("Posting Date", PostDate);
      IF FilterCostCenter <> '' THEN
        PurchInvLine.SETFILTER("Shortcut Dimension 1 Code", FilterCostCenter);
      IF FilterJob <> '' THEN
        PurchInvLine.SETFILTER("Job No.", FilterJob);
      //>>ITERO.GH 140630
      //IF DimCostObject <> '' THEN
      //  PurchInvLine.SETFILTER("Shortcut Dimension 2 Code", DimCostObject);
      //<<ITERO.GH 140630
      PurchInvLine.SETFILTER("Amount (LCY)", '<>%1', 0);
      IF PurchInvLine.FINDSET(FALSE) THEN REPEAT

        //If the record already exists, add the amount otherwise create a new record
        IF TempVendorPurchaseBuffer.GET(USERID, PurchInvLine."Buy-from Vendor No.",
                           PurchInvLine."Shortcut Dimension 1 Code", PurchInvLine."Shortcut Dimension 2 Code") THEN

          AddExistentPurchVendorCalc(TempVendorPurchaseBuffer, TRUE, PurchInvLine."Amount (LCY)")

        ELSE
          CreateNewPurchVendorCalc(PurchInvLine."Buy-from Vendor No.", PurchInvLine."Shortcut Dimension 1 Code",
                              PurchInvLine."Shortcut Dimension 2 Code", PurchInvLine."Amount (LCY)", TRUE);

      UNTIL PurchInvLine.NEXT = 0;

      //*** Create Credit Data ***
      IF VendorFilter <> '' THEN
        PurchCrMemoLine.SETFILTER("Buy-from Vendor No.", VendorFilter);
      IF PostDate <> '' THEN
        PurchCrMemoLine.SETFILTER("Posting Date", PostDate);
      IF FilterCostCenter <> '' THEN
        PurchCrMemoLine.SETFILTER("Shortcut Dimension 1 Code", FilterCostCenter);
      IF FilterJob <> '' THEN
        PurchCrMemoLine.SETFILTER("Job No.", FilterJob);
      //>>ITERO.GH 140630
      //IF DimCostObject <> '' THEN
      //  PurchCrMemoLine.SETFILTER("Shortcut Dimension 2 Code", DimCostObject);
      //<<ITERO.GH 140630
      PurchCrMemoLine.SETFILTER("Amount (LCY)", '<>%1', 0);

      IF PurchCrMemoLine.FINDSET(FALSE) THEN REPEAT

        //If the record already exists, add the amount otherwise create a new record
        IF TempVendorPurchaseBuffer.GET(USERID, PurchCrMemoLine."Buy-from Vendor No.",
                           PurchCrMemoLine."Shortcut Dimension 1 Code", PurchCrMemoLine."Shortcut Dimension 2 Code") THEN

          AddExistentPurchVendorCalc(TempVendorPurchaseBuffer, FALSE, PurchCrMemoLine."Amount (LCY)")

        ELSE
          CreateNewPurchVendorCalc(PurchCrMemoLine."Buy-from Vendor No.", PurchCrMemoLine."Shortcut Dimension 1 Code",
                              PurchCrMemoLine."Shortcut Dimension 2 Code", PurchCrMemoLine."Amount (LCY)", FALSE);

      UNTIL PurchCrMemoLine.NEXT = 0;
    END;

    PROCEDURE CreateNewPurchVendorCalc@1100285006(VendorNo@1100285008 : Code[20];CostCenter@1100285002 : Code[10];CostObject@1100285003 : Code[10];Amt@1100285004 : Decimal;Invoice@1100285005 : Boolean);
    VAR
      PurchaseVendorBonusCalcul@1100285000 : Record 81608;
      Vendor@1100285001 : Record 23;
      GeneralLedgerSetup@1100285006 : Record 98;
      DimensionValueChains@1100285011 : Record 11128020;
      DimensionValue@1100285007 : Record 349;
      DimCode@1100285009 : Code[20];
      DimValue@1100285010 : Code[20];
    BEGIN
      GeneralLedgerSetup.GET;
      DimCode := '';
      DimValue := '';

      TempVendorPurchaseBuffer.INIT;
      TempVendorPurchaseBuffer.UserID := USERID;
      TempVendorPurchaseBuffer."Vendor No." := VendorNo;
      TempVendorPurchaseBuffer."Cost Center" := CostCenter;
      TempVendorPurchaseBuffer."Cost Object" := CostObject;
      IF  Invoice THEN
        TempVendorPurchaseBuffer.Invoiced := Amt
      ELSE
        TempVendorPurchaseBuffer.Credited := Amt;

      IF Vendor.GET(VendorNo) THEN
        TempVendorPurchaseBuffer."Vendor Name" := Vendor.Name;

      DimensionValueChains.GetDimNextInChain(GeneralLedgerSetup."Global Dimension 1 Code", CostCenter,DimCode,DimValue);
      WHILE (DimCode <> '') AND (DimCode <> GeneralLedgerSetup."Shortcut Dimension 6 Code") DO
        DimensionValueChains.GetDimNextInChain(DimCode,DimValue,DimCode,DimValue);
      TempVendorPurchaseBuffer.Divison := DimValue;

      DimensionValueChains.GetDimNextInChain(GeneralLedgerSetup."Global Dimension 1 Code", CostCenter,DimCode,DimValue);
      WHILE (DimCode <> '') AND (DimCode <> GeneralLedgerSetup."Shortcut Dimension 5 Code") DO
        DimensionValueChains.GetDimNextInChain(DimCode,DimValue,DimCode,DimValue);
      TempVendorPurchaseBuffer.Region := DimValue;

      IF DimensionValue.GET(GeneralLedgerSetup."Global Dimension 1 Code", CostCenter) THEN
        TempVendorPurchaseBuffer."Cost Center Name" := DimensionValue.Name;

      TempVendorPurchaseBuffer.INSERT;
    END;

    PROCEDURE AddExistentPurchVendorCalc@1100285007(VAR PurchaseVendorBonusCalcul@1100285000 : Record 81608;Invoice@1100285001 : Boolean;Amt@1100285002 : Decimal);
    BEGIN
      WITH TempVendorPurchaseBuffer DO BEGIN
        IF Invoice THEN
          Invoiced += Amt
        ELSE
          Credited += Amt;
        MODIFY;
      END;
    END;
