    LOCAL PROCEDURE PostEntries@1100529602();
    VAR
      GenJournalLine@1100529600 : Record 81;
      CustLedgerEntry@1100529603 : Record 21;
      GenJnlPostLine@1100529601 : Codeunit 12;
    BEGIN
      IF NOT Post THEN
        EXIT;

      TmpGenJournalLine.SETFILTER(Amount, '<>%1', 0);
      IF TmpGenJournalLine.ISEMPTY THEN
        EXIT;

      TmpGenJournalLine.FINDSET;
      REPEAT
        CLEAR(GenJournalLine);
        GenJournalLine."Posting Date" := TmpGenJournalLine."Posting Date";
        GenJournalLine."Document No." := TmpGenJournalLine."Document No.";
        GenJournalLine.VALIDATE("Account No.", TmpGenJournalLine."Account No.");
        GenJournalLine.VALIDATE("Bal. Account No.", TmpGenJournalLine."Bal. Account No.");
        GenJournalLine."Gen. Bus. Posting Group" := '';
        GenJournalLine."VAT Bus. Posting Group" := '';
        GenJournalLine."Gen. Prod. Posting Group" := '';
        GenJournalLine.VALIDATE("VAT Prod. Posting Group", '');
        GenJournalLine."Bal. Gen. Bus. Posting Group" := '';
        GenJournalLine."Bal. VAT Bus. Posting Group" := '';
        GenJournalLine."Bal. Gen. Prod. Posting Group" := '';
        GenJournalLine.VALIDATE("Bal. VAT Prod. Posting Group", '');
        IF (GenJournalLine."Shortcut Dimension 1 Code" = '') AND (TmpGenJournalLine."Shortcut Dimension 1 Code" <> '') THEN
          GenJournalLine.VALIDATE("Shortcut Dimension 1 Code", TmpGenJournalLine."Shortcut Dimension 1 Code");
        IF (GenJournalLine."Shortcut Dimension 2 Code" = '') AND (TmpGenJournalLine."Shortcut Dimension 2 Code" <> '') THEN
          GenJournalLine.VALIDATE("Shortcut Dimension 2 Code", TmpGenJournalLine."Shortcut Dimension 2 Code");
        GenJournalLine.Description := TmpGenJournalLine.Description;
        GenJournalLine.VALIDATE(Amount, TmpGenJournalLine.Amount);
        GenJournalLine."System-Created Entry" := TRUE;

        GenJnlPostLine.RUN(GenJournalLine);

        IF TmpGenJournalLine."Incoming Document Entry No." <> 0 THEN BEGIN
          CustLedgerEntry.GET(TmpGenJournalLine."Incoming Document Entry No.");
          IF PostDubious THEN
            CustLedgerEntry."Dubious Posted on" := TODAY
          ELSE
            CustLedgerEntry."Addition Dubious Posted on" := TODAY;
          CustLedgerEntry.MODIFY;
        END;
      UNTIL TmpGenJournalLine.NEXT = 0;
    END;

    BEGIN
    END.
