    PROCEDURE CopyTechSpecsFromPlantType@1100485001();
    VAR
      lvPlanttypeRec@1100485004 : Record 11012551;
      lvPlantProfileAnswerRec@1100485000 : Record 11012582;
      lvPlantProfileAnswerRec2@1100485001 : Record 11012582;
      lvText001@1100485002 : TextConst 'DEU=Es sind keine technischen Spezifikationen bei Werkzeugart ''%1'' vorhanden.;ENU=No Technical Specifications present of Plant Type ''%1''.;NLD=Er zijn geen technische specificatie aanwezig bij materieelsoort ''%1''.;NOR=Det finnes ingen tekniske Spesifikasjoner for maskintype ''%1''.;SVE=Det finns inga tekniska specifikationer fîr Maskinstyp ''%1''.';
      lvText002@1100485003 : TextConst 'DEU=Technische Spezifikationen von einer anderen Werkzeugart kopieren?;ENU=Copy Techninical Specifications from an other Plant Type?;NLD=Technische specificaties van een andere materieelsoort kopiâren?;NOR=Kopier tekniske Spesifikasjoner fra en annen maskintype?;SVE=Kopiera tekniska specifikationer frÜn en annan Maskinstyp?';
    BEGIN
      TESTFIELD(Code);

      IF NOT CONFIRM(lvText002, TRUE) THEN
        EXIT;

      lvPlanttypeRec.FILTERGROUP(6);
      lvPlanttypeRec.SETFILTER(Code, '<>%1', Code);  //* Alleen kopieren van een andere materieelsoort
      lvPlanttypeRec.FILTERGROUP(0);
      lvPlanttypeRec.Code := Code;  //* Ivm pointer in overzichtscherm (in de buurt van huidige materieelsoort)
      IF PAGE.RUNMODAL(0, lvPlanttypeRec) = ACTION::LookupOK THEN BEGIN
        IF (lvPlanttypeRec.Code <> Code) AND (lvPlanttypeRec.Code <> '') THEN BEGIN
          lvPlantProfileAnswerRec.SETRANGE(Level, lvPlantProfileAnswerRec.Level::PlantType);
          lvPlantProfileAnswerRec.SETRANGE("Plant Type", lvPlanttypeRec.Code);
          IF NOT lvPlantProfileAnswerRec.FIND('-') THEN
            ERROR(lvText001, lvPlanttypeRec.Code);

          REPEAT
