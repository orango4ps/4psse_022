    PROCEDURE FillTmpInvPropRubricTable@1210190005(IProjectNo@1210190000 : Code[20];IVersion@1210190001 : Code[10]);
    VAR
      lvInvPropRubricRec@1210190002 : Record 11012913;
    BEGIN
      WITH lvInvPropRubricRec DO BEGIN
        SETRANGE("Project No.", IProjectNo);
        SETRANGE(Version, IVersion);
        IF (RubricFilterTxt <> '') THEN
          SETFILTER("Rubric Code", RubricFilterTxt);
        IF FINDSET THEN BEGIN
          REPEAT
            IF NOT TmpInvPropRubricRec.GET(IProjectNo, 'TOTAL', "Rubric Code") THEN BEGIN
              TmpInvPropRubricRec := lvInvPropRubricRec;
              TmpInvPropRubricRec.Version := 'TOTAL';
              TmpInvPropRubricRec.INSERT;
            END;
          UNTIL NEXT = 0;
        END;
      END;
    END;

    PROCEDURE CalculateRubricActual@1210190007(INumber@1210190000 : Integer) : Decimal;
    BEGIN
      WITH InvPropRubricRec DO BEGIN
        CASE Type OF
          Type::Costs,
          Type::Revenues,
          Type::"Percentage Costs",
          Type::"Percentage Revenues",
          Type::"Costs as Perc. Revenues",
          Type::"Revenues as Perc. Costs",
          Type::"Costs to Distr. as Perc. Rev.",
          Type::Subtotal,
          Type::Balance:
            CASE INumber OF
              1:
                EXIT(ProjDevelopmentCU.CalculateActualByRubric(InvPropRubricRec, 0D, 0D, RefDate1));
              2:
                EXIT(ProjDevelopmentCU.CalculateActualByRubric(InvPropRubricRec, PeriodStartDate, PeriodEndDate, RefDate2));
            END;
          END;
        EXIT(0);
      END;
    END;

    PROCEDURE CalculateRubricBudget@1210190011(INumber@1210190000 : Integer) : Decimal;
    BEGIN
      WITH InvPropRubricRec DO BEGIN
        CALCFIELDS("Reference Date");
        CASE Type OF
          Type::Costs,
          Type::Revenues,
          Type::"Percentage Costs",
          Type::"Percentage Revenues",
          Type::"Costs as Perc. Revenues",
          Type::"Revenues as Perc. Costs",
          Type::"Costs to Distr. as Perc. Rev.",
          Type::Subtotal,
          Type::Balance:
            CASE INumber OF
              1:
                EXIT(ProjDevelopmentCU.CalculateBudgetByRubric(InvPropRubricRec, 0D, 0D, 0D));
              2:
                EXIT(ProjDevelopmentCU.CalculateBudgetByRubric(InvPropRubricRec, 0D, 0D, RefDate2));
            END;
          Type::ROS:
            EXIT(ProjDevelopmentCU.CalculateTotalRos("Project No.", Version, "Reference Date", TRUE));
          Type::EPV:
            EXIT(ProjDevelopmentCU.CalculateTotalEPV("Project No.", Version, "Reference Date", PeriodType::Year, TRUE));
          Type::PRI:
            EXIT(ProjDevelopmentCU.CalculateTotalPRI("Project No.", Version, "Reference Date", PeriodType::Year, TRUE));
        END;
        EXIT(0);
      END;
    END;
