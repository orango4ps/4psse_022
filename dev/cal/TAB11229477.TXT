    LOCAL PROCEDURE CheckMaxAllowedWithBudget@1100528206(LocBudget@1100528201 : Decimal;VAR LocAllowed@1100528200 : Decimal);
    BEGIN
      IF JobsSetup."Progress Range" < JobsSetup."Progress Range"::"No Progress or Allowed Max." THEN BEGIN
        //Allowed may not exceed Budget
        IF LocBudget >= 0 THEN BEGIN
          IF LocAllowed > LocBudget THEN
            LocAllowed := LocBudget;
        END ELSE BEGIN
          IF LocAllowed < LocBudget THEN
            LocAllowed := LocBudget;
        END;
      END;
      IF (LocAllowed < 0) AND (JobsSetup."Fill Allowed Cost" = JobsSetup."Fill Allowed Cost"::"Negative not allowed") THEN
        LocAllowed := 0;
    END;

    LOCAL PROCEDURE DetermineAllowed@1100528200(LocBudget@1100529601 : Decimal;LocAmount@1100529600 : Decimal) LocAllowed : Decimal;
    BEGIN
      IF LocBudget >= 0 THEN BEGIN
        IF LocBudget < LocAmount THEN
          LocAllowed := LocBudget
        ELSE
          IF (LocAmount >= 0) OR (JobsSetup."Fill Allowed Cost" = JobsSetup."Fill Allowed Cost"::"No restriction") THEN
            LocAllowed := LocAmount;
      END ELSE BEGIN
        IF (LocBudget > LocAmount) AND (JobsSetup."Fill Allowed Cost" = JobsSetup."Fill Allowed Cost"::"No restriction") THEN
          LocAllowed := LocBudget
        ELSE
          IF JobsSetup."Fill Allowed Cost" = JobsSetup."Fill Allowed Cost"::"No restriction" THEN
            LocAllowed := LocAmount;
      END;
    END;
