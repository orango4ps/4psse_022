    VAR
      ArrNo@1100525000 : Integer;
    BEGIN
      ArrNo := MaxArrayNo;

        IF (ShipAddressFromArray[ArrNo] <> '') OR (ShipAddressToArray[ArrNo] <> '') THEN
          EXIT(ArrNo);
        ArrNo := ArrNo - 1;
      UNTIL ArrNo < 4;

      EXIT(3);
    END;

Public Function InNewGroup(group1 As Object, group2 As Object) As Boolean
    newgroup = False
    If Not (group1 = currentgroup1)
        newgroup = True
        currentgroup2 = group2
        currentgroup1 = group1
    Else
        If Not (group2 = currentgroup2)
            newgroup = True
            currentgroup2 = group2
        End If
    End If
    Return newgroup
End Function
