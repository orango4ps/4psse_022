    LOCAL PROCEDURE GetLines@1100409001();
    VAR
      SalesCrMemoHeader@1100409000 : Record 114;
      SalesCrMemoLine@1100409001 : Record 115;
      StringFunctions@1100409002 : Codeunit 11012273;
    BEGIN
      IF (StringFunctions.IsFilterString(PlantLocation)) THEN
        SalesCrMemoLine.SETFILTER("Plant Location", PlantLocation)
      ELSE
        SalesCrMemoLine.SETRANGE("Plant Location", COPYSTR(PlantLocation, 1, MAXSTRLEN(SalesCrMemoLine."Plant Location")));

      IF (SalesCrMemoLine.FINDSET) THEN BEGIN
        REPEAT
          IF (SalesCrMemoHeader.GET(SalesCrMemoLine."Document No.")) THEN BEGIN
            COPY(SalesCrMemoHeader);
            IF (INSERT) THEN;
          END;
        UNTIL (SalesCrMemoLine.NEXT = 0);
      END;
      IF FINDSET THEN;
    END;

    LOCAL PROCEDURE GetRemainingAmount@1100409002() : Decimal;
    VAR
      CustLedgerEntry@1100409000 : Record 21;
    BEGIN
      CustLedgerEntry.SETCURRENTKEY("Document No.");

      CustLedgerEntry.SETRANGE("Document No.", "No.");
      CustLedgerEntry.SETRANGE("Document Type", CustLedgerEntry."Document Type"::"Credit Memo");
      CustLedgerEntry.SETRANGE("Customer No.", "Sell-to Customer No.");
      IF (CustLedgerEntry.FINDSET) THEN BEGIN
        CustLedgerEntry.CALCFIELDS("Remaining Amount");
        EXIT(CustLedgerEntry."Remaining Amount");
      END;
      EXIT(0);
    END;

    BEGIN
    END.
