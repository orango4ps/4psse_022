    LOCAL PROCEDURE FillBuffer@1100285501();
    BEGIN
      LineNo:= 1;
      Buffer.RESET;
      Buffer.DELETEALL;
      Buffer.SETCURRENTKEY("Project No.", Principal, "Installment No.");
      IF ProjectNo <>'' THEN
        ProjectInstallment.SETRANGE("Project No.",ProjectNo);
      IF PrincipalNo <> '' THEN
        ProjectInstallment.SETRANGE(Principal,PrincipalNo);
      ProjectInstallment.SETFILTER("Withheld Amount",'<>%1',0);
      ProjectInstallment.SETFILTER("Invoiced Price",'<>%1',0);
      IF ProjectInstallment.FIND('-') THEN REPEAT
        IF NOT IsFinal THEN BEGIN
          SetJobLedgerEntryFilters;
          IF JobLedgerEntry.FIND('-') THEN BEGIN
            Buffer.SETRANGE("Project No.",ProjectInstallment."Project No.");
            Buffer.SETRANGE(Principal, ProjectInstallment.Principal);
            Buffer.SETRANGE("Installment No.",JobLedgerEntry."Installment No.");
            IF NOT Buffer.FINDFIRST THEN BEGIN
              Buffer.INIT;
              Buffer."Project No." := ProjectInstallment."Project No.";
              Buffer.Principal := ProjectInstallment.Principal;
              Buffer."Installment No." := JobLedgerEntry."Installment No.";
              Buffer.Description := JobLedgerEntry."Document No.";
              Buffer."Plot No." := IntToCode10(LineNo);
              IF  ProjectInstallment."Currency Code" = '' THEN
                Buffer."Currency Code" := GLSetup."LCY Code"
              ELSE
                Buffer."Currency Code" := ProjectInstallment."Currency Code";
              Buffer."Withheld Amount" := ProjectInstallment."Withheld Amount";
              Buffer.INSERT;
              LineNo +=1;
            END;
          END;
        END;
      UNTIL ProjectInstallment.NEXT=0;
    END;

    LOCAL PROCEDURE SetJobLedgerEntryFilters@1100285507();
    BEGIN
      JobLedgerEntry.RESET;
      JobLedgerEntry.SETCURRENTKEY("Project Invoice", "Installment Invoice", "Job No.",
                                    Principal, "Installment No.", "Plot No.");
      JobLedgerEntry.SETRANGE("Project Invoice",TRUE);
      JobLedgerEntry.SETRANGE("Installment Invoice",TRUE);
      JobLedgerEntry.SETRANGE("Job No.",ProjectInstallment."Project No.");
      JobLedgerEntry.SETRANGE(Principal, ProjectInstallment.Principal);
      JobLedgerEntry.SETRANGE("Installment No.",ProjectInstallment."Installment No.");
    END;

    LOCAL PROCEDURE IsFinal@1100285503() : Boolean;
    BEGIN

      ProjectInstallment2.RESET;
      ProjectInstallment2.SETRANGE("Project No.",ProjectInstallment."Project No.");
      ProjectInstallment2.SETRANGE(Principal, ProjectInstallment.Principal);
      ProjectInstallment2.SETRANGE("Cost Object", JobSetup."Final Installment Cost Object");
      IF ProjectInstallment2.FINDFIRST THEN BEGIN
        ProjectInstallment2.CALCFIELDS("Invoiced Price");
        IF ProjectInstallment2."Invoiced Price" <> 0 THEN
          EXIT(TRUE)
        ELSE
          EXIT(FALSE);
      END;
    END;

    END;
