    PROCEDURE GetBuyBackRate@1000000000(ItemNo@1000000001 : Code[20]) BuyBackRate : Decimal;
    VAR
      ItemSalesBuyBackRate@1000000000 : Record 11020318;
    BEGIN
      ItemSalesBuyBackRate.SETRANGE("Item No.", ItemNo);
      ItemSalesBuyBackRate.SETFILTER("Starting Date", '..%1', TODAY);
      IF ItemSalesBuyBackRate.FINDLAST THEN
        BuyBackRate := ItemSalesBuyBackRate."Buy Back Rate"
      ELSE
        BuyBackRate := 0;
    END;

    PROCEDURE CreateTmpItem@1000000001(ItemNo@1000000000 : Code[20]);
    VAR
      Item@1000000001 : Record 27;
      InventoryPostingGroup@1000000002 : Record 94;
    BEGIN
      IF NOT TmpItem.GET(ItemNo) THEN BEGIN
        Item.GET(ItemNo);
        TmpItem.INIT;
        TmpItem."No." := Item."No.";
        TmpItem."Inventory Posting Group" := Item."Inventory Posting Group";
        TmpItem.INSERT;

        IF NOT TmpInventoryPostingGroup.GET(Item."Inventory Posting Group") THEN BEGIN
          IF NOT InventoryPostingGroup.GET(Item."Inventory Posting Group") THEN InventoryPostingGroup.INIT;
          TmpInventoryPostingGroup.INIT;
          TmpInventoryPostingGroup.Code := Item."Inventory Posting Group";
          TmpInventoryPostingGroup.Description := InventoryPostingGroup.Description;
          TmpInventoryPostingGroup.INSERT;
        END;
      END;
    END;

    BEGIN
    END.
