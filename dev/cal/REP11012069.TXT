    PROCEDURE FillPeriodInfo@1210190017();
    BEGIN
      PeriodStartDateArray[1] := 0D;
      CASE ReportType OF
        ReportType::"Age Composition":
          BEGIN
            SwitchSignDateFormula(PeriodLength);
            PeriodStartDateArray[NoOfPeriods] := ReferenceDate;
            FOR PeriodNo := (NoOfPeriods - 1) DOWNTO 2 DO
              PeriodStartDateArray[PeriodNo] := CALCDATE(PeriodLength, PeriodStartDateArray[PeriodNo+1]);
          END;
        ReportType::"Cash Flow":
          BEGIN
            PeriodStartDateArray[2] := ReferenceDate;
            FOR PeriodNo := 3 TO NoOfPeriods DO
              PeriodStartDateArray[PeriodNo] := CALCDATE(PeriodLength, PeriodStartDateArray[PeriodNo-1]);
          END;
      END;

      FOR PeriodNo := 1 TO (NoOfPeriods - 1) DO
        PeriodEndDateArray[PeriodNo] := PeriodStartDateArray[PeriodNo+1] - 1;

    PROCEDURE SwitchSignDateFormula@1210190004(VAR IOPeriodLength@1210190000 : DateFormula);
    VAR
      Len@1210190004 : Integer;
      I@1210190003 : Integer;
      Char@1210190002 : Text[1];
      PeriodLengthTxt@1210190001 : Text[100];
    BEGIN
      //* In een DateFormula kunnen geen haakjes gebruikt worden (deze woren automatisch weer weggehaald) daarom werkt
      //* de volgende constructie niet: DateFormula := -(DateFormula)
      //* Dus alle tekens omkeren (als eerste positie geen '+' of '-' dan '-' toevoegen op eerste positie)

      PeriodLengthTxt := FORMAT(IOPeriodLength);
      Len := STRLEN(PeriodLengthTxt);
      FOR I := 1 TO (Len - 1) DO BEGIN
        Char := COPYSTR(PeriodLengthTxt, I, 1);
        IF Char = '+' THEN
          Char := '-'
        ELSE BEGIN
          IF Char = '-' THEN
            Char := '+';
        END;
        IF (Char = '+') OR (Char = '-') THEN BEGIN
          IF I = 1 THEN
            PeriodLengthTxt := Char + COPYSTR(PeriodLengthTxt, (I + 1))
          ELSE
            PeriodLengthTxt := COPYSTR(PeriodLengthTxt, 1, (I - 1)) + Char + COPYSTR(PeriodLengthTxt, (I + 1));
        END;
      END;
      Char := COPYSTR(PeriodLengthTxt, 1, 1);
      IF (Char <> '+') AND (Char <> '-') THEN
        PeriodLengthTxt := '-' + PeriodLengthTxt;
      EVALUATE(IOPeriodLength, PeriodLengthTxt);
    END;
