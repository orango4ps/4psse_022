    LOCAL PROCEDURE FindServiceContractArrangement@1100529609(VAR LocServiceContractCtrlBudget@1100529601 : Record 11071759;VAR LocServiceContractArrangement@1100529600 : Record 11071705) : Boolean;
    BEGIN
      LocServiceContractArrangement.RESET;
      LocServiceContractArrangement.SETRANGE("Service Contract No.", LocServiceContractCtrlBudget."Service Contract No.");
      LocServiceContractArrangement.SETRANGE("Object No.", LocServiceContractCtrlBudget."Service Object No.");
      LocServiceContractArrangement.SETRANGE("Service Order No.", LocServiceContractCtrlBudget."Service Order No.");
      LocServiceContractArrangement.SETRANGE("Maintenance Proposal No.", LocServiceContractCtrlBudget."Maintenance Proposal No.");
      LocServiceContractArrangement.SETRANGE("Service Category", LocServiceContractCtrlBudget."Service Category");
      LocServiceContractArrangement.SETRANGE("Theoretical Maintenance Date", LocServiceContractCtrlBudget."Reference Date");
      IF LocServiceContractArrangement.ISEMPTY THEN
        EXIT(FALSE);
      LocServiceContractArrangement.FINDFIRST;
      IF LocServiceContractArrangement.COUNT > 1 THEN BEGIN
        IF LocServiceContractCtrlBudget.ISTEMPORARY THEN
          EXIT(FALSE);
        IF NOT TmpDoubleServiceContractArrangement.GET(LocServiceContractArrangement."Service Contract No.", LocServiceContractArrangement."Object No.",
          LocServiceContractArrangement."Maintenance Scheme", LocServiceContractArrangement."Line No.")
        THEN BEGIN
          TmpDoubleServiceContractArrangement := LocServiceContractArrangement;
          TmpDoubleServiceContractArrangement.INSERT;
        END;
        EXIT(FALSE);
      END;
      IF NOT LocServiceContractCtrlBudget.ISTEMPORARY THEN
        IF TmpServiceContractArrangement.GET(LocServiceContractArrangement."Service Contract No.", LocServiceContractArrangement."Object No.",
          LocServiceContractArrangement."Maintenance Scheme", LocServiceContractArrangement."Line No.")
        THEN
          EXIT(FALSE);
      EXIT(TRUE);
    END;

    LOCAL PROCEDURE GetDateCalcFormula@1100529603();
    BEGIN
      DateCalcFormula := '<+' + FORMAT(PeriodQuantity);

      CASE PeriodType OF
        PeriodType::Day:
          DateCalcFormula := DateCalcFormula + 'D>';
        PeriodType::Week:
          DateCalcFormula := DateCalcFormula + 'W>';
        PeriodType::Quarter:
          DateCalcFormula := DateCalcFormula + 'Q>';
        PeriodType::Month:
          DateCalcFormula := DateCalcFormula + 'M>';
        PeriodType::Year:
          DateCalcFormula := DateCalcFormula + 'Y>';
      END;
    END;

    LOCAL PROCEDURE SetNewDatesServiceContractArrangement@1100529617(LocServiceContractArrangement@1100529600 : Record 11071705);
    VAR
      LocContractObjectMaintenance@1100529604 : Record 11071692;
      NewTheoreticalMaintenanceDate@1100529602 : Date;
      LocYearNo@1100529601 : Integer;
      LocWeekNo@1100529603 : Integer;
      LocDayNo@1100529605 : Integer;
    BEGIN
      NewTheoreticalMaintenanceDate := CALCDATE(DateCalcFormula, LocServiceContractArrangement."Theoretical Maintenance Date");
      LocDayNo := DATE2DWY(NewTheoreticalMaintenanceDate, 1);
      LocWeekNo := DATE2DWY(NewTheoreticalMaintenanceDate, 2);
      LocYearNo := DATE2DWY(NewTheoreticalMaintenanceDate, 3);

      TmpServiceContractArrangement := LocServiceContractArrangement;
      TmpServiceContractArrangement."Maintenance Year" := LocYearNo;
      TmpServiceContractArrangement."Maintenance Week" := LocWeekNo;
      TmpServiceContractArrangement."Maintenance Day" := LocDayNo;
      TmpServiceContractArrangement.RecalcTheoreticalMaintDate;
      TmpServiceContractArrangement.INSERT;

      IF Finalize THEN
        IF NOT TmpContractObjectMaintenance.GET(LocServiceContractArrangement."Service Contract No.",
          LocServiceContractArrangement."Object No.", LocServiceContractArrangement."Maintenance Scheme")
        THEN BEGIN
          LocContractObjectMaintenance.GET(LocServiceContractArrangement."Service Contract No.",
            LocServiceContractArrangement."Object No.", LocServiceContractArrangement."Maintenance Scheme");
          TmpContractObjectMaintenance := LocContractObjectMaintenance;
          TmpContractObjectMaintenance.INSERT;
        END;
    END;

    LOCAL PROCEDURE CheckUniqueDWY@1100526202(CheckServiceContractArrangement@1100529600 : Record 11071705);
    VAR
      LocServiceContractArrangement@1100526203 : Record 11071705;
      LocTmpServiceContractArrangement@1100529601 : TEMPORARY Record 11071705;
    BEGIN
      LocServiceContractArrangement.SETRANGE("Service Contract No.", CheckServiceContractArrangement."Service Contract No.");
      LocServiceContractArrangement.SETRANGE("Object No.", CheckServiceContractArrangement."Object No.");
      LocServiceContractArrangement.SETRANGE("Maintenance Scheme", CheckServiceContractArrangement."Maintenance Scheme");
      LocServiceContractArrangement.SETFILTER("Line No.", '<>%1', CheckServiceContractArrangement."Line No.");
      IF LocServiceContractArrangement.FINDSET THEN
        REPEAT
          IF TmpServiceContractArrangement.GET(
            LocServiceContractArrangement."Service Contract No.",
            LocServiceContractArrangement."Object No.",
            LocServiceContractArrangement."Maintenance Scheme",
            LocServiceContractArrangement."Line No.")
          THEN
            LocTmpServiceContractArrangement := TmpServiceContractArrangement
          ELSE
            LocTmpServiceContractArrangement := LocServiceContractArrangement;
          LocTmpServiceContractArrangement.INSERT;
        UNTIL LocServiceContractArrangement.NEXT = 0;

      TmpServiceContractArrangement.GET(
        CheckServiceContractArrangement."Service Contract No.",
        CheckServiceContractArrangement."Object No.",
        CheckServiceContractArrangement."Maintenance Scheme",
        CheckServiceContractArrangement."Line No.");

      LocTmpServiceContractArrangement.SETRANGE("Maintenance Year", CheckServiceContractArrangement."Maintenance Year");
      LocTmpServiceContractArrangement.SETRANGE("Maintenance Week", CheckServiceContractArrangement."Maintenance Week");
      LocTmpServiceContractArrangement.SETRANGE("Maintenance Day", CheckServiceContractArrangement."Maintenance Day");
      IF NOT LocTmpServiceContractArrangement.ISEMPTY THEN
        ProcessError(Text006);
    END;

    LOCAL PROCEDURE FinalizeRecords@1100529619();
    VAR
      ServiceContractArrangement@1100529600 : Record 11071705;
      ContractObjectMaintenance@1100529601 : Record 11071692;
      ServiceContractCtrlBudget@1100529602 : Record 11071759;
    BEGIN
      IF NOT Finalize THEN
        EXIT;

      IF TmpContractObjectMaintenance.ISEMPTY THEN
        EXIT;

      TmpServiceContractArrangement.RESET;
      TmpServiceContractArrangement.SETRANGE("Firm Planned", FALSE);
      TmpServiceContractArrangement.DELETEALL;
      TmpServiceContractArrangement.RESET;

      TmpContractObjectMaintenance.FINDSET;
      REPEAT
        TmpServiceContractArrangement.SETRANGE("Service Contract No.", TmpContractObjectMaintenance."Service Contract No.");
        TmpServiceContractArrangement.SETRANGE("Object No.", TmpContractObjectMaintenance."Object No.");
        TmpServiceContractArrangement.SETRANGE("Maintenance Scheme", TmpContractObjectMaintenance."Maintenance Scheme");
        IF TmpServiceContractArrangement.FINDSET THEN BEGIN
          REPEAT
            ServiceContractArrangement.GET(TmpServiceContractArrangement."Service Contract No.", TmpServiceContractArrangement."Object No.",
              TmpServiceContractArrangement."Maintenance Scheme", TmpServiceContractArrangement."Line No.");
            ServiceContractArrangement."Maintenance Year" := TmpServiceContractArrangement."Maintenance Year";
            ServiceContractArrangement."Maintenance Week" := TmpServiceContractArrangement."Maintenance Week";
            ServiceContractArrangement."Maintenance Day" := TmpServiceContractArrangement."Maintenance Day";
            ServiceContractArrangement.RecalcTheoreticalMaintDate;
            ServiceContractArrangement.MODIFY;
          UNTIL TmpServiceContractArrangement.NEXT = 0;

          ContractObjectMaintenance.GET(TmpContractObjectMaintenance."Service Contract No.", TmpContractObjectMaintenance."Object No.", TmpContractObjectMaintenance."Maintenance Scheme");
          ContractObjectMaintenance.UpdateLastAndNextMaintDate;
          ContractObjectMaintenance.MODIFY;
        END;
      UNTIL TmpContractObjectMaintenance.NEXT = 0;

      NewTmpServiceContractCtrlBudget.RESET;
      NewTmpServiceContractCtrlBudget.SETRANGE(Finished, FALSE);
      NewTmpServiceContractCtrlBudget.DELETEALL;
      NewTmpServiceContractCtrlBudget.RESET;
      IF NewTmpServiceContractCtrlBudget.ISEMPTY THEN
        EXIT;
      NewTmpServiceContractCtrlBudget.FINDSET;
      REPEAT
        ServiceContractCtrlBudget.GET(NewTmpServiceContractCtrlBudget."Reporting Date", NewTmpServiceContractCtrlBudget."Service Contract No.", NewTmpServiceContractCtrlBudget."Line No.");
        ServiceContractCtrlBudget."Reference Date" := NewTmpServiceContractCtrlBudget."Reference Date";
        ServiceContractCtrlBudget.MODIFY;
      UNTIL NewTmpServiceContractCtrlBudget.NEXT = 0;

      PassDatabetweenPages.SetServiceContractCtrlBudget(NewTmpServiceContractCtrlBudget);
    END;

    BEGIN
    END.
