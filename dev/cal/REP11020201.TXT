    PROCEDURE DetermineStockFactor@1100525010();
    VAR
      ProjectPreclosureResult@1100525000 : Record 11012093;
    BEGIN
      //Get Stock factor
      //It is difficult to get stockfactor from G/L, get it from Project Preclosure Results.
      //when a project is closed we assume there is no stock anymore.
      StockFactor := 0;
      IF Job."Preclosure Result" <> Job."Preclosure Result"::No THEN BEGIN
        ProjectPreclosureResult.SETCURRENTKEY("Project No.","Posting Date");
        ProjectPreclosureResult.SETRANGE("Project No.", Job."No.");
        ProjectPreclosureResult.SETFILTER("Posting Date", '>%1', EndDate);
        ProjectPreclosureResult.SETRANGE(Posted, TRUE);
        IF Job."Project Status" >= Job."Project Status"::Finished THEN
          IF NOT ProjectPreclosureResult.FINDFIRST THEN
            EXIT; //Job already is closed (we don't know when but somewhere between EndDate and TODAY) don't display a stock

        ProjectPreclosureResult.SETFILTER("Posting Date", '..%1', EndDate);
        IF ProjectPreclosureResult.FINDLAST THEN BEGIN
          IF Job."Preclosure Result" = Job."Preclosure Result"::"Sales Progress" THEN BEGIN
            IF ProjectPreclosureResult."Units Saleable" <> 0 THEN
              StockFactor := 1 - (ProjectPreclosureResult."Units Sold" / ProjectPreclosureResult."Units Saleable");
          END;
        END;
      END;
    END;

    PROCEDURE CalcProjectCostsAndRevenues@1100525001(JobNo@1100525001 : Code[20]);
    VAR
      Job@1100525002 : Record 11072003;
      ProjectType@1100525003 : Record 11012009;
      TempGLAccount@1100528400 : TEMPORARY Record 15;
      I@1100525004 : Integer;
    BEGIN
      // Assumptions:
      //
      // The fields "WIP Account Labor", "WIP Account Material", "WIP Account Subcontracting"
      // "WIP Account Plant" and "WIP Account Sundry" of the project type are filled with different account no.'s.
      //
      // The fields "WIP Account Revenue", "WIP Account Revenue Labor", "WIP Account Revenue Material",
      // "WIP Account Revenue Subcontr.", "WIP Account Revenue Plant" and "WIP Account Revenue Sundry"
      // may be filled with the same account no., but not with one of the "cost" WIP Account no's.
      //
      // TempGLAccount is used to make sure that the amount of a G/L entry will not be multiplied
      // by the number of times the account no. occurs in the WIP Account Revenue fields of the project type.

      Job.GET(JobNo);
      ProjectType.GET(Job."Project Type");

      FOR I := 1 TO 6 DO BEGIN
        CASE I OF
          1: IF ProjectType."WIP Account Labor" <> '' THEN
               CalcProjectGL(Job."No.", ProjectType."WIP Account Labor", I, TempGLAccount);
          2: IF ProjectType."WIP Account Material" <> '' THEN
               CalcProjectGL(Job."No.", ProjectType."WIP Account Material", I, TempGLAccount);
          3: IF ProjectType."WIP Account Subcontracting" <> '' THEN
               CalcProjectGL(Job."No.", ProjectType."WIP Account Subcontracting", I, TempGLAccount);
          4: IF ProjectType."WIP Account Plant" <> '' THEN
               CalcProjectGL(Job."No.", ProjectType."WIP Account Plant", I, TempGLAccount);
          5: IF ProjectType."WIP Account Sundry" <> '' THEN
               CalcProjectGL(Job."No.", ProjectType."WIP Account Sundry", I, TempGLAccount);
          6: BEGIN
               IF ProjectType."WIP Account Revenue" <> '' THEN
                 CalcProjectGL(Job."No.", ProjectType."WIP Account Revenue", I, TempGLAccount);
               IF ProjectType."WIP Account Revenue Labor" <> '' THEN
                 CalcProjectGL(Job."No.", ProjectType."WIP Account Revenue Labor", I, TempGLAccount);
               IF ProjectType."WIP Account Revenue Material" <> '' THEN
                 CalcProjectGL(Job."No.", ProjectType."WIP Account Revenue Material", I, TempGLAccount);
               IF ProjectType."WIP Account Revenue Subcontr." <> '' THEN
                 CalcProjectGL(Job."No.", ProjectType."WIP Account Revenue Subcontr.", I, TempGLAccount);
               IF ProjectType."WIP Account Revenue Plant" <> '' THEN
                 CalcProjectGL(Job."No.", ProjectType."WIP Account Revenue Plant", I, TempGLAccount);
               IF ProjectType."WIP Account Revenue Sundry" <> '' THEN
                 CalcProjectGL(Job."No.", ProjectType."WIP Account Revenue Sundry", I, TempGLAccount);
             END;
        END;
      END;
    END;

    PROCEDURE CalcProjectGL@1210190031(JobNo@1100525001 : Code[20];AccountNo@1210190000 : Code[20];CostType@1210190001 : ',Labor,Material,Subcontracting,Plant,Sundry,Revenues';VAR TempGLAccount@1100528400 : TEMPORARY Record 15);
    VAR
      GLEntry@1100525000 : Record 17;
    BEGIN
      IF TempGLAccount.GET(AccountNo) THEN
        EXIT;

      TempGLAccount."No." := AccountNo;
      TempGLAccount.INSERT;

      GLEntry.SETCURRENTKEY("G/L Account No.","Posting Date");
      GLEntry.SETRANGE("G/L Account No.", AccountNo);
      GLEntry.SETFILTER("Posting Date", '%1..%2', StartDate, EndDate);
      GLEntry.SETRANGE("Job No.", JobNo);
      GLEntry.DPA_SetFilters(CurrReport.OBJECTID(FALSE));
      IF GLEntry.FINDSET THEN
        REPEAT
          IF CostType <> CostType::Revenues THEN
            ProjectGL[CostType] := ProjectGL[CostType] + GLEntry.Amount
          ELSE
            ProjectGL[7] := ProjectGL[7] + (GLEntry.Amount * -1);
        UNTIL GLEntry.NEXT = 0;
    END;

    PROCEDURE CalcPreclosurePerJob@1100525004(JobNo@1100525002 : Code[20];CreditAccountNo@1210190000 : Code[20];DebitAccountNo@1100525000 : Code[20]) : Decimal;
    VAR
      GLEntry@1100525001 : Record 17;
      PreclosureResult@1210190002 : Decimal;
    BEGIN
      PreclosureResult := 0;

      GLEntry.SETCURRENTKEY("G/L Account No.","Posting Date");
      GLEntry.SETFILTER("G/L Account No.", '%1|%2', CreditAccountNo, DebitAccountNo);
      GLEntry.SETFILTER("Posting Date", '%1..%2', StartDate, EndDate);
      GLEntry.SETRANGE("Job No.", JobNo);
      IF GLEntry.FINDSET THEN
        REPEAT
          PreclosureResult := PreclosureResult + GLEntry.Amount;
        UNTIL GLEntry.NEXT = 0;

      EXIT(PreclosureResult);
    END;

    BEGIN
    END.
