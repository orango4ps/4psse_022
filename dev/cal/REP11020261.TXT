    PROCEDURE IsValidSettlementSheet@1100528609(SettlementSheet@1100528601 : Record 11012021;VAR ErrorMessage@1100528600 : Text[250]) : Boolean;
    VAR
      Customer@1100528602 : Record 18;
      Commission@1100528200 : Record 11012020;
      CollectInvoicesBy@1100409000 : Option;
    BEGIN
      IF SettlementSheet.Principal = '' THEN
        ErrorMessage := STRSUBSTNO(Text001, SettlementSheet.FIELDCAPTION(Principal));

      IF NOT Customer.GET(SettlementSheet.Principal) THEN
        ErrorMessage := STRSUBSTNO(Text002, SettlementSheet.FIELDCAPTION(Principal), SettlementSheet.Principal);

      Commission.GET(SettlementSheet."Project No.",SettlementSheet.Principal,SettlementSheet."Commission No.");
      CollectInvoicesBy := Commission."Collect Cost Plus Lines By";

      CASE CollectInvoicesBy OF
        Commission."Collect Cost Plus Lines By"::Supervisor:
          IF SettlementSheet.Supervisor = '' THEN
            ErrorMessage := STRSUBSTNO(Text001, SettlementSheet.FIELDCAPTION(Supervisor));
        Commission."Collect Cost Plus Lines By"::"Main Project":
          IF SettlementSheet."Main Project" = '' THEN
            ErrorMessage := STRSUBSTNO(Text001, SettlementSheet.FIELDCAPTION("Main Project"));
      END;

      EXIT(ErrorMessage = '');
    END;

    PROCEDURE IsValidBillToCustomer@1100528603(BillToCustomerNo@1100528601 : Code[20];VAR ErrorMessage@1100528600 : Text[250]) : Boolean;
    VAR
      Customer@1100528602 : Record 18;
      SettlementSheet@1100528200 : Record 11012021;
    BEGIN
      IF NOT Customer.GET(BillToCustomerNo) THEN
        ErrorMessage := STRSUBSTNO(Text002, SettlementSheet.FIELDCAPTION("Bill-to Customer No."), BillToCustomerNo);

      EXIT(ErrorMessage = '');
    END;

    PROCEDURE DPA_DepartmentInFilter@1100525001(TableID@1100525000 : Integer;No@1100525003 : Code[20]) : Boolean;

      Proj@1100525002 : Record 11072003;

         EXIT(DPA_Management.CodeInFilter(Proj."Global Dimension 1 Code", DPA_UserDepartmentFilter));

      EXIT(FALSE);
    END;

    BEGIN
    END.
