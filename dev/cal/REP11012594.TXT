    PROCEDURE ValidateReferenceDate@1100485003();
    BEGIN
      IF (ReferenceDate = 0D) THEN
        ERROR(Text000);

      IF (ReferenceDate > TODAY) THEN
        ERROR(Text001);

      CASE InternalExternal OF
        InternalExternal::Internal:
          QuantityLabelTxt := Text003;
        InternalExternal::External:
          QuantityLabelTxt := Text004;
      END;
    END;

    PROCEDURE DetermineQuantityOnDate@1100485000(IPlantType@1100485000 : Code[20]);
    VAR
      lvExternal@1100485001 : Boolean;
    BEGIN
      WITH PlantInventory DO BEGIN
        PlantTypeQuantityInternal := 0;
        PlantTypeQuantityExternal := 0;

        RESET;
        //SETCURRENTKEY(Removed, Location, "Plant Type", "Plant No", ...);
        //* This is the first key so no need to switch, and also set filter on location here (change regarding to R11012591)
        SETRANGE(Location, "Plant Location".Code);
        SETRANGE("Plant Type", IPlantType);
        SETFILTER("Present from", '..%1', ReferenceDate);

        SETRANGE(Removed, FALSE);
        IF FINDSET THEN BEGIN
          REPEAT
            IF NOT (Sold AND ("Plant Location"."Ending Date" < ReferenceDate) AND ("Plant Location"."Ending Date" <> 0D)) THEN BEGIN
              IF CheckInternalExternal("Plant Type", "Plant No.", lvExternal) THEN BEGIN
                IF NOT lvExternal THEN
                  PlantTypeQuantityInternal := PlantTypeQuantityInternal + Quantity
                ELSE
                  PlantTypeQuantityExternal := PlantTypeQuantityExternal + Quantity;
              END;
            END;
          UNTIL NEXT = 0;
        END;

        SETRANGE(Removed, TRUE);
        SETFILTER("Removed on", '>=%1', ReferenceDate);  //* now needed '>=%1' (after this extra test on 'Depot')

        IF FINDSET THEN BEGIN
          REPEAT
            IF NOT (("Removed on" = ReferenceDate) AND Depot) THEN BEGIN  //* Don't count transport day of depot
              IF CheckInternalExternal("Plant Type", "Plant No.", lvExternal) THEN BEGIN
                IF NOT lvExternal THEN
                  PlantTypeQuantityInternal := PlantTypeQuantityInternal + Quantity
                ELSE
                  PlantTypeQuantityExternal := PlantTypeQuantityExternal + Quantity;
              END;
            END;
          UNTIL NEXT = 0;
        END;
      END;
    END;

    PROCEDURE CheckInternalExternal@1100485002(IPlantType@1100485000 : Code[20];IPlantNo@1100485001 : Code[10];VAR OExternal@1100485002 : Boolean) : Boolean;
    VAR
      lvPlantNoRec@1100485003 : Record 11012552;
      lvVendorNo@1100485004 : Code[20];
    BEGIN
      OExternal := "Plant Type".External;

      lvVendorNo := '';
      IF (IPlantNo <> '0') AND (IPlantNo <> '') THEN BEGIN
        IF (IPlantType <> PlantNo."Plant Type") OR (IPlantNo <> PlantNo."No.") THEN BEGIN
          IF NOT PlantNo.GET(IPlantType, IPlantNo) THEN BEGIN
            PlantNo."Plant Type" := IPlantType;
            PlantNo."No." := IPlantNo;
            PlantNo.External := "Plant Type".External;
            PlantNo.Vendor := '';
          END ELSE BEGIN
            IF PlantNo.External AND (ExtVendorFilter <> '') AND (InternalExternal <> InternalExternal::Internal) THEN BEGIN
              IF NOT Vendor.GET(PlantNo.Vendor) THEN BEGIN
                Vendor.INIT;
                Vendor."No." := PlantNo.Vendor;
                lvPlantNoRec.SETRANGE("Plant Type", IPlantType);
                lvPlantNoRec.SETRANGE("No.", IPlantNo);
                lvPlantNoRec.SETFILTER(Vendor, ExtVendorFilter);
                IF lvPlantNoRec.FINDFIRST THEN
                  Vendor."Subcontractor Type" := Vendor."Subcontractor Type"::Subcontractor;  //* Use field to determine if vendor fits in filter.
                Vendor.INSERT;
              END;
            END;
          END;
        END;
        OExternal := PlantNo.External;
        lvVendorNo := PlantNo.Vendor;
      END;

      CASE InternalExternal OF
        InternalExternal::Internal:
          IF OExternal THEN
            EXIT(FALSE);
        InternalExternal::External:
          IF NOT OExternal THEN
            EXIT(FALSE);
      END;

      IF OExternal AND (ExtVendorFilter <> '') THEN BEGIN
        IF lvVendorNo = '' THEN  //* Base assumption is that vendor is filled at external plant.
          EXIT(FALSE);           //* otherwise difficult: then make difference between vendor empty and plant no. /0/empty/unknown.
        IF NOT Vendor.GET(lvVendorNo) THEN
          EXIT(FALSE);
        IF Vendor."Subcontractor Type" = Vendor."Subcontractor Type"::" " THEN
          EXIT(FALSE);
      END;

      EXIT(TRUE);
    END;

    BEGIN
    {
      Copied from report 11012591 (per plant type/ date).
      Basic assumption is that this will be printed for a specific location (or only some locations).
      So for each location, the whole table with plant type is read (within setup filters).
    }
    END.
