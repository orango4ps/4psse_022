      IF DtldCustLedgEntry.FINDSET THEN
        REPEAT
          IF (DtldCustLedgEntry."Entry Type" <> DtldCustLedgEntry."Entry Type"::"Initial Entry") AND

        UNTIL DtldCustLedgEntry.NEXT = 0;
    END;
