    PROCEDURE BudgetURL@1100525004() : Text;
    VAR
      BudgetLine@1100525000 : Record 11012001;
      RecRef@1100525003 : RecordRef;
      HyperlinkManagement@1100525002 : Codeunit 11012351;
    BEGIN
      BudgetLine.SETRANGE("Main Project No.", Job."Main Project");
      Job.COPYFILTER("Project Filter", BudgetLine."Project No.");
      Job.COPYFILTER("Cumulation Filter", BudgetLine."Version Date");
      BudgetLine.SETFILTER("Extension Contract Status", '%1|%2|%3',
        BudgetLine."Extension Contract Status"::"Not Applicable",
        BudgetLine."Extension Contract Status"::Order,
        BudgetLine."Extension Contract Status"::"Invoicing Allowed");

      RecRef.GETTABLE(BudgetLine);

      EXIT(HyperlinkManagement.CreatePageURLWithFilter(RecRef));
    END;
