    LOCAL PROCEDURE GetReportName@1100525003() : Text;
    BEGIN
      EXIT(DELSTR(CurrReport.OBJECTID, 1, STRPOS(CurrReport.OBJECTID, ' ')));
    END;

    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Investment Proposal".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', "Investment Proposal".TABLECAPTION, "Investment Proposal".GETFILTERS));
    END;

    PROCEDURE CalculateRubricBudget@1210190011() : Decimal;
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        CASE Type OF
          Type::Subtotal:
            EXIT(ProjectDevelopmentFunctions.CalculateBudgetByRubricCostRev(InvestmProposalRubricLine, 0D, 0D, ReferenceDate, CalcCost));
          Type::Balance:
            EXIT(ProjectDevelopmentFunctions.CalculateBudgetByRubric(InvestmProposalRubricLine, 0D, 0D, ReferenceDate));
          Type::ROS:
            EXIT(ProjectDevelopmentFunctions.CalculateTotalRos("Project No.", Version, ReferenceDate, TRUE));
          Type::EPV:
            EXIT(ProjectDevelopmentFunctions.CalculateTotalEPV("Project No.", Version, ReferenceDate, PeriodType, TRUE));
          Type::PRI:
            EXIT(ProjectDevelopmentFunctions.CalculateTotalPRI("Project No.", Version, ReferenceDate, PeriodType, TRUE));
        END;
        EXIT(0);
      END;
    END;

    PROCEDURE CalculatePeriodAmounts@1210190006();
    VAR
      Counter@1100525000 : Integer;
      SplitType@1100525001 : 'Actual,Budget,Both';
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        CLEAR(AmntArray);
        FOR Counter := 1 TO NoOfColumn DO BEGIN
          CASE Type OF
            Type::Subtotal:
              AmntArray[Counter] := ProjectDevelopmentFunctions.CalculateBudgetByRubricCostRev(InvestmProposalRubricLine,
                                          StartDateArray[Counter], GetEndDate(Counter), ReferenceDate, CalcCost);
            Type::Balance:
              AmntArray[Counter] := ProjectDevelopmentFunctions.CalculateBudgetByRubric(InvestmProposalRubricLine,
                                          StartDateArray[Counter], GetEndDate(Counter), ReferenceDate);
            Type::EPV:
              BEGIN
                DateEndPeriod := EndDateArray[Counter];
                IF Counter = NoOfColumn THEN
                  DateEndPeriod := 0D;  //* Hierdoor EPV vanaf laatste periode (kolom 'Period..')
                AmntArray[Counter] :=
                  ProjectDevelopmentFunctions.CalculatePeriodEPV(
                    "Project No.", Version, ReferenceDate, PeriodType,
                    StartDateArray[Counter], DateEndPeriod, SplitType::Budget, TRUE);
              END;
          END;
        END;
      END;
    END;

    PROCEDURE GetEndDate@1210190002(Counter@1210190001 : Integer) : Date;
    BEGIN
      IF Counter < NoOfColumn THEN
        EXIT(EndDateArray[Counter])
      ELSE
        EXIT(NORMALDATE(EndDateArray[Counter]) + 36500);
    END;

    BEGIN
    END.
