    LOCAL PROCEDURE CheckSelections@1100528500();
    VAR
      PlantNumber@1100528502 : Record 11012552;
      PrecLedgEntry@1100528501 : Record 11020516;
      PrecedingPresent@1100528500 : Boolean;
    BEGIN
      ValidateDateTo();
      ValidatePostingDate(TRUE);
      ReadJournalTemplateAndBatch();

      IF (NOT Definitive) AND (PrintType = PrintType::No) THEN
        ERROR(Text011);

      PrecedingPresent := FALSE;
      PlantNumber.COPYFILTERS("Plant Number");
      IF PlantNumber.FINDSET THEN BEGIN
        REPEAT
          PrecLedgEntry.SETRANGE("Plant Type", PlantNumber."Plant Type");
          PrecLedgEntry.SETRANGE("Plant No.", PlantNumber."No.");
          IF NOT PrecLedgEntry.ISEMPTY THEN
            PrecedingPresent := TRUE;
        UNTIL (PlantNumber.NEXT = 0) OR PrecedingPresent;
      END;
      IF PrecedingPresent THEN BEGIN
        IF Definitive AND (ProcessType = ProcessType::Posting) THEN
          ERROR(Text001);
      END ELSE BEGIN
        IF (ProcessType = ProcessType::"Contra Posting") THEN
          ERROR(Text002);
      END;
    END;

    PROCEDURE ValidateDateTo@1100485004();
    BEGIN
      IF (ProcessType = ProcessType::Posting) THEN BEGIN
        IF (DateTo = 0D) THEN
          ERROR(Text003, PrecLedgEntryRec.FIELDCAPTION("Date To"));
      END;

      IF (DateTo > WORKDATE) THEN
        ERROR(Text004);
    END;

    PROCEDURE ValidatePostingDate@1100485005(ICheckPostingAllowed@1100485000 : Boolean);
    VAR
      lvUserSetupRec@1100485003 : Record 91;
      lvAllowPostingFrom@1100485002 : Date;
      lvAllowPostingTo@1100485001 : Date;
    BEGIN
      IF Definitive THEN BEGIN
        IF (PostingDate = 0D) THEN
          ERROR(Text003, PrecLedgEntryRec.FIELDCAPTION("Posting Date"));

        IF ICheckPostingAllowed THEN BEGIN
          IF PostingDate <> NORMALDATE(PostingDate) THEN
            ERROR(Text005, PrecLedgEntryRec.FIELDCAPTION("Posting Date"));

          IF USERID <> '' THEN
            IF lvUserSetupRec.GET(USERID) THEN BEGIN
              lvAllowPostingFrom := lvUserSetupRec."Allow Posting From";
              lvAllowPostingTo := lvUserSetupRec."Allow Posting To";
            END;
          IF (lvAllowPostingFrom = 0D) AND (lvAllowPostingTo = 0D) THEN BEGIN
            GLSetupRec.GET;
            lvAllowPostingFrom := GLSetupRec."Allow Posting From";
            lvAllowPostingTo := GLSetupRec."Allow Posting To";
          END;
          IF lvAllowPostingTo = 0D THEN
            lvAllowPostingTo := 99991231D;

          IF (PostingDate < lvAllowPostingFrom) OR (PostingDate > lvAllowPostingTo) THEN
            ERROR(Text006, PrecLedgEntryRec.FIELDCAPTION("Posting Date"));
        END;
      END;
    END;

    PROCEDURE ReadPlantPostGrpData@1100485016();
    VAR
      lvGLAccRec@1100485000 : Record 15;
    BEGIN
      WITH PlantPostGrpRec DO BEGIN
        IF PostingGroupExternal = '' THEN
          PlantTypeRec.TESTFIELD("Posting Group External");

        IF "Plant Number".Vendor = '' THEN BEGIN
          VendorRec."No." := '';
          VendorRec.INIT;
        END ELSE BEGIN
          IF ("Plant Number".Vendor <> VendorRec."No.") THEN
            VendorRec.GET( "Plant Number".Vendor);
        END;

        IF (PostingGroupExternal <> Code) OR
           (VendorRec."Vendor Posting Group" <> "Vendor Posting Group No.") THEN
        BEGIN

          IF NOT GET(PostingGroupExternal, VendorRec."Vendor Posting Group") THEN
            GET(PostingGroupExternal, '');

          TESTFIELD("PLE Account No.");
          TESTFIELD("PLE Balance Account No.");
          IF NOT lvGLAccRec.GET("PLE Account No.") THEN
            ERROR(Text010, FIELDCAPTION("PLE Account No."), "PLE Account No.", Code);
          IF NOT lvGLAccRec.GET("PLE Balance Account No.") THEN
            ERROR(Text010, FIELDCAPTION("PLE Balance Account No."), "PLE Balance Account No.", Code);

        END;
      END;
    END;
