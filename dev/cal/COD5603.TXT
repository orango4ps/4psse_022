OBJECT Codeunit 5603 FA Get Balance Account
{
  OBJECT-PROPERTIES
  {
    Date=;
    Time=;
    Version List=NAVW111.00;
  }
  PROPERTIES
  {
    OnRun=BEGIN
          END;

  }
  CODE
  {
    VAR
      Text000@1000 : TextConst 'ENU=Do you want to insert a line for the balancing account that is related to the selected lines?;NOR=Vil du sette inn en line for motkontoen som er relatert til de valgte linjene?;SVE=Vill du l„gga till en rad f”r det balanskonto som ber”r valda rader?';
      FAInsertGLAcc@1001 : Codeunit 5601;

    [External]
    PROCEDURE InsertAcc@1(VAR GenJnlLine@1000 : Record 81);
    BEGIN
      CLEARALL;
      WITH GenJnlLine DO BEGIN
        IF COUNT > 1 THEN
          IF NOT CONFIRM(Text000) THEN
            EXIT;
        IF FIND('+') THEN
          REPEAT
            FAInsertGLAcc.GetBalAcc(GenJnlLine);
          UNTIL NEXT(-1) = 0;
      END;
    END;

    [External]
    PROCEDURE InsertAccWithBalAccountInfo@2(VAR GenJnlLine@1000 : Record 81;BalAccountType@1001 : Option;BalAccountNo@1002 : Code[20]);
    BEGIN
      CLEARALL;
      WITH GenJnlLine DO BEGIN
        IF COUNT > 1 THEN
          IF NOT CONFIRM(Text000) THEN
            EXIT;
        IF FIND('+') THEN
          REPEAT
            FAInsertGLAcc.GetBalAccWithBalAccountInfo(GenJnlLine,BalAccountType,BalAccountNo);
          UNTIL NEXT(-1) = 0;
      END;
    END;

    BEGIN
    END.
  }
}
