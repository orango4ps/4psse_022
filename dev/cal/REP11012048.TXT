    PROCEDURE GetReportName@1100525003() : Text;
    BEGIN
      EXIT(DELSTR(CurrReport.OBJECTID, 1, STRPOS(CurrReport.OBJECTID, ' ')));
    END;

    PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Progress Level".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', "Progress Level".TABLECAPTION, "Progress Level".GETFILTERS));
    END;

    PROCEDURE GetProgressDate@1100525002() : Date;
    VAR
      Date@1100525000 : Record 2000000007;
    BEGIN
      IF "Progress Level".GETFILTER("Progress Date") <> '' THEN BEGIN
        Date.SETRANGE("Period Type", Date."Period Type"::Date);
        Date.SETFILTER("Period Start", "Progress Level".GETFILTER("Progress Date"));
        IF Date.FINDSET THEN
          IF Date.NEXT = 0 THEN
            EXIT("Progress Level".GETRANGEMAX("Progress Date"));
      END;

      ERROR(Text000);
    END;

    PROCEDURE CumStandOfPerc@1() : Decimal;
    BEGIN
      IF "Progress Line"."Progress Registry" = "Progress Line"."Progress Registry"::Quantities THEN
        EXIT("Progress Line"."Previous Posture")
      ELSE
        EXIT("Progress Line"."Prev.Cumulative Progr.%");
    END;

    PROCEDURE CreateTempProgressLines@3();
    VAR
      Job@1100525000 : Record 11072003;
      CreateProgress@1100525001 : Codeunit 11012031;
    BEGIN
      "Progress Level".COPYFILTER("Project No.", Job."No.");
      IF Job.FINDSET THEN
        REPEAT
          IF NOT "Progress Level".GET(Job."No.", ProgressDate) THEN BEGIN
            "Progress Level".INIT;
            "Progress Level"."Project No." := Job."No.";
            "Progress Level"."Progress Date" := ProgressDate;
            "Progress Level".INSERT; //Run Table Trigger Not Necessary
            TempProgressLevel := "Progress Level";
            TempProgressLevel.INSERT;
          END;
          IF "Progress Level".FIND('=') THEN BEGIN
            //FIND is needed in case filters are set on Progress Level.
            CODEUNIT.RUN(CODEUNIT::"Create Progress Lines", "Progress Level");

            IF Job."Progress Registry" = Job."Progress Registry"::Percentages THEN
              CreateProgress.CreateProgressPerc("Progress Level", SelectedLevel - 1);
          END;
        UNTIL Job.NEXT = 0;
    END;

    PROCEDURE RemoveTempProgressLines@18();
    BEGIN
      WHILE TempProgressLevel.FINDFIRST DO BEGIN
        "Progress Level".RESET;
        "Progress Level" := TempProgressLevel;
        IF "Progress Level".FIND('=') THEN
          "Progress Level".DELETE(TRUE);
        TempProgressLevel.DELETE;
      END;
    END;

    PROCEDURE BudgetAmount@10() : Decimal;
    BEGIN
      CASE "Progress Line"."Progress Method" OF
        "Progress Line"."Progress Method"::"Budget Line":
          EXIT(BudgetLine."Amount (LCY)");
        "Progress Line"."Progress Method"::"Element - Cost Object":
          BEGIN
            "Progress Line".CALCFIELDS("Budget Amount");
            EXIT("Progress Line"."Budget Amount");
          END;
        "Progress Line"."Progress Method"::"Cost Object":
          BEGIN
            "Progress Line".CALCFIELDS("Budget Amount Cost Obj.");
            EXIT("Progress Line"."Budget Amount Cost Obj.");
          END;
      END;
      EXIT(0);
    END;

    PROCEDURE BudgetHours@13() : Decimal;
    BEGIN
      CASE "Progress Line"."Progress Method" OF
        "Progress Line"."Progress Method"::"Budget Line":
          EXIT(BudgetLine.Hours);
        "Progress Line"."Progress Method"::"Element - Cost Object":
          BEGIN
            "Progress Line".CALCFIELDS("Budgetted Hours");
            EXIT("Progress Line"."Budgetted Hours");
          END;
        "Progress Line"."Progress Method"::"Cost Object":
          BEGIN
            "Progress Line".CALCFIELDS("Budgetted Hours Cost Obj.");
            EXIT("Progress Line"."Budgetted Hours Cost Obj.");
          END;
      END;
      EXIT(0);
    END;

    PROCEDURE NormBudget@4() : Decimal;
    BEGIN
      IF Job."Progress Method" <> Job."Progress Method"::"Budget Line" THEN
        EXIT(0);

      IF "Progress Line"."Progress Method" = "Progress Line"."Progress Method"::"Budget Line" THEN
        EXIT(BudgetLine.Norm)
      ELSE
