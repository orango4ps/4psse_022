    PROCEDURE HeaderMarginCheck@1210190001(IAmountLines@1210190000 : Decimal) : Boolean;
    BEGIN
      IF (MarginAmnt = 0) AND (MarginPerc = 0) THEN
        EXIT(TRUE);  //* No extra warning if margins not used.
      IF ((MarginAmnt <> 0) AND (ABS(DiffDec) > MarginAmnt)) THEN
        EXIT(FALSE);
      IF (MarginPerc <> 0) THEN BEGIN
        IF (IAmountLines = 0) THEN
          EXIT(FALSE)
        ELSE BEGIN
          IF (ABS(DiffDec / IAmountLines * 100)  > MarginPerc) THEN
            EXIT(FALSE);
        END;
      END;

      EXIT(TRUE);
    END;

          ErrorText[ErrorCounterLine] := Text;
      END;
    END;

    BEGIN
