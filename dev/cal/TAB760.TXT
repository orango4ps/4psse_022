    VAR
      StartDate@1000 : Date;
    BEGIN
      IF "Use Work Date as Base" THEN
        StartDate := WORKDATE
      ELSE
        StartDate := TODAY;
      IF "Show Orders" = "Show Orders"::"All Orders" THEN BEGIN
        CALCFIELDS("Latest Order Document Date");
        StartDate := "Latest Order Document Date";
      END;

      EXIT(StartDate);
    END;
