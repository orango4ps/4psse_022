    PROCEDURE ExplodeBOM@1100528504();
    VAR
      EstimateRecipeLine@1100525000 : Record 11072075;
    BEGIN
      EstimateRecipeLine.COPY(Rec);
      EstimateRecipeLine.MARKEDONLY(TRUE);
      IF EstimateRecipeLine.ISEMPTY THEN
        EstimateRecipeLine.MARK(TRUE);
      IF EstimateRecipeLine.FINDSET THEN BEGIN
        REPEAT
          ChangedCostAmountBOM := EstimateRecipeLine."Cost Amount";
          ChangedSalesAmountBOM := EstimateRecipeLine."Sales Amount";  //C028603
          CODEUNIT.RUN(CODEUNIT::"EstimateRecipeLine-Explode BOM",EstimateRecipeLine);  //DP00381
          UpgradeBOMAmountAfter(EstimateRecipeLine);
          COMMIT;
        UNTIL EstimateRecipeLine.NEXT = 0;
      END;
      CLEARMARKS;
    END;
