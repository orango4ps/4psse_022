    LOCAL PROCEDURE GetTotalAssessmentLines@1100528200();
    VAR
      VendorAssessmentLineNew@1100528200 : Record 11072420;
      SaveDate@1100528201 : Date;
    BEGIN
      "Vendor Assessment Vendor".COPYFILTER(Date, VendorAssessmentLineNew.Date);
      VendorAssessmentLineNew.SETRANGE("Vendor No.", "Vendor Assessment Vendor"."Vendor No.");
      VendorAssessmentLineNew.SETFILTER(Score, '<>%1', 0);
      VendorAssessmentLineNew.SETFILTER("Weight Factor", '<>%1', 0);
      VendorAssessmentLineNew.SETCURRENTKEY("Vendor No.",Date);
      IF VendorAssessmentLineNew.FINDSET THEN BEGIN
        REPEAT
          IF SaveDate <> VendorAssessmentLineNew.Date THEN BEGIN
            NoOfAssess := NoOfAssess + 1;
            SaveDate := VendorAssessmentLineNew.Date;
          END;
          TotalScore := TotalScore + (VendorAssessmentLineNew."Weight Factor" * VendorAssessmentLineNew.Score);
          MaxScore := MaxScore + (VendorAssessmentLineNew."Weight Factor" * PurSetupRec."Max. Score Assessment");
        UNTIL VendorAssessmentLineNew.NEXT = 0;
      END;
    END;

    BEGIN
    END.
