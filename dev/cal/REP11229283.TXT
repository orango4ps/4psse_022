    LOCAL PROCEDURE GetVATPostingSetup@1100529608(VATBusPostingGroup@1100529600 : Code[10];VATProdPostingGroup@1100529601 : Code[10];CheckSales@1100529602 : Boolean);
    BEGIN
      IF (VATPostingSetup."VAT Bus. Posting Group" <> VATBusPostingGroup) OR (VATPostingSetup."VAT Prod. Posting Group" <> VATProdPostingGroup) THEN
        VATPostingSetup.GET(VATBusPostingGroup, VATProdPostingGroup);
      IF CheckSales THEN BEGIN
        VATPostingSetup.TESTFIELD("VAT Irrecoverable Invoices");
        VATPostingSetup.TESTFIELD("Sales VAT Account");
      END ELSE BEGIN
        VATPostingSetup.TESTFIELD("VAT Unpaid Invoices");
        VATPostingSetup.TESTFIELD("Purchase VAT Account");
      END;
    END;

    LOCAL PROCEDURE PostEntries@1100529602();
    VAR
      GenJournalLine@1100529600 : Record 81;
      GenJnlPostLine@1100529601 : Codeunit 12;
    BEGIN
      IF NOT Post THEN
        EXIT;
      TmpGenJournalLine.SETRANGE("Document No.", '');
      TmpGenJournalLine.DELETEALL;
      TmpGenJournalLine.RESET;
      IF TmpGenJournalLine.ISEMPTY THEN BEGIN
        UpdateCustLedgerEntry;
        UpdateVendLedgerEntry;
        EXIT;
      END;

      CLEAR(TmpCustLedgerEntry);
      CLEAR(TmpVendorLedgerEntry);

      COMMIT;
      TmpGenJournalLine.SETCURRENTKEY("Line No.");
      TmpGenJournalLine.FINDSET;
      REPEAT
        CLEAR(GenJournalLine);
        GenJournalLine."Posting Date" := TmpGenJournalLine."Posting Date";
        GenJournalLine."Document No." := TmpGenJournalLine."Document No.";
        GenJournalLine.VALIDATE("Account No.", TmpGenJournalLine."Account No.");
        GenJournalLine."VAT Posting" := GenJournalLine."VAT Posting"::"Manual VAT Entry";
        GenJournalLine."Gen. Bus. Posting Group" := TmpGenJournalLine."Gen. Bus. Posting Group";
        GenJournalLine.VALIDATE("Gen. Prod. Posting Group", TmpGenJournalLine."Gen. Prod. Posting Group");
        GenJournalLine."VAT Bus. Posting Group" := TmpGenJournalLine."VAT Bus. Posting Group";
        GenJournalLine.VALIDATE("VAT Prod. Posting Group", TmpGenJournalLine."VAT Prod. Posting Group");
        IF (GenJournalLine."Shortcut Dimension 1 Code" = '') AND (TmpGenJournalLine."Shortcut Dimension 1 Code" <> '') THEN
          GenJournalLine.VALIDATE("Shortcut Dimension 1 Code", TmpGenJournalLine."Shortcut Dimension 1 Code");
        IF (GenJournalLine."Shortcut Dimension 2 Code" = '') AND (TmpGenJournalLine."Shortcut Dimension 2 Code" <> '') THEN
          GenJournalLine.VALIDATE("Shortcut Dimension 2 Code", TmpGenJournalLine."Shortcut Dimension 2 Code");
        GenJournalLine.Description := TmpGenJournalLine.Description;
        GenJournalLine."VAT Base Amount" := TmpGenJournalLine."VAT Base Amount";
        GenJournalLine."VAT Base Amount (LCY)" := TmpGenJournalLine."VAT Base Amount";
        GenJournalLine."VAT Amount" := TmpGenJournalLine."VAT Amount";
        GenJournalLine."VAT Amount (LCY)" := TmpGenJournalLine."VAT Amount";
        GenJournalLine.Amount := - GenJournalLine."VAT Amount";
        GenJournalLine."Amount (LCY)" := - GenJournalLine."VAT Amount (LCY)";
        IF TmpGenJournalLine."Source Type" = TmpGenJournalLine."Source Type"::Customer THEN
          GenJournalLine."Gen. Posting Type" := GenJournalLine."Gen. Posting Type"::Sale
        ELSE
          GenJournalLine."Gen. Posting Type" := GenJournalLine."Gen. Posting Type"::Purchase;
        GenJournalLine."System-Created Entry" := TRUE;

        GenJnlPostLine.RUN(GenJournalLine);
      UNTIL TmpGenJournalLine.NEXT = 0;

      UpdateCustLedgerEntry;
      UpdateVendLedgerEntry;
      COMMIT;
    END;

    LOCAL PROCEDURE UpdateCustLedgerEntry@1100529601();
    VAR
      CustLedgerEntry@1100529600 : Record 21;
    BEGIN
      IF NOT Post THEN
        EXIT;
      IF TmpCustLedgerEntry.ISEMPTY THEN
        EXIT;
      CustLedgerEntry.LOCKTABLE;
      TmpCustLedgerEntry.FINDSET;
      REPEAT
        IF CustLedgerEntry.GET(TmpCustLedgerEntry."Entry No.") THEN BEGIN
          IF UndoReversalVAT THEN BEGIN
            CustLedgerEntry."Irrecoverable Invoice" := FALSE;
            CustLedgerEntry."Reversal VAT" := FALSE;
            CustLedgerEntry."Reversal VAT Posted on" := 0D;
          END ELSE BEGIN
            CustLedgerEntry."Reversal VAT" := TRUE;
            CustLedgerEntry."Reversal VAT Posted on" := PostingDate;
          END;
          CustLedgerEntry.MODIFY;
        END;
      UNTIL TmpCustLedgerEntry.NEXT = 0;
    END;

    LOCAL PROCEDURE UpdateVendLedgerEntry@1100529618();
    VAR
      VendorLedgerEntry@1100529600 : Record 25;

        EXIT;
      IF TmpVendorLedgerEntry.ISEMPTY THEN

    END;

    LOCAL PROCEDURE ProcessLedgerEntry@1100529607(ProcessCustomerEntry@1100529600 : Boolean) : Boolean;
    VAR
      GLEntry@1100529617 : Record 17;
      VATEntry@1100529607 : Record 254;
      VATLineCount@1100529614 : Integer;
      TotalVATLineCount@1100529613 : Integer;
      OriginalAmount@1100529601 : Decimal;
      RemainingAmount@1100529602 : Decimal;
      Factor@1100529603 : Decimal;
      TotalVATBaseAmount@1100529605 : Decimal;
      TotalVATAmount@1100529604 : Decimal;
      VATBaseAmount@1100529608 : Decimal;
      VATAmount@1100529606 : Decimal;
      SumVATBaseAmount@1100529610 : Decimal;
      SumVATAmount@1100529609 : Decimal;
      ReminderVATBaseAmount@1100529612 : Decimal;
      ReminderVATAmount@1100529611 : Decimal;
      TotalOrigVATBaseAmount@1100529616 : Decimal;
      TotalOrigVATAmount@1100529615 : Decimal;
    BEGIN
      IF ProcessCustomerEntry THEN BEGIN
        VATEntry.SETRANGE("Document No.", "Cust. Ledger Entry"."Document No.");
        VATEntry.SETRANGE("Posting Date", "Cust. Ledger Entry"."Posting Date");
      END ELSE BEGIN
        VATEntry.SETRANGE("Document No.", "Vendor Ledger Entry"."Document No.");
        VATEntry.SETRANGE("Posting Date", "Vendor Ledger Entry"."Posting Date");
      END;
      VATEntry.SETFILTER(Amount, '<>0');
      IF VATEntry.ISEMPTY THEN BEGIN
        CLEAR(VATEntry);
        InsertTmpGenJournalLine(ProcessCustomerEntry, VATEntry, 0, 0, 0, 0);
        EXIT(FALSE);
      END;

      IF UndoReversalVAT THEN BEGIN
        GLEntry.SETCURRENTKEY("Document No.", "Posting Date");
        IF ProcessCustomerEntry THEN BEGIN
          GLEntry.SETRANGE("Document No.", "Cust. Ledger Entry"."Document No.");
          GLEntry.SETRANGE("Posting Date", "Cust. Ledger Entry"."Reversal VAT Posted on");
        END ELSE BEGIN
          GLEntry.SETRANGE("Document No.", "Vendor Ledger Entry"."Document No.");
          GLEntry.SETRANGE("Posting Date", "Vendor Ledger Entry"."Reversal VAT Posted on");
        END;
        IF GLEntry.ISEMPTY THEN BEGIN
          CLEAR(VATEntry);
          InsertTmpGenJournalLine(ProcessCustomerEntry, VATEntry, 0, 0, 0, 0);
          EXIT(FALSE);
        END;
        GLEntry.FINDLAST;
        VATEntry.RESET;
        VATEntry.SETRANGE("Document No.", GLEntry."Document No.");
        VATEntry.SETRANGE("Posting Date", GLEntry."Posting Date");
        VATEntry.SETRANGE("Transaction No.", GLEntry."Transaction No.");
        IF VATEntry.ISEMPTY THEN BEGIN
          CLEAR(VATEntry);
          InsertTmpGenJournalLine(ProcessCustomerEntry, VATEntry, 0, 0, 0, 0);
          EXIT(FALSE);
        END;
      END ELSE
        IF ProcessCustomerEntry THEN BEGIN
          OriginalAmount := "Cust. Ledger Entry"."Original Amt. (LCY)";
          RemainingAmount := "Cust. Ledger Entry"."Remaining Amt. (LCY)";
        END ELSE BEGIN
          OriginalAmount := "Vendor Ledger Entry"."Original Amt. (LCY)";
          RemainingAmount := "Vendor Ledger Entry"."Remaining Amt. (LCY)";
        END;

      VATEntry.CALCSUMS(Base, Amount);
      TotalOrigVATBaseAmount := ROUND(VATEntry.Base, GLSetup."Amount Rounding Precision");
      TotalOrigVATAmount := ROUND(VATEntry.Amount, GLSetup."Amount Rounding Precision");
      IF (OriginalAmount = RemainingAmount) OR UndoReversalVAT THEN BEGIN
        Factor := 1;
        TotalVATBaseAmount := TotalOrigVATBaseAmount;
        TotalVATAmount := TotalOrigVATAmount;
      END ELSE BEGIN
        Factor := RemainingAmount / OriginalAmount;
        TotalVATBaseAmount := ROUND(VATEntry.Base * Factor, GLSetup."Amount Rounding Precision");
        TotalVATAmount := ROUND(VATEntry.Amount * Factor, GLSetup."Amount Rounding Precision");
        ReminderVATBaseAmount := 0;
        ReminderVATAmount := 0;
        SumVATBaseAmount := 0;
        SumVATAmount := 0;
        TotalVATLineCount := VATEntry.COUNT;
        VATLineCount := 0;
      END;

      IF (RemainingAmount = 0) AND NOT UndoReversalVAT THEN BEGIN
        CLEAR(VATEntry);
        InsertTmpGenJournalLine(ProcessCustomerEntry, VATEntry, TotalVATBaseAmount, TotalVATAmount, TotalOrigVATBaseAmount, TotalOrigVATAmount);
        EXIT(TRUE);
      END;

      VATEntry.FINDSET;
      REPEAT
        GetVATPostingSetup(VATEntry."VAT Bus. Posting Group", VATEntry."VAT Prod. Posting Group", ProcessCustomerEntry);
        IF Factor <> 1 THEN BEGIN
          VATLineCount += 1;
          IF VATLineCount = TotalVATLineCount THEN BEGIN
            VATEntry.Base := TotalVATBaseAmount - SumVATBaseAmount;
            VATEntry.Amount := TotalVATAmount - SumVATAmount;
          END ELSE BEGIN
            VATEntry.Base := VATEntry.Base * Factor + ReminderVATBaseAmount;
            VATEntry.Amount := VATEntry.Amount * Factor + ReminderVATAmount;
            VATBaseAmount := ROUND(VATEntry.Base, GLSetup."Amount Rounding Precision");
            VATAmount := ROUND(VATEntry.Amount, GLSetup."Amount Rounding Precision");
            ReminderVATBaseAmount := VATEntry.Base - VATBaseAmount;
            ReminderVATAmount := VATEntry.Amount - VATAmount;
            SumVATBaseAmount += VATBaseAmount;
            SumVATAmount += VATAmount;
          END;
        END;
        InsertTmpGenJournalLine(ProcessCustomerEntry, VATEntry, TotalVATBaseAmount, TotalVATAmount, TotalOrigVATBaseAmount, TotalOrigVATAmount);
      UNTIL VATEntry.NEXT = 0;
      EXIT(TRUE);
    END;

    LOCAL PROCEDURE GetDateFilter@1100529611();
    VAR
      Date@1100529600 : Record 2000000007;
    BEGIN
      IF UndoReversalVAT AND (FilterDate = 0D) THEN BEGIN
        DateFilter := '';
        EXIT;
      END;
      Date.SETRANGE("Period Type", Date."Period Type"::Date);
      Date.SETFILTER("Period Start", '<=%1', CALCDATE('<-1Y>', FilterDate));
      Date.FINDLAST;
      Date.SETFILTER("Period Start", '..%1', Date."Period Start");
      DateFilter := Date.GETFILTER("Period Start");
    END;

    LOCAL PROCEDURE SetFiltersOnCustomerLedgerEntry@1100529612();
    BEGIN
      IF SourceType = SourceType::Vendors THEN
        EXIT;

        "Cust. Ledger Entry".SETFILTER("Due Date", DateFilter);
      "Cust. Ledger Entry".SETRANGE("Reversal VAT", UndoReversalVAT);

      ELSE

    END;

        "Vendor Ledger Entry".SETRANGE("Unpaid Invoice", TRUE)
      ELSE
        IF UndoVendorDocNo <> '' THEN
          "Vendor Ledger Entry".SETRANGE("Document No.", UndoVendorDocNo);
    END;

    BEGIN
    END.
