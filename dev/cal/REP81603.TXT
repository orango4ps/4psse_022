                BEGIN
                  CompanyInformation.GET;
                  CompanyInformation.CALCFIELDS(Picture);

                  TEMPHourAccoutBuffer.DELETEALL;

                  //*** Filters on cost center from dimension filter ***
                  GLSetup.GET;
                  FilterCostCenter := '';

                  IF DimensionValueCode = '' THEN BEGIN
                    ERROR(Text002);
                    //REPORT.RUN(81603);
                  END;

                  IF DimensionCode = GLSetup."Shortcut Dimension 1 Code" THEN BEGIN

                    FilterCostCenter := DimensionValueCode;

                  END ELSE BEGIN

                    IF DimensionValueCode <> '' THEN
                      TEMPDimValueChains.getSourceDimensions(DimensionCode, DimensionValueCode, GLSetup."Shortcut Dimension 1 Code", TEMPDimValueChains);

                    First := TRUE;
                    IF TEMPDimValueChains.FINDSET(FALSE) THEN REPEAT

                      IF First THEN BEGIN
                        FilterCostCenter := TEMPDimValueChains."Start Dimension Value Code";
                        First := FALSE
                      END ELSE
                        FilterCostCenter += '|' + TEMPDimValueChains."Start Dimension Value Code";
                    UNTIL TEMPDimValueChains.NEXT = 0;

                  END;

                  //*** Filters on the employees from cost center, manager and employee ***
                  IF (ManagernNo <> '') OR (FilterCostCenter <> '') OR (EmployeeNo <> '') THEN BEGIN
                    First := TRUE;

                    IF EmployeeNo <> '' THEN
                      Employee.SETFILTER("No.", EmployeeNo);

                    IF ManagernNo <> '' THEN
                      Employee.SETFILTER("Manager No.", ManagernNo);

                    IF FilterCostCenter <> '' THEN
                      Employee.SETFILTER("Global Dimension 1 Code", FilterCostCenter);

                    IF Employee.FINDSET(FALSE) THEN REPEAT

                      IF First THEN BEGIN
                        FilterEmployee := Employee."No.";
                        First := FALSE
                      END ELSE
                        FilterEmployee += '|' + Employee."No.";

                    UNTIL Employee.NEXT = 0;

                    IF FilterEmployee = '' THEN
                      CurrReport.SKIP;
                  END;

                  // *** Convert filter date ***

                  IF FromWeek <> 0 THEN
                    PeriodStart := DWY2DATE(1, FromWeek, FromYear);
                  IF ToWeek <> 0 THEN
                    PeriodEnd := DWY2DATE(7, ToWeek, ToYear);

                  CreateBuffer;
                  TEMPHourAccoutBuffer.RESET;
                  TEMPHourAccoutBuffer.SETCURRENTKEY("Last Name","First Name",Date);
                END;

    PROCEDURE CreateBuffer@1100285501();
    VAR
      HourAccounting@1100285508 : Record 11012038;
      PostedHourHeader@1100285506 : Record 11012084;
      PersonalWorkSchedule@1100285505 : Record 11128051;
      WorkSchedule@1100285504 : Record 11128050;
      HourAccountingLine@1100285501 : Record 11012039;
      PostedHourLine@1100285503 : Record 11012085;
      Employee@1100285502 : Record 5200;
      WageCompRec@1100285507 : Record 11012014;
      EntryNo@1100285500 : Integer;
    BEGIN
      EntryNo := 1;

      //*** Sets filter on requested period ***
      IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
        //>>141104 ENH050
        //HourAccounting.SETRANGE("Posting Date", PeriodStart, PeriodEnd)
        HourAccounting.SETRANGE("Document Date", PeriodStart, PeriodEnd)
        //<<141104 ENH050

      ELSE IF (PeriodStart <> 0D) THEN
        //>>141104 ENH050
        //HourAccounting.SETFILTER("Posting Date", '%1..', PeriodStart)
        HourAccounting.SETFILTER("Document Date", '%1..', PeriodStart)
        //<<141104 ENH050

      ELSE IF (PeriodEnd <> 0D) THEN
        //>>141104 ENH050
        //HourAccounting.SETFILTER("Posting Date", '..%1', PeriodEnd);
        HourAccounting.SETFILTER("Document Date", '..%1', PeriodEnd);
        //<<141104 ENH050

      //*** Filter on employee ***
      IF FilterEmployee <> '' THEN
        HourAccounting.SETFILTER("Employee No.", FilterEmployee);

      //*** If requested to only show open or released weeks, filter is set ****
      // >> 141127 REP-100-043 ITERO.MK
      IF (ShowOpenHours = FALSE) OR (ShowReleasedHours = FALSE) THEN BEGIN

        IF ShowOpenHours THEN
          HourAccounting.SETRANGE(Status, HourAccounting.Status::Open)

        ELSE IF ShowReleasedHours THEN
          HourAccounting.SETRANGE(Status, HourAccounting.Status::Released)

      //    ELSE
      //    CurrReport.SKIP;                               //KONTROLLERA OM DETTA ŽR KORREKT

      END;

      IF ShowOpenHours OR ShowReleasedHours THEN BEGIN
      // << 141127 REP-100-043 ITERO.MK

        IF HourAccounting.FINDSET(FALSE) THEN REPEAT

          WITH TEMPHourAccoutBuffer DO BEGIN
            INIT;

            "Entry No." := EntryNo;
            Week := HourAccounting.Week;
            Year := HourAccounting.Year;
            "No." := HourAccounting."Employee No.";

            IF Employee.GET(HourAccounting."Employee No.") THEN BEGIN

              "Last Name" :=  Employee."Last Name";
              "First Name" := Employee."First Name";
              "Wage Component /Cost Code" := Employee."Global Dimension 1 Code";  //Field used to store Cost Center
              "Project No." := Employee."Manager No."; //Field used to store manager no
              "Agda No." := Employee."Salary No.";

            END;

            //***Calc Scheduled Hours from personal or general work schedual

            PersonalWorkSchedule.SETRANGE("Employee number", HourAccounting."Employee No.");
            PersonalWorkSchedule.SETRANGE(Date, DWY2DATE(1, HourAccounting.Week,HourAccounting.Year),
                                      DWY2DATE(7, HourAccounting.Week,HourAccounting.Year));
            IF PersonalWorkSchedule.FINDSET(FALSE) THEN BEGIN

              REPEAT
                Hours += PersonalWorkSchedule."Norm hours";
              UNTIL PersonalWorkSchedule.NEXT = 0;

            END ELSE BEGIN

              WorkSchedule.SETRANGE(Date, DWY2DATE(1, HourAccounting.Week,HourAccounting.Year),
                                      DWY2DATE(7, HourAccounting.Week,HourAccounting.Year));

              IF WorkSchedule.FINDSET(FALSE) THEN REPEAT
                Hours += WorkSchedule."Norm Hours";
              UNTIL WorkSchedule.NEXT = 0;

            END;

            //*** Calc Open Hours from Hour Accounting Line
            HourAccountingLine.RESET;
            HourAccountingLine.SETRANGE(Year, HourAccounting.Year);
            HourAccountingLine.SETRANGE(Week, HourAccounting.Week);
            HourAccountingLine.SETRANGE("Employee No.", HourAccounting."Employee No.");


            IF HourAccounting.Status = HourAccounting.Status::Open THEN BEGIN
              IF HourAccountingLine.FINDSET(FALSE) THEN REPEAT

                // 4PSSE.MK 140325 >>
                //** Only include normal hours
                IF WageCompRec.GET(HourAccountingLine."Wage Component") THEN BEGIN
                  IF (WageCompRec."Component Type" = WageCompRec."Component Type"::Hours) AND
                                      (WageCompRec.Overtime = FALSE) THEN BEGIN
                // 4PSSE.MK 140325 <<

                    "Open Hours" += HourAccountingLine."Total Line";
                  END;
                END;
              UNTIL HourAccountingLine.NEXT = 0;

            //*** Calc Released Hours from Hour Accounting Line

            END ELSE IF HourAccounting.Status = HourAccounting.Status::Released THEN BEGIN
              IF HourAccountingLine.FINDSET(FALSE) THEN REPEAT

                // 4PSSE.MK 140325 >>
                //** Only include normal hours
                IF WageCompRec.GET(HourAccountingLine."Wage Component") THEN BEGIN
                  IF (WageCompRec."Component Type" = WageCompRec."Component Type"::Hours) AND
                                      (WageCompRec.Overtime = FALSE) THEN BEGIN
                // 4PSSE.MK 140325 <<

                    "Released Hours" += HourAccountingLine."Total Line";
                  END;
                END;

              UNTIL HourAccountingLine.NEXT = 0;

            END;

            INSERT;

          END;

          EntryNo += 1;

        UNTIL HourAccounting.NEXT = 0;

      END; // 141127

      //***Get Posted Hours
      IF ShowPostedHours THEN BEGIN

        //*** Sets filter on requested period ***
        IF (PeriodStart <> 0D) AND (PeriodEnd <> 0D) THEN
          //>>141104 ENH050
          //PostedHourLine.SETRANGE("Posting Date", PeriodStart, PeriodEnd)
          PostedHourLine.SETRANGE("Actual date of line", PeriodStart, PeriodEnd)
          //<<141104 ENH050
        ELSE IF (PeriodStart <> 0D) THEN
          //>>141104 ENH050
          //PostedHourLine.SETFILTER("Posting Date", '%1..', PeriodStart)
          PostedHourLine.SETFILTER("Actual date of line", '%1..', PeriodStart)
          //<<141104 ENH050

        ELSE IF (PeriodEnd <> 0D) THEN
          //>>141104 ENH050
          //PostedHourLine.SETFILTER("Posting Date", '..%1', PeriodEnd);
          PostedHourLine.SETFILTER("Actual date of line", '..%1', PeriodEnd);
          //<<141104 ENH050

        //*** Filter on employee ***
        IF FilterEmployee <> '' THEN
          PostedHourLine.SETFILTER("Employee No.", FilterEmployee);

        IF PostedHourLine.FINDSET(FALSE) THEN REPEAT

          // 4PSSE.MK 140325 >>
          //** Only include normal hours
          IF WageCompRec.GET(PostedHourLine."Wage Component") THEN BEGIN
            IF (WageCompRec."Component Type" = WageCompRec."Component Type"::Hours) AND
                                    (WageCompRec.Overtime = FALSE) THEN BEGIN
              // 4PSSE.MK 140325 <<


              TEMPHourAccoutBuffer.RESET;
              TEMPHourAccoutBuffer.SETRANGE(Year, PostedHourLine.Year);
              TEMPHourAccoutBuffer.SETRANGE(Week, PostedHourLine.Week);
              TEMPHourAccoutBuffer.SETRANGE("No.", PostedHourLine."Employee No.");
              IF TEMPHourAccoutBuffer.FINDSET THEN BEGIN

                TEMPHourAccoutBuffer."Posted Hours" += PostedHourLine."Total Line";
                TEMPHourAccoutBuffer.MODIFY;

              END ELSE BEGIN

                WITH TEMPHourAccoutBuffer DO BEGIN
                  INIT;
                  "Entry No." := EntryNo;
                  Year := PostedHourLine.Year;
                  Week := PostedHourLine.Week;
                  "No." := PostedHourLine."Employee No.";

                  IF Employee.GET(PostedHourLine."Employee No.") THEN BEGIN

                    "Last Name" :=  Employee."Last Name";
                    "First Name" := Employee."First Name";
                    "Wage Component /Cost Code" := Employee."Global Dimension 1 Code";  //Field used to store Cost Center
                    "Project No." := Employee."Manager No."; //Field used to store manager no
                    "Agda No." := Employee."Salary No.";

                  END;

                  //***Calc Scheduled Hours from personal or general work schedual

                  PersonalWorkSchedule.SETRANGE("Employee number", PostedHourLine."Employee No.");
                  PersonalWorkSchedule.SETRANGE(Date, DWY2DATE(1, PostedHourLine.Week,PostedHourLine.Year),
                                            DWY2DATE(7, PostedHourLine.Week,PostedHourLine.Year));
                  IF PersonalWorkSchedule.FINDSET(FALSE) THEN BEGIN

                    REPEAT
                      Hours += PersonalWorkSchedule."Norm hours";
                    UNTIL PersonalWorkSchedule.NEXT = 0;

                  END ELSE BEGIN

                    WorkSchedule.SETRANGE(Date, DWY2DATE(1, PostedHourLine.Week, PostedHourLine.Year),
                                            DWY2DATE(7, PostedHourLine.Week, PostedHourLine.Year));

                    IF WorkSchedule.FINDSET(FALSE) THEN REPEAT
                      Hours += WorkSchedule."Norm Hours";
                    UNTIL WorkSchedule.NEXT = 0;

                  END;

                  "Posted Hours" += PostedHourLine."Total Line";

                  INSERT;

                END;
              END;
              EntryNo += 1;

            END;
          END;

        UNTIL PostedHourLine.NEXT = 0;

      END;
    END;

    PROCEDURE GetReportFilter1@1100525000() : Text;
    VAR
      FilterString@1100285000 : Text;
    BEGIN
      IF (FromWeek <> 0) OR (ToWeek <> 0) THEN BEGIN

        FilterString := TextDateFilter;

        IF (FromWeek <> 0) AND (ToWeek <> 0) THEN
          FilterString += ' '+ TextWeek +' '+ FORMAT(FromWeek) +' '+ TextYear +' '+ FORMAT(FromYear) +' ... '+
                                TextWeek +' '+ FORMAT(ToWeek) +' '+ TextYear +' '+ FORMAT(ToYear)
        ELSE IF (FromWeek <> 0) THEN
          FilterString += ' '+ TextWeek +' '+ FORMAT(FromWeek) +' '+ TextYear +' '+ FORMAT(FromYear) +' ... '
        ELSE IF (ToWeek <> 0) THEN
          FilterString += ' ' + TextWeek +' '+ FORMAT(ToWeek) +' '+ TextYear +' '+ FORMAT(ToYear);

      END ELSE
        FilterString := '';

      EXIT(FilterString);
    END;

    PROCEDURE GetReportFilter2@1100285503() : Text;
    VAR
      FilterString@1100285500 : Text;
    BEGIN
      IF DimensionValueCode <> '' THEN
        FilterString := DimensionCode + ': ' + DimensionValueCode
      ELSE
        FilterString := '';

      IF FilterString <> '' THEN
        FilterString += ' / ';

      FilterString += STRSUBSTNO(TextHours, ShowOpenHours, ShowReleasedHours, ShowPostedHours, ShowOnlyDeviations);
