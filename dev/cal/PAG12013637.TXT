          AppSetupNew."Path to Used Invoices":= '';

          AppSetupNew.INSERT;
        END;
      END;

      // Users
      IF Users THEN
        ExFlow.CopyExFlowSetup(Comp,'USER');

      // UserGroup
      IF UserGroup THEN
        ExFlow.CopyExFlowSetup(Comp,'USERGROUP');

      // Templates
      IF Templates THEN
        ExFlow.CopyExFlowSetup(Comp,'TEMPLATES');

      // Vendor ID's

        ExFlow.CopyExFlowSetup(Comp,'VENDORID');

      // Advance Vend. Opts.

      // Line Advanced

      // Invoicing Codes
