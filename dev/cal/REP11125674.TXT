    PROCEDURE InitSurcharges@1210190002();
    VAR
      EstElemRec@1210190000 : Record 11125679;
      EstLineRec@1210190001 : Record 11125698;
    BEGIN
      EstElemRec.SETRANGE("Estimate No.", CurrEst);
      EstElemRec.SETFILTER(Type, '%1|%2',
        EstElemRec.Type::"Commitment Description (Main Line)", EstElemRec.Type::"Commitment (Main Line)");
      EstElemRec.SETRANGE(Title, TRUE);
      EstElemRec.SETRANGE("Lock Actual Amount for Updates", FALSE);
      IF EstElemRec.FINDSET(TRUE, FALSE) THEN BEGIN
        REPEAT
          IF EstElemRec."Surcharge Amount Labor" <> 0 THEN
            EstElemRec.VALIDATE("Surcharge Amount Labor", 0);
          IF EstElemRec."Surcharge Amount Material" <> 0 THEN
            EstElemRec.VALIDATE("Surcharge Amount Material", 0);
          IF EstElemRec."Surcharge Amount Subcontr." <> 0 THEN
            EstElemRec.VALIDATE("Surcharge Amount Subcontr.", 0);
          IF EstElemRec."Surcharge Amount Plant" <> 0 THEN
            EstElemRec.VALIDATE("Surcharge Amount Plant", 0);
          EstElemRec."Total Surcharge Amount":= 0;
          EstElemRec.MODIFY;
        UNTIL EstElemRec.NEXT = 0;
      END;

      EstLineRec.SETRANGE("Estimate No.", CurrEst);
      EstLineRec.SETRANGE("Adjustment Line", TRUE);
      EstLineRec.DELETEALL;

      IF UpdateMethod = UpdateMethod::Restore THEN BEGIN
        CalcEstCu.RecalculateEstimateRAW(CurrEst);
        CLEAR(CalcEstCu);

        NewEst.GET(CurrEst);
        NewEst.CALCFIELDS("Labor (RAW)", "Material (RAW)", "Subcontracting (RAW)", "Plant (RAW)", "Sundry (RAW)", "Total (RAW)",
          "Surcharge Amount Labor", "Surcharge Amount Material", "Surcharge Amount Subcontr.", "Surcharge Amount Plant",
          "Total Surcharge Amount");
      END;
    END;

    PROCEDURE UpdateSurcharges@1210190003();
    VAR
      EstimateRAWSetup@1210190005 : Record 11125673;
      EstElemRec@1210190001 : Record 11125679;
      BaseAmntOld@1210190007 : Decimal;
      BaseAmntNew@1210190008 : Decimal;
      SumOldPerc1@1210190009 : Decimal;
      SumOldPerc2@1210190010 : Decimal;
      SumNewPerc1@1210190013 : Decimal;
      SumNewPerc2@1210190012 : Decimal;
      ElemTotal@1210190015 : Decimal;
      ElemFixed@1210190014 : Decimal;
      SumTotal@1210190011 : Decimal;
      CalcAmnt@1210190004 : Decimal;
      AdjustFactor@1210190003 : Decimal;
      CostTypeFactor@1210190016 : Decimal;
      RestAmnt@1210190019 : Decimal;
      i@1210190020 : Integer;
    BEGIN
      EstimateRAWSetup.GET;

      EstElemRec.SETRANGE("Estimate No.", CurrEst);
      EstElemRec.SETFILTER(Type, '%1|%2',
        EstElemRec.Type::"Commitment Description (Main Line)", EstElemRec.Type::"Commitment (Main Line)");
      EstElemRec.SETRANGE(Title, TRUE);
      IF EstElemRec.FINDSET THEN BEGIN
        REPEAT
          IF EstElemRec."Tender Paragraph No." < '9' THEN BEGIN
            ElemTotal := ElemTotal +  EstElemRec."Calculated Amount";
            IF EstElemRec."Lock Actual Amount for Updates" THEN
              ElemFixed := ElemFixed + EstElemRec."Calculated Amount";
          END ELSE BEGIN
            IF EstElemRec."Tender Paragraph No." <> '919990' THEN BEGIN
              SumTotal := SumTotal + EstElemRec."Calculated Amount";
            END;
          END;
        UNTIL EstElemRec.NEXT = 0;
      END;

      BaseAmntOld := ContrAmntOld - SumTotal;
      BaseAmntNew := ContrAmntNew - SumTotal;

      SumOldPerc1 := Estimate.CalculateAdditionalCost(1, BaseAmntOld);
      SumOldPerc2 := Estimate.CalculateAdditionalCost(2, BaseAmntOld);
      SumNewPerc1 := Estimate.CalculateAdditionalCost(1, BaseAmntNew);
      SumNewPerc2 := Estimate.CalculateAdditionalCost(2, BaseAmntNew);

      BaseAmntOld := BaseAmntOld - ElemFixed;
      BaseAmntNew := BaseAmntNew - (SumNewPerc1-SumOldPerc1) - (SumNewPerc2-SumOldPerc2) - ElemFixed;

      REPEAT
        AdjustFactor := (BaseAmntNew - BaseAmntOld) / BaseAmntOld;

        EstElemRec.RESET;
        EstElemRec.SETRANGE("Estimate No.", CurrEst);
        EstElemRec.SETFILTER("Tender Paragraph No.", '<9');
        EstElemRec.SETFILTER(Type, '%1|%2',
          EstElemRec.Type::"Commitment Description (Main Line)", EstElemRec.Type::"Commitment (Main Line)");
        EstElemRec.SETRANGE(Title, TRUE);
        EstElemRec.SETRANGE("Lock Actual Amount for Updates", FALSE);
        IF EstElemRec.FINDSET(TRUE,FALSE) THEN BEGIN
          REPEAT
            EstElemRec.CALCFIELDS("Calculated Amount Labor", "Calculated Amount Material",
              "Calculated Amount Subcontr.", "Calculated Amount Plant", "Estimate Line Amount");
            CalcAmnt := EstElemRec."Calculated Amount Labor" + EstElemRec."Calculated Amount Material" +
              EstElemRec."Calculated Amount Subcontr." + EstElemRec."Calculated Amount Plant";
            IF CalcAmnt <> 0 THEN BEGIN
              IF EstElemRec."Calculated Amount Labor" <> 0 THEN BEGIN
                IF EstElemRec."Estimate Line Amount" = 0 THEN
                  CostTypeFactor := 1
                ELSE
                  CostTypeFactor := EstElemRec."Calculated Amount Labor" / EstElemRec."Estimate Line Amount";
                EstElemRec.VALIDATE("Surcharge Amount Labor", CalcAmnt * AdjustFactor * CostTypeFactor);
              END;
              IF EstElemRec."Calculated Amount Material" <> 0 THEN BEGIN
                IF EstElemRec."Calculated Amount Material" <> 0 THEN
                  IF EstElemRec."Estimate Line Amount" = 0 THEN
                    CostTypeFactor := 1
                  ELSE
                    CostTypeFactor := EstElemRec."Calculated Amount Material" / EstElemRec."Estimate Line Amount";
                EstElemRec.VALIDATE("Surcharge Amount Material", CalcAmnt * AdjustFactor * CostTypeFactor);
              END;
              IF EstElemRec."Calculated Amount Subcontr." <> 0 THEN BEGIN
                IF EstElemRec."Calculated Amount Subcontr." <> 0 THEN
                  IF EstElemRec."Estimate Line Amount" = 0 THEN
                    CostTypeFactor := 1
                  ELSE
                    CostTypeFactor := EstElemRec."Calculated Amount Subcontr." / EstElemRec."Estimate Line Amount";
                EstElemRec.VALIDATE("Surcharge Amount Subcontr.", CalcAmnt * AdjustFactor * CostTypeFactor);
              END;
              IF EstElemRec."Calculated Amount Plant" <> 0 THEN BEGIN
                IF EstElemRec."Calculated Amount Plant" <> 0 THEN
                  IF EstElemRec."Estimate Line Amount" = 0 THEN
                    CostTypeFactor := 1
                  ELSE
                    CostTypeFactor := EstElemRec."Calculated Amount Plant" / EstElemRec."Estimate Line Amount";
                EstElemRec.VALIDATE("Surcharge Amount Plant", CalcAmnt * AdjustFactor * CostTypeFactor);
              END;
              EstElemRec.MODIFY;
            END;
          UNTIL EstElemRec.NEXT = 0;
        END;

        CalcEstCu.RecalculateEstimateRAW(CurrEst);
        CLEAR(CalcEstCu);

        NewEst.GET(CurrEst);
        NewEst.CALCFIELDS("Labor (RAW)", "Material (RAW)", "Subcontracting (RAW)", "Plant (RAW)", "Sundry (RAW)", "Total (RAW)",
          "Surcharge Amount Labor", "Surcharge Amount Material", "Surcharge Amount Subcontr.", "Surcharge Amount Plant",
          "Total Surcharge Amount");


        RestAmnt := ContrAmntNew - NewEst."Contract Amount (Actual)";
        BaseAmntNew := BaseAmntNew + RestAmnt;
        i := i +1;
      UNTIL i >= NumCalc;
    END;

    BEGIN
    END.
