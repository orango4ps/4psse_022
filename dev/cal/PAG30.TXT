    PROCEDURE HandleLookupUnit@1100485000(lvType@1100485001 : Integer);
    VAR
      ItmUnitRec@1100485000 : Record 5404;
    BEGIN
      //**4PS
      CurrPage.SAVERECORD;
      COMMIT;
      //**due to central itemmanagement and change of subtable (unit/variant/etc) in related company
      //**is started in subform update-trigger of itemtable, calling codeunit central itemmanagement)
      //**this method results in message that record has been changed and data must be introduced again.
      ItmUnitRec.SETRANGE("Item No.", "No.");
      IF PAGE.RUNMODAL(PAGE::"Item Units of Measure", ItmUnitRec) = ACTION::LookupOK THEN BEGIN
        GET("No.");
        CASE lvType OF
          1: VALIDATE("Base Unit of Measure", ItmUnitRec.Code);
          2: VALIDATE("Purch. Unit of Measure", ItmUnitRec.Code);
          3: VALIDATE("Sales Unit of Measure", ItmUnitRec.Code);
        END;
      END;
      CurrPage.UPDATE(TRUE);
    END;
