    PROCEDURE SetText@1100525000(InstallmentMotivationLine@1100525000 : Record 11012472);
    BEGIN
      InstallmentMotivationLine.CALCFIELDS(Finished);
      IF InstallmentMotivationLine.Finished THEN
        FinishedText := 'G'
      ELSE
        FinishedText := '';

      CASE InstallmentMotivationLine."Settlement Type" OF
        InstallmentMotivationLine."Settlement Type"::"Non Settling": SettlementTypeText := 'N';
        InstallmentMotivationLine."Settlement Type"::Settling: SettlementTypeText := 'V';
        InstallmentMotivationLine."Settlement Type"::Approved: SettlementTypeText := 'A';
        InstallmentMotivationLine."Settlement Type"::Reduction: SettlementTypeText := 'M';
        InstallmentMotivationLine."Settlement Type"::Open: SettlementTypeText := 'O';
      ELSE
         SettlementTypeText := '';
      END;
    END;

    BEGIN
    END.
