    PROCEDURE SetFieldsEditable@11012000();
    VAR
      Editable@1210190001 : Boolean;
      PermExtended@1210190002 : Boolean;
      ServCollListInvoiceTextLine@1100525000 : Boolean;
    BEGIN
      //**4PS
      IF SalesHeader.GET("Document Type","Document No.") THEN BEGIN
        Editable := (NOT SalesHeader."Plant Invoice") AND
          (SalesSetup."Generated Invoices Editable" OR
           (NOT SalesHeader."Project Invoice" AND NOT SalesHeader."Service Invoice" AND NOT SalesHeader.ResourceRequestInvoice));
        PermExtended := NOT Editable AND (Type = Type::" ");
        ServCollListInvoiceTextLine := SalesHeader."Service Invoice" AND ("Cost Plus Line No." <> 0) AND (Type = Type::" ");
        FieldsEditable := Editable;

        DescriptionEditable := (Editable OR PermExtended OR ServCollListInvoiceTextLine);
        "No.Editable" := (Editable OR PermExtended);
        IF ("Job No." <> '') OR ("Service Order No." <> '') OR ("Plant Type" <> '') OR ("Loan Code" <> '') THEN
          "No.Editable" := FALSE;
        "Unit Cost (LCY)Editable" := (Editable OR PermExtended);
        LoanIsEditable := Editable AND ("Job No." = '') AND ("Service Contract No." = '') AND ("Service Order No." = '') AND ("Plant Type" = '');
        VATGroupsEditable := SalesSetup."Generated Invoices Editable" OR
           (NOT SalesHeader."Project Invoice" AND NOT SalesHeader."Service Invoice" AND NOT SalesHeader.ResourceRequestInvoice);
      END;
    END;
