    PROCEDURE EvaluateVendorCategory@1100528901();
    VAR
      Vendor@1100528901 : Record 23;
      VendorCategory@1100528902 : Record 11071892;
    BEGIN
      //**4PS
      MARKEDONLY(FALSE);
      CLEARMARKS;
      IF CodingSystem + VendorCategoryFilter + VendorQualificationCodeFilter <> '' THEN BEGIN

        IF CodingSystem <> '' THEN
          VendorCategory.SETRANGE("Coding System", CodingSystem);
        IF VendorCategoryFilter <> '' THEN
          VendorCategory.SETFILTER(Code, VendorCategoryFilter);
        IF VendorQualificationCodeFilter <> '' THEN
          VendorCategory.SETFILTER("Vendor Qualification Code", VendorQualificationCodeFilter);

        Vendor := Rec;
        Vendor.COPYFILTERS(Rec);
        RESET;
        IF FINDSET THEN
          REPEAT
            VendorCategory.SETRANGE("Vendor No.", "No.");
            IF NOT VendorCategory.ISEMPTY THEN
              MARK := TRUE;
          UNTIL NEXT = 0;
        COPYFILTERS(Vendor);
        MARKEDONLY(TRUE);
      END;
      CurrPage.UPDATE(FALSE);
    END;

    PROCEDURE GetVendCategoryAndQualificByVendCategory@1100529101();
    VAR
      VendorCategory@1100528902 : Record 11071892;
    BEGIN
      //**4PS
      CLEAR(VendorCategoryField);
      CLEAR(VendorCategoryDescription);
      CLEAR(VendorQualifByVendCategoryField);
      VendorCategory.SETRANGE("Vendor No.", "No.");
      IF CodingSystem <> '' THEN
        VendorCategory.SETRANGE("Coding System", CodingSystem);
      IF VendorCategoryFilter <> '' THEN
        VendorCategory.SETFILTER(Code, VendorCategoryFilter);
      IF VendorCategory.FINDFIRST THEN BEGIN
        IF VendorCategory.COUNT = 1 THEN BEGIN
          VendorCategory.CALCFIELDS(Description);
          VendorCategoryField := VendorCategory.Code;
          VendorCategoryDescription := VendorCategory.Description;
        END ELSE BEGIN
          VendorCategoryField := '[' + FORMAT(VendorCategory.COUNT) + ']';
          VendorCategoryDescription := '[' + FORMAT(VendorCategory.COUNT) + ']';
        END;
      END;

      IF VendorQualificationCodeFilter <> '' THEN
        VendorCategory.SETFILTER("Vendor Qualification Code", VendorQualificationCodeFilter);
      IF VendorCategory.FINDFIRST THEN BEGIN
        IF VendorCategory.COUNT = 1 THEN BEGIN
          VendorQualifByVendCategoryField := VendorCategory."Vendor Qualification Code";
        END ELSE BEGIN
          VendorQualifByVendCategoryField := '[' + FORMAT(VendorCategory.COUNT) + ']';
        END;
      END;
    END;
