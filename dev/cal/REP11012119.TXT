    PROCEDURE FillTmpDimensionValueTable@1210190000();
    VAR
      DimensionValue@1210190000 : Record 349;
      Company2@1210190001 : Record 2000000006;
    BEGIN
      TempDimensionValue.RESET;
      TempDimensionValue.DELETEALL;

      DimensionValue.SETFILTER(Code, "G/L Account".GETFILTER("Global Dimension 1 Filter"));
      DimensionValue.SETRANGE("Global Dimension No.", 1);
      IF DimensionValue.FINDSET THEN BEGIN
        REPEAT
          TempDimensionValue := DimensionValue;
          TempDimensionValue."Dimension Code" := '';
          TempDimensionValue.INSERT;
        UNTIL DimensionValue.NEXT = 0;
      END;

      Company2.COPY(Company);
      Company2.FILTERGROUP(6);
      Company2.SETFILTER(Name, '<>%1', COMPANYNAME);
      Company2.FILTERGROUP(0);
      IF Company2.FINDSET THEN BEGIN
        REPEAT
          DimensionValue.CHANGECOMPANY(Company2.Name);
          DimensionValue.SETFILTER(Code, "G/L Account".GETFILTER("Global Dimension 1 Filter"));
          DimensionValue.SETRANGE("Global Dimension No.", 1);
          IF DimensionValue.FINDSET THEN BEGIN
            REPEAT
              IF NOT TempDimensionValue.GET('', DimensionValue.Code) THEN BEGIN
                TempDimensionValue := DimensionValue;
                TempDimensionValue."Dimension Code" := '';
                TempDimensionValue.INSERT;
              END;
            UNTIL DimensionValue.NEXT = 0;
          END;
        UNTIL Company2.NEXT = 0;
      END;

      IF NOT TempDimensionValue.GET('', '') THEN BEGIN
        TempDimensionValue.INIT;
        TempDimensionValue."Dimension Code" := '';
        TempDimensionValue.Code := '';
        TempDimensionValue.Name := Text001 + ' ' + Text002;
        TempDimensionValue.INSERT;
        TempDimensionValue.SETFILTER(Code, "G/L Account".GETFILTER("Global Dimension 1 Filter"));
        IF TempDimensionValue.FINDFIRST THEN BEGIN
          IF (TempDimensionValue.Code <> '') THEN BEGIN
            TempDimensionValue.GET('', '');
            TempDimensionValue.DELETE;
          END;
        END ELSE BEGIN
          TempDimensionValue.GET('', '');
          TempDimensionValue.DELETE;
        END;
      END;
      TempDimensionValue.RESET;
    END;
