    PROCEDURE SetEditable@1100525003();

    BEGIN

          (NOT PurchasesPayablesSetup."Approve Purchase Invoices")));

        ShortcutDimension2CodeEditable := TRUE;

          ("Plant Type" = '') AND

          (NOT PurchasesPayablesSetup."Approve Purchase Invoices")));
