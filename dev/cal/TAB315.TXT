    PROCEDURE CheckIfCodeIsAccordingToFormat@1100528401(Code@11012005 : Code[20];FormatCode@1100528401 : Code[20];ObjectCaption@1100528400 : Text[100]);
    VAR
      x1@1000000001 : Integer;
      x2@1000000002 : Integer;
      TempCode1@11012007 : Code[20];
      TempCode2@11012006 : Code[20];
    BEGIN
      //**4PS
      IF STRLEN(Code) <> STRLEN(FormatCode) THEN
        ERROR(Text11012010, ObjectCaption, Code);
      TempCode1 := Code;
      TempCode2 := FormatCode;
      REPEAT
        x1 := STRPOS(TempCode1, '.');
        x2 := STRPOS(TempCode2, '.');
        IF x1 <> x2 THEN
          ERROR(Text11012010, ObjectCaption, Code);
        IF x1 <> 0 THEN
          TempCode1 := DELSTR(TempCode1, x1, 1);
        IF x2 <> 0 THEN
          TempCode2 := DELSTR(TempCode2, x2, 1);
      UNTIL (x1 = 0) OR (x2 = 0);
    END;
