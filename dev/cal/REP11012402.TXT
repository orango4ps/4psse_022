    LOCAL PROCEDURE CheckAllNotWorkedHours@1100528301();
    BEGIN
      IF (("Frost Out Request Line"."Hours Not Worked Monday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Monday" < 5)) THEN
        CreateMessageNotWorked(1, "Frost Out Request Line"."Hours Not Worked Monday");
      IF (("Frost Out Request Line"."Hours Not Worked Tuesday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Tuesday" < 5)) THEN
        CreateMessageNotWorked(2, "Frost Out Request Line"."Hours Not Worked Tuesday");
      IF (("Frost Out Request Line"."Hours Not Worked Wednesday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Wednesday" < 5)) THEN
        CreateMessageNotWorked(3, "Frost Out Request Line"."Hours Not Worked Wednesday");
      IF (("Frost Out Request Line"."Hours Not Worked Thursday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Thursday" < 5)) THEN
        CreateMessageNotWorked(4, "Frost Out Request Line"."Hours Not Worked Thursday");
      IF (("Frost Out Request Line"."Hours Not Worked Friday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Friday" < 5)) THEN
        CreateMessageNotWorked(5, "Frost Out Request Line"."Hours Not Worked Friday");
      IF (("Frost Out Request Line"."Hours Not Worked Saturday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Saturday" < 5)) THEN
        CreateMessageNotWorked(6, "Frost Out Request Line"."Hours Not Worked Saturday");
      IF (("Frost Out Request Line"."Hours Not Worked Sunday" > 0) AND
          ("Frost Out Request Line"."Hours Not Worked Sunday" < 5)) THEN
        CreateMessageNotWorked(7, "Frost Out Request Line"."Hours Not Worked Sunday");
    END;

    LOCAL PROCEDURE CheckAllWageComponents@1100528320();
    VAR
      PostedHourLine@1100528300 : Record 11012085;
      WageComponent@1100528301 : Record 11012014;
      HoursArr@1100528302 : ARRAY [7] OF Decimal;
    BEGIN
      PostedHourLine.SETRANGE(Year, "Frost Out Request Line".Year);
      PostedHourLine.SETRANGE(Week, "Frost Out Request Line".Week);
      PostedHourLine.SETRANGE("Employee No.", "Frost Out Request Line"."Employee No.");
      PostedHourLine.SETRANGE(Overtime, FALSE);
      PostedHourLine.SETRANGE("Component Type", PostedHourLine."Component Type"::Hours);
      IF (PostedHourLine.FINDSET) THEN BEGIN
        REPEAT
          WageComponent.GET(PostedHourLine."Wage Component");
          IF (WageComponent."Frost Delay") THEN BEGIN
            CheckWageComponent(PostedHourLine, HoursArr);
          END;
        UNTIL(PostedHourLine.NEXT = 0);

        CheckHourAccountingHours(HoursArr);
        CheckHourAccRegistrations(HoursArr);
      END;
    END;

          END;
        END;
      END;

      FrostOutRegistration@1100528302 : Record 11020670;

      CheckDate@1100528303 : Date;
    BEGIN
      FOR Counter := 1 TO 7 DO BEGIN
        CheckDate := DWY2DATE(Counter, "Frost Out Request Line".Week, "Frost Out Request Line".Year);

        IF (HoursArr[Counter] <> 0) THEN
          FrostOutRegistration.SETRANGE("Employee No.", "Frost Out Request Line"."Employee No.");
          FrostOutRegistration.SETFILTER("Register Date", '<=%1', CheckDate);
          FrostOutRegistration.SETFILTER("Unregister Date", '%1|>=%2', 0D, CheckDate);

            AddMessage(STRSUBSTNO(Text005,
                "Frost Out Request Line".FIELDCAPTION("Employee No."),
                "Frost Out Request Line"."Employee No.",
                GetWeekDay(Counter),
                "Frost Out Request Line".FIELDCAPTION(Year),
                "Frost Out Request Line".Year,
                "Frost Out Request Line".FIELDCAPTION(Week),
                "Frost Out Request Line".Week,
                FrostOutRegistration.TABLECAPTION),
              'Error-1', 1, HoursArr[Counter]);
      END;
    END;
