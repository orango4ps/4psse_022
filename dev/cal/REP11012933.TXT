    PROCEDURE BuildDebriefHours@1100525017(ResourceNo@1100525007 : Code[20]);
    VAR
      CompanyPlanGroup@1100528701 : Record 11229300;
      CompanyPlanGroupLine@1100528702 : Record 11229301;
      Company@1100528700 : Record 2000000006;
      DebriefFSStatusMobile@1100525009 : Record 11012683;
      DebriefFSStatusMobile2@1100525008 : Record 11012683;
      WorkOrder@1100528704 : Record 11229279;
      Date1@1100525006 : Date;
      Date2@1100525005 : Date;
      Travel@1100525004 : Boolean;
      Found@1100525003 : Boolean;
      TableNo@1100525002 : Integer;
      Hours1@1100525001 : Decimal;
      Hours2@1100525000 : Decimal;
      CompanyPlanGroupCode@1100528703 : Code[20];
      ServiceOrderNo@1100528705 : Code[20];
      ProjectNo@1100528706 : Code[20];
    BEGIN
      //* Travel hours : Time between status 50 and 60 or 90 (50=Start travel, 60=Start work / 90=Break)
      //* Working hours: Time between status 60 and 65 or 70 or 90 (60=Start work, 65=Finished but followup work needed
      //*                                                       70=Technical finished / 90=Break)
      CompanyPlanGroupCode := CompanyPlanGroup.GetCurrentCompanyPlanGroupCode;
      IF (CompanyPlanGroupCode <> '') THEN BEGIN
        CompanyPlanGroupLine.SETRANGE("Group Code", CompanyPlanGroupCode);
        IF CompanyPlanGroupLine.FINDSET THEN BEGIN
          REPEAT
            Company.GET(CompanyPlanGroupLine."Company Name");

            TableNo := DATABASE::"Debrief FS Status (Mobile)";

            DebriefFSStatusMobile2.CHANGECOMPANY(Company.Name);
            DebriefFSStatusMobile2.SETCURRENTKEY("Resource No. (Employee)", "Date/Time");
            DebriefFSStatusMobile2.SETRANGE("Resource No. (Employee)", ResourceNo);  //db, 15-09-14: C016845

            DebriefFSStatusMobile.CHANGECOMPANY(Company.Name);
            DebriefFSStatusMobile.SETCURRENTKEY("Resource No. (Employee)", "Date/Time");
            DebriefFSStatusMobile.SETRANGE("Resource No. (Employee)", ResourceNo);  //db, 15-09-14: C016845
            DebriefFSStatusMobile.SETRANGE("Date/Time", CREATEDATETIME(StartingDate-1,0T), CREATEDATETIME(EndingDate+1,0T)-1);
            DebriefFSStatusMobile.SETFILTER("Field Service Status", '%1|%2', '50', '60');
            DebriefFSStatusMobile.SETFILTER("Work Order No.", '<>%1', '');
            IF DebriefFSStatusMobile.FINDSET(FALSE, FALSE) THEN BEGIN
              REPEAT
                DebriefFSStatusMobile2 := DebriefFSStatusMobile;
                DebriefFSStatusMobile2.SETRANGE("Field Service Status");
                DebriefFSStatusMobile2.SETRANGE("Work Order No.");
                DebriefFSStatusMobile2.FIND;
                IF MobileSetup."Daysheet Presentation 2" IN
                  [MobileSetup."Daysheet Presentation 2"::DaySheetHours,
                   MobileSetup."Daysheet Presentation 2"::WeekSheetHours,
                   MobileSetup."Daysheet Presentation 2"::WeekDaySheetHours] THEN
                  DebriefFSStatusMobile2.SETFILTER("Field Service Status", '%1|%2|%3|%4', '60', '65', '70', '90')
                ELSE
                  DebriefFSStatusMobile2.SETFILTER("Field Service Status", '%1|%2|%3', '60', '90', '100');
                DebriefFSStatusMobile2.SETRANGE("Work Order No.", DebriefFSStatusMobile."Work Order No.");
                IF DebriefFSStatusMobile2.NEXT <> 0 THEN BEGIN
                  Found := FALSE;
                  CASE DebriefFSStatusMobile."Field Service Status" OF
                    '50':
                      BEGIN
                        IF DebriefFSStatusMobile2."Field Service Status" IN ['60', '90'] THEN BEGIN
                          Found := TRUE;
                          Travel := TRUE;
                        END;
                      END;
                    '60':
                      BEGIN
                        IF MobileSetup."Daysheet Presentation 2" IN
                          [MobileSetup."Daysheet Presentation 2"::DaySheetHours,
                           MobileSetup."Daysheet Presentation 2"::WeekSheetHours,
                           MobileSetup."Daysheet Presentation 2"::WeekDaySheetHours] THEN BEGIN
                          IF DebriefFSStatusMobile2."Field Service Status" IN ['65', '70', '90'] THEN BEGIN
                            Found := TRUE;
                            Travel := FALSE;
                          END;
                        END ELSE BEGIN
                          IF DebriefFSStatusMobile2."Field Service Status" IN ['90', '100'] THEN BEGIN
                            Found := TRUE;
                            Travel := FALSE;
                          END;
                        END;
                      END;
                  END;
                  IF Found THEN BEGIN
                    IF NOT WorkOrder.GET(DebriefFSStatusMobile."Work Order No.") THEN WorkOrder.INIT;  //db, 15-09-14: C016845
                    ServiceOrderNo := '';
                    ProjectNo := '';
                    IF WorkOrder."Source Type" = WorkOrder."Source Type"::ServiceOrder THEN
                      ServiceOrderNo := WorkOrder."Source No.";
                    IF WorkOrder."Source Type" = WorkOrder."Source Type"::Project THEN
                      ProjectNo := WorkOrder."Source No.";
                    CalcHrsMobile(DebriefFSStatusMobile."Date/Time", DebriefFSStatusMobile2."Date/Time", Date1, Hours1, Date2, Hours2);
                    IF (Date1 <> 0D) AND (Hours1 > 0) THEN
                      WriteSortTable(TableNo, WorkOrder."Source Company", ServiceOrderNo, ProjectNo, Date1, Hours1, Travel);  //db, 15-09-14: C016845
                    IF (Date2 <> 0D) AND (Hours2 > 0) THEN
                      WriteSortTable(TableNo, WorkOrder."Source Company", ServiceOrderNo, ProjectNo, Date2, Hours2, Travel);  //db, 15-09-14: C016845
                  END;
                END;
              UNTIL DebriefFSStatusMobile.NEXT = 0;
            END;
          UNTIL CompanyPlanGroupLine.NEXT = 0;
        END;
      END;
    END;

    PROCEDURE BuildPostedHrsAcc@1100525021(EmployeeNo@1100525000 : Code[20]);
    VAR
      PostedHourLine@1100525004 : Record 11012085;
      Date@1100525003 : Record 2000000007;
      Travel@1100525002 : Boolean;
      TableNo@1100525001 : Integer;
      SourceCompany@1100528700 : Text;
    BEGIN
      TableNo := DATABASE::"Posted Hour Line";

      Date.SETRANGE("Period Type", Date."Period Type"::Week);
      Date.SETRANGE("Period Start", StartingDate, EndingDate-6);
      IF Date.FINDSET(FALSE, FALSE) THEN BEGIN
        REPEAT
          PostedHourLine.CHANGECOMPANY(Resource.Company);
          PostedHourLine.SETRANGE(Year, DATE2DWY(Date."Period Start", 3));
          PostedHourLine.SETRANGE(Week, DATE2DWY(Date."Period Start", 2));
          PostedHourLine.SETRANGE("Employee No.", EmployeeNo);
          PostedHourLine.SETFILTER(Type, '%1|%2', PostedHourLine.Type::Service, PostedHourLine.Type::Project);
          IF PostedHourLine.FINDSET(FALSE, FALSE) THEN BEGIN
            REPEAT
              IF ((PostedHourLine.Type = PostedHourLine.Type::Service) AND (PostedHourLine."Service Order No." <> '')) OR
                 ((PostedHourLine.Type = PostedHourLine.Type::Project) AND (PostedHourLine."Project No." <> '')) THEN BEGIN
                Travel := CheckTravel(PostedHourLine."Cost Object");
                IF PostedHourLine."Receiving Company" = '' THEN
                  SourceCompany := PostedHourLine.CURRENTCOMPANY
                ELSE
                  SourceCompany := PostedHourLine."Receiving Company";
                IF NOT PrintDetails THEN BEGIN
                  IF PostedHourLine."Total Line" <> 0 THEN
                    WriteSortTable(TableNo, SourceCompany, PostedHourLine."Service Order No.", PostedHourLine."Project No.", DWY2DATE(1, PostedHourLine.Week, PostedHourLine.Year), PostedHourLine."Total Line", Travel);
                END ELSE BEGIN
                  IF PostedHourLine.Monday <> 0 THEN
                    WriteSortTable(TableNo, SourceCompany, PostedHourLine."Service Order No.", PostedHourLine."Project No.", DWY2DATE(1, PostedHourLine.Week, PostedHourLine.Year), PostedHourLine.Monday, Travel);
                  IF PostedHourLine.Tuesday <> 0 THEN
                    WriteSortTable(TableNo, SourceCompany, PostedHourLine."Service Order No.", PostedHourLine."Project No.", DWY2DATE(2, PostedHourLine.Week, PostedHourLine.Year), PostedHourLine.Tuesday, Travel);
                  IF PostedHourLine.Wednesday <> 0 THEN

    PROCEDURE WriteSortTable@1100525002(TableNo@1100525004 : Integer;SourceCompany@1100528700 : Text;ServiceOrderNo@1100525001 : Code[20];ProjectNo@1100528703 : Code[10];SortDate@1100525002 : Date;Quantity@1100525003 : Decimal;Travel@1100525016 : Boolean);
    VAR
      ServiceOrder@1210190000 : Record 11012823;
      ServiceOrderIndication@1210190001 : Record 11012837;
      SortDate2@1100525005 : Date;
      WeekDayNo@1100525015 : Integer;
      Quan1@1100525009 : Decimal;
      Quan2@1100525014 : Decimal;
      Quan3@1100525013 : Decimal;
      Quan4@1100525012 : Decimal;

      Quan6@1100525010 : Decimal;

    BEGIN
      IF Quantity = 0 THEN
        EXIT;

      IF NOT ServiceOrder.GET(ServiceOrderNo) THEN
        ServiceOrder.INIT;
      CASE PrintBy OF
        PrintBy::Employee:
          SortCode1 := TempEmployee."No.";
        PrintBy::Department:
          SortCode1 := TempEmployee."Global Dimension 1 Code";
        PrintBy::OrderKind:
          SortCode1 := ServiceOrder."Order Kind";
        PrintBy::OrderType:
          SortCode1 := ServiceOrder."Order Type";
      END;

      IF PrintBy IN [PrintBy::OrderKind, PrintBy::OrderType] THEN BEGIN
        IF NOT TempServiceOrderIndication.GET(0, SortCode1) THEN BEGIN
          TempServiceOrderIndication.INIT;
          TempServiceOrderIndication.Indication := 0;
          TempServiceOrderIndication.Code := SortCode1;
          ServiceOrderIndication.CHANGECOMPANY(SourceCompany);
          IF ServiceOrderIndication.GET(PrintBy - 2, SortCode1) THEN
            TempServiceOrderIndication.Description := ServiceOrderIndication.Description;
          TempServiceOrderIndication.INSERT;
        END;
      END;

      IF PrintDetails THEN BEGIN
        SortCode2 := ServiceOrderNo + ProjectNo;
        SortDate2 := SortDate;
      END ELSE BEGIN
        SortCode2 := '';
        WeekDayNo := DATE2DWY(SortDate, 1);
        SortDate2 := SortDate - (WeekDayNo - 1);
      END;

      CASE TableNo OF
        DATABASE::"Debrief FS Status (Mobile)":
          IF NOT Travel THEN
            Quan1 := Quantity
          ELSE
            Quan4 := Quantity;
        DATABASE::"Service Order Cost Plus Entry":
          IF NOT Travel THEN
            Quan2 := Quantity
          ELSE
            Quan5 := Quantity;
        DATABASE::"Hour Accounting Line",
        DATABASE::"Posted Hour Line" :
          IF NOT Travel THEN
            Quan3 := Quantity
          ELSE
            Quan6 := Quantity;
      END;

      IF NOT TempSortingTableHoursMobile.GET(SortCode1, SourceCompany, SortCode2, SortDate2) THEN BEGIN
        TempSortingTableHoursMobile.INIT;
        TempSortingTableHoursMobile."Sorting Code 1" := SortCode1;
        TempSortingTableHoursMobile.Company := SourceCompany;
        TempSortingTableHoursMobile."Sorting Code 2" := SortCode2;
        TempSortingTableHoursMobile."Sorting Date" := SortDate2;
        TempSortingTableHoursMobile."Quantity 1" := Quan1;
        TempSortingTableHoursMobile."Quantity 2" := Quan2;
        TempSortingTableHoursMobile."Quantity 3" := Quan3;
        TempSortingTableHoursMobile."Quantity 4" := Quan4;
        TempSortingTableHoursMobile."Quantity 5" := Quan5;
        TempSortingTableHoursMobile."Quantity 6" := Quan6;
        TempSortingTableHoursMobile."Sorting Year" := DATE2DWY(SortDate, 3);
        TempSortingTableHoursMobile."Sorting Period" := DATE2DWY(SortDate, 2);
        TempSortingTableHoursMobile.INSERT;
      END ELSE BEGIN
        TempSortingTableHoursMobile."Quantity 1" := TempSortingTableHoursMobile."Quantity 1" + Quan1;
        TempSortingTableHoursMobile."Quantity 2" := TempSortingTableHoursMobile."Quantity 2" + Quan2;
        TempSortingTableHoursMobile."Quantity 3" := TempSortingTableHoursMobile."Quantity 3" + Quan3;
        TempSortingTableHoursMobile."Quantity 4" := TempSortingTableHoursMobile."Quantity 4" + Quan4;
        TempSortingTableHoursMobile."Quantity 5" := TempSortingTableHoursMobile."Quantity 5" + Quan5;
        TempSortingTableHoursMobile."Quantity 6" := TempSortingTableHoursMobile."Quantity 6" + Quan6;
        IF (TempSortingTableHoursMobile."Quantity 1" = 0) AND (TempSortingTableHoursMobile."Quantity 2" = 0) AND (TempSortingTableHoursMobile."Quantity 3" = 0) AND
           (TempSortingTableHoursMobile."Quantity 4" = 0) AND (TempSortingTableHoursMobile."Quantity 5" = 0) AND (TempSortingTableHoursMobile."Quantity 6" = 0)
        THEN
          TempSortingTableHoursMobile.DELETE
        ELSE
          TempSortingTableHoursMobile.MODIFY;
      END;
    END;

    PROCEDURE CalcHrsMobile@1100525004(StartDateTime@1100525000 : DateTime;EndDateTime@1100525001 : DateTime;VAR Date1@1100525003 : Date;VAR Hours1@1100525002 : Decimal;VAR Date2@1100525004 : Date;VAR Hours2@1100525005 : Decimal);
    VAR
      lvDate1@1100525009 : Date;
      lvDate2@1100525008 : Date;
      lvHours@1100525007 : Decimal;
    BEGIN
      Date1 := 0D;
      Date2 := 0D;
      Hours1 := 0;
      Hours2 := 0;

      lvDate1 := DT2DATE(StartDateTime);
      lvDate2 := DT2DATE(EndDateTime);
      IF (lvDate1 = lvDate2) THEN BEGIN
        IF (EndDateTime > StartDateTime) AND (lvDate1 >= StartingDate) AND (lvDate1 <= EndingDate) THEN BEGIN
          lvHours := (EndDateTime - StartDateTime) / (1000*60*60);
          lvHours := ROUND(lvHours, 0.01);
          IF lvHours > 0 THEN BEGIN
            Date1 := lvDate1;
            Hours1 := lvHours;
          END;
        END;
      END ELSE BEGIN
        IF (lvDate1 = lvDate2 - 1) THEN BEGIN
          IF (lvDate1 >= StartingDate) AND (lvDate1 <= EndingDate) THEN BEGIN
            lvHours := (CREATEDATETIME(lvDate2,0T) - StartDateTime) / (1000*60*60);
            lvHours := ROUND(lvHours, 0.01);
            IF lvHours > 0 THEN BEGIN
              Date1 := lvDate1;
              Hours1 := lvHours;
            END;
          END;
          IF (lvDate2 >= StartingDate) AND (lvDate2 <= EndingDate) THEN BEGIN
            lvHours := (EndDateTime - CREATEDATETIME(lvDate2,0T)) / (1000*60*60);
            lvHours := ROUND(lvHours, 0.01);
            IF lvHours > 0 THEN BEGIN
              Date2 := lvDate2;
              Hours2 := lvHours;
            END;
          END;
        END;
      END;
    END;
