    PROCEDURE Explode@3(Verify@11012000 : Boolean;SelectedLines@1100529000 : Boolean;VAR SelectedOrderLine@1100529001 : Record 11012557);
    VAR
      lvPlantOrderCostMgtCU@1100485001 : Codeunit 11012581;
      lvQuan@1100525000 : Decimal;
    BEGIN
      PlantSetup.GET;
      IF Verify THEN BEGIN
        TESTFIELD(Posted,FALSE);
        IF Status = Status::Released THEN
          FIELDERROR(Status);
      END ELSE
        IF Posted OR (Status >= Status::Released) THEN
          EXIT;

      IF Verify AND GUIALLOWED THEN
        IF NOT CONFIRM(Text002) THEN
          EXIT;

      OrderRec.COPY(Rec);

      OrderLineRec.RESET;
      IF SelectedLines THEN
        OrderLineRec.COPY(SelectedOrderLine);
      OrderLineRec.SETRANGE("Plant Order No.",OrderRec."No.");
      OrderLineRec.SETRANGE(Type, OrderLineRec.Type::Plant);
      OrderLineRec.SETRANGE("No.",'');
      IF NOT (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN
        OrderLineRec.SETFILTER(Quantity,'>1');
      OrderLineRec.SETRANGE("Set Code", '');
      OrderLineRec.SETRANGE(Bulk, FALSE);
      IF OrderLineRec.FINDSET(TRUE, FALSE) THEN BEGIN
        REPEAT
          lvQuan := OrderLineRec.Quantity;
          IF PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival) THEN BEGIN
            IF (OrderLineRec."Asked Quantity" > lvQuan) THEN
              lvQuan := OrderLineRec."Asked Quantity";
          END ELSE BEGIN
            IF (OrderLineRec."Asked Quantity" <> 0) THEN
              OrderLineRec."Asked Quantity" := 0;
          END;
          IF lvQuan > 1 THEN BEGIN
            //Test whole amount
            IF (lvQuan MOD 1) <> 0 THEN
              ERROR(
                Text003,
                OrderLineRec."Plant Type", OrderLineRec."No.");

            //Determine Line Spacing
            OrderLineRec2 := OrderLineRec;
            OrderLineRec2.SETRANGE("Plant Order No.",OrderRec."No.");
            IF OrderLineRec2.FIND('>') THEN BEGIN
              LineSpacing := (OrderLineRec2."Line No." - OrderLineRec."Line No.") DIV lvQuan;
              IF LineSpacing = 0 THEN
                ERROR(Text004);
            END ELSE
              LineSpacing := 10000;

            //Add Lines;
            NextLineNo := OrderLineRec."Line No.";
            FOR Counter := 1 TO (lvQuan - 1) DO BEGIN
              NextLineNo := NextLineNo + LineSpacing;
              OrderLineRec2 := OrderLineRec;
              OrderLineRec2."Line No." := NextLineNo;
              OrderLineRec2."Asked Quantity" := 0;
              OrderLineRec2.Quantity := 0;
              IF PlantSetup."Extended Picking Procedure" AND (OrderRec.Type = Type::Arrival) AND
                 (OrderLineRec."Asked Quantity" > 1)
              THEN BEGIN
                OrderLineRec2."Asked Quantity" := 1;
                OrderLineRec."Asked Quantity" := OrderLineRec."Asked Quantity" - 1;
              END;
              IF (OrderLineRec.Quantity > 1) THEN BEGIN
                OrderLineRec2.Quantity := 1;
                OrderLineRec.Quantity := OrderLineRec.Quantity - 1;
              END;
              IF (OrderRec.Type = Type::Removal) AND (OrderLineRec."Qty. to Receive" > 1) THEN BEGIN
                OrderLineRec."Qty. to Receive" := OrderLineRec."Qty. to Receive" - 1;
                OrderLineRec2."Qty. to Receive" := 1;
              END;
              OrderLineRec2."Quantity Inventory" := 0;
              OrderLineRec2."Quantity Damaged" := 0;
              OrderLineRec2."Quantity Exit" := 0;
              OrderLineRec2.INSERT;
              lvPlantOrderCostMgtCU.RunUpdatePOLineLinkedCostLine('I',OrderRec,OrderLineRec2);
            END;
            OrderLineRec.MODIFY;
            lvPlantOrderCostMgtCU.RunUpdatePOLineLinkedCostLine('I',OrderRec,OrderLineRec);
          END;
        UNTIL OrderLineRec.NEXT = 0;
      END;
    END;

    PROCEDURE ExplodeSet@1100485000(IVerify@1100485005 : Boolean;SelectedLines@1100529001 : Boolean;VAR SelectedOrderLine@1100529000 : Record 11012557);
    VAR
      lvSetLineRec@1100485002 : Record 11012588;
      lvNewOrderLineRec@1100485003 : Record 11012557;
      PlantToLoc@1210190000 : Record 11012554;
      lvPlantOrderCostMgtCU@1100485010 : Codeunit 11012581;
      lvExplodeUnique@1100485000 : Boolean;
      lvNoOfLinesNeeded@1100485001 : Integer;
      lvQuantity@1100485004 : Decimal;
      lvAskedQuantity@1100485007 : Decimal;
      lvOrderQuantity@1100485006 : Decimal;
      lvQtyToReceive@1100525000 : Decimal;
    BEGIN
      PlantSetup.GET;
      IF IVerify THEN BEGIN
        TESTFIELD(Posted,FALSE);
        IF Status = Status::Released THEN
          FIELDERROR(Status);
      END ELSE
        IF Posted OR (Status >= Status::Released) THEN
          EXIT;

      IF NOT (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN BEGIN
        IF (NOT IVerify) THEN
          lvExplodeUnique := TRUE
        ELSE BEGIN
          CASE STRMENU(Text007, 1) OF
            1: lvExplodeUnique := TRUE;
            2: lvExplodeUnique := FALSE;
            ELSE
              EXIT;
          END;
        END;
      END ELSE BEGIN
        IF IVerify AND GUIALLOWED THEN
          IF NOT CONFIRM(Text009) THEN
            EXIT;
      END;

      PlantToLoc.GET("To Location");

      OrderLineRec.RESET;
      IF SelectedLines THEN
        OrderLineRec.COPY(SelectedOrderLine);
      OrderLineRec.SETRANGE("Plant Order No.", "No.");
      OrderLineRec.SETRANGE("Plant Type", '');
      OrderLineRec.SETRANGE(Type, OrderLineRec.Type::Plant);
      IF NOT (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN
        OrderLineRec.SETFILTER(Quantity, '<>%1', 0)
      ELSE
        OrderLineRec.SETFILTER("Asked Quantity", '<>%1', 0);
      OrderLineRec.SETFILTER("Set Code", '<>%1', '');
      IF OrderLineRec.FINDSET THEN BEGIN
        REPEAT
          OrderLineRec2 := OrderLineRec;  //* Save record
          IF NOT (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN
            lvOrderQuantity := OrderLineRec.Quantity
          ELSE
            lvOrderQuantity := OrderLineRec."Asked Quantity";

          //* First insert the cost lines of a set (before OrderLineRec is modified/deleted).
          lvPlantOrderCostMgtCU.RunInsertPOStandCostlinesSet(Rec, OrderLineRec, lvOrderQuantity);

          //* Plant Lines
          lvSetLineRec.RESET;
          lvSetLineRec.SETRANGE("Set Code", OrderLineRec."Set Code");
          //lvSetLineRec.SETRANGE(Type, lvSetLineRec.Type::Plant); //Also items
          lvSetLineRec.SETFILTER(Quantity, '<>%1', 0);
          IF lvSetLineRec.FINDSET THEN BEGIN
            lvNoOfLinesNeeded := 0;
            REPEAT
              lvSetLineRec.CALCFIELDS(Bulk);
              IF (NOT lvSetLineRec.Bulk) AND (lvSetLineRec."Plant No." = '') AND (lvSetLineRec.Type = lvSetLineRec.Type::Plant) THEN
                lvNoOfLinesNeeded := lvNoOfLinesNeeded + ROUND(lvSetLineRec.Quantity * lvOrderQuantity, 1)
              ELSE
                lvNoOfLinesNeeded := lvNoOfLinesNeeded + 1;
            UNTIL lvSetLineRec.NEXT = 0;

            //Determine Line Spacing
            OrderLineRec3.RESET;
            OrderLineRec3.SETRANGE("Plant Order No.", OrderLineRec."Plant Order No.");
            OrderLineRec3."Plant Order No." := OrderLineRec."Plant Order No.";
            OrderLineRec3."Line No." := OrderLineRec."Line No.";
            IF OrderLineRec3.FIND('>') THEN
              LineSpacing := (OrderLineRec3."Line No." - OrderLineRec."Line No.") DIV (lvNoOfLinesNeeded)
            ELSE
              LineSpacing := 10000;
            IF LineSpacing <> 0 THEN
              NextLineNo := OrderLineRec."Line No."
            ELSE BEGIN
              LineSpacing := 10000;
              OrderLineRec3.FINDLAST;
              NextLineNo := OrderLineRec3."Line No." + 10000;
            END;

            OrderLineRec.DELETE(TRUE);  //* Delete Plant Order - Set Line

            //Add Lines;
            lvSetLineRec.FIND('-');
            REPEAT
              CASE lvSetLineRec.Type OF
                lvSetLineRec.Type::Plant:
                  BEGIN
                    lvSetLineRec.CALCFIELDS(Bulk);
                    IF lvSetLineRec.Bulk THEN BEGIN
                      lvQuantity := ROUND(lvSetLineRec.Quantity * OrderLineRec2.Quantity, 0.00001);
                      lvAskedQuantity := ROUND(lvSetLineRec.Quantity * OrderLineRec2."Asked Quantity", 0.00001);
                      lvQtyToReceive := ROUND(lvSetLineRec.Quantity * OrderLineRec2."Qty. to Receive", 0.00001);
                    END ELSE BEGIN
                      IF (lvSetLineRec."Plant No." <> '') THEN BEGIN
                        lvQuantity := 1;
                        lvAskedQuantity := 1;
                        IF OrderLineRec2."Qty. to Receive" > 0 THEN
                          lvQtyToReceive := 1;
                      END ELSE BEGIN
                        lvQuantity := ROUND(lvSetLineRec.Quantity * OrderLineRec2.Quantity, 1);
                        lvAskedQuantity := ROUND(lvSetLineRec.Quantity * OrderLineRec2."Asked Quantity", 1);
                        lvQtyToReceive := ROUND(lvSetLineRec.Quantity * OrderLineRec2."Qty. to Receive", 1);
                      END;
                    END;
                    //
                    lvNewOrderLineRec.INIT;
                    lvNewOrderLineRec."Plant Order No." := OrderLineRec2."Plant Order No.";
                    lvNewOrderLineRec."Line No." := NextLineNo;
                    lvNewOrderLineRec.Type := OrderLineRec2.Type::Plant;
                    lvNewOrderLineRec."Plant Type" := lvSetLineRec."Plant Type";
                    IF lvSetLineRec.Bulk THEN
                      lvNewOrderLineRec."No." := '0'
                    ELSE
                      lvNewOrderLineRec."No." := lvSetLineRec."Plant No.";
                    lvNewOrderLineRec.GetPlantDescription();  //* Fill lvNewOrderLineRec.Description
                    lvNewOrderLineRec.VALIDATE("Unit of Measure");  //Fill with Unit of th plant
                    lvNewOrderLineRec."Set Code" := '';
                    lvNewOrderLineRec."Set (Exploded)" := lvSetLineRec."Set Code";
                    lvNewOrderLineRec."Transfer Date" := OrderLineRec2."Transfer Date";
                    lvNewOrderLineRec."From Location" := OrderLineRec2."From Location";
                    //lvNewOrderLineRec."To Rate Code" := OrderLineRec2."To Rate Code";
                    IF OrderLineRec2."To Rate Code" <> '' THEN  //*23814
                      lvNewOrderLineRec."To Rate Code" := OrderLineRec2."To Rate Code"
                    ELSE BEGIN
                      IF (lvSetLineRec."Default Rate Code" <> '') AND
                         (Type IN [Type::Arrival, Type::"Other Transfers"]) AND (NOT PlantToLoc.Depot)
                      THEN
                        lvNewOrderLineRec."To Rate Code" := lvSetLineRec."Default Rate Code";
                    END;
                    lvNewOrderLineRec.GetDefaults();  // Here also 'FillDefaultToRateCode' (only filled when empty) is called,
                                                      // but there also check rate code allowed (if not then made empty).
                    lvNewOrderLineRec.Element := OrderLineRec2.Element;
                    lvNewOrderLineRec."Expected return on" := OrderLineRec2."Expected return on";
                    lvNewOrderLineRec."To Employee Company" := OrderLineRec2."To Employee Company";
                    lvNewOrderLineRec."To Employee No." := OrderLineRec2."To Employee No.";
                    IF NOT (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN
                      lvNewOrderLineRec.Quantity := lvQuantity
                    ELSE BEGIN
                      IF OrderLineRec2.Quantity > 0 THEN
                        lvNewOrderLineRec.Quantity := lvQuantity;
                      lvNewOrderLineRec."Asked Quantity" := lvAskedQuantity;
                    END;
                    IF (Type = Type::Removal) AND (PlantSetup."Removal with Qty. to Receive") THEN
                      lvNewOrderLineRec."Qty. to Receive" := lvQtyToReceive;
            //      lvNewOrderLineRec."Rental Type"  Already filled in GetDefaults, value from order lins is not taken into account ?
                    IF (lvNewOrderLineRec."Rental Type" = OrderLineRec."Rental Type"::Rental) THEN BEGIN
                      lvNewOrderLineRec."Rental Startdate (To Location)" := OrderLineRec2."Rental Startdate (To Location)";
                      lvNewOrderLineRec."Rental Enddate (From Location)" := OrderLineRec2."Rental Enddate (From Location)";
                      lvNewOrderLineRec.VALIDATE("Rental Shift", OrderLineRec2."Rental Shift");
                    END;
                    lvNewOrderLineRec."Quantity Inventory" := 0;
                    lvNewOrderLineRec."Quantity Damaged" := 0;
                    lvNewOrderLineRec."Quantity Exit" := 0;
                    IF (NOT lvSetLineRec.Bulk) AND (lvSetLineRec."Plant No." <> '') THEN
                      lvNewOrderLineRec.CheckInspection();
                    lvNewOrderLineRec."Comment CP Order" := OrderLineRec2."Comment CP Order";
                    lvNewOrderLineRec."Picker No." := OrderLineRec2."Picker No.";
                    lvNewOrderLineRec."Picking Status" := OrderLineRec2."Picking Status";
                    lvNewOrderLineRec.INSERT;
                    lvPlantOrderCostMgtCU.RunUpdatePOLineLinkedCostLine('I',Rec,lvNewOrderLineRec);
                    NextLineNo := NextLineNo + LineSpacing;
                    //
                    IF (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN
                      lvQuantity := lvAskedQuantity;
                    IF (NOT lvSetLineRec.Bulk) AND (lvSetLineRec."Plant No." = '') AND (lvQuantity > 1) THEN BEGIN
                      IF NOT lvExplodeUnique THEN
                          NextLineNo := NextLineNo + ROUND((lvQuantity - 1) * LineSpacing, 1)
                      ELSE BEGIN
                        lvNewOrderLineRec.Quantity := 1;
                        IF lvQtyToReceive >= 1 THEN BEGIN
                          lvNewOrderLineRec."Qty. to Receive" := 1;
                          lvQtyToReceive := lvQtyToReceive - 1;
                        END;
                        lvNewOrderLineRec.MODIFY;
                        lvPlantOrderCostMgtCU.RunUpdatePOLineLinkedCostLine('I',Rec,lvNewOrderLineRec);
                        REPEAT
                          lvQuantity := lvQuantity - 1;
                          lvNewOrderLineRec."Line No." := NextLineNo;
                          lvNewOrderLineRec.Quantity := 1;
                          IF lvQtyToReceive >= 1 THEN BEGIN
                            IF lvQuantity <= 1 THEN BEGIN // IF last of repeat-until (see check on UNTIL)
                              lvNewOrderLineRec."Qty. to Receive" := lvQtyToReceive;
                              lvQtyToReceive := 0;
                            END ELSE BEGIN
                              lvNewOrderLineRec."Qty. to Receive" := 1;
                              lvQtyToReceive := lvQtyToReceive - 1;
                            END;
                          END;
                          lvNewOrderLineRec.INSERT;
                          lvPlantOrderCostMgtCU.RunUpdatePOLineLinkedCostLine('I',Rec,lvNewOrderLineRec);
                          NextLineNo := NextLineNo + LineSpacing;
                        UNTIL lvQuantity <= 1;
                      END;
                    END;
                  END;
                lvSetLineRec.Type::Item:
                  BEGIN
                    //* Item Lines
                    //* Items on plant order exist only in plant extended. Still no testing on 'ReadPermission' here
                    //* because when not in license, then no item lines exist at set.
                    //* When item lines exist in set and the user, who explode the set, has no permission then
                    //* program should give an error, otherwise the items will not exist on PO.
                    lvSetLineRec.CALCFIELDS("Sales Unit of Measure");
                    lvQuantity := ROUND(lvSetLineRec.Quantity * OrderLineRec2.Quantity, 0.00001);
                    lvAskedQuantity := ROUND(lvSetLineRec.Quantity * OrderLineRec2."Asked Quantity", 0.00001);
                    //
                    lvNewOrderLineRec.INIT;
                    lvNewOrderLineRec."Plant Order No." := OrderLineRec2."Plant Order No.";
                    lvNewOrderLineRec."Line No." := NextLineNo;
                    lvNewOrderLineRec.Type := lvNewOrderLineRec.Type::Item;
                    lvNewOrderLineRec."Set Code" := '';
                    lvNewOrderLineRec."Set (Exploded)" := lvSetLineRec."Set Code";
                    lvNewOrderLineRec."Transfer Date" := OrderLineRec2."Transfer Date";
                    lvNewOrderLineRec."From Location" := OrderLineRec2."From Location";
                    lvNewOrderLineRec.VALIDATE("Item No.", lvSetLineRec."Item No.");
                    lvNewOrderLineRec.Element := OrderLineRec2.Element;
                    lvNewOrderLineRec."To Employee Company" := OrderLineRec2."To Employee Company";
                    lvNewOrderLineRec."To Employee No." := OrderLineRec2."To Employee No.";
                    IF NOT (PlantSetup."Extended Picking Procedure" AND (Type = Type::Arrival)) THEN
                      lvNewOrderLineRec.VALIDATE(Quantity, lvQuantity)
                    ELSE BEGIN
                      IF OrderLineRec2.Quantity > 0 THEN
                        lvNewOrderLineRec.VALIDATE(Quantity, lvQuantity);
                      lvNewOrderLineRec.VALIDATE("Asked Quantity", lvAskedQuantity);
                    END;
                    lvNewOrderLineRec."Comment CP Order" := OrderLineRec2."Comment CP Order";
                    lvNewOrderLineRec."Picker No." := OrderLineRec2."Picker No.";
                    lvNewOrderLineRec."Picking Status" := OrderLineRec2."Picking Status";
                    lvNewOrderLineRec.INSERT(TRUE);
                    NextLineNo := NextLineNo + LineSpacing;
                  END;
              END;
            UNTIL lvSetLineRec.NEXT = 0;
          END;
        UNTIL OrderLineRec.NEXT = 0;
      END;
    END;

            END;
          END;
        END;
      END;

        PlantLocRec.Depot := TRUE;
      IF (NOT PlantLocRec.Depot) AND (PlantLocRec."Employee No." = '')  THEN BEGIN
        IF NOT "To Location Invoiced" THEN BEGIN
          IF NOT CONFIRM(STRSUBSTNO(Text005, PlantLocRec.TABLECAPTION,"To Location",TABLECAPTION)) THEN
            EXIT(FALSE);
        END ELSE BEGIN
          IF "To Location Invoice Status" IN
             ["To Location Invoice Status"::Proposal,"To Location Invoice Status"::"Temporary"]
          THEN BEGIN
            IF NOT CONFIRM(STRSUBSTNO(Text006, PlantLocRec.TABLECAPTION,"To Location",TABLECAPTION)) THEN
              EXIT(FALSE);
          END;
        END;
        IF (Type = Type::Arrival) OR (Type = Type::"Other Transfers") THEN BEGIN
          IF NOT "Order Cost Invoiced" THEN BEGIN
            IF NOT CONFIRM(STRSUBSTNO(Text005,PlantLocRec.TABLECAPTION+lvText000,"To Location",TABLECAPTION)) THEN
              EXIT(FALSE);
          END ELSE BEGIN
            IF "Order Cost Invoice Status" IN
               ["Order Cost Invoice Status"::Proposal,"Order Cost Invoice Status"::"Temporary"]
            THEN BEGIN
              IF NOT CONFIRM(STRSUBSTNO(Text006,PlantLocRec.TABLECAPTION+lvText000,"To Location",TABLECAPTION)) THEN
                EXIT(FALSE);
            END;
          END;
        END;
      END;

      EXIT(TRUE);
    END;

    PROCEDURE UpdateOrderlineToLocation@1100485002();
    VAR
      lvPlantTypeRec@1100485000 : Record 11012551;
      lvPlantNoRec@1100525000 : Record 11012552;
      ReturnDepots@1100409001 : Record 11012668;
      ExternalPlant@1100525001 : Boolean;
      DeviatingReturnDepotForPlant@1100525002 : Boolean;
      ReturnDepot@1100409000 : Code[20];
    BEGIN
      OrderLineRec.RESET;
      OrderLineRec.SETRANGE("Plant Order No.", "No.");
      OrderLineRec.SETRANGE(Type, OrderLineRec.Type::Plant);
      IF Type <> Type::Removal THEN BEGIN
        OrderLineRec.SETFILTER("To Location", '<>%1', '');
        OrderLineRec.MODIFYALL("To Location", '');
        EXIT;
      END ELSE BEGIN
        OrderLineRec.SETFILTER("Plant Type", '<>%1', '');
        OrderLineRec.SETFILTER("No.", '<>%1', '');
        IF NOT OrderLineRec.FINDSET(TRUE, FALSE) THEN
          EXIT;
        //DP0152.sn
        PlantSetup.GET;
        IF ("To Location" = '') AND
           (PlantSetup."Deviating Return Depot base on" = PlantSetup."Deviating Return Depot base on"::"Multiple Depots")
        THEN
          EXIT;
        //DP0152.en

    PROCEDURE ReceiptRegistrationTransportAllowed@1100529019(ErrorMode@1100529001 : Boolean) : Boolean;
    VAR
      PlantLocation@1100529000 : Record 11012554;
    BEGIN
      PlantSetup.GET;
      IF ErrorMode THEN BEGIN
        PlantSetup.TESTFIELD("Receipt Registration PO (TR)", TRUE);
        IF NOT (Type IN [Type::Arrival, Type::"Other Transfers"]) THEN
          FIELDERROR(Type);
        IF Type = Type::"Other Transfers" THEN BEGIN  //Only if from Loc1 to Loc2
          TESTFIELD("From Location");
          TESTFIELD("To Location");
          IF "From Location" = "To Location" THEN

          PlantLocation.GET("From Location");
          PlantLocation.TESTFIELD(Depot,FALSE);
          PlantLocation.GET("To Location");
          PlantLocation.TESTFIELD(Depot,FALSE);
        END;
      END ELSE BEGIN
        IF NOT PlantSetup."Receipt Registration PO (TR)" THEN
          EXIT(FALSE);
        IF NOT (Type IN [Type::Arrival, Type::"Other Transfers"]) THEN
          EXIT(FALSE);
        IF Type = Type::"Other Transfers" THEN BEGIN  //Only if from Loc1 to Loc2
          IF ("From Location" = '') OR ("To Location" ='') OR ("From Location" = "To Location") THEN
            EXIT(FALSE);
          IF NOT PlantLocation.GET("From Location") THEN
            EXIT(FALSE);
          IF PlantLocation.Depot THEN
            EXIT(FALSE);
          IF NOT PlantLocation.GET("To Location") THEN
            EXIT(FALSE);
          IF PlantLocation.Depot THEN
            EXIT(FALSE);
        END;
      END;
      EXIT(TRUE);
    END;

    LOCAL PROCEDURE AssignPlantOrdersToRide_SetSoftFilters@1100528501(VAR PlantOrder2@1100528500 : Record 11012556;VAR PlantTransportRide@1100528501 : Record 11020517);
    VAR
      PlantOrder@1100528502 : Record 11012556;
      Warning1@1100528503 : Text;
      Warning2@1100528504 : Text;
    BEGIN
      PlantOrder.COPY(PlantOrder2);
      PlantOrder.FINDFIRST;
      IF PlantOrder.COUNT = 1 THEN BEGIN
        PlantTransportRide.SETRANGE(Date, PlantOrder."Transfer Date");
        IF PlantOrder."Shipping Agent" <> '' THEN
          PlantTransportRide.SETRANGE("Shipping Agent", PlantOrder."Shipping Agent");
      END ELSE BEGIN
        PlantOrder.FILTERGROUP(11);
        PlantOrder.SETFILTER("Transfer Date", '<>%1', PlantOrder."Transfer Date");
        IF PlantOrder.ISEMPTY THEN
          PlantTransportRide.SETRANGE(Date, PlantOrder."Transfer Date")
        ELSE
          Warning1 := STRSUBSTNO(Text027a, PlantTransportRide.FIELDCAPTION(Date));

        PlantOrder.SETRANGE("Transfer Date");
        //
        PlantOrder.SETFILTER("Shipping Agent", '<>%1', '');
        IF PlantOrder.FINDFIRST THEN BEGIN
          PlantOrder.SETFILTER("Shipping Agent", '<>%1&<>%2', '', PlantOrder."Shipping Agent");
          IF PlantOrder.ISEMPTY THEN
            PlantTransportRide.SETRANGE("Shipping Agent", PlantOrder."Shipping Agent")
          ELSE
            Warning2 := STRSUBSTNO(Text027b, PlantTransportRide.FIELDCAPTION("Shipping Agent"));
        END;
        PlantOrder.SETRANGE("Shipping Agent");
        PlantOrder.FILTERGROUP(0);
        IF (Warning1 <> '') OR (Warning2 <> '') THEN
          MESSAGE(Text027 + Warning1 + Warning2, PlantTransportRide.TABLECAPTION);
      END;
    END;
