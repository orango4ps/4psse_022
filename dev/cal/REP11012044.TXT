    PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Payment Agreement".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', "Payment Agreement".TABLECAPTION, "Payment Agreement".GETFILTERS));
    END;

    PROCEDURE VendorLedgerEntryFCYToLCY@1(AmountFCY@11012000 : Decimal) : Decimal;
    VAR
      CurrencyExchangeRate@1100525000 : Record 330;
    BEGIN
      WITH "Vendor Ledger Entry" DO BEGIN
        IF "Currency Code" = '' THEN
          EXIT(AmountFCY);
        EXIT(
          ROUND(
            CurrencyExchangeRate.ExchangeAmtFCYToLCY(
              0, '', "Posting Date", "Currency Code", AmountFCY,
              CurrencyExchangeRate.ExchangeRate(
                0, '', "Posting Date", "Currency Code", FALSE), FALSE)));
      END;
    END;

    PROCEDURE PurchInvHeaderFCYToLCY@2(AmountFCY@11012000 : Decimal) : Decimal;
    VAR
      CurrencyExchangeRate@1100525000 : Record 330;
    BEGIN
      WITH "Purch. Inv. Header" DO BEGIN
        IF "Currency Code" = '' THEN
          EXIT(AmountFCY);
        EXIT(
          ROUND(
            CurrencyExchangeRate.ExchangeAmtFCYToLCY(
              0, '', "Posting Date", "Currency Code", AmountFCY,
      //      "Currency Factor", FALSE), 0));   //DP00727.o
              "Currency Factor", FALSE)));   //DP00727.n
      END;
    END;

    BEGIN
    {
      // Hiding the tablix pur. inv. lines gives problems.
      // When you hide the tablix using the PrintInvoices field the tablix is also not printed when there are no invoice lines and PrintInvoices is True.
      // Therefore the static rows of this tablix are printed always and we do not set the Hidden property of the Tablix.
    }
    END.
