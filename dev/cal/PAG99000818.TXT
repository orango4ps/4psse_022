    VAR
      Item@1000 : Record 27;
    BEGIN
      OnBeforeReserveComp(Rec,xRec);

      IF (xRec."Remaining Qty. (Base)" <> "Remaining Qty. (Base)") OR
         (xRec."Item No." <> "Item No.") OR
         (xRec."Location Code" <> "Location Code")

            CurrPage.SAVERECORD;
            AutoReserve;

          END;
    END;
