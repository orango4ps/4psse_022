    PROCEDURE SetFormFilter@1100285001();
    BEGIN
      ExDoc.RESET;
      ExDoc.SETCURRENTKEY("Document Type","Document No.");

      IF AppSetup."Purchase Order" THEN
        ExDoc.SETFILTER("Document Type",'<=%1', "Document Type"::"Return Order")
      ELSE
        ExDoc.SETFILTER("Document Type",'<>%1&<=%2',"Document Type"::Order,"Document Type"::"Return Order");

      ExDoc.SETFILTER(Status, '<>%1&<>%2', Status::Received, Status::Invoiced);
      ExDoc.SETRANGE("Change Jnl Processed PO", FALSE);
    END;

    LOCAL PROCEDURE GetDocuments@1100285002();
    VAR
      ExUserComp@1100285000 : Record 12013641;
      Company@1100285002 : Record 2000000006;
      WebCompID@1100285005 : Record 12013630;
      ExFlow@1100285003 : Codeunit 12013601;
      AllCompAccess@1100285001 : Boolean;
      Window@1100285004 : Dialog;
    BEGIN
      Window.OPEN(Text000);

      RESET;
      DELETEALL;

      AllCompAccess := FALSE;
      IF ExFlow.GetUserComp(ExUserComp,USERID) THEN
        IF ExUserComp."All-Company Access to Ex Docs" THEN BEGIN
          AllCompAccess := TRUE;
          Company.RESET;
          Company.FINDSET;
          REPEAT
            WebCompID.GET(Company.Name);
            ExDoc.CHANGECOMPANY(Company.Name);
            IF ExDoc.FINDSET THEN
              REPEAT
                Window.UPDATE(1,ExDoc.ID);

                INIT;
                TRANSFERFIELDS(ExDoc);
                ID := ExDoc.ID + WebCompID."Web Company ID";
                "Temp Company Name" := Company.Name;
                INSERT;
              UNTIL ExDoc.NEXT = 0;
          UNTIL Company.NEXT = 0;
        END;

      IF NOT AllCompAccess THEN
        IF ExDoc.FINDSET THEN
          REPEAT
            Window.UPDATE(1,ExDoc.ID);

            INIT;
            TRANSFERFIELDS(ExDoc);
            "Temp Company Name" := COMPANYNAME;
            INSERT;
          UNTIL ExDoc.NEXT = 0;

      Window.CLOSE;
    END;
