    LOCAL PROCEDURE GetReportFilter@1100525000() ReportFilter : Text;
    BEGIN
      IF "Bank Account".GETFILTER("No.") <> '' THEN
        ReportFilter := STRSUBSTNO('%1: %2', "Bank Account".TABLECAPTION, "Bank Account".GETFILTER("No."));
      IF "Bank Account Ledger Entry".GETFILTER("Posting Date") <> '' THEN BEGIN
        IF ReportFilter <> '' THEN
          ReportFilter += ', ';
        ReportFilter += STRSUBSTNO('%1: %2', Text000, "Bank Account Ledger Entry".GETFILTER("Posting Date"));
      END;
    END;

    LOCAL PROCEDURE GetPostingDate@1100527018(PeriodFilter@1100527000 : Text) EndDate : Date;
    BEGIN
      IF STRLEN(DELCHR(PeriodFilter, '=', '.|&')) = STRLEN(PeriodFilter) THEN
        EVALUATE(EndDate, PeriodFilter)
      ELSE
        IF (STRPOS(PeriodFilter, '..') <> 0) AND ((STRPOS(PeriodFilter, '..') + 2) < STRLEN(PeriodFilter)) THEN
          EVALUATE(EndDate, COPYSTR(PeriodFilter, STRPOS(PeriodFilter, '..') + 2));
    END;

    BEGIN
    END.
