    LOCAL PROCEDURE CheckDescriptions@1100485000();
    BEGIN
      WITH "Service Order Budget Line" DO BEGIN
        IF "Service Order No." <> ServiceOrder."No." THEN BEGIN
          IF NOT ServiceOrder.GET("Service Order No.") THEN
            ServiceOrder.INIT;
          ServiceOrder.CALCFIELDS("Service Location Description", "Service Contract Description");
        END;
        IF "Trade Item" <> '' THEN
          ItemCode := "Trade Item"
        ELSE
          IF "Basic Item" <> '' THEN
            ItemCode := "Basic Item"
          ELSE
            IF "Basic Item" <> '' THEN
              ItemCode := "Item No."
            ELSE
              ItemCode := "Cost Object";
      END;
    END;

    BEGIN
    END.
