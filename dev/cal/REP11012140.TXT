    PROCEDURE PostCommitments@1100485001();
    VAR
      EntryNo@1100485000 : Integer;
    BEGIN
      WITH GenJnlLine DO BEGIN

        TempGenJnlLineBuffer.RESET;
        TempGenJnlLineBuffer.SETFILTER(Amount, '<>%1',0);
        IF NOT TempGenJnlLineBuffer.FIND('-') THEN
          EXIT;

        //Fill general data and get Document No
        INIT;
        "Journal Template Name" := GenLedgerSetup."Commitments Journal Template";
        "Account Type" := "Account Type"::"G/L Account";
        "Posting Date" := PostingDate;
        "Document Date" := PostingDate;
        IF GenJnlTemplate."No. Series" <> '' THEN BEGIN
          CLEAR(NoSeriesMgt);
          "Document No." := NoSeriesMgt.GetNextNo(GenJnlTemplate."No. Series","Posting Date",TRUE);
        END ELSE
          "Document No." := DocumentNo;
        "Source Code" := GenJnlTemplate."Source Code";
        "Reason Code" := GenJnlTemplate."Reason Code";
        "Posting No. Series" := GenJnlTemplate."Posting No. Series";
        "System-Created Entry" := TRUE;

        //Post General Ledger Entries
        IF TempGenJnlLineBuffer.FIND('-') THEN
          REPEAT
            LineNo := LineNo + 10000;
            "Line No." := LineNo;
            Description := '';
            "Account No." := '';
            "Bal. Account No." := '';
            VALIDATE("Account No.", TempGenJnlLineBuffer."G/L Account");
            VALIDATE("Bal. Account No.", TempGenJnlLineBuffer."Bal. Account No.");
            VALIDATE(Amount, TempGenJnlLineBuffer.Amount);

            //Call C013268 sn
            "Gen. Bus. Posting Group" := '';
            VALIDATE("Gen. Prod. Posting Group",'');
            "Bal. Gen. Bus. Posting Group" := '';
            VALIDATE("Bal. Gen. Prod. Posting Group",'');
            //Call C013268 en

            "VAT Bus. Posting Group" := '';
            VALIDATE("VAT Prod. Posting Group",'');
            "Bal. VAT Bus. Posting Group" := '';
            VALIDATE("Bal. VAT Prod. Posting Group",'');
            VALIDATE("Shortcut Dimension 1 Code", TempGenJnlLineBuffer."Job No.");   //Job No is used to store Dim1
            PostGenJnl;
          UNTIL TempGenJnlLineBuffer.NEXT = 0;

        //Reverse post on next day
        IF TempGenJnlLineBuffer.FIND('-') THEN
          REPEAT
            LineNo := LineNo + 10000;
            "Line No." := LineNo;
            Description := '';
            "Posting Date" := CALCDATE('<1D>',PostingDate);
            Description := '';
            "Account No." := '';
            "Bal. Account No." := '';
            VALIDATE("Account No.", TempGenJnlLineBuffer."G/L Account");
            VALIDATE("Bal. Account No.", TempGenJnlLineBuffer."Bal. Account No.");
            VALIDATE(Amount, -TempGenJnlLineBuffer.Amount);

            //Call C013268 sn
            "Gen. Bus. Posting Group" := '';
            VALIDATE("Gen. Prod. Posting Group",'');
            "Bal. Gen. Bus. Posting Group" := '';
            VALIDATE("Bal. Gen. Prod. Posting Group",'');
            //Call C013268 en

            "VAT Bus. Posting Group" := '';
            VALIDATE("VAT Prod. Posting Group",'');
            "Bal. VAT Bus. Posting Group" := '';
            VALIDATE("Bal. VAT Prod. Posting Group",'');
            VALIDATE("Shortcut Dimension 1 Code", TempGenJnlLineBuffer."Job No.");  //Job No is used to store Dim1
            PostGenJnl;
          UNTIL TempGenJnlLineBuffer.NEXT = 0;

        //Post Commitment Entries
        IF TempCommitmentEntry.FIND('-') THEN BEGIN
          IF CommitmentEntry.FIND('+') THEN
            EntryNo := CommitmentEntry."Entry No.";
          REPEAT
            CommitmentEntry := TempCommitmentEntry;
            EntryNo := EntryNo + 1;
            CommitmentEntry."Entry No." := EntryNo;
            CommitmentEntry."Posting Date" := PostingDate;
            CommitmentEntry."Document No." := "Document No.";
            CommitmentEntry.INSERT(TRUE);
          UNTIL TempCommitmentEntry.NEXT = 0;
        END;

      END;
    END;

    PROCEDURE PurchaseOrderControlCheckData@1100525022() : Boolean;
    VAR
      InCorrectData@1100525000 : Boolean;
    BEGIN
      //Check Data
      InCorrectData := FALSE;

      IF ("Purchase Order Control Line"."Project No." = '') AND
         ("Purchase Order Control Line"."Service Order No." = '') THEN
        InCorrectData := TRUE;

      IF "Purchase Order Control Line"."Project No." <> '' THEN
        IF ProjRec."No." <> "Purchase Order Control Line"."Project No." THEN BEGIN
          IF NOT ProjRec.GET("Purchase Order Control Line"."Project No.") THEN BEGIN
            InCorrectData := TRUE;
          END ELSE BEGIN
            ProjRec.TESTFIELD("Project Type");
            ProjectTypeRec.GET(ProjRec."Project Type");
            IF NOT (ProjectTypeRec."Post Soft Commitments" OR ProjectTypeRec."Post Firm Commitments") THEN
              InCorrectData := TRUE;
          END;
        END;

      IF "Purchase Order Control Line"."Service Order No." <> '' THEN
        IF ServOrderRec."No." <> "Purchase Order Control Line"."Service Order No." THEN BEGIN
          IF NOT ServOrderRec.GET("Purchase Order Control Line"."Service Order No.") THEN BEGIN
            InCorrectData := TRUE;
          END ELSE BEGIN
            ServOrderRec.TESTFIELD("Service Type");
            ServTypeRec.GET(ServOrderRec."Service Type");
            IF NOT (ServTypeRec."Post Soft Commitments" OR ServTypeRec."Post Firm Commitments") THEN
              InCorrectData := TRUE;
          END;
        END;

      EXIT(NOT InCorrectData);
    END;

    PROCEDURE PurchaseLineCheckData@1100525027() : Boolean;
    VAR
      InCorrectData@1100525000 : Boolean;
    BEGIN
      InCorrectData := FALSE;

      IF ("Purchase Line"."Job No." <> '') OR
         ("Purchase Line"."Service Order No." <> '') THEN
        InCorrectData := TRUE;

      IF "Purchase Line"."Employee No." = '' THEN
        InCorrectData := TRUE;

      IF "Purchase Line"."Amnt. Received" = 0 THEN
        InCorrectData := TRUE;

      IF "Purchase Line".Type <> "Purchase Line".Type::"G/L Account" THEN
        InCorrectData := TRUE;


      EXIT(NOT InCorrectData);
    END;

    PROCEDURE HandleProject@1100485002(VAR GlobalDimension1Code@1100528400 : Code[20]) : Boolean;
    VAR
      Dim1@1100485001 : Code[20];
      Dim2@1100485000 : Code[20];
    BEGIN
      WITH "Purchase Order Control Line" DO BEGIN
        GlobalDimension1Code := ProjRec."Global Dimension 1 Code";
        PurchLineRec.GET(PurchLineRec."Document Type"::Order, "Order No.", "Order Line No.");

        PurchOrderContrLine.SETCURRENTKEY("Project No.",Date);
        PurchOrderContrLine.SETRANGE("Service Order No.");  //** Remove filter
        PurchOrderContrLine.SETRANGE("Project No.", "Project No.");
        PurchOrderContrLine.SETRANGE(Date,0D,PostingDate);
        PurchOrderContrLine.SETRANGE("Order No.", "Order No.");
        PurchOrderContrLine.SETRANGE("Order Line No.", "Order Line No.");
        PurchOrderContrLine.CALCSUMS("Outstanding Amount", "Amt. Rcd. Not Invoiced");
        IF (PurchOrderContrLine."Outstanding Amount" = 0) AND
           (PurchOrderContrLine."Amt. Rcd. Not Invoiced" = 0)
        THEN
          EXIT(FALSE);

        IF NOT ( (ProjectTypeRec."Post Soft Commitments" AND (PurchOrderContrLine."Outstanding Amount" <> 0) ) OR
                 (ProjectTypeRec."Post Firm Commitments" AND (PurchOrderContrLine."Amt. Rcd. Not Invoiced" <> 0)) ) THEN
          EXIT(FALSE);

        IF PurchHeaderRec."No." <> "Order No." THEN BEGIN
          PurchHeaderRec.GET(PurchHeaderRec."Document Type"::Order,"Order No.");
          PurchHeaderRec.TESTFIELD("Pay-to Vendor No.");
          VendRec.GET(PurchHeaderRec."Pay-to Vendor No.");
          VendRec.TESTFIELD("Vendor Posting Group");
          VendPostingGroup.GET(VendRec."Vendor Posting Group");
          IF NOT VendPostGrpLedgerAccProjectRec.GET(ProjectTypeRec.Code, VendRec."Vendor Posting Group") THEN
            VendPostGrpLedgerAccProjectRec.INIT;
        END;

        IF Post THEN BEGIN

          IF PostDimension THEN BEGIN

            Dim1 := ProjRec."Global Dimension 1 Code";
            IF "Coverage Account Surcharge" <> '' THEN BEGIN
              Dim1 := "Coverage Department Surcharge";
              IF Dim1 = '' THEN
                Dim1 := ProjRec."Global Dimension 1 Code";
            END;

            Dim2 := '';
            IF VendRec."Global Dimension 1 Code" <> '' THEN
              Dim2 := VendRec."Global Dimension 1 Code"
            ELSE
              Dim2 := ProjRec."Global Dimension 1 Code" ;
          END;

          GetWIPCoverAccountsProject;

          CreateCommitmentEntry := FALSE;

          IF ProjectTypeRec."Post Firm Commitments" AND ("Amt. Rcd. Not Invoiced" <> 0) THEN BEGIN
            //WIP Firm Commitment
            IF ProjectTypeRec."Post Compl. WIP Firm Comm." THEN BEGIN
              FillTempBuffer(WIPAccountFirm, ProjRec."Global Dimension 1 Code", "Amt. Rcd. Not Invoiced", 0);
              FillTempBuffer(CoverAccountFirm, Dim1, "Amt. Rcd. Not Invoiced" * -1, 0);
              FillTempBuffer(CostAccountFirm, Dim1, "Amt. Rcd. Not Invoiced", 1);
              FillTempBuffer(InvoiceToReceiveAccount, Dim2, "Amt. Rcd. Not Invoiced" * -1, 1);
            END ELSE BEGIN
              FillTempBuffer(WIPAccountFirm, ProjRec."Global Dimension 1 Code", "Amt. Rcd. Not Invoiced", 0);
              FillTempBuffer(InvoiceToReceiveAccount, Dim2, "Amt. Rcd. Not Invoiced" * -1, 0);
            END;
          END;

          IF (ProjectTypeRec."Post Ovh. Surch. Firm Comm.") AND ("Overhead Surcharge Firm" <> 0) THEN BEGIN
            //Post Ovh. Surch. Firm Commitments,
            TESTFIELD("Coverage Account Surcharge");
            IF ProjectTypeRec."Post Compl Surcharge Firm" THEN BEGIN
              FillTempBuffer(WIPAccountFirm, ProjRec."Global Dimension 1 Code", "Overhead Surcharge Firm", 0);
              FillTempBuffer(
                ProjectTypeRec.GetCoverageAccFirmSurcharge("Cost Type" - 1,VendRec."Vendor Posting Group"),
                ProjRec."Global Dimension 1 Code", "Overhead Surcharge Firm" * -1, 0);
              FillTempBuffer(
                ProjectTypeRec.GetCostAccFirmSurcharge("Cost Type" - 1,VendRec."Vendor Posting Group"),
                ProjRec."Global Dimension 1 Code", "Overhead Surcharge Firm", 1);
              FillTempBuffer("Coverage Account Surcharge", Dim1, "Overhead Surcharge Firm" * -1, 1);
            END ELSE BEGIN
              FillTempBuffer(WIPAccountFirm, ProjRec."Global Dimension 1 Code", "Overhead Surcharge Firm", 0);
              FillTempBuffer("Coverage Account Surcharge", Dim1, "Overhead Surcharge Firm" * -1, 0);
            END;
          END;

    PROCEDURE FillTempBuffer@1100485005(AccNo@1100485000 : Code[10];lDim@1100485003 : Code[20];Amnt@1100485002 : Decimal;Sorter@1100485001 : Option);
    VAR
      lGenJournalLine@1100525000 : Record 81;
    BEGIN
      lGenJournalLine.VALIDATE(Amount, Amnt);
      Amnt := lGenJournalLine.Amount;

      IF Amnt = 0 THEN
        EXIT;

      CreateCommitmentEntry := TRUE;

      TempGenJnlLineBuffer.RESET;
      TempGenJnlLineBuffer.SETRANGE(Type,Sorter);
      TempGenJnlLineBuffer.SETRANGE("G/L Account",AccNo);
      IF PostDimension THEN
        TempGenJnlLineBuffer.SETFILTER("Job No.", '%1', lDim);  //KeyField Job No is used to store Global Dim
      IF TempGenJnlLineBuffer.FIND('-') THEN BEGIN
        TempGenJnlLineBuffer.Amount := TempGenJnlLineBuffer.Amount + Amnt;
        TempGenJnlLineBuffer.MODIFY;
      END ELSE BEGIN
        TempGenJnlLineBuffer.INIT;
        TempGenJnlLineBuffer.Type := Sorter;
        TempGenJnlLineBuffer."G/L Account" := AccNo;
        IF PostDimension THEN
          TempGenJnlLineBuffer."Job No." := lDim;  //KeyField Job No is used to store Global Dim
        TempGenJnlLineBuffer.Amount := Amnt;
        TempGenJnlLineBuffer.INSERT;
      END;
    END;

    PROCEDURE HandleExternalEmployee@1100525032() : Boolean;
    VAR
      lvPurchReceiptRec@1100525000 : Record 120;
      Dim2@1100525001 : Code[20];
    BEGIN
      WITH "Purchase Line" DO BEGIN

        IF Post THEN BEGIN

          GetCoverAccountExternal;

          IF PostDimension THEN BEGIN
            Dim2 := '';
            IF VendRec."Global Dimension 1 Code" <> '' THEN
              Dim2 := VendRec."Global Dimension 1 Code"
            ELSE
              Dim2 := "Shortcut Dimension 1 Code" ;
          END;
          CreateCommitmentEntry := FALSE;
          //Firm Commitment
          FillTempBuffer("No.", "Shortcut Dimension 1 Code", OrderAmount-InvoicedAmt, 0);
          FillTempBuffer(InvoiceToReceiveAccount, Dim2, (OrderAmount-InvoicedAmt) * -1, 0);

          IF CreateCommitmentEntry THEN BEGIN
            TempCommitmentEntry.INIT;
            TempCommitmentEntry."Entry No." := TempCommitmentEntry."Entry No." + 1;
            TempCommitmentEntry."Employee No." := "Employee No.";
            TempCommitmentEntry."Order No." := "Document No.";
            TempCommitmentEntry."Order Line No." := "Line No.";
            TempCommitmentEntry.Description := Description;
            TempCommitmentEntry."Line Type" := TempCommitmentEntry."Line Type"::Receipt;
            TempCommitmentEntry.Date := "Posting Date";

            lvPurchReceiptRec.RESET;
            lvPurchReceiptRec.SETCURRENTKEY("Order No.");
            lvPurchReceiptRec.SETRANGE("Order No.","Document No.");
            IF lvPurchReceiptRec.FINDFIRST THEN
              TempCommitmentEntry."Receipt No." := lvPurchReceiptRec."No.";
            "Receipt No." := lvPurchReceiptRec."No.";

            TempCommitmentEntry."Outstanding Amount" := 0;
            TempCommitmentEntry."Amt. Rcd. Not Invoiced" := (OrderAmount-InvoicedAmt);
            TempCommitmentEntry."G/L Account No." := "No.";
            IF PostDimension THEN
              TempCommitmentEntry."Global Dimension 1 Code" := "Shortcut Dimension 1 Code";
            TempCommitmentEntry.INSERT;
          END;
        END;

      END;
      EXIT(TRUE);
    END;

    PROCEDURE CheckTotalExternalEmployee@1100525031() : Boolean;
    BEGIN
      //Don't RESET, because filter is set in BeforeDataItem
      PurchInvlineRec.SETRANGE("Purchase Route Reference", "Purchase Line"."Purchase Route Reference");

      OrderAmount := "Purchase Line".Amount;
      OrderQuanty := "Purchase Line".Quantity;

      InvoicedAmt := 0;
      InvoicedQty := 0;
      OpenAmount  := 0;

      IF NOT PurchInvlineRec.ISEMPTY THEN BEGIN
        PurchInvlineRec.FINDSET;
        REPEAT
          InvoicedAmt := InvoicedAmt + PurchInvlineRec."Amount (LCY)";
          InvoicedQty := InvoicedQty + PurchInvlineRec.Quantity;
        UNTIL PurchInvlineRec.NEXT = 0;
      END;

      IF (InvoicedAmt >= OrderAmount) OR
         (InvoicedQty >= OrderQuanty) THEN
        EXIT(FALSE);

      OpenAmount := OrderAmount - InvoicedAmt;

      EXIT(TRUE)
    END;
