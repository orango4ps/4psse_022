    { 1100527800;;DataItem;                  ;
               DataItemTable=Table11012746;

               OnPreDataItem=VAR
                               DocumentManagement@1100528401 : Codeunit 11012406;

               OnAfterGetRecord=VAR
                                  FileManagement@1100527800 : Codeunit 419;

                                  IF SourceStorageFileExists("Document Properties"."Storage Type", "Document Properties".GetStorageURI) AND
                                     NOT TargetStorageFileExists(TargetStorageType, TargetStorageURI) AND
                                     FileMustBePutInDateSubfolder("Document Properties", TargetStorageURI)
                                  THEN BEGIN
                                    IF NOT TestRun THEN BEGIN
                                      DocumentManagement.CreateStorageLocationIfNotExists(TargetStorageType, DestinationFolder);

                                        "Document Properties"."Storage Type", TargetStorageType,
                                        "Document Properties".GetStorageURI, TargetStorageURI, ErrorMessage)
                                      THEN BEGIN
                                        FileCopied := TRUE;

                                        MODIFY;
                                      END;
                                    END;
                                  END ELSE

                                END;

    LOCAL PROCEDURE SourceStorageFileExists@1100528402(StorageType@1100528401 : Option;StorageURI@1100528402 : Text) : Boolean;
    VAR
      DocumentProperties@1100528403 : Record 11012746;
      FileManagement@1100528400 : Codeunit 419;
    BEGIN
      IF StorageType = DocumentProperties."Storage Type"::"SharePoint Online" THEN
        EXIT(TRUE) //In SharePoint Online it cannot be checked if the file exists.
      ELSE
        EXIT(FileManagement.StorageFileExists(StorageURI, StorageType));
    END;

    LOCAL PROCEDURE TargetStorageFileExists@1100528403(StorageType@1100528401 : Option;StorageURI@1100528402 : Text) : Boolean;
    VAR
      DocumentProperties@1100528403 : Record 11012746;
      FileManagement@1100528400 : Codeunit 419;
    BEGIN
      IF StorageType = DocumentProperties."Storage Type"::"SharePoint Online" THEN
        EXIT(FALSE) //In SharePoint Online it cannot be checked if the file exists.
      ELSE
        EXIT(FileManagement.StorageFileExists(StorageURI, StorageType));
    END;

    LOCAL PROCEDURE FileMustBePutInDateSubfolder@1100528411(DocumentProperties@1100528400 : Record 11012746;TargetStorageURI@1100528402 : Text) : Boolean;
    VAR
      DocumentMgtSetup@1100528401 : Record 11071831;
    BEGIN
      DocumentMgtSetup.GET;
      IF NOT DocumentMgtSetup."Use Date Subfolders" THEN
        EXIT(FALSE);

      IF DocumentMgtSetup."Starting Date Subfolders" > 0D THEN
        IF DocumentProperties."Creation Date" < DocumentMgtSetup."Starting Date Subfolders" THEN
          EXIT(FALSE);

      IF LOWERCASE("Document Properties".GetStorageURI) = LOWERCASE(TargetStorageURI) THEN
        EXIT(FALSE);

      IF "Document Properties"."Storage Type" = "Document Properties"."Storage Type"::"SharePoint Online" THEN
        // In case of SharePoint Online, TargetStorageURI does not contain the complete URI, but only a
        // "subfolder" and file name.
        IF STRPOS(UrlDecode("Document Properties".GetStorageURI), UrlDecode(TargetStorageURI)) > 0 THEN
          EXIT(FALSE);

      EXIT(TRUE);
    END;

    LOCAL PROCEDURE UrlDecode@1100528400(Url@1100528400 : Text) DecodedUrl : Text;
    VAR
      TypeHelper@1100528401 : Codeunit 10;
    BEGIN
      DecodedUrl := TypeHelper.UrlDecode(Url);
      DecodedUrl := CONVERTSTR(DecodedUrl, '\', '/');
      DecodedUrl := LOWERCASE(DecodedUrl);
    END;

    BEGIN
    END.
