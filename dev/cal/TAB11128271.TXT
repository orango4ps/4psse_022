        This.RESET;
        This.SETRANGE("User ID",USERID);
        This.DELETEALL;

          This."Estimate No." := Estimate."No.";

          This.INSERT;
          Added += 1;
        UNTIL (Estimate.NEXT = 0) OR (Added > 9);
