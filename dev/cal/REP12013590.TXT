    PROCEDURE CalculateExFlowStats@1100285005();
    VAR
      ExDocApprover@1100285011 : Record 12013610;
      PostedExDocApprover@1100285000 : Record 12013628;
      TempExDocApprover@1100285010 : TEMPORARY Record 12013610;
      TempExDoc@1100285013 : TEMPORARY Record 12013608;
      TempApprover@1100285012 : Code[50];
      FromDate@1100285003 : DateTime;
      ToDate@1100285004 : DateTime;
      PrevFromDate@1100285009 : DateTime;
      PrevToDate@1100285008 : DateTime;
    BEGIN
      FromDate := CREATEDATETIME(DMY2DATE(1,1,DATE2DMY(TODAY,3)),0T);
      ToDate := CREATEDATETIME(DMY2DATE(31,12,DATE2DMY(TODAY,3)),235959T);
      PrevFromDate := CREATEDATETIME(DMY2DATE(1,1,DATE2DMY(TODAY,3)-1),0T);
      PrevToDate := CREATEDATETIME(DMY2DATE(31,12,DATE2DMY(TODAY,3)-1),235959T);

      TempExDocApprover.SETCURRENTKEY("Approval Date","Approved By");
      TempExDoc.SETCURRENTKEY("Create Date");

      TempExDoc.RESET;
      TempExDoc.DELETEALL;

      ExDocApprover.RESET;
      ExDocApprover.CHANGECOMPANY(Company.Name);
      ExDocApprover.SETCURRENTKEY("Approval Date","Approved By");
      ExDocApprover.SETRANGE("Approval Date", PrevFromDate, ToDate);
      IF ExDocApprover.FINDSET THEN
        REPEAT
          TempExDoc.SETRANGE(ID, ExDocApprover."Document ID");
          IF NOT TempExDoc.FINDFIRST THEN BEGIN
            TempExDoc.ID := ExDocApprover."Document ID";

            IF ExDocApprover."Approval Date" >= FromDate THEN
              TempExDoc."Create Date" := DT2DATE(FromDate)
            ELSE
              TempExDoc."Create Date" := DT2DATE(PrevFromDate);

            IF TempExDoc.INSERT THEN {};
          END;

          IF TempApprover <> ExDocApprover."Approved By" THEN BEGIN
            TempApprover := ExDocApprover."Approved By";

            IF ExDocApprover."Approval Date" >= FromDate THEN BEGIN
              TempExDocApprover.SETRANGE("Approval Date", FromDate);
              TempExDocApprover.SETRANGE("Approved By", ExDocApprover."Approved By");
              IF NOT TempExDocApprover.FINDFIRST THEN BEGIN
                TempExDocApprover.ID := ExDocApprover.ID;
                TempExDocApprover."Approval Date" := FromDate;
                TempExDocApprover."Approved By" := ExDocApprover."Approved By";
                IF TempExDocApprover.INSERT THEN {};
              END;
            END
            ELSE BEGIN
              TempExDocApprover.SETRANGE("Approval Date", PrevFromDate);
              TempExDocApprover.SETRANGE("Approved By", ExDocApprover."Approved By");
              IF NOT TempExDocApprover.FINDFIRST THEN BEGIN
                TempExDocApprover.ID := ExDocApprover.ID;
                TempExDocApprover."Approval Date" := PrevFromDate;
                TempExDocApprover."Approved By" := ExDocApprover."Approved By";
                IF TempExDocApprover.INSERT THEN {};
              END;
            END;
          END;
        UNTIL ExDocApprover.NEXT = 0;

      PostedExDocApprover.RESET;
      PostedExDocApprover.CHANGECOMPANY(Company.Name);
      PostedExDocApprover.SETCURRENTKEY("Approval Date","Approved By");
      PostedExDocApprover.SETRANGE("Approval Date", PrevFromDate, ToDate);
      IF PostedExDocApprover.FINDSET THEN
        REPEAT
          TempExDoc.SETRANGE(ID, PostedExDocApprover."Document ID");
          IF NOT TempExDoc.FINDFIRST THEN BEGIN
            TempExDoc.ID := PostedExDocApprover."Document ID";

            IF PostedExDocApprover."Approval Date" >= FromDate THEN
              TempExDoc."Create Date" := DT2DATE(FromDate)
            ELSE
              TempExDoc."Create Date" := DT2DATE(PrevFromDate);

            IF TempExDoc.INSERT THEN {};
          END;

          IF TempApprover <> PostedExDocApprover."Approved By" THEN BEGIN
            TempApprover := PostedExDocApprover."Approved By";

            IF PostedExDocApprover."Approval Date" >= FromDate THEN BEGIN
              TempExDocApprover.SETRANGE("Approval Date", FromDate);
              TempExDocApprover.SETRANGE("Approved By", PostedExDocApprover."Approved By");
              IF NOT TempExDocApprover.FINDFIRST THEN BEGIN
                TempExDocApprover.ID := ExDocApprover.ID;
                TempExDocApprover."Approval Date" := FromDate;
                TempExDocApprover."Approved By" := PostedExDocApprover."Approved By";
                IF TempExDocApprover.INSERT THEN {};
              END;
            END
            ELSE BEGIN
              TempExDocApprover.SETRANGE("Approval Date", PrevFromDate);
              TempExDocApprover.SETRANGE("Approved By", PostedExDocApprover."Approved By");
              IF NOT TempExDocApprover.FINDFIRST THEN BEGIN
                TempExDocApprover.ID := ExDocApprover.ID;
                TempExDocApprover."Approval Date" := PrevFromDate;
                TempExDocApprover."Approved By" := PostedExDocApprover."Approved By";
                IF TempExDocApprover.INSERT THEN {};
              END;
            END;
          END;
        UNTIL PostedExDocApprover.NEXT = 0;

      TempExDocApprover.RESET;
      TempExDocApprover.SETCURRENTKEY("Approval Date","Approved By");
      TempExDoc.RESET;
      TempExDoc.SETCURRENTKEY("Create Date");

      TempExDoc.SETRANGE("Create Date", DT2DATE(FromDate));
      TempExDocApprover.SETRANGE("Approval Date", FromDate);

      CurrDateFilter := FORMAT(DT2DATE(FromDate)) + '..' + FORMAT(DT2DATE(ToDate));
      CurrNoOfDocuments := TempExDoc.COUNT;
      CurrNoofApprovers := TempExDocApprover.COUNT;
      TotCurrNoOfDocuments := TotCurrNoOfDocuments + CurrNoOfDocuments;
      TotCurrNoofApprovers := TotCurrNoofApprovers + CurrNoofApprovers;

      TempExDoc.SETRANGE("Create Date", DT2DATE(PrevFromDate));
      TempExDocApprover.SETRANGE("Approval Date", PrevFromDate);
      PrevDateFilter := FORMAT(DT2DATE(PrevFromDate)) + '..' + FORMAT(DT2DATE(PrevToDate));
      PrevNoOfDocuments := TempExDoc.COUNT;
      PrevNoofApprovers := TempExDocApprover.COUNT;

      TotPrevNoOfDocuments := TotPrevNoOfDocuments + PrevNoOfDocuments;
      TotPrevNoofApprovers := TotPrevNoofApprovers + PrevNoofApprovers;
    END;

    BEGIN
    {
      2017-03-08  CHMO  EXF#347
    }
    END.
