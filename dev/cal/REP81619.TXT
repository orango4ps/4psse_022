    PROCEDURE GetProjectNo@1100285502(VendLedgEntry@1100285500 : Record 25) : Code[20];
    VAR
      PurchInvLine@1100285502 : Record 123;
      PurchCrMemoLine@1100285503 : Record 125;
      JobNo@1100285504 : Code[20];
      AccountNo@1100285501 : Code[20];
    BEGIN
      //**Go throught the invoice lines to find the first project no.
      CASE VendLedgEntry."Document Type" OF

        VendLedgEntry."Document Type"::Invoice:
          BEGIN
            PurchInvLine.SETRANGE("Document No.", VendLedgEntry."Document No.");
            IF PurchInvLine.FINDSET(FALSE) THEN REPEAT

              IF PurchInvLine."Job No." <> '' THEN
                JobNo := PurchInvLine."Job No.";

            UNTIL (PurchInvLine.NEXT = 0) OR (JobNo <> '');

            //**If the lines doesn't have a Job No., prints cost center + account or location code
            IF JobNo = '' THEN BEGIN
              JobNo := VendLedgEntry."Global Dimension 1 Code";

              IF PurchInvLine.FINDSET(FALSE) THEN REPEAT
                IF PurchInvLine.Type = PurchInvLine.Type::"G/L Account" THEN

                  AccountNo := PurchInvLine."No."
                ELSE
                  AccountNo := PurchInvLine."Location Code";

              UNTIL (PurchInvLine.NEXT = 0) OR (AccountNo <> '');

              JobNo += '_' + AccountNo;

            END;

          END; //**end invoice

        VendLedgEntry."Document Type"::"Credit Memo":
          BEGIN
            PurchCrMemoLine.SETRANGE("Document No.", VendLedgEntry."Document No.");
            IF PurchCrMemoLine.FINDSET(FALSE) THEN REPEAT

              IF PurchCrMemoLine."Job No." <> '' THEN
                JobNo := PurchCrMemoLine."Job No.";

            UNTIL (PurchCrMemoLine.NEXT = 0) OR (JobNo <> '');

            //**If the lines doesn't have a Job No., prints cost center + account or location code
            IF JobNo = '' THEN BEGIN
              JobNo := VendLedgEntry."Global Dimension 1 Code";

              IF PurchCrMemoLine.FINDSET(FALSE) THEN REPEAT
                IF PurchCrMemoLine.Type = PurchCrMemoLine.Type::"G/L Account" THEN

                  AccountNo := PurchCrMemoLine."No."
                ELSE
                  AccountNo := PurchCrMemoLine."Location Code";

              UNTIL (PurchCrMemoLine.NEXT = 0) OR (AccountNo <> '');

              JobNo += '_' + AccountNo;

            END;

          END; //**end credit memo

      END;

      EXIT(JobNo);
    END;
