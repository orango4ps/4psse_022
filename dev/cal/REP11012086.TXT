                "Progress % Base" :=  ("Actual Cost"/"Prognosis Revenue") * 100;

                "Progress % Base" := ("Actual Cost"/("Prognosis Revenue"-PrognEndResVersusContractPr)) * 100;

                IF includeFinishedJobs THEN //IME403
                  JobRec.CALCFIELDS("Received/Not Invoiced", "Open Ovh. Surch. Firm (Purch.)");  // C-026777

      //<<IME403
                FirmCommitments := JobRec."Received/Not Invoiced" + JobRec."Open Ovh. Surch. Firm (Purch.)";  // C-026777

            "Units Saleable" := PlotRec.COUNT;
            PlotRec.SETFILTER("Sales Status", '%1..', PlotRec."Sales Status"::Sold);

            "Units Sold" := PlotRec.COUNT;
          END;

          IF "Units Saleable" <> 0 THEN
            "Sales Progress" := ROUND(100 * "Units Sold" / "Units Saleable")

            "Sales Progress" := 0;
          "Progress %" := ROUND("Progress % Base" * ("Sales Progress" / 100));

          "Progress %" := "Progress % Base";

        IF JobRec."Forecast Level" = JobRec."Forecast Level"::"Cost Component" THEN BEGIN
          "Progress % Base" := 0;

    LOCAL PROCEDURE CalculateActualHours@1100525015("JobNo."@1100525000 : Code[20];PostingDate@1100525002 : Date) Result : Decimal;
    VAR
      JobLedgerEntry@1100525001 : Record 11072005;
    BEGIN
      Result := 0;
      JobLedgerEntry.SETRANGE("Job No.","JobNo.");
      JobLedgerEntry.SETRANGE("Cost Type",JobLedgerEntry."Cost Type"::Labor);
      JobLedgerEntry.SETFILTER("Posting Date", '..%1', PostingDate);
      IF JobLedgerEntry.FINDSET THEN
        REPEAT
          IF CostObjectIsRecordProgress(JobLedgerEntry."Global Dimension 2 Code") THEN
            Result := Result + JobLedgerEntry.Quantity;
        UNTIL JobLedgerEntry.NEXT = 0;
      EXIT(Result);
    END;

    LOCAL PROCEDURE CalculateBudgetHours@1100525016("JobNo."@1100525000 : Code[20]) Result : Decimal;

      BudgetLine@1100525001 : Record 11012001;
    BEGIN
      Result := 0;
      BudgetLine.SETRANGE("Project No.","JobNo.");
      BudgetLine.SETRANGE("Cost Type",BudgetLine."Cost Type"::Labor);
      IF BudgetLine.FINDSET THEN
        REPEAT
          IF CostObjectIsRecordProgress(BudgetLine."Cost Object") THEN
            Result := Result + BudgetLine.Hours;
        UNTIL BudgetLine.NEXT = 0;
      EXIT(Result);
    END;
