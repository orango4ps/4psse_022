    LOCAL PROCEDURE AggregateSource@12047958(VendorLedgerEntry@12047959 : Record 25);
    BEGIN

        EXIT;
      VendorLedgerEntry.CALCFIELDS("Remaining Amount");
      tSourceAggregate."Source Type" := tSourceAggregate."Source Type"::Vendor;
