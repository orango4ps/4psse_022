    CaptionML=[DEU=Werkzeugmietposten abrufen;
               ENU=Get Plant Rent Entry;
               NLD=Materieel inhuurposten ophalen];
    OnInitReport=BEGIN
                   PlantSetup.GET;
                   PlantSetup.TESTFIELD("Cost Component Rent");
                 END;

    OnPreReport=BEGIN
                  NextLineNo := 0;
                  MaxEndingDate := 0D;
                  PlantNoAmount := 0;
                  TotalAmount := 0;

                  TempPurchaseLine.RESET;
                  TempPurchaseLine.DELETEALL;
                  TempInvoiceApprovalLine.RESET;
                  TempInvoiceApprovalLine.DELETEALL;

                  IF (NOT OnlyTotalRentAmt) THEN BEGIN
                    IF (InvoiceDateFrom = 0D) OR (InvoiceDateTo = 0D) THEN
                      ERROR(Text000);
                    IF (InvoiceDateTo < InvoiceDateFrom) THEN
                      ERROR(Text001);
                  END ELSE BEGIN
                    IF (TotalRentAmount = 0) THEN
                      ERROR(Text002);
                  END;

                  ValidateTotalRentAmount;

                  CASE TableNo OF
                    DATABASE::"Purchase Line":
                      BEGIN
                        IF (NOT CreditMemo) THEN
                          PurchaseHeader.GET(PurchaseHeader."Document Type"::Invoice, InvoiceNo)
                        ELSE
                          PurchaseHeader.GET(PurchaseHeader."Document Type"::"Credit Memo", InvoiceNo);
                        PurchaseHeader.TESTFIELD("No.");
                        PurchaseHeader.TESTFIELD("Buy-from Vendor No.");
                        VendorNo := PurchaseHeader."Buy-from Vendor No.";
                      END;
                    DATABASE::"Purch. Inv. Header":
                      BEGIN
                        PurchInvHeader.GET(InvoiceNo);
                        IF PurchInvHeader."DMS Status" > PurchInvHeader."DMS Status"::"Awaiting Line Entry" THEN
                          PurchInvHeader.FIELDERROR("DMS Status");
                        PurchInvHeader.TESTFIELD("No.");
                        PurchInvHeader.TESTFIELD("Buy-from Vendor No.");
                        VendorNo := PurchInvHeader."Buy-from Vendor No.";
                      END;
                    DATABASE::"Purch. Cr. Memo Hdr.":
                      BEGIN
                        PurchCrMemoHdr.GET(InvoiceNo);
                        IF PurchCrMemoHdr."DMS Status" > PurchCrMemoHdr."DMS Status"::"Awaiting Line Entry" THEN
                          PurchCrMemoHdr.FIELDERROR("DMS Status");
                        PurchCrMemoHdr.TESTFIELD("No.");
                        PurchCrMemoHdr.TESTFIELD("Buy-from Vendor No.");
                        VendorNo := PurchCrMemoHdr."Buy-from Vendor No.";
                      END;
                    ELSE
                      ERROR('');
                  END;

                  IF NOT Vendor.GET(VendorNo) THEN
                    CLEAR(Vendor);

                  PlantLocationFilters.COPYFILTERS("Plant Location");

                  Window.OPEN(Text003);
                END;

    OnPostReport=VAR
                   PurchaseLine@1100528702 : Record 39;
                   InvoiceApprovalLine@1100528703 : Record 11012032;
                   PosLinePresent@1100528701 : Boolean;
                   NegLinePresent@1100528700 : Boolean;
                 BEGIN
                   Window.CLOSE;

                   IF (TotalRentAmount <> 0) THEN BEGIN
                     Window.OPEN(Text004);
                     PosLinePresent := FALSE;
                     NegLinePresent := FALSE;

                     CASE TableNo OF
                       DATABASE::"Purchase Line":
                         BEGIN
                           IF OnlyTotalRentAmt THEN BEGIN
                             PurchaseLine.RESET;
                             IF (NOT CreditMemo) THEN
                               PurchaseLine.SETRANGE("Document Type", PurchaseLine."Document Type"::Invoice)
                             ELSE
                               PurchaseLine.SETRANGE("Document Type", PurchaseLine."Document Type"::"Credit Memo");
                             PurchaseLine.SETRANGE("Document No.", PurchaseHeader."No.");
                             PurchaseLine.SETRANGE("Cost Component Plant", PlantSetup."Cost Component Rent");
                             PurchaseLine.SETFILTER("Line Amount", '<>%1', 0);
                             IF PurchaseLine.FINDSET THEN BEGIN
                               REPEAT
                                 IF (PurchaseLine."Line Amount" > 0) THEN PosLinePresent := TRUE;
                                 IF (PurchaseLine."Line Amount" < 0) THEN NegLinePresent := TRUE;
                                 TempPurchaseLine := PurchaseLine;
                                 TempPurchaseLine.INSERT;
                               UNTIL PurchaseLine.NEXT = 0;
                             END;
                           END;
                           DetermineDivideOver(PosLinePresent, NegLinePresent);
                           UpdatePurchLineAmt();
                         END;
                       DATABASE::"Purch. Inv. Header",
                       DATABASE::"Purch. Cr. Memo Hdr.":
                         BEGIN
                           IF OnlyTotalRentAmt THEN BEGIN
                             InvoiceApprovalLine.RESET;
                             IF (NOT CreditMemo) THEN BEGIN
                               InvoiceApprovalLine.SETRANGE("Journal Template Name", PurchInvHeader."Journal Template");
                               InvoiceApprovalLine.SETRANGE("Document Type", InvoiceApprovalLine."Document Type"::"Purchase Invoice");
                               InvoiceApprovalLine.SETRANGE("Document No.", PurchInvHeader."No.");
                             END ELSE BEGIN
                               InvoiceApprovalLine.SETRANGE("Journal Template Name", PurchCrMemoHdr."Journal Template");
                               InvoiceApprovalLine.SETRANGE("Document Type", InvoiceApprovalLine."Document Type"::"Credit Memo");
                               InvoiceApprovalLine.SETRANGE("Document No.", PurchCrMemoHdr."No.");
                             END;
                             InvoiceApprovalLine.SETRANGE("Cost Component Plant", PlantSetup."Cost Component Rent");
                             InvoiceApprovalLine.SETFILTER(Amount, '<>%1', 0);
                             IF InvoiceApprovalLine.FINDSET THEN BEGIN
                               REPEAT
                                 IF (InvoiceApprovalLine.Amount > 0) THEN PosLinePresent := TRUE;
                                 IF (InvoiceApprovalLine.Amount < 0) THEN NegLinePresent := TRUE;
                                 TempInvoiceApprovalLine := InvoiceApprovalLine;
                                 TempInvoiceApprovalLine.INSERT;
                               UNTIL InvoiceApprovalLine.NEXT = 0;
                             END;
                           END;
                           DetermineDivideOver(PosLinePresent, NegLinePresent);
                           UpdateApprovalLineAmt();
                         END;
                     END;
                     Window.CLOSE;
                   END;
                 END;

    BEGIN

        IF TotalRentAmount < 0 THEN
          ERROR(Text005);

      END;

    PROCEDURE UpdatePurchLineAmt@1100485001();
    VAR
      PurchaseLine@1100528700 : Record 39;
      LastLineNo@1100485015 : Integer;
      LastDebitLineNo@1100485014 : Integer;
      LastCreditLineNo@1100485013 : Integer;
      Amt@1100485012 : Decimal;
      SubTotalAmt@1100485011 : Decimal;
      TotalDebitAmt@1100485010 : Decimal;
      TotalCreditAmt@1100485009 : Decimal;
      TotalAmt@1100485008 : Decimal;
      WritenAmt@1100485007 : Decimal;
      RentAmt@1100485006 : Decimal;
      AmtFilterCond@1100485005 : Text[1];
    BEGIN
      TempPurchaseLine.RESET;
      TempPurchaseLine.SETFILTER("Line Amount", '<>%1', 0);
      IF TempPurchaseLine.FINDSET THEN BEGIN
        REPEAT
          CalculateTotalsInTmpTable(
            TempPurchaseLine."Line Amount", TempPurchaseLine."Line No.",
            TotalDebitAmt, TotalCreditAmt, LastDebitLineNo, LastCreditLineNo);
        UNTIL TempPurchaseLine.NEXT = 0;

        FillUpdateFields(
          TotalDebitAmt, TotalCreditAmt, LastDebitLineNo, LastCreditLineNo, TotalAmt, LastLineNo, RentAmt, AmtFilterCond);
        IF AmtFilterCond IN ['>','<'] THEN
          TempPurchaseLine.SETFILTER("Line Amount", AmtFilterCond + '%1', 0);

        IF CheckConversionFactor(TotalAmt, RentAmt) THEN BEGIN
          IF TempPurchaseLine.FINDSET THEN BEGIN
            REPEAT
              Window.UPDATE(1, TempPurchaseLine."Plant Type" + '-' + TempPurchaseLine."Plant No.");
              SubTotalAmt += TempPurchaseLine."Line Amount";
              IF (TempPurchaseLine."Line No." <> LastLineNo) THEN
                Amt := (SubTotalAmt * (RentAmt / TotalAmt)) - WritenAmt
              ELSE
                Amt := RentAmt - WritenAmt;
              PurchaseLine.GET(
                TempPurchaseLine."Document Type", TempPurchaseLine."Document No.", TempPurchaseLine."Line No.");
              PurchaseLine.VALIDATE("Line Amount", Amt);
              PurchaseLine.MODIFY(TRUE);
              WritenAmt += PurchaseLine."Line Amount";
            UNTIL TempPurchaseLine.NEXT = 0;
          END;
        END;
      END;

      TempPurchaseLine.RESET;
      TempPurchaseLine.DELETEALL;
    END;

    PROCEDURE UpdateApprovalLineAmt@1100485011();
    VAR
      InvoiceApprovalLine@1100528700 : Record 11012032;
      LastLineNo@1100485001 : Integer;
      LastDebitLineNo@1100485008 : Integer;

    PROCEDURE CalculateTotalsInTmpTable@1100485002(Amt@1100485006 : Decimal;LineNo@1100485000 : Integer;VAR TotalDebitAmt@1100485002 : Decimal;VAR TotalCreditAmt@1100485001 : Decimal;VAR LastDebitLineNo@1100485004 : Integer;VAR LastCreditLineNo@1100485003 : Integer);
    BEGIN
      IF (NOT CreditMemo) THEN BEGIN
        IF (Amt >= 0) THEN BEGIN
          TotalDebitAmt += Amt;
          LastDebitLineNo := LineNo;
        END ELSE BEGIN
          TotalCreditAmt += Amt;
          LastCreditLineNo := LineNo;
        END;
      END ELSE BEGIN
        IF (Amt < 0) THEN BEGIN
          TotalDebitAmt += Amt;
          LastDebitLineNo := LineNo;
        END ELSE BEGIN
          TotalCreditAmt += Amt;
          LastCreditLineNo := LineNo;
        END;
      END;
    END;

    PROCEDURE FillUpdateFields@1100485012(TotalDebitAmt@1100485012 : Decimal;TotalCreditAmt@1100485011 : Decimal;LastDebitLineNo@1100485009 : Integer;LastCreditLineNo@1100485008 : Integer;VAR TotalAmt@1100485003 : Decimal;VAR LastLineNo@1100485010 : Integer;VAR RentAmt@1100485002 : Decimal;VAR AmtFilterCond@1100485000 : Text[1]);
    BEGIN
      TotalAmt := TotalDebitAmt + TotalCreditAmt;
      IF (LastDebitLineNo >= LastCreditLineNo) THEN
        LastLineNo := LastDebitLineNo
      ELSE
        LastLineNo := LastCreditLineNo;
      RentAmt := TotalRentAmount;
      IF CreditMemo THEN
        RentAmt := -RentAmt;
      AmtFilterCond := '';

      CASE DivideOver OF
        DivideOver::Debit:
          BEGIN
            TotalAmt := TotalDebitAmt;
            LastLineNo := LastDebitLineNo;
            RentAmt -= TotalCreditAmt;
            IF (NOT CreditMemo) THEN
              AmtFilterCond := '>'
            ELSE
              AmtFilterCond := '<';
          END;
        DivideOver::Credit:
          BEGIN
            TotalAmt := TotalCreditAmt;
            LastLineNo := LastCreditLineNo;
            RentAmt -= TotalDebitAmt;
            IF (NOT CreditMemo) THEN
              AmtFilterCond := '<'
            ELSE
              AmtFilterCond := '>';
          END;
      END;
    END;

    PROCEDURE DetermineDivideOver@1100485004(PosLinesPresent@1100485000 : Boolean;NegLinesPresent@1100485001 : Boolean);
    BEGIN
      DivideOver := DivideOver::Both;
      IF OnlyTotalRentAmt AND PosLinesPresent AND NegLinesPresent THEN BEGIN
        CASE STRMENU(Text007, 3) OF
          1: DivideOver := DivideOver::Debit;
          2: DivideOver := DivideOver::Credit;
          3: DivideOver := DivideOver::Both;
          ELSE
            ERROR('');  //* Anulate
        END;
      END;
    END;

    PROCEDURE CheckConversionFactor@1100485016(AmountEntries@1100485000 : Decimal;RentAmt@1100485001 : Decimal) : Boolean;
    BEGIN
      //* If conversion factor is zero or negative then only a 'MESSAGE', so no 'ERROR'. Only do NOT devided the difference, inserting lines (rental entries) must be done (if that is selected)

      IF (AmountEntries = RentAmt) THEN
        EXIT(FALSE);  //* Factor is 1, divide difference (=0) over the lines not needed

      IF (AmountEntries = 0) OR (RentAmt = 0) THEN BEGIN
        MESSAGE(Text008);
        EXIT(FALSE);
      END;

      IF ((AmountEntries > 0) AND (RentAmt < 0)) OR ((AmountEntries < 0) AND (RentAmt > 0)) THEN BEGIN
        MESSAGE(Text009);
        EXIT(FALSE);
      END;

      EXIT(TRUE);
    END;

    BEGIN
    {
      This report is called from purchase invoices and purchase credit memos (both register and approve).
      the following is agreed with BNBM (Edwin Prinsen) on  13-06-2006:
       - Only get the debit rent entries, so not the credit part.
       - Get rental entries in a credit memo's then these will become negative
      Reasons why not get the credit-rental entries:
       - Difficult what shoud be get, is not (only) part that in the entered period
       - Prepayment part would give a credit entry, but if only a part of a plant lot is returned then it is not allowedj to modify 'Ext. Rented Invoiced to'
         in the Plant Number table. For the part that is not returned then no 'VJP' (credit) is calculated anymore when this date is reset (set to a lower date).
         Or if date is not resetas then the  credit part will always come back deel when getting the rent entries.
       - Credit when returned. F.i. if plant is returned just before the end of a periode, but the vendor has not registered this in his systeem.
         Then you get a invoice for the total period, In the next period then thit part is credited.
         This you will never get well because yoy do not know what the vendor has registerd in his system.
         Also same problem as in previous item if only a part of a lot is returned.
      It is not desirable that in some cases and not in other case the credit entries are get (depending on if all of a lot is returned at once),
      therefore never get credit entries.
      So the credit lines must always be inserted by the user (often not so much). Also then the user must decide if this lines if these lines
      should be on a credit memo (pos) or on a invoice (neg), depending on other lines on the credit memo(total amt must be neg) / invoice(total amount must be pos).
    }
    END.
