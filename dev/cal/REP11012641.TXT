    LOCAL PROCEDURE CalculateDepreciation@1100528500(Int@1100528500 : Integer);
    BEGIN
      IF NOT Printed THEN BEGIN
        CLEAR(DepreciationCum);
        FixedAsset.RESET;
        IF Int = 1 THEN
          FixedAsset.SETRANGE("Plant Type","Plant Type".Code)
        ELSE BEGIN
          FixedAsset.SETRANGE("Plant Type","Plant Number"."Plant Type");
          FixedAsset.SETRANGE("Plant No.","Plant Number"."No.");
        END;
        IF FixedAsset.FINDSET THEN BEGIN
          REPEAT
            FixedAsset1.RESET;
            FixedAsset1.SETRANGE("Main Asset/Component",FixedAsset1."Main Asset/Component"::Component);
            FixedAsset1.SETRANGE("Component of Main Asset",FixedAsset."No.");
            IF FixedAsset1.FINDFIRST THEN BEGIN
              REPEAT
                FALedgerEntry.SETRANGE("FA No.", FixedAsset1."No.");
                FALedgerEntry.SETFILTER("Posting Date", PeriodFilter);
                IF FALedgerEntry.FINDSET THEN BEGIN
                  REPEAT
                    CumDepreciation := CumDepreciation + (FALedgerEntry.Amount * -1);
                    DepreciationCum := DepreciationCum + (FALedgerEntry.Amount * -1);
                    FADescription := Text001 + FixedAsset1."No.";
                  UNTIL FALedgerEntry.NEXT = 0;
                END;
              UNTIL FixedAsset1.NEXT = 0;
            END;
          UNTIL FixedAsset.NEXT = 0;
        END;
        Printed := TRUE;
      END ELSE
        CurrReport.SKIP;
    END;

    BEGIN
    END.
