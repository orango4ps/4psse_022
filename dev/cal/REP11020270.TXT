    PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Vendor Category".GETFILTERS <> '' THEN

    END;

    PROCEDURE FillTurnover@1100528900(iVendor@1100528901 : Code[20];iCodingSystem@1100528902 : Code[20];iCode@1100528903 : Code[20]);
    BEGIN
      IF TmpRec.GET(iVendor,'','','','','') THEN EXIT;

      TmpRec.INIT;
      TmpRec."Key Code1" := iVendor;
      TmpRec.Dec1 := GetTurnover(iVendor);
      TmpRec.INSERT;

      TurnoverTotal := TurnoverTotal + TmpRec.Dec1;
    END;

    PROCEDURE GetTurnover@1100528901(iVendor@1100528901 : Code[20]) Turnover : Decimal;
    VAR
      VatEntryRec@1100528900 : Record 254;
    BEGIN
      VatEntryRec.SETRANGE("Bill-to/Pay-to No.", iVendor);
      VatEntryRec.SETFILTER("Posting Date", PeriodFilterTxt);
      IF VatEntryRec.FINDSET THEN BEGIN
        REPEAT
          Turnover := Turnover + VatEntryRec.Base;
        UNTIL VatEntryRec.NEXT = 0;
      END;
      EXIT(Turnover);
    END;

    BEGIN
    END.
