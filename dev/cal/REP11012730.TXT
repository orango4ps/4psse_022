    PROCEDURE ProcessCostPlusEntries@16(Quantity@11012000 : Decimal;"Invoice Price"@1100485000 : Decimal;"VAT Prod. Posting Group"@11012003 : Code[10];Invoiced@1210190000 : Boolean;Chargeable@1100525000 : Boolean;CostPlusServOrder@1100525002 : Code[20];CostPlusLine@1100525001 : Integer) : Boolean;
    VAR
      ServiceLedgerEntry@1100525005 : Record 11012819;
      SalesInvoiceHeader@1100525004 : Record 112;
      ServiceCollectiveListMgt@1100525006 : Codeunit 11012850;
      VATBusPostingGroup@1100525003 : Code[20];
    BEGIN
      IF Invoiced AND NOT PrintProcessedLines THEN EXIT(FALSE);

      IF NOT Invoiced AND NOT Chargeable AND PrintChargeableLinesOnly THEN EXIT(FALSE);

      //Calculate Amounts per Settlement Sheet
      AmountInc := "Invoice Price";
      AmountCTInc += AmountInc;
      AmountSOInc += AmountInc;
      AmountSSInc += AmountInc;

      //VAT
      // sn CALL C005313
      IF Invoiced THEN BEGIN
        VATBusPostingGroup := CustomerRec."VAT Bus. Posting Group";
        IF (CostPlusServOrder <> '') THEN BEGIN
          ServiceLedgerEntry.SETCURRENTKEY("Service Order No.", "Cost Plus Line No.");
          ServiceLedgerEntry.SETRANGE("Document Type", ServiceLedgerEntry."Document Type"::"Sales Invoice");
          ServiceLedgerEntry.SETRANGE("Service Order No.", CostPlusServOrder);
          ServiceLedgerEntry.SETRANGE("Cost Plus Line No.", CostPlusLine);
          IF ServiceLedgerEntry.FINDFIRST THEN
            IF SalesInvoiceHeader.GET(ServiceLedgerEntry."Document No.") THEN
              IF SalesInvoiceHeader."VAT Bus. Posting Group" <> '' THEN
                VATBusPostingGroup := SalesInvoiceHeader."VAT Bus. Posting Group";
          IF NOT VATPostingSetup.GET(VATBusPostingGroup,"VAT Prod. Posting Group") THEN
            VATPostingSetup.INIT;
        END ELSE BEGIN
          ServiceCollectiveListMgt.FilterSCInvLinesInvoicedAmt("Collective-List SC Inv. Line", ServiceLedgerEntry);
          ServiceLedgerEntry.SETRANGE("Document Type", ServiceLedgerEntry."Document Type"::"Sales Invoice");
          IF ServiceLedgerEntry.FINDFIRST THEN
            IF SalesInvoiceHeader.GET(ServiceLedgerEntry."Document No.") THEN
              IF SalesInvoiceHeader."VAT Bus. Posting Group" <> '' THEN
                VATBusPostingGroup := SalesInvoiceHeader."VAT Bus. Posting Group";
          IF NOT VATPostingSetup.GET(VATBusPostingGroup,"VAT Prod. Posting Group") THEN
            VATPostingSetup.INIT;
        END;
      END ELSE BEGIN
      // en CAL C005313
        IF NOT VATPostingSetup.GET(CustomerRec."VAT Bus. Posting Group","VAT Prod. Posting Group") THEN
          VATPostingSetup.INIT;
      END; //n CAL C005313
      "VAT %" := VATPostingSetup."VAT %";
      CASE VATPostingSetup."VAT Calculation Type" OF
        VATPostingSetup."VAT Calculation Type"::"Reverse Charge VAT",
        VATPostingSetup."VAT Calculation Type"::"Sales Tax":
          "VAT %" := 0;
      END;

      CASE VATPostingSetup."VAT Calculation Type" OF
        VATPostingSetup."VAT Calculation Type"::"Normal VAT",
        VATPostingSetup."VAT Calculation Type"::"Reverse Charge VAT":
          BEGIN
            "VAT Base" := AmountInc;
            "Amount Including VAT" := ROUND(AmountInc + "VAT Base" * "VAT %" / 100,0.00001);
          END;
        VATPostingSetup."VAT Calculation Type"::"Full VAT":
          BEGIN
            "VAT Base" := 0;
            "Amount Including VAT" := AmountInc;
          END;
        VATPostingSetup."VAT Calculation Type"::"Sales Tax":
      //  Voorlopig niet ondersteund. Is alleen van toepassing op de Amerikaanse markt
          BEGIN
            "VAT Base" := AmountInc;
            "Amount Including VAT" := AmountInc;
          END;
      END;

      "AmountSS incl.VAT" += "Amount Including VAT";

      VATAmountLine.INIT;
      VATAmountLine."VAT Identifier" := VATPostingSetup."VAT Identifier";
      VATAmountLine."VAT Calculation Type" := VATPostingSetup."VAT Calculation Type";
      VATAmountLine."VAT %" := "VAT %";
      VATAmountLine."VAT Base" := "VAT Base";
      VATAmountLine."Amount Including VAT" := "Amount Including VAT";
      VATAmountLine.InsertLine;

      PrintTotal := TRUE;

      EXIT(TRUE);
    END;

    If Not (group1 = currentgroup1)
