    PROCEDURE InsertFeature@1100285006(pJobNo@1100285000 : Code[20];pFeatureGroup@1100285001 : Text[50];pFeature@1100285002 : Code[10]);
    VAR
      ContactFeatureType@1100285003 : Record 11012753;
      lvLineNo@1100285004 : Integer;
    BEGIN
      WITH ContactFeatureType DO BEGIN
        SETRANGE("Relation Type", "Relation Type"::Job);
        SETRANGE(Code, Job."No.");
        IF FINDLAST THEN
          lvLineNo := ContactFeatureType."Line No." + 10000
        ELSE
          lvLineNo := 10000;

        SETRANGE("Feature Group", pFeatureGroup);
        IF FINDFIRST THEN BEGIN
          VALIDATE(Feature, pFeature);
          MODIFY(TRUE);
        END ELSE BEGIN
          INIT;
          VALIDATE("Relation Type", "Relation Type"::Job);
          VALIDATE(Code, pJobNo);
          VALIDATE("Line No.", lvLineNo);
          VALIDATE("Feature Group", pFeatureGroup);
          VALIDATE(Feature, pFeature);
          INSERT(TRUE);
        END;
      END;
    END;
