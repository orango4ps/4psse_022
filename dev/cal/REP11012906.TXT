    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    BEGIN
      IF "Project Type".GETFILTERS <> '' THEN

    PROCEDURE CalculateAmounts@1210190000();
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        //* Cost
        CalcCost := TRUE;
        Type := Type::Subtotal;
        Totaling := '..' + LastRubric;
        ActualCost := ActualCost + CalculateRubricActual();
        BudgetCost := BudgetCost + CalculateRubricBudget();
        CalculatePeriodAmounts();

        //* Revenues
        CalcCost := FALSE;
        Type := Type::Subtotal;
        Totaling := '..' + LastRubric;
        ActualRev := ActualRev + CalculateRubricActual();
        BudgetRev := BudgetRev + CalculateRubricBudget();
        CalculatePeriodAmounts();

        //* EPV
        Type := Type::EPV;
        Totaling := '';
        AmntEPV := AmntEPV + ProjectDevelopmentFunctions.CalculateTotalEPV("Project No.", Version, ReferenceDate, PeriodType, FALSE);
        ProjectDevelopmentFunctions.AddFromEPVCalcTableToTotal(TempEPVCalculationProjectTypeTotal);
        ProjectDevelopmentFunctions.AddFromEPVCalcTableToTotal(TempEPVCalculationGeneralTotal);
      END;
    END;

    PROCEDURE CalculateRubricActual@1210190007() : Decimal;
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        CASE Type OF
          Type::Subtotal:
            EXIT(ProjectDevelopmentFunctions.CalculateActualByRubricCostRev(InvestmProposalRubricLine, 0D, 0D, ReferenceDate, CalcCost));
          Type::Balance:
            EXIT(ProjectDevelopmentFunctions.CalculateActualByRubric(InvestmProposalRubricLine, 0D, 0D, ReferenceDate));
        END;
        EXIT(0);
      END;
    END;

    PROCEDURE CalculateRubricBudget@1210190011() : Decimal;
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        CASE Type OF
          Type::Subtotal:
            EXIT(ProjectDevelopmentFunctions.CalculateBudgetByRubricCostRev(InvestmProposalRubricLine, 0D, 0D, ReferenceDate, CalcCost));
          Type::Balance:
            EXIT(ProjectDevelopmentFunctions.CalculateBudgetByRubric(InvestmProposalRubricLine, 0D, 0D, ReferenceDate));
        END;
        EXIT(0);
      END;
    END;

    PROCEDURE CalculatePeriodAmounts@1210190006();
    VAR
      Amount@1210190000 : Decimal;
      Counter@1100525000 : Integer;
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        FOR Counter := 1 TO NoOfColumn DO BEGIN
          IF Counter <= LastActualColumn THEN
            Amount := ProjectDevelopmentFunctions.CalculateActualByRubricCostRev(InvestmProposalRubricLine,
                          GetStartDate(Counter), EndDateArray[Counter], ReferenceDate, CalcCost)
          ELSE
            Amount := ProjectDevelopmentFunctions.CalculateBudgetByRubricCostRev(InvestmProposalRubricLine,
                          StartDateArray[Counter], GetEndDate(Counter), ReferenceDate, CalcCost);
          IF CalcCost THEN
            YearAmntCostArray[Counter] := YearAmntCostArray[Counter] + Amount
          ELSE
            YearAmntRevArray[Counter] := YearAmntRevArray[Counter] + Amount;
        END;
      END;
    END;

    PROCEDURE GetStartDate@1210190002(Counter@1210190001 : Integer) : Date;
    BEGIN
      IF Counter > 1 THEN
        EXIT(StartDateArray[Counter])
      ELSE
      //  EXIT(DMY2DATE(1, 1, 1));  //* Not 0D !
        EXIT(0D);
    END;

    PROCEDURE GetEndDate@1210190001(Counter@1210190001 : Integer) : Date;
    BEGIN
      IF Counter < NoOfColumn THEN
        EXIT(EndDateArray[Counter])
      ELSE
        EXIT(NORMALDATE(EndDateArray[Counter]) + 36500);
    END;

    BEGIN
    END.
