    PROCEDURE GetReportFilter1@1100525000() : Text;

      IF Job.GETFILTER("No.") <> '' THEN

        IF FilterString <> '' THEN

        FilterString := Job.FIELDCAPTION(Description) + ': ' +  Job.GETFILTER(Description);
      END;

        IF FilterString <> '' THEN
          FilterString += ' / ';

        IF FilterString <> '' THEN
          FilterString += ' / ';

        IF FilterString <> '' THEN
          FilterString += ' / ';
        FilterString += PaidCaption;
      END;

        IF FilterString <> '' THEN
          FilterString += ' / ';
        FilterString += OverdueCaption;
      END;

      IF NotApproved THEN BEGIN
        IF FilterString <> '' THEN
          FilterString += ' / ';
