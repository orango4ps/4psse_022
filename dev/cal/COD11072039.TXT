OBJECT Codeunit 11072039 FA Reclass. Jnl.-B.Transfer
{
  OBJECT-PROPERTIES
  {
    Date=;
    Time=;
    Version List=NAVW17.00,4PS14.00 (5645);
  }
  PROPERTIES
  {
    TableNo=11072083;
    OnRun=BEGIN
            FAReclassJnlBatch.COPY(Rec);
            Code;
            COPY(FAReclassJnlBatch);
          END;

  }
  CODE
  {
    VAR
      Text000@1000 : TextConst 'DEU=Mchten Sie die Buch.-Bltter umbuchen?;ENU=Do you want to reclassify the journals?;NLD=Wilt u de dagboeken herindelen?;NOR=Vil du reklassifisere journalene?;SVE=Vill du gruppera journalerna?';
      Text001@1001 : TextConst 'DEU=Die Buch.-Bltter wurden erfolgreich umgebucht.;ENU=The journals were successfully reclassified.;NLD=De dagboeken werden probleemloos heringedeeld.;NOR=Reklassifiseringen av journalene er klar.;SVE=Grupperingen av journalerna r klar.';
      Text002@1002 : TextConst 'DEU="Es war nicht mglich alle Buch.-Bltter umzubuchen. ";ENU="It was not possible to reclassify all of the journals. ";NLD="Niet alle dagboeken konden heringedeeld worden. ";NOR=Det var ikke mulig  reklassifisere alle journaler.;SVE="Det var inte mjligt att gruppera alla journaler. "';
      Text003@1003 : TextConst 'DEU=Die ungebuchten Umb. Buch.-Bltter wurden markiert.;ENU=The journals that were not successfully reclassified are now marked.;NLD=De dagboeken die niet zonder problemen konden heringedeeld worden, zijn gemarkeerd.;NOR=Journalene som ikke ble reklassifisert er n markerte.;SVE=Journalerna som inte blev grupperade r nu mrkta.';
      FAReclassJnlTempl@1004 : Record 11072082;
      FAReclassJnlBatch@1005 : Record 11072083;
      FAReclassJnlLine@1006 : Record 11072084;
      FAReclassTransferBatch@1007 : Codeunit 11072037;
      JournalWithErrors@1008 : Boolean;

    LOCAL PROCEDURE Code@1();
    BEGIN
      WITH FAReclassJnlBatch DO BEGIN
        FAReclassJnlTempl.GET("Journal Template Name");

        IF NOT CONFIRM(Text000,FALSE) THEN
          EXIT;

        FIND('-');
        REPEAT
          FAReclassJnlLine."Journal Template Name" := "Journal Template Name";
          FAReclassJnlLine."Journal Batch Name" := Name;
          FAReclassJnlLine."Line No." := 1;

          CLEAR(FAReclassTransferBatch);
          IF FAReclassTransferBatch.RUN(FAReclassJnlLine) THEN
            MARK(FALSE)
          ELSE BEGIN
            MARK(TRUE);
            JournalWithErrors := TRUE;
          END;
        UNTIL NEXT = 0;

        IF NOT JournalWithErrors THEN
          MESSAGE(Text001)
        ELSE
          MESSAGE(
            Text002 +
            Text003);

        IF NOT FIND('=><') THEN BEGIN
          RESET;
          FILTERGROUP := 2;
          SETRANGE("Journal Template Name","Journal Template Name");
          FILTERGROUP := 0;
          Name := '';
        END;
      END;
    END;

    BEGIN
    END.
  }
}
