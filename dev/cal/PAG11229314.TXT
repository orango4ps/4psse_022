    PROCEDURE UpdateQuantity@1210190001(NewQuantity@1100525000 : Decimal);
    VAR
      EstimateModelLineQuantity@1100525001 : Record 11012197;
    BEGIN
      MATRIX_ColumnTempRec.TESTFIELD(Model);

      IF EstimateModelLineQuantity.GET(
