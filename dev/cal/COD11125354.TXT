    PROCEDURE CheckDoubleWebServiceNames@1100528307();
    VAR
      WebServiceSetLine@1100528300 : Record 11229309;
      TenantWebService@1100528301 : Record 2000000168;
      WebService@1100528302 : Record 2000000076;
      DoubleServiceNames@1100528303 : Text;
      ErrorFound@1100528304 : Boolean;
    BEGIN
      IF (NOT WebServiceSetLine.FINDSET(TRUE)) THEN
        EXIT;

      REPEAT
        ErrorFound := FALSE;
        IF (PermissionManager.SoftwareAsAService) THEN BEGIN
          TenantWebService.SETFILTER("Service Name", '%1', '@'+WebServiceSetLine."Service Name");
          IF (TenantWebService.COUNT > 1) THEN BEGIN
            TenantWebService.DELETEALL;
            WebServiceSetLine.Published := FALSE;
            WebServiceSetLine.MODIFY;
            ErrorFound := TRUE;
          END;
        END ELSE BEGIN
          WebService.SETRANGE("Service Name", WebServiceSetLine."Service Name");
          IF (WebService.COUNT > 1) THEN BEGIN
            WebService.DELETEALL;
            WebServiceSetLine.Published := FALSE;
            WebServiceSetLine.MODIFY;
            ErrorFound := TRUE;
          END;
        END;
        IF (ErrorFound) THEN BEGIN
          IF (DoubleServiceNames <> '') THEN
            DoubleServiceNames += ', ';
          DoubleServiceNames += WebServiceSetLine."Service Name";
        END;
      UNTIL (WebServiceSetLine.NEXT() = 0);

      IF GUIALLOWED AND (DoubleServiceNames <> '') THEN
        MESSAGE(DoubleWebServicesFoundTxt, DoubleServiceNames);
    END;

    LOCAL PROCEDURE CleanupWebServiceSet@1100528310();
    VAR
      WebServiceSet@1100528300 : Record 11229298;
      WebServiceSetLine@1100528301 : Record 11229309;
    BEGIN
      WebServiceSet.SETRANGE("Is Standard", TRUE);
      IF (NOT WebServiceSet.FINDSET) THEN
        EXIT;

      REPEAT
        WebServiceSet.TogglePublishWebService(WebServiceSet, FALSE);
      UNTIL (WebServiceSet.NEXT = 0);

      IF (NOT WebServiceSet.FINDSET(TRUE, TRUE)) THEN
        EXIT;

      REPEAT
        WebServiceSetLine.SETRANGE("Web Service Set Code", WebServiceSet.Code);
        IF (WebServiceSetLine.COUNT = 0) THEN
          WebServiceSet.DELETE;
      UNTIL (WebServiceSet.NEXT = 0);
    END;

    LOCAL PROCEDURE CleanupWebServiceSetLine@1100528311();
    VAR
      WebServiceSetLine@1100528300 : Record 11229309;
    BEGIN
      WebServiceSetLine.SETFILTER("Service Name", '<>%1', '');
      WebServiceSetLine.SETRANGE("Object Id", 0);
      WebServiceSetLine.DELETEALL;
    END;

    LOCAL PROCEDURE FillTempWebServiceSetLine@1100528312();
    VAR
      WebServiceSetLine@1100528300 : Record 11229309;
    BEGIN
      TempWebServiceSetLine.DELETEALL;
      WebServiceSetLine.SETRANGE(Published, TRUE);
      IF (NOT WebServiceSetLine.FINDSET) THEN
          EXIT;

      REPEAT
        TempWebServiceSetLine.COPY(WebServiceSetLine);
        TempWebServiceSetLine."Object Type" := 0;
        TempWebServiceSetLine."Service Name" := '';
        IF (TempWebServiceSetLine.INSERT) THEN;
      UNTIL (WebServiceSetLine.NEXT = 0);
    END;

    PROCEDURE AddPossibleWebServiceObject@1100525027(ObjectType@1100525001 : Option;ObjectId@1100525002 : Integer;ServiceName@1100525003 : Text);
    VAR
      PossibleWebService@1100525000 : Record 11229317;
    BEGIN
      IF (NOT PossibleWebService.GET(ObjectType, ServiceName)) THEN BEGIN
        PossibleWebService."Object Type" := ObjectType;
        PossibleWebService."Service Name" := ServiceName;
        PossibleWebService.INSERT;
      END;
      PossibleWebService."Object Id" := ObjectId;
      PossibleWebService."Is Standard" := TRUE;
      PossibleWebService.GetInLicense;
      PossibleWebService.MODIFY;

      AddWebServiceSetLine(PossibleWebService);
    END;

    END;
