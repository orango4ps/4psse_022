    { 1100525000;;DataItem;                  ;
               DataItemTable=Table5200;
               DataItemTableView=SORTING(No.)

               OnPreDataItem=BEGIN

               OnAfterGetRecord=VAR
                                  PostedHourLine@1100525001 : Record 11012085;
                                  HourAccountingLine@1100525000 : Record 11012039;
                                  DimensionValue@1100525002 : Record 349;
                                BEGIN
                                  CostCenterDescription := '';
                                  DimensionValue.SETRANGE("Global Dimension No.",1);
                                  DimensionValue.SETRANGE(Code,"Global Dimension 1 Code");
                                  IF DimensionValue.FINDFIRST THEN
                                    CostCenterDescription := DimensionValue.Name;

                                  TotalHours := 0;
                                  TotaCumulativelHours := 0;
                                  PostedHourLine.SETFILTER(Year, '%1', Year);
                                  PostedHourLine.SETFILTER(Week, '%1', WeekNumber);
                                  PostedHourLine.SETRANGE("Employee No.", "No.");
                                  PostedHourLine.SETRANGE(Type, PostedHourLine.Type::Project);
                                  PostedHourLine.SETRANGE("Component Type",PostedHourLine."Component Type"::Hours);
                                  PostedHourLine.CALCSUMS("Total Line");
                                  TotalHours += PostedHourLine."Total Line";

                                  HourAccountingLine.SETFILTER(Year, '%1', Year);
                                  HourAccountingLine.SETFILTER(Week, '%1', WeekNumber);
                                  HourAccountingLine.SETRANGE("Employee No.", "No.");
                                  HourAccountingLine.SETRANGE(Type, HourAccountingLine.Type::Project);
                                  HourAccountingLine.SETRANGE("Component Type",HourAccountingLine."Component Type"::Hours);
                                  HourAccountingLine.CALCSUMS("Total Line");
                                  TotalHours += HourAccountingLine."Total Line";

                                  IF "Norm Hours" <> 0 THEN
                                    TotalHoursPercent := TotalHours / "Norm Hours" * 100;

                                  PostedHourLine.SETFILTER(Year, '%1', Year);
                                  PostedHourLine.SETFILTER(Week, '..%1', WeekNumber);
                                  PostedHourLine.SETRANGE("Employee No.", "No.");
                                  PostedHourLine.SETRANGE(Type, PostedHourLine.Type::Project);
                                  PostedHourLine.SETRANGE("Component Type",PostedHourLine."Component Type"::Hours);
                                  PostedHourLine.CALCSUMS("Total Line");
                                  TotaCumulativelHours += PostedHourLine."Total Line";

                                  HourAccountingLine.SETFILTER(Year, '%1', Year);
                                  HourAccountingLine.SETFILTER(Week, '..%1', WeekNumber);
                                  HourAccountingLine.SETRANGE("Employee No.", "No.");
                                  HourAccountingLine.SETRANGE(Type, HourAccountingLine.Type::Project);
                                  HourAccountingLine.SETRANGE("Component Type",HourAccountingLine."Component Type"::Hours);
                                  HourAccountingLine.CALCSUMS("Total Line");
                                  TotaCumulativelHours += HourAccountingLine."Total Line";

                                  IF "Norm Hours" <> 0 THEN
                                    TotalCumulativeHoursPercent := 100 * TotaCumulativelHours / ("Norm Hours" * WeekNumber);
                                END;
                                 }
