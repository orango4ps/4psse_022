    PROCEDURE CheckRelatedSalesOrderReleased@1100485000();
    VAR
      lvSalesRelOrderRec@1100485000 : Record 11012787;
      lvSalesOrderHeadRec@1100485001 : Record 36;
      lvText002@1100485002 : TextConst 'DEU=Drucken Werkzeugauftrag ''%1'' nicht zul„ssig, verbundener Vermietungsauftrag ''%2'' ist in Verkaufsbetrieb ''%3'' nicht freigegeben worden.;ENU=Printing Plant Order ''%1'' not allowed, related Sales Rental Order ''%2'' is not released in Sales Company ''%3''.;NLD=Afdrukken materieelorder ''%1'' niet toegestaan, gerelateerde verhuurorder ''%2'' is niet vrijgegeven in verkoopbedrijf ''%3''.;NOR=Det er ikke tillatt † skrive ut maskinordre ''%1''. Den relaterte utleieordren (salg) ''%2'' er ikke frigitt i salgsbedriften ''%3''.;SVE=Det „r inte till†tet att skriva ut Maskinsorder ''%1''. Den relaterade uthyrningsordern (f”rs„ljning) ''%2'' har inte sl„ppts i f”rs„ljningsf”retaget ''%3''.';
      lvSalesRemovalRec@1100525002 : Record 11012789;
      lvText003@1100525000 : TextConst 'DEU=Drucken Werkzeugauftrag ''%1'' nicht zul„ssig, Abfuhrzeilen des verbundenen Vermietungsauftrags ''%2'' ist in Verkaufsbetrieb ''%3'' nicht freigegeben worden.;ENU=Printing Plant Order ''%1'' not allowed, removal lines of related Sales Rental Order ''%2'' not released in Sales Company ''%3''.;NLD=Afdrukken materieelorder ''%1'' niet toegestaan, afvoerregels van gerelateerde verhuurorder ''%2'' niet vrijgegeven in verkoopbedrijf ''%3''.;NOR=Det er ikke tillatt † skrive ut maskinordre ''%1''. Fjerningsradene i den relaterte utleieordren (salg) ''%2'' er ikke frigitt i salgsbedriften ''%3''.;SVE=Det „r inte till†tet att skriva ut Maskinsorder ''%1''. Borttagningsraderna i den relaterade uthyrningsordern (f”rs„ljning) ''%2'' har inte sl„ppts i f”rs„ljningsf”retaget ''%3''.';
      lvText004@1100525003 : TextConst 'DEU=Drucken Werkzeugauftrag ''%1'' nicht zul„ssig, Abfuhrzeilen des verbundenen Vermietungsauftrags ''%2'' (Verkaufsbetrieb ''%3'') sind noch nicht nochmals in Logistikbetrieb erstellt worden.;ENU=Printing Plant Order ''%1'' not allowed, removal lines of related Sales Rental Order ''%2'' (Sales Company ''%3'') not yet recreated in Logistic Company.;NLD=Afdrukken materieelorder ''%1'' niet toegestaan, afvoerregels van gerelateerde verhuurorder ''%2'' (verkoopbedrijf ''%3'') nog niet opnieuw aangemaakt in logistiek bedrijf.;NOR=Det er ikke tillatt † skrive ut maskinordre ''%1''. Fjerningsradene i den relaterte utleieordren (salg) ''%2'' (Salgsbedriften ''%3'') er ikke gjenopprettet i Logistikkbedriften.;SVE=Det „r inte till†tet att skriva ut Maskinsorder ''%1''. Borttagningsraderna i den relaterade uthyrningsordern (f”rs„ljning) ''%2'' (f”rs„ljningsf”retaget ''%3'') har inte skapats „n i logistikf”retaget.';
    BEGIN
      IF "Plant Order"."Sales Rental Order No." = '' THEN  //* Sales Rental Contract or Sales Rental Order
        EXIT;

      //* Sales Logistics Separated
      lvSalesRelOrderRec.SETCURRENTKEY(Level,"Plant Document No.");
      lvSalesRelOrderRec.SETRANGE(Level, lvSalesRelOrderRec.Level::PlantOrder);
      lvSalesRelOrderRec.SETRANGE("Plant Document No.", "Plant Order"."No.");
      IF lvSalesRelOrderRec.FINDFIRST THEN BEGIN
        lvSalesOrderHeadRec.CHANGECOMPANY(lvSalesRelOrderRec."Sales Company Name");
        IF lvSalesOrderHeadRec.GET(lvSalesOrderHeadRec."Document Type"::Order, lvSalesRelOrderRec."Sales Order No.") THEN BEGIN
          IF lvSalesOrderHeadRec."Sales Document Type" =
            lvSalesOrderHeadRec."Sales Document Type"::"Sales Logistics Separated"
          THEN BEGIN
            IF lvSalesOrderHeadRec.Status <> lvSalesOrderHeadRec.Status::Released THEN
              ERROR(lvText002, "Plant Order"."No.", lvSalesRelOrderRec."Sales Order No.", lvSalesRelOrderRec."Sales Company Name");
            IF "Plant Order".Type = "Plant Order".Type::Removal THEN BEGIN
              lvSalesRemovalRec.CHANGECOMPANY(lvSalesRelOrderRec."Sales Company Name");
              lvSalesRemovalRec.SETRANGE("Document No.", lvSalesRelOrderRec."Sales Order No.");
              lvSalesRemovalRec.SETRANGE(Type, lvSalesRemovalRec.Type::Removal);
              lvSalesRemovalRec.SETRANGE("Removal Order Type",lvSalesRemovalRec."Removal Order Type"::"Plant Order");
              lvSalesRemovalRec.SETRANGE("Removal Order No.", "Plant Order"."No.");
              lvSalesRemovalRec.SETRANGE(Status, lvSalesRemovalRec.Status::Open, lvSalesRemovalRec.Status::Released);
              //* All lines of the PO must be processed (released in sales and create PO in logistics).
              //* If the removal lines are reopend and then modified and released again then PO is filled in the removal lines,
              //* but first a new PO must be created in logistics (then old-PO will be deleted)
              IF lvSalesRemovalRec.FINDFIRST THEN BEGIN
                IF lvSalesRemovalRec.Status = lvSalesRemovalRec.Status::Open THEN
                  ERROR(lvText003, "Plant Order"."No.", lvSalesRelOrderRec."Sales Order No.", lvSalesRelOrderRec."Sales Company Name")
                ELSE
                  ERROR(lvText004, "Plant Order"."No.", lvSalesRelOrderRec."Sales Order No.", lvSalesRelOrderRec."Sales Company Name");
              END;
            END;
          END;
        END;
      END;
    END;

    BEGIN
