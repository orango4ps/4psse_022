    LOCAL PROCEDURE GetReportFilter@1100525004() : Text;
    BEGIN
      IF "Import Discount Term".GETFILTERS <> '' THEN
        EXIT(STRSUBSTNO('%1: %2', "Import Discount Term".TABLECAPTION, "Import Discount Term".GETFILTERS));
    END;

    PROCEDURE FillTmpVendor@1100525000();
    BEGIN
      //db, 04-11-13
      IF TmpVendorRec.GET("Import Discount Term".Vendor) THEN EXIT;
      TmpVendorRec.Code := "Import Discount Term".Vendor;
      TmpVendorRec.INSERT;

      IF VendorFilterTxt = '' THEN BEGIN
        VendorFilterTxt := TmpVendorRec.Code;
      END ELSE BEGIN
        IF STRLEN(VendorFilterTxt) < 1000 THEN
          VendorFilterTxt := VendorFilterTxt + '|' + TmpVendorRec.Code;
      END;
    END;

    PROCEDURE VendorInFilter@1100525001() : Boolean;
    BEGIN
      //db, 04-11-13
      IF TmpVendorRec.GET("Discount Term History".Vendor) THEN
        EXIT(TRUE)
      ELSE
        EXIT(FALSE);
    END;

    BEGIN
    END.
