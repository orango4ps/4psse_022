    PROCEDURE GetReasonCodeExit@1100409001(PlantType@1100409000 : Code[20];PlantNo@1100409001 : Code[20]) retValue : Code[10];
    VAR
      PlantInventory@1100409002 : Record 11012555;
      PlantOrderLine@1100409003 : Record 11012557;
    BEGIN
      PlantInventory.SETCURRENTKEY(Removed, "Plant Type", "Plant No.");
      PlantInventory.SETRANGE(Removed, TRUE);
      PlantInventory.SETRANGE("Plant Type", PlantType);
      PlantInventory.SETRANGE("Plant No.", PlantNo);
      PlantInventory.SETRANGE("Removal Order Type", PlantInventory."Removal Order Type"::"Exit Order");
      IF NOT PlantInventory.FINDFIRST THEN
        EXIT;
      PlantOrderLine.SETRANGE("Plant Order No.", PlantInventory."Arrival Order");
      PlantOrderLine.SETRANGE("Line No.", PlantInventory."Arrival Order Line");
      IF NOT PlantOrderLine.FINDFIRST THEN
        EXIT;
      EXIT(PlantOrderLine."Reason Code (Exit)");
    END;

    BEGIN
    END.
