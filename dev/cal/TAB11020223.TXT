    VAR
      DimValRec@11012000 : Record 349;
      DimMgt@11012001 : Codeunit 408;

    PROCEDURE CalculateAllowed@1(VAR AllowedAmount@11012000 : Decimal;VAR AllowedHours@11012001 : Decimal;VAR PrevAllowedAmount@11012002 : Decimal;VAR PrevAllowedHours@11012003 : Decimal);
    VAR
      BudgetLineRec@11012004 : Record 11012001;
      ProjLedgerRec@11012005 : Record 11072005;
      PreviousCumulFilter@11012006 : Text[250];
      FiltGroup@11012007 : Integer;
    BEGIN
      AllowedAmount := 0;
      AllowedHours := 0;
      PrevAllowedAmount := 0;
      PrevAllowedHours := 0;

      DimMgt.GetDimValueRec(2, "Cost Object", DimValRec, FALSE, "Project No.");
      IF NOT DimValRec."Record Progress" THEN BEGIN
        //Determine Budget for Cost Objects wherefor no progress will be recorded
        BudgetLineRec.RESET;
        BudgetLineRec.SETCURRENTKEY(
          "Project No.","Extension Contract","Cost Type","Cost Object",
          "Extension Contract Status","Version Date");
        BudgetLineRec.SETRANGE("Project No.", "Project No.");
        BudgetLineRec.SETRANGE("Extension Contract", "Extension Contract");
        BudgetLineRec.SETRANGE("Cost Type", "Cost Type");
        BudgetLineRec.SETRANGE("Cost Object", "Cost Object");
        BudgetLineRec.SETFILTER(
          "Extension Contract Status",'%1|(>=%2&<%3)',
          BudgetLineRec."Extension Contract Status"::"Not Applicable",
          BudgetLineRec."Extension Contract Status"::Order,
          BudgetLineRec."Extension Contract Status"::Expired);
        FOR FiltGroup := 0 TO 4 DO BEGIN
          FILTERGROUP(FiltGroup);
          BudgetLineRec.FILTERGROUP(FiltGroup);
          BudgetLineRec.SETFILTER("Version Date",GETFILTER("Cumulation Filter"));
        END;
        FILTERGROUP(0);
        BudgetLineRec.FILTERGROUP(0);
        BudgetLineRec.CALCSUMS("Amount (LCY)", Hours);

        //Determine Actual for Cost Objects wherefor no progress will be recorded.
        ProjLedgerRec.RESET;
        ProjLedgerRec.SETCURRENTKEY("Job No.","Extension Contract","Cost Type","Global Dimension 2 Code","Entry Type","Posting Date");
        ProjLedgerRec.SETRANGE("Job No.", "Project No.");
        ProjLedgerRec.SETRANGE("Extension Contract", "Extension Contract");
        ProjLedgerRec.SETRANGE("Global Dimension 2 Code", "Cost Object");
        ProjLedgerRec.SETRANGE("Entry Type", ProjLedgerRec."Entry Type"::Usage);
        FOR FiltGroup := 0 TO 4 DO BEGIN
          FILTERGROUP(FiltGroup);
          ProjLedgerRec.FILTERGROUP(FiltGroup);
          ProjLedgerRec.SETFILTER("Posting Date",GETFILTER("Cumulation Filter"));
        END;
        FILTERGROUP(0);
        ProjLedgerRec.FILTERGROUP(0);
        ProjLedgerRec.CALCSUMS("Total Cost (LCY)", Quantity);

        //Determine Allowed
        IF BudgetLineRec."Amount (LCY)" < ProjLedgerRec."Total Cost (LCY)" THEN
          AllowedAmount := BudgetLineRec."Amount (LCY)"
        ELSE
          AllowedAmount := ProjLedgerRec."Total Cost (LCY)";

        IF BudgetLineRec.Hours < ProjLedgerRec.Quantity THEN
          AllowedHours := BudgetLineRec.Hours
        ELSE
          AllowedHours := ProjLedgerRec.Quantity;
      END ELSE BEGIN
        //Determine Allowed
        CALCFIELDS("Allowed Costs", "Allowed Hours");
        AllowedAmount := "Allowed Costs";
        AllowedHours := "Allowed Hours";
      END;
      IF AllowedAmount < 0 THEN
        AllowedAmount := 0;
      IF AllowedHours < 0 THEN
        AllowedHours :=0;


      //Determine Previous Dates
      PreviousCumulFilter := '';
      FOR FiltGroup := 4 DOWNTO 0 DO BEGIN
        FILTERGROUP(FiltGroup);
        IF (GETFILTER("Period Filter") <> '') THEN
          PreviousCumulFilter := '<'+FORMAT(GETRANGEMIN("Period Filter"));
      END;
      IF PreviousCumulFilter = '' THEN
        EXIT;

      IF NOT DimValRec."Record Progress" THEN BEGIN
        //Determine Budget previous periode for Cost Objects wherefor no progress will be recorded.
        BudgetLineRec.SETFILTER("Version Date",PreviousCumulFilter);
        BudgetLineRec.CALCSUMS("Amount (LCY)", Hours);

        //Determine Actual previous Period for Cost Objects for which no Progress will be recorded.
        ProjLedgerRec.SETFILTER("Posting Date",PreviousCumulFilter);
        ProjLedgerRec.CALCSUMS("Total Cost (LCY)", Quantity);

        //Determine Allowed
        IF BudgetLineRec."Amount (LCY)" < ProjLedgerRec."Total Cost (LCY)" THEN
          PrevAllowedAmount := BudgetLineRec."Amount (LCY)"
        ELSE
          PrevAllowedAmount := ProjLedgerRec."Total Cost (LCY)";

        IF BudgetLineRec.Hours < ProjLedgerRec.Quantity THEN
          PrevAllowedHours := BudgetLineRec.Hours
        ELSE
          PrevAllowedHours := ProjLedgerRec.Quantity;
      END ELSE BEGIN
        //Determine Allowed previous Period
        CALCFIELDS("Allowed Previous Period", "Allowed Hours Previous Period");
        PrevAllowedAmount := "Allowed Previous Period";
        PrevAllowedHours := "Allowed Hours Previous Period";
      END;
      IF PrevAllowedAmount < 0 THEN
        PrevAllowedAmount := 0;
      IF PrevAllowedHours < 0 THEN
        PrevAllowedHours :=0;
    END;

    BEGIN
    {
      140815 ITERO.DL IME155 exclude "Component Type"::'expense hours' from "Open Amount (Hours)", "Open Ovh. Surch. (Hours)", "Open Hours"
      190110 ORANGO.PR RFC226 Added optionvalue to "Wage Component Type Filter" - Addition
      190131 ORANGO.SB RFC-226 Added Wage Component Type Filter in field Actual Hours and Actual Hours Period
    }
    END.
