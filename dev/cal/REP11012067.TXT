    PROCEDURE DetermineProjectText@1100525001(CustLedgerEntry@1100525000 : Record 21) RetVal : Text[50];
    VAR
      SalesInvoiceLine@1100525004 : Record 113;
      SalesCrMemoLine@1100525002 : Record 115;
      ProjectNo@1100525006 : Code[20];
    BEGIN
      RetVal := '';
      CASE CustLedgerEntry."Document Type" OF
        CustLedgerEntry."Document Type"::Invoice:
          BEGIN
            SalesInvoiceLine.SETRANGE("Document No.", CustLedgerEntry."Document No.");
            SalesInvoiceLine.SETFILTER(Type, '<>%1', SalesInvoiceLine.Type::" ");
            SalesInvoiceLine.SETFILTER("Job No.", '<>%1', '');
            IF SalesInvoiceLine.FINDSET(FALSE, FALSE) THEN BEGIN
              ProjectNo := '';
              REPEAT
                IF (ProjectNo <> SalesInvoiceLine."Job No.") AND (ProjectNo <> '') THEN BEGIN
                  RetVal := Text001;
                  EXIT(RetVal);
                END ELSE
                  RetVal := SalesInvoiceLine."Job No.";
                ProjectNo := SalesInvoiceLine."Job No.";
              UNTIL SalesInvoiceLine.NEXT = 0;
            END;
          END;
        CustLedgerEntry."Document Type"::"Credit Memo":
          BEGIN
            SalesCrMemoLine.SETRANGE("Document No.", CustLedgerEntry."Document No.");
            SalesCrMemoLine.SETFILTER(Type, '<>%1', SalesCrMemoLine.Type::" ");
            SalesCrMemoLine.SETFILTER("Job No.", '<>%1', '');
            IF SalesCrMemoLine.FINDSET(FALSE, FALSE) THEN BEGIN
              ProjectNo := '';
              REPEAT
                IF (ProjectNo <> SalesCrMemoLine."Job No.") AND (ProjectNo <> '') THEN BEGIN
                  RetVal := Text001;
                  EXIT(RetVal);
                END ELSE
                  RetVal := SalesCrMemoLine."Job No.";
                ProjectNo := SalesCrMemoLine."Job No.";
              UNTIL SalesCrMemoLine.NEXT = 0;
            END;
          END;
      END;
    END;

    BEGIN
    END.
