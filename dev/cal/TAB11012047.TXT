    PROCEDURE CalculateAllowedAndPrevious@1(VAR AllowedAmount@11012000 : Decimal;VAR AllowedHours@11012001 : Decimal;VAR PrevAllowedAmount@11012002 : Decimal;VAR PrevAllowedHours@11012003 : Decimal);
    BEGIN

    PROCEDURE CalculateAllowedOnly@1100525003(VAR AllowedAmount@1100525001 : Decimal;VAR AllowedHours@1100525000 : Decimal);
    VAR
      DummyDec@1100525002 : Decimal;
    BEGIN
      CalculateAllowed(FALSE, AllowedAmount, AllowedHours, DummyDec, DummyDec);
    END;

    PROCEDURE CalculateAllowed@1100525002(ReadPreviousAllowed@1100525009 : Boolean;VAR AllowedAmount@1100525008 : Decimal;VAR AllowedHours@1100525007 : Decimal;VAR PrevAllowedAmount@1100525006 : Decimal;VAR PrevAllowedHours@1100525005 : Decimal);
    VAR
      lvJobRec@1100485000 : Record 11072003;
      lvAllowedAmount@1100485004 : Decimal;
      lvAllowedHours@1100485003 : Decimal;
      lvPrevAllowedAmount@1100485002 : Decimal;
      lvPrevAllowedHours@1100485001 : Decimal;
    BEGIN
      AllowedAmount := 0;
      AllowedHours := 0;
      PrevAllowedAmount := 0;
      PrevAllowedHours := 0;

      SetCompany;

      DimMgt.GetDimValueRec(2, "Cost Object", DimValRec, FALSE, '');

      IF DimValRec."Record Progress" THEN
        IF (GETFILTER("Progress Filter") = '''''') AND (GETFILTER(PrevProgressFilter) = '''''') THEN
          EXIT;

      ProjSetup.GET;

      CalculateAllowedSingleProject(
        ReadPreviousAllowed, AllowedAmount, AllowedHours, PrevAllowedAmount, PrevAllowedHours,
        "Project No.", DimValRec."Record Progress");

      CALCFIELDS("Single/Main/Sub Project");
      IF "Single/Main/Sub Project" = Rec."Single/Main/Sub Project"::"Main Project" THEN BEGIN
        //This is a main project, also add the allowed amounts/hours of the sub projects
        lvJobRec.RESET;
        IF CURRENTCOMPANY <> COMPANYNAME THEN
          lvJobRec.CHANGECOMPANY(CURRENTCOMPANY);
        lvJobRec.SETCURRENTKEY("Main Project");
        lvJobRec.SETFILTER("Main Project", "Project No.");
        //take filter into account because in some cases we do not want to cumulate over the
        //subprojects (e.g. in report 11012028)
        lvJobRec.SETFILTER("No.", GETFILTER("Project Filter"));
        //skip the main project now
        lvJobRec.SETFILTER("Single/Main/Sub Project",'%1', Rec."Single/Main/Sub Project"::"Sub Project");
        IF lvJobRec.FINDSET(FALSE, FALSE) THEN BEGIN
          REPEAT
            CalculateAllowedSingleProject(
              ReadPreviousAllowed, lvAllowedAmount, lvAllowedHours, lvPrevAllowedAmount,
              lvPrevAllowedHours, lvJobRec."No.", DimValRec."Record Progress");
            AllowedAmount := AllowedAmount + lvAllowedAmount;
            AllowedHours := AllowedHours + lvAllowedHours;
            PrevAllowedAmount := PrevAllowedAmount + lvPrevAllowedAmount;
            PrevAllowedHours := PrevAllowedHours + lvPrevAllowedHours;
          UNTIL lvJobRec.NEXT = 0;
        END;
      END;
    END;

    LOCAL PROCEDURE CalculateAllowedSingleProject@1100485000(ReadPreviousAllowed@1100525004 : Boolean;VAR AllowedAmount@1100485003 : Decimal;VAR AllowedHours@1100485002 : Decimal;VAR PrevAllowedAmount@1100485001 : Decimal;VAR PrevAllowedHours@1100485000 : Decimal;ProjNo@1100485008 : Code[20];RecordProgress@1100485009 : Boolean);
    VAR
      CostControlRec2@1100485013 : Record 11012047;
      CostContMgt@1100485010 : Codeunit 11012006;
      FiltGroup@1100485005 : Integer;
      PreviousCumulFilter@1100485004 : Text[250];
      TotCostAmnt@1100485011 : Decimal;
      TotHours@1100485012 : Decimal;
      Job@1100525000 : Record 11072003;
      DetermineProgressDate@1100525001 : Codeunit 11012014;
      DeterminePrognosisDate@1100525002 : Codeunit 11012029;
    BEGIN
      AllowedAmount := 0;
      AllowedHours := 0;
      PrevAllowedAmount := 0;
      PrevAllowedHours := 0;

      IF CURRENTCOMPANY <> COMPANYNAME THEN
        CostControlRec2.CHANGECOMPANY(CURRENTCOMPANY);

      //Check if Cost Control present on SubProject
      IF "Project No." <> ProjNo THEN BEGIN
        CostControlRec2 := Rec;
        CostControlRec2."Project No." := ProjNo;
        IF NOT CostControlRec2.FIND('=') THEN
          EXIT;
      END;
      //

      //Determine Budget
      CostControlRec2.COPY(Rec);
      CostControlRec2.SETRANGE("Project Filter", ProjNo);
      CostControlRec2.CALCFIELDS(Budget, "Budget Hours");

      //Progress dates and prognosis dates may differ for each subroject.
      IF Job.GET(ProjNo) THEN BEGIN
        FOR FiltGroup := 4 DOWNTO 0 DO BEGIN
          FILTERGROUP(FiltGroup);
          Job.FILTERGROUP(FiltGroup);
          IF GETFILTER("Period Filter") = '' THEN
            Job.SETFILTER("Period Filter",GETFILTER("Cumulation Filter"))
          ELSE
            Job.SETFILTER("Period Filter",GETFILTER("Period Filter"));
          Job.SETFILTER("Cumulation Filter",GETFILTER("Cumulation Filter"));
        END;
        IF ReadPreviousAllowed THEN BEGIN
          DetermineProgressDate.DetermineBothDates(Job,TRUE);
          DeterminePrognosisDate.DetermineBothDates(Job);
        END ELSE BEGIN
          DetermineProgressDate.DetermineDate(Job,TRUE);
          DeterminePrognosisDate.DetermineDate(Job);
        END;
        FOR FiltGroup := 4 DOWNTO 0 DO BEGIN
          CostControlRec2.FILTERGROUP(FiltGroup);
          Job.FILTERGROUP(FiltGroup);
          CostControlRec2.SETFILTER("Progress Filter",Job.GETFILTER("Progress Filter"));
          CostControlRec2.SETFILTER(PrevProgressFilter,Job.GETFILTER(PrevProgressFilter));
          CostControlRec2.SETFILTER("Prognosis Filter",Job.GETFILTER("Prognosis Filter"));
          CostControlRec2.SETFILTER(PrevPrognosisFilter,Job.GETFILTER(PrevPrognosisFilter));
        END;
      END;

      IF NOT RecordProgress THEN BEGIN
        //Determine Total Cost for Cost Objects where no progress will be recorded for.
        TotCostAmnt := CostContMgt.TotCostCostObj(CostControlRec2,TRUE);
        TotHours := CostContMgt.TotHoursCostObj(CostControlRec2,TRUE);
        //Determine Allowed
        AllowedAmount := DetermineAllowed(CostControlRec2.Budget,TotCostAmnt);
        AllowedHours := DetermineAllowed(CostControlRec2."Budget Hours",TotHours);
      END ELSE BEGIN
        //Determine Allowed
        CostControlRec2.CALCFIELDS("Allowed Costs", "Allowed Hours");
        AllowedAmount := CostControlRec2."Allowed Costs";
        AllowedHours := CostControlRec2."Allowed Hours";
        CheckMaxAllowedWithBudget(CostControlRec2.Budget, AllowedAmount);
        CheckMaxAllowedWithBudget(CostControlRec2."Budget Hours", AllowedHours);
      END;

      IF NOT ReadPreviousAllowed THEN
        EXIT;

      //Determine Previous Cumulation Filter
      PreviousCumulFilter := '';
      FOR FiltGroup := 4 DOWNTO 0 DO BEGIN
        FILTERGROUP(FiltGroup);
        IF (GETFILTER("Period Filter") <> '') THEN
          PreviousCumulFilter := '<'+FORMAT(GETRANGEMIN("Period Filter"));
      END;
      IF PreviousCumulFilter IN ['','<'] THEN //C040378.c
        EXIT;

      //Determine previous Budget
      CostControlRec2.SETFILTER("Cumulation Filter", PreviousCumulFilter);
      CostControlRec2.CALCFIELDS(Budget, "Budget Hours");

      IF NOT RecordProgress THEN BEGIN
        //Determine Total Cost previous periode for Cost Objects for which no progress will be recorded.
        TotCostAmnt := CostContMgt.TotCostCostObj(CostControlRec2,TRUE);
        TotHours := CostContMgt.TotHoursCostObj(CostControlRec2,TRUE);
        //Determine Allowed
        PrevAllowedAmount := DetermineAllowed(CostControlRec2.Budget,TotCostAmnt);
        PrevAllowedHours := DetermineAllowed(CostControlRec2."Budget Hours",TotHours);
      END ELSE BEGIN
        //Determine Allowed previous Period
        CostControlRec2.CALCFIELDS("Allowed Previous Period", "Allowed Hours Previous Period");
        PrevAllowedAmount := CostControlRec2."Allowed Previous Period";
        PrevAllowedHours := CostControlRec2."Allowed Hours Previous Period";
        CheckMaxAllowedWithBudget(CostControlRec2.Budget, PrevAllowedAmount);
        CheckMaxAllowedWithBudget(CostControlRec2."Budget Hours", PrevAllowedHours);
      END;
    END;

    PROCEDURE CalculatePurchaseResults@1210190000(AllowedDec@1210190007 : Decimal;VAR PurchaseDec@1210190009 : Decimal;VAR EstimPurchResultDec@1210190003 : Decimal;VAR PurchCostDec@1210190002 : Decimal;VAR ActualPurchResultDec@1210190001 : Decimal;VAR TotPurchResultDec@1210190000 : Decimal);
    VAR
      CostControlRec2@1210190004 : Record 11012047;
      ProjSetup@1210190005 : Record 315;
      ProjectCostObjectRec@1100485000 : Record 11012065;
      ProjectInProcess@1210190006 : Boolean;
    BEGIN
      CALCFIELDS("Purchase Budget", Budget, "Actual Costs", Prognosis, "Risc Coverage");

      SetCompany;
      IF CURRENTCOMPANY <> COMPANYNAME THEN BEGIN
        CostControlRec2.CHANGECOMPANY(CURRENTCOMPANY);
        ProjSetup.CHANGECOMPANY(CURRENTCOMPANY);
        ProjectCostObjectRec.CHANGECOMPANY(CURRENTCOMPANY);
      END;
      ProjSetup.GET;

      CostControlRec2.COPY(Rec);
      CostControlRec2.SETFILTER("Purchase Action Filter", '<>%1', '');
      CostControlRec2.CALCFIELDS(Purchase, "Actual Costs");

      PurchaseDec := CostControlRec2.Purchase;

      IF ProjSetup."Prognosis Total Risk Coverage" THEN BEGIN
        EstimPurchResultDec := ("Purchase Budget" - CostControlRec2.Purchase) - "Risc Coverage";
      END ELSE BEGIN
        EstimPurchResultDec := "Purchase Budget" - CostControlRec2.Purchase;
      END;

      PurchCostDec := CostControlRec2."Actual Costs";

      IF "Cost Object" <> DimValRec.Code THEN
        DimMgt.GetDimValueRec(2, "Cost Object", DimValRec, FALSE, '');

      IF ProjSetup."Act. purchase result by action" THEN BEGIN
        //ABB
        IF (DimValRec."Record Progress" ) AND (Budget <> 0) THEN
          ActualPurchResultDec := ((Budget - "Purchase Budget") * (AllowedDec / Budget)) -
                                  ("Actual Costs" - CostControlRec2."Actual Costs")
        ELSE
          ActualPurchResultDec := (Budget - "Purchase Budget") - ("Actual Costs" - CostControlRec2."Actual Costs");
      END ELSE BEGIN
        //BVR
        IF (DimValRec."Record Progress" ) THEN
          IF Budget <> 0 THEN
            ActualPurchResultDec := AllowedDec - "Actual Costs" - ((AllowedDec / Budget) * EstimPurchResultDec)
          ELSE
            ActualPurchResultDec := AllowedDec - "Actual Costs" - EstimPurchResultDec
        ELSE BEGIN
          ActualPurchResultDec := Budget - "Actual Costs" - EstimPurchResultDec;
          IF (ActualPurchResultDec > 0) THEN BEGIN
            //Determine project in process
            IF "Project No." <> ProjRec."No." THEN
              ProjRec.GET("Project No.");

            IF ("Project No." <> ProjectCostObjectRec."Project No.") OR
               ("Cost Object" <> ProjectCostObjectRec."Cost Object") THEN
              IF NOT ProjectCostObjectRec.GET("Project No.","Cost Object") THEN
                ProjectCostObjectRec."Technical Finished Date"  := 0D;

            IF ProjectCostObjectRec."Technical Finished Date"  <> 0D THEN
              ProjectInProcess := ProjectCostObjectRec."Technical Finished Date" > GETRANGEMAX("Cumulation Filter")
            ELSE
              ProjectInProcess := ((ProjRec."Technical Finished Date" = 0D) OR
                                   (ProjRec."Technical Finished Date" > GETRANGEMAX("Cumulation Filter")));

            IF ProjectInProcess THEN
              ActualPurchResultDec := 0;
          END;
        END;
      END;

      TotPurchResultDec := EstimPurchResultDec + ActualPurchResultDec - Prognosis;
    END;

    LOCAL PROCEDURE DetermineAllowed@1100528200(BudgetDec@1100528200 : Decimal;AmountDec@1100528201 : Decimal) AllowedDec : Decimal;
    BEGIN
      IF BudgetDec >= 0 THEN BEGIN
        IF BudgetDec < AmountDec THEN
          AllowedDec := BudgetDec
        ELSE
          IF (AmountDec >= 0) OR (ProjSetup."Fill Allowed Cost" = ProjSetup."Fill Allowed Cost"::"No restriction") THEN
            AllowedDec := AmountDec;
      END ELSE BEGIN
        IF (BudgetDec > AmountDec) AND (ProjSetup."Fill Allowed Cost" = ProjSetup."Fill Allowed Cost"::"No restriction") THEN
          AllowedDec := BudgetDec
        ELSE
          IF (ProjSetup."Fill Allowed Cost" = ProjSetup."Fill Allowed Cost"::"No restriction") THEN
            AllowedDec := AmountDec;
      END;
    END;

    LOCAL PROCEDURE CheckMaxAllowedWithBudget@1100528206(BudgetDec@1100528201 : Decimal;VAR AllowedDec@1100528200 : Decimal);
    BEGIN
      IF ProjSetup."Progress Range" < ProjSetup."Progress Range"::"No Progress or Allowed Max." THEN BEGIN
        //Allowed may not exceed Budget
        IF BudgetDec >= 0 THEN BEGIN
          IF AllowedDec > BudgetDec THEN
            AllowedDec := BudgetDec;
        END ELSE BEGIN
          IF AllowedDec < BudgetDec THEN
            AllowedDec := BudgetDec;
        END;
      END;
      IF (AllowedDec < 0) AND (ProjSetup."Fill Allowed Cost" = ProjSetup."Fill Allowed Cost"::"Negative not allowed") THEN
        AllowedDec := 0;
    END;
