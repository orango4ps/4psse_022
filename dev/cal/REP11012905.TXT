    PROCEDURE CalculateRubricTotal@1210190010() : Decimal;

    PROCEDURE CalculatePeriodAmounts@1210190006();
    BEGIN
      WITH InvestmProposalRubricLine DO BEGIN
        CLEAR(YearAmntArray);
        FOR Counter := 1 TO NoOfColumn DO BEGIN
          CASE Type OF
            Type::Subtotal:
              BEGIN
                IF Counter <= LastActualColumn THEN
                  YearAmntArray[Counter] := ProjDevelopmentCU.CalculateActualByRubricCostRev(InvestmProposalRubricLine,
                                              GetStartDate(Counter), EndDateArray[Counter], ReferenceDate, CalcCostBln)
                ELSE
                  YearAmntArray[Counter] := ProjDevelopmentCU.CalculateBudgetByRubricCostRev(InvestmProposalRubricLine,
                                              StartDateArray[Counter], GetEndDate(Counter), ReferenceDate, CalcCostBln);
              END;
            Type::Balance:
              BEGIN
                IF Counter <= LastActualColumn THEN
                  YearAmntArray[Counter] := ProjDevelopmentCU.CalculateActualByRubric(InvestmProposalRubricLine,
                                              GetStartDate(Counter), EndDateArray[Counter], ReferenceDate)
                ELSE
                  YearAmntArray[Counter] := ProjDevelopmentCU.CalculateBudgetByRubric(InvestmProposalRubricLine,
                                              StartDateArray[Counter], GetEndDate(Counter), ReferenceDate);
              END;
          END;
        END;
      END;
    END;
