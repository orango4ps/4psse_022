    LOCAL PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      ProjectFilter0@1100529601 : Text;
      ProjectFilter2@1100529600 : Text;
    BEGIN
      ProjectFilter0 := Project.GETFILTERS;
      Project.FILTERGROUP(2);
      ProjectFilter2 := Project.GETFILTERS;
      Project.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Project.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Project.TABLECAPTION, ProjectFilter0));
    END;

    LOCAL PROCEDURE GetPrognosisDate@1100529602();
    VAR
      Prognosis@1100529600 : Record 11012034;
    BEGIN
      PrognosisDate := FilterDate;
      PrevPrognosisDate := 0D;

      Prognosis.SETRANGE("Project No.", Project."No.");
      Prognosis.SETRANGE("Prognosis Date", 0D, FilterDate);
      Prognosis.SETRANGE("Select For Reporting", TRUE);
      IF Prognosis.FINDLAST THEN BEGIN
        PrognosisDate := Prognosis."Prognosis Date";

        Prognosis.SETFILTER("Prognosis Date", '<%1', PrognosisDate);
        IF Prognosis.FINDLAST THEN
          PrevPrognosisDate := Prognosis."Prognosis Date";
      END;
    END;

    LOCAL PROCEDURE GetReportLines@1100529601();
    VAR
      PrognosisLine@1100529603 : Record 11012035;
      ProjectPurchaseAction@1100529600 : Record 11012026;
      TmpCostObject@1100529601 : TEMPORARY Record 349;
      TextFunctionManagement@1100525000 : Codeunit 11012357;
      DescrArray@1100529602 : ARRAY [5] OF Text;
      VendNameArray@1100529604 : ARRAY [5] OF Text;
    BEGIN
      TmpReportLine.RESET;
      TmpReportLine.DELETEALL;
      TmpCommentLine.RESET;
      TmpCommentLine.DELETEALL;

      TmpTotalReportLine.DELETEALL;
      TmpTotalReportLine.INIT;

      PrognosisLine.SETRANGE("Project No.", Project."No.");
      PrognosisLine.SETRANGE("Prognosis Date", PrognosisDate);
      IF PrognosisLine.ISEMPTY THEN
        EXIT;

      ProjectPurchaseAction.SETRANGE("Project No.", Project."No.");
      ProjectPurchaseAction.SETFILTER("Default Cost Object Purchase", '<>%1', '');
      IF ProjectPurchaseAction.ISEMPTY THEN
        EXIT;

      PrognosisLine.SETFILTER("Previous Date Filter", FORMAT(PrevPrognosisDate));
      PrognosisLine.SETAUTOCALCFIELDS("Prev Prognosis End Result");

      ProjectPurchaseAction.FINDSET;
      REPEAT
        TmpCostObject.SETRANGE(Code, ProjectPurchaseAction."Default Cost Object Purchase");
        IF TmpCostObject.ISEMPTY THEN BEGIN
          TmpCostObject.Code := ProjectPurchaseAction."Default Cost Object Purchase";
          TmpCostObject.INSERT;

          TmpReportLine.INIT;
          TmpReportLine."Key Code1" := ProjectPurchaseAction."Default Cost Object Purchase";
          TmpReportLine."Key Code2" := '';
          TmpReportLine."Key Code3" := '';
          TmpReportLine.Text1 := ProjectPurchaseAction.Description;
          TmpReportLine.INSERT;

          GetPurchOrderInfo(ProjectPurchaseAction);
          UpdatePurchActionBudget(ProjectPurchaseAction);
          TmpReportLine.GET(ProjectPurchaseAction."Default Cost Object Purchase");

          IF (NOT TmpReportLine.Bool1) OR ShowPurchaseActionsWithoutPO THEN BEGIN
            PrognosisLine.SETRANGE("Cost Object", ProjectPurchaseAction."Default Cost Object Purchase");
            IF PrognosisLine.FINDSET THEN
              REPEAT
                TmpReportLine.Dec1 += PrognosisLine."Budget Amount";
                TmpReportLine.Dec2 += PrognosisLine."Purchase Amount";
                TmpReportLine.Dec3 += PrognosisLine."Total Risc Coverage Amount";
                TmpReportLine.Dec4 += PrognosisLine."Risc Coverage";

                TmpReportLine.Dec6 += PrognosisLine."Actual Costs (Purchase Action)";
                TmpReportLine.Dec7 += PrognosisLine."Actual Costs" - PrognosisLine."Actual Costs (Purchase Action)";
                TmpReportLine.Dec8 += PrognosisLine."Actual Costs";

                TmpReportLine.Dec10 += PrognosisLine."Prognosis End Result";
                TmpReportLine.Dec11 += PrognosisLine."Prognosis End Result" - PrognosisLine."Prev Prognosis End Result";

                InsertComment(PrognosisLine.Comment);
              UNTIL PrognosisLine.NEXT = 0;

            IF NOT TmpReportLine.Bool1 THEN
              TmpReportLine.Dec5 := TmpReportLine.Dec1 - TmpReportLine.Dec2 - TmpReportLine.Dec3;
            TmpReportLine.Dec9 := TmpReportLine.Dec1 - TmpReportLine.Dec10 - TmpReportLine.Dec8;
            TmpReportLine.MODIFY;

            TmpTotalReportLine.Dec1 += TmpReportLine.Dec1;
            TmpTotalReportLine.Dec2 += TmpReportLine.Dec2;
            TmpTotalReportLine.Dec3 += TmpReportLine.Dec3;
            TmpTotalReportLine.Dec4 += TmpReportLine.Dec4;
            TmpTotalReportLine.Dec5 += TmpReportLine.Dec5;
            TmpTotalReportLine.Dec6 += TmpReportLine.Dec6;
            TmpTotalReportLine.Dec7 += TmpReportLine.Dec7;
            TmpTotalReportLine.Dec8 += TmpReportLine.Dec8;
            TmpTotalReportLine.Dec9 += TmpReportLine.Dec9;
            TmpTotalReportLine.Dec10 += TmpReportLine.Dec10;
            TmpTotalReportLine.Dec11 += TmpReportLine.Dec11;
          END ELSE
            TmpReportLine.DELETE;
        END ELSE
          UpdatePurchActionBudget(ProjectPurchaseAction);
      UNTIL ProjectPurchaseAction.NEXT= 0;

      TmpTotalReportLine.INSERT;

      UpdatePurchOrderInfo;

      IF TmpReportLine.FINDSET THEN
        REPEAT
          CLEAR(DescrArray);
          CLEAR(VendNameArray);
          TextFunctionManagement.Text2Array(TmpReportLine.Text1, 25, DescrArray);
          TextFunctionManagement.Text2Array(TmpReportLine.Text2, 25, VendNameArray);
          TmpReportLine.SplitByTwoArrays(TmpReportLine, DescrArray, VendNameArray, DescrRounding);
          IF PrintPrognosisRemarks THEN BEGIN
            TmpCommentLine.SETRANGE("Key Code1", TmpReportLine."Key Code1");
            IF TmpCommentLine.FINDSET THEN
              REPEAT
                TmpReportLine := TmpCommentLine;
                TmpReportLine.INSERT;
              UNTIL TmpCommentLine.NEXT = 0;
          END;
        UNTIL TmpReportLine.NEXT = 0;
    END;

    LOCAL PROCEDURE GetPurchOrderInfo@1100529605(ProjectPurchaseAction@1100529602 : Record 11012026);
    VAR
      PurchaseHeader@1100529604 : Record 38;
      PurchaseLine@1100529600 : Record 39;
      OnePurchOrder@1100529601 : Boolean;
    BEGIN
      PurchaseLine.SETRANGE("Document Type", PurchaseLine."Document Type"::Order);
      PurchaseLine.SETRANGE("Job No.", Project."No.");
      PurchaseLine.SETRANGE("Shortcut Dimension 2 Code", ProjectPurchaseAction."Default Cost Object Purchase");
      PurchaseLine.SETRANGE("Order Date", 0D, PrognosisDate);
      IF PurchaseLine.ISEMPTY THEN BEGIN
        TmpReportLine.Bool1 := TRUE;
        TmpReportLine.MODIFY;
        EXIT;
      END;

      PurchaseLine.FINDFIRST;
      PurchaseLine.SETFILTER("Document No.", '<>%1', PurchaseLine."Document No.");
      OnePurchOrder := PurchaseLine.ISEMPTY;
      PurchaseLine.SETRANGE("Document No.");

      IF OnePurchOrder THEN BEGIN
        PurchaseHeader.GET(PurchaseLine."Document Type", PurchaseLine."Document No.");
        TmpReportLine.Bool2 := TRUE;
        TmpReportLine.Code1 := PurchaseLine."Document No.";
        TmpReportLine.Text2 := PurchaseHeader."Buy-from Vendor Name";
        TmpReportLine.MODIFY;
        EXIT;
      END;

      IF NOT ShowPurchaseOrderLines THEN
        EXIT;

      TmpReportLine.Text1 := '';
      TmpReportLine.MODIFY;

      PurchaseLine.SETCURRENTKEY("Document Type", "Document No.");
      PurchaseLine.FINDSET;
      REPEAT
        IF NOT OnePurchOrder THEN
          IF TmpReportLine."Key Code3" <> PurchaseLine."Document No." THEN BEGIN
            PurchaseHeader.GET(PurchaseLine."Document Type", PurchaseLine."Document No.");
            TmpReportLine.INIT;
            TmpReportLine."Key Code2" := '';
            TmpReportLine."Key Code3" := PurchaseHeader."No.";
            TmpReportLine.Code1 := PurchaseHeader."No.";
            TmpReportLine.Text2 := PurchaseHeader."Buy-from Vendor Name";
            TmpReportLine.Bool3 := TRUE;
            TmpReportLine.INSERT;
          END;
        TmpReportLine.Dec2 += PurchaseLine."Amount (LCY)";
        TmpReportLine.Dec3 += PurchaseLine."Risc Coverage Amount (LCY)";
        TmpReportLine.MODIFY;
      UNTIL PurchaseLine.NEXT = 0;
    END;

    LOCAL PROCEDURE UpdatePurchActionBudget@1100529628(ProjectPurchaseAction@1100529602 : Record 11012026);
    VAR
      PurchaseLine@1100529600 : Record 39;
      BudgetLine@1100529605 : Record 11012001;
      JobLedgerEntry@1100529601 : Record 11072005;
      BudgetCost@1100529603 : Decimal;
      ActualCost@1100529604 : Decimal;
    BEGIN
      IF NOT ShowPurchaseOrderLines THEN
        EXIT;
      IF NOT TmpReportLine.GET(ProjectPurchaseAction."Default Cost Object Purchase") THEN
        EXIT;
      IF TmpReportLine.Bool2 THEN
        EXIT;

      PurchaseLine.SETRANGE("Document Type", PurchaseLine."Document Type"::Order);
      PurchaseLine.SETRANGE("Job No.", Project."No.");
      PurchaseLine.SETRANGE("Shortcut Dimension 2 Code", ProjectPurchaseAction."Default Cost Object Purchase");
      PurchaseLine.SETRANGE("Order Date", 0D, PrognosisDate);
      PurchaseLine.SETRANGE("Purchase Action", ProjectPurchaseAction.Code);
      IF PurchaseLine.ISEMPTY THEN
        EXIT;

      BudgetLine.SETRANGE("Project No.", ProjectNo);
      BudgetLine.SETRANGE("Cost Object", ProjectPurchaseAction."Default Cost Object Purchase");
      BudgetLine.SETRANGE("Version Date", 0D, PrognosisDate);
      BudgetLine.SETRANGE("Purchase Action", ProjectPurchaseAction.Code);
      BudgetLine.SETFILTER("Extension Contract Status", '%1|(>=%2&<%3)',
        BudgetLine."Extension Contract Status"::"Not Applicable", BudgetLine."Extension Contract Status"::Order, BudgetLine."Extension Contract Status"::Expired);
      IF NOT BudgetLine.ISEMPTY THEN BEGIN
        BudgetLine.CALCSUMS("Amount (LCY)");
        BudgetCost := BudgetLine."Amount (LCY)";
      END;

      JobLedgerEntry.SETRANGE("Job No.", ProjectNo);
      JobLedgerEntry.SETRANGE("Entry Type", JobLedgerEntry."Entry Type"::Usage);
      JobLedgerEntry.SETRANGE("Global Dimension 2 Code", ProjectPurchaseAction."Default Cost Object Purchase");
      JobLedgerEntry.SETRANGE("Purchase Action", ProjectPurchaseAction.Code);
      JobLedgerEntry.SETRANGE("Posting Date", 0D, PrognosisDate);
      IF NOT JobLedgerEntry.ISEMPTY THEN BEGIN
        JobLedgerEntry.CALCSUMS("Total Cost (LCY)");
        ActualCost := JobLedgerEntry."Total Cost (LCY)";
      END;

      TmpReportLine.SETRANGE("Key Code1", ProjectPurchaseAction."Default Cost Object Purchase");
      TmpReportLine.SETFILTER("Key Code3", '<>%1', '');
      IF TmpReportLine.FINDSET THEN
        REPEAT
          PurchaseLine.SETRANGE("Document No.", TmpReportLine."Key Code3");
          IF NOT PurchaseLine.ISEMPTY THEN BEGIN
            TmpReportLine.Text1 := ProjectPurchaseAction.Description;
            TmpReportLine.Dec1 := BudgetCost;

    LOCAL PROCEDURE GetFilterDate@1100529604();
    VAR
      Prognosis@1100529600 : Record 11012034;
    BEGIN
      IF ProjectNo = '' THEN BEGIN
        FilterDate := 0D;
        EXIT;
      END;

      IF NOT Prognosis.GET(ProjectNo, FilterDate) THEN
        FilterDate := 0D
      ELSE
        IF NOT Prognosis."Select For Reporting" THEN
          FilterDate := 0D;

      IF FilterDate = 0D THEN BEGIN
        Prognosis.SETRANGE("Project No.", ProjectNo);
        Prognosis.SETRANGE("Select For Reporting", TRUE);
        IF Prognosis.FINDLAST THEN
          FilterDate := Prognosis."Prognosis Date";
        EXIT;
      END;
    END;

    LOCAL PROCEDURE RoundAmount@1100529634(Value@1102 : Decimal) : Text;
    BEGIN
      CASE RoundingFactor OF

        RoundingFactor::"1000000":
          EXIT(FORMAT(ROUND(Value / 1000000, 0.1), 0, Text010));
      END;
    END;
