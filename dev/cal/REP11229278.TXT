    PROCEDURE FillTempCompany@1100409000();
    VAR
      CompanyPlanGroupLine@1100528600 : Record 11229301;
      Company@1100409000 : Record 2000000006;
    BEGIN
      TempCompany.RESET;
      TempCompany.DELETEALL;

      CompanyPlanGroupLine.SETRANGE("Group Code", "Company Plan Group".Code);
      IF CompanyPlanGroupLine.ISEMPTY THEN BEGIN
        IF Company.FINDSET THEN
          REPEAT
            TempCompany.COPY(Company);
            TempCompany.INSERT;
          UNTIL Company.NEXT = 0;
      END ELSE BEGIN
        IF CompanyPlanGroupLine.FINDSET THEN
          REPEAT
            TempCompany.Name := CompanyPlanGroupLine."Company Name";
            TempCompany.INSERT;
          UNTIL CompanyPlanGroupLine.NEXT = 0;
      END;
      IF TempCompany.ISEMPTY THEN
        ERROR(Text000, "Company Plan Group".Code);
    END;

    BEGIN

    newgroup = False
    If Not (group1 = currentgroup1)
        newgroup = True
        currentgroup2 = group2
        currentgroup1 = group1

        If Not (group2 = currentgroup2)
            newgroup = True
            currentgroup2 = group2

    End If
