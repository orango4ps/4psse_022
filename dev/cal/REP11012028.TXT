    PROCEDURE GetReportFilter@1100525000() : Text;
    VAR
      ProjectFilter0@1100525000 : Text;
      ProjectFilter2@1100525001 : Text;
    BEGIN
      ProjectFilter0 := Job.GETFILTERS;
      Job.FILTERGROUP(2);
      ProjectFilter2 := Job.GETFILTERS;
      Job.FILTERGROUP(0);

      IF (ProjectFilter0 = '') AND (ProjectFilter2 = '') THEN
        EXIT;

      IF ProjectFilter2 <> '' THEN
        EXIT(STRSUBSTNO('%1: %2, %3', Job.TABLECAPTION, ProjectFilter0, ProjectFilter2))
      ELSE
        EXIT(STRSUBSTNO('%1: %2', Job.TABLECAPTION, ProjectFilter0));
    END;

    PROCEDURE RemoveFlowFieldFilters@1100485002();
    VAR
      FieldRec@1100485004 : Record 2000000041;
      RecRef@1100485001 : RecordRef;
      FldRef@1100485003 : FieldRef;
      i@1100485002 : Integer;
    BEGIN
      RecRef.GETTABLE(Job);
      FOR i := 1 TO RecRef.FIELDCOUNT DO BEGIN
        FldRef := RecRef.FIELDINDEX(i);
        FieldRec.GET(RecRef.NUMBER, FldRef.NUMBER);
        IF FieldRec.Class = FieldRec.Class::FlowField THEN
          FldRef.SETRANGE;
      END;
      Job.SETVIEW(RecRef.GETVIEW);
    END;

    PROCEDURE CheckDataWithinProjFilters@1100485001() : Boolean;

      i@1210190000 : Integer;
    BEGIN

      EXIT(RecRef2.FIND('='));
    END;

    LOCAL PROCEDURE RoundAmount@1102(Value@1102 : Decimal) ValueAsText : Text[30];
    VAR
      GLSetup@1100529600 : Record 98;
    BEGIN
      IF NormalFormatString = '' THEN BEGIN
        GLSetup.GET;
        NormalFormatString := Text001 + GLSetup."Amount Decimal Places" + Text002;
      END;

      CASE RoundingFactor OF
        RoundingFactor::None:
          ValueAsText := FORMAT(Value, 0, NormalFormatString);
        RoundingFactor::"1":
          ValueAsText := FORMAT(ROUND(Value, 1));
        RoundingFactor::"1000":
          ValueAsText := FORMAT(ROUND(Value / 1000, 0.1), 0, Text003);
        RoundingFactor::"1000000":
          ValueAsText := FORMAT(ROUND(Value / 1000000, 0.1), 0, Text003);
      END;
    END;
