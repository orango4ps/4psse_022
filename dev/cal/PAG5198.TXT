                         ("Attendee No." <> "Attendee No.") OR

                          ("Attendance Type" <> "Attendance Type"::"To-do Organizer")))
                     THEN
                       ERROR(Text001,TABLECAPTION);

                     COPY(xAttendee);
                     MODIFY;
                     EXIT(FALSE);
                   END;
